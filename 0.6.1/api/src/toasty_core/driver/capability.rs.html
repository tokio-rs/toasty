3/// Describes what a database driver supports.
4///
5/// The query planner reads these flags to decide which [`Operation`](super::Operation)
6/// variants to generate. For example, a SQL driver sets `sql: true` and
7/// receives `QuerySql` operations, while DynamoDB sets `sql: false` and
8/// receives key-value operations like `GetByKey` and `QueryPk`.
9///
10/// Pre-built configurations are available as associated constants:
11/// [`SQLITE`](Self::SQLITE), [`POSTGRESQL`](Self::POSTGRESQL),
12/// [`MYSQL`](Self::MYSQL), and [`DYNAMODB`](Self::DYNAMODB).
13///
14/// # Examples
15///
16/// ```
17/// use toasty_core::driver::Capability;
18///
19/// let cap = &Capability::SQLITE;
20/// assert!(cap.sql);
21/// assert!(cap.returning_from_mutation);
22/// assert!(!cap.select_for_update);
23/// ```
24#[derive(Debug)]
25pub struct Capability {
26    /// When `true`, the database uses a SQL-based query language and the
27    /// planner will emit [`QuerySql`](super::operation::QuerySql) operations.
28    pub sql: bool,
29
30    /// Column storage types supported by the database.
31    pub storage_types: StorageTypes,
32
33    /// What the database is able to change about its own schema. See
34    /// [`SchemaMutations`] for the individual fields; the migration
35    /// generator branches on them to choose between an in-place
36    /// `ALTER COLUMN` and a table rebuild, and between one combined
37    /// alter statement and several single-property ones.
38    pub schema_mutations: SchemaMutations,
39
40    /// SQL: supports update statements in CTE queries.
41    pub cte_with_update: bool,
42
43    /// SQL: Supports row-level locking. If false, then the driver is expected
44    /// to serializable transaction-level isolation.
45    pub select_for_update: bool,
46
47    /// SQL: Mysql doesn't support returning clauses from insert / update queries
48    pub returning_from_mutation: bool,
49
50    /// DynamoDB does not support != predicates on the primary key.
51    pub primary_key_ne_predicate: bool,
52
53    /// Whether the database has an auto increment modifier for integer columns.
54    pub auto_increment: bool,
55
56    /// Whether the database supports `VARCHAR(n)` column types natively.
57    ///
58    /// Must be consistent with [`StorageTypes::varchar`]: when `true`,
59    /// `varchar` must be `Some`; when `false`, `varchar` must be `None`.
60    /// Use [`Capability::validate`] to check this invariant.
61    pub native_varchar: bool,
62
63    /// Whether the database has native support for Timestamp types.
64    pub native_timestamp: bool,
65
66    /// Whether the database has native support for Date types.
67    pub native_date: bool,
68
69    /// Whether the database has native support for Time types.
70    pub native_time: bool,
71
72    /// Whether the database has native support for DateTime types.
73    pub native_datetime: bool,
74
75    /// Whether the database supports native enum types.
76    ///
77    /// - PostgreSQL: `true` — `CREATE TYPE ... AS ENUM`
78    /// - MySQL: `true` — inline `ENUM('a', 'b')` column type
79    /// - SQLite: `false` — uses `TEXT` + `CHECK` constraint
80    /// - DynamoDB: `false` — plain string attribute
81    pub native_enum: bool,
82
83    /// Whether enum types are standalone named objects requiring separate DDL.
84    ///
85    /// When `true`, migrations must emit `CREATE TYPE` / `ALTER TYPE` for enum
86    /// types. When `false`, enum definitions are inline in column types.
87    ///
88    /// - PostgreSQL: `true` — `CREATE TYPE <name> AS ENUM (...)`
89    /// - MySQL: `false` — inline `ENUM('a', 'b')` on the column
90    /// - SQLite: `false`
91    /// - DynamoDB: `false`
92    pub named_enum_types: bool,
93
94    /// Whether the database has native support for Decimal types.
95    pub native_decimal: bool,
96
97    /// Whether BigDecimal driver support is implemented.
98    /// TODO: Remove this flag when PostgreSQL BigDecimal support is implemented.
99    /// Currently only MySQL has implemented BigDecimal driver support.
100    pub bigdecimal_implemented: bool,
101
102    /// Whether the database's decimal type supports arbitrary precision.
103    /// When false, the decimal type requires fixed precision and scale to be specified upfront.
104    /// - PostgreSQL: true (NUMERIC supports arbitrary precision)
105    /// - MySQL: false (DECIMAL requires fixed precision/scale)
106    /// - SQLite/DynamoDB: false (no native decimal support, stored as TEXT)
107    pub decimal_arbitrary_precision: bool,
108
109    /// Whether OR is supported in index key conditions (e.g. DynamoDB KeyConditionExpression).
110    /// DynamoDB: false. All other backends: true (SQL backends never use index key conditions).
111    pub index_or_predicate: bool,
112
113    /// Whether the database has a native prefix-match operator that does not
114    /// require LIKE-style escaping. When `true`, `starts_with` is left in the
115    /// AST and the driver renders it natively (DynamoDB's `begins_with()`,
116    /// PostgreSQL's `^@`). When `false`, the lowering rewrites it to a
117    /// `LIKE` expression — which requires `native_like` to be `true`.
118    pub native_starts_with: bool,
119
120    /// Whether the database has a native `LIKE` expression. When `false`,
121    /// `Expr::Like` cannot be sent to the driver; `starts_with` lowering
122    /// will not produce one.
123    pub native_like: bool,
124
125    /// Whether the driver can answer queries that don't match any primary key
126    /// or index — i.e. supports unindexed full-table reads.
127    ///
128    /// SQL drivers set this to `true`: unindexed queries go through
129    /// [`QuerySql`](super::operation::QuerySql), so the SQL engine handles
130    /// them transparently. DynamoDB also sets this to `true`; the planner
131    /// emits [`Operation::Scan`](super::Operation::Scan) for the unindexed
132    /// case. A hypothetical pure key-value store with no full-scan capability
133    /// would set this to `false`.
134    pub scan: bool,
135
136    /// Whether scan operations support ordering results.
137    ///
138    /// SQL drivers do not use `Operation::Scan`, so this is `true` for them
139    /// (ordering is handled inside `QuerySql`). DynamoDB's `Scan` API returns
140    /// items in an arbitrary order with no server-side sort, so this is `false`
141    /// for DynamoDB. When `false`, the planner rejects queries that combine a
142    /// scan path with `ORDER BY`.
143    pub scan_supports_sort: bool,
144
145    /// Whether to test connection pool behavior.
146    /// TODO: We only need this for the `connection_per_clone.rs` test, come up with a better way.
147    pub test_connection_pool: bool,
148
149    /// Whether the backend can walk a paginated query in reverse from a
150    /// cursor.
151    ///
152    /// Gates the `prev_cursor` field on a `Page` returned to user code.
153    /// When `true`, the executor extracts a previous-page cursor from the
154    /// first row of every page (see `apply_sql_pagination` in
155    /// `toasty/src/engine/exec/exec_statement.rs`). When `false`, the
156    /// executor leaves `prev_cursor` as `None`, so
157    /// `Page::has_prev()` returns `false` and `Page::prev(&db)` resolves
158    /// to `Ok(None)` without issuing a query. `Paginate::before(cursor)`
159    /// itself is not rejected — users who already hold a cursor can walk
160    /// backwards explicitly — but a driver that returns `false` is
161    /// declaring that it has no way to *produce* such a cursor.
162    ///
163    /// Drivers should set this to `true` when the backend can answer a
164    /// query equivalent to "rows ordered by K, descending from K = c,
165    /// limited to N" — i.e. the same `ORDER BY` clause reversed plus a
166    /// strict inequality on the cursor key. SQL backends meet this
167    /// trivially. DynamoDB does not: a `Query` with `ScanIndexForward =
168    /// false` returns rows in the opposite direction but cannot be
169    /// rooted at an arbitrary client-supplied cursor without an extra
170    /// `KeyConditionExpression`, and `Scan` has no order guarantee at
171    /// all.
172    pub backward_pagination: bool,
173
174    /// The driver's bind layer accepts a single parameter whose value is
175    /// `Value::List(items)` and type is `Type::List(elem)`, sending it as
176    /// one protocol-level parameter (not N separate scalars).
177    /// Property of the driver bind impl, not the SQL dialect.
178    pub bind_list_param: bool,
179
180    /// The SQL dialect parses `expr <op> ANY(<array>)` and `expr <op> ALL(<array>)`
181    /// as predicates against an array-valued operand.
182    /// Property of the dialect, not the bind layer.
183    pub predicate_match_any: bool,
184
185    /// Whether the database can store a `Vec<scalar>` model field as a native
186    /// array column (e.g. PostgreSQL `text[]`, `int8[]`).
187    ///
188    /// When `true`, schema build maps `Type::List(elem)` to `db::Type::List(elem)`
189    /// and the driver's bind layer accepts `Value::List(items)` as a single
190    /// array-valued parameter.
191    ///
192    /// When `false`, `Vec<T>` model fields use whatever fallback the backend
193    /// provides (JSON column on MySQL/SQLite, native List `L` on DynamoDB).
194    /// See [`Self::vec_scalar`] for the schema-build gate.
195    pub native_array: bool,
196
197    /// Whether the driver supports `Vec<scalar>` model fields, by whatever
198    /// representation (native typed array column, JSON column, key-value
199    /// list attribute, ...). Used by the schema builder as the gate for
200    /// accepting `stmt::Type::List(_)` fields.
201    pub vec_scalar: bool,
202
203    /// Whether the driver natively renders `IsSuperset` / `Intersects` array
204    /// predicates over an arbitrary right-hand-side expression.
205    ///
206    /// SQL drivers set this to `true`: each dialect has a single operator
207    /// (`@>` on PostgreSQL, `JSON_CONTAINS` on MySQL, a `json_each`
208    /// subquery on SQLite) that takes the rhs as a bound expression
209    /// regardless of its shape.
210    ///
211    /// DynamoDB sets this to `false`: it has no equivalent operator and
212    /// emulates the predicates by emitting one `contains(path, vN)` clause
213    /// per rhs element, which requires the rhs to be a concrete list of
214    /// values at filter-construction time. The capability check rejects
215    /// any other rhs shape before the driver is invoked.
216    pub native_array_set_predicates: bool,
217
218    /// Whether the driver supports atomic in-place removal of every element
219    /// equal to a given value from a `Vec<scalar>` field (`stmt::remove`).
220    ///
221    /// - PostgreSQL `text[]`: `true` — `array_remove(col, v)`.
222    /// - MySQL / SQLite JSON: `false` — no value-removal operator.
223    /// - DynamoDB List: `false` — no value-removal on Lists.
224    pub vec_remove: bool,
225
226    /// Whether the driver supports atomic in-place removal of the last
227    /// element of a `Vec<scalar>` field (`stmt::pop`).
228    ///
229    /// - PostgreSQL: `true` — array slicing.
230    /// - MySQL / SQLite: `false`.
231    /// - DynamoDB: `false` — `UpdateExpression` indices must be literal
232    ///   integers, so the last index cannot be expressed in one statement.
233    pub vec_pop: bool,
234
235    /// Whether the driver supports atomic in-place removal of an element at a
236    /// given index from a `Vec<scalar>` field (`stmt::remove_at`).
237    ///
238    /// - PostgreSQL: `true` — array slicing.
239    /// - MySQL / SQLite: `false`.
240    /// - DynamoDB: `false`.
241    pub vec_remove_at: bool,
242}

244/// Maps application-level types to the concrete database column types used for
245/// storage.
246///
247/// Each database has different native type support. For example, PostgreSQL has
248/// a native `UUID` type while SQLite stores UUIDs as `BLOB`. This struct
249/// captures those mappings so the schema layer can generate correct DDL and the
250/// driver can encode/decode values appropriately.
251///
252/// Pre-built configurations: [`SQLITE`](Self::SQLITE),
253/// [`POSTGRESQL`](Self::POSTGRESQL), [`MYSQL`](Self::MYSQL),
254/// [`DYNAMODB`](Self::DYNAMODB).
255///
256/// # Examples
257///
258/// ```
259/// use toasty_core::driver::StorageTypes;
260///
261/// let st = &StorageTypes::POSTGRESQL;
262/// // PostgreSQL stores UUIDs natively
263/// assert!(matches!(st.default_uuid_type, toasty_core::schema::db::Type::Uuid));
264/// ```
265#[derive(Debug)]
266pub struct StorageTypes {
267    /// The default storage type for a string.
268    pub default_string_type: db::Type,
269
270    /// When `Some` the database supports varchar types with the specified upper
271    /// limit.
272    pub varchar: Option<u64>,
273
274    /// The default storage type for a UUID.
275    pub default_uuid_type: db::Type,
276
277    /// The default storage type for Bytes (Vec<u8>).
278    pub default_bytes_type: db::Type,
279
280    /// The default storage type for a Decimal (fixed-precision decimal).
281    pub default_decimal_type: db::Type,
282
283    /// The default storage type for a BigDecimal (arbitrary-precision decimal).
284    pub default_bigdecimal_type: db::Type,
285
286    /// The default storage type for a Timestamp (instant in time).
287    pub default_timestamp_type: db::Type,
288
289    /// The default storage type for a Zoned (timezone-aware instant).
290    pub default_zoned_type: db::Type,
291
292    /// The default storage type for a Date (civil date).
293    pub default_date_type: db::Type,
294
295    /// The default storage type for a Time (wall clock time).
296    pub default_time_type: db::Type,
297
298    /// The default storage type for a DateTime (civil datetime).
299    pub default_datetime_type: db::Type,
300
301    /// Maximum value for unsigned integers. When `Some`, unsigned integers
302    /// are limited to this value. When `None`, full u64 range is supported.
303    pub max_unsigned_integer: Option<u64>,
304}

306/// The database's capabilities to mutate the schema (tables, columns, indices).
307///
308/// Used by the migration generator to decide how to express each
309/// column change. `alter_column_type` gates whether an in-place
310/// `ALTER COLUMN` is possible at all — SQLite has it set to `false`,
311/// and a type change there triggers a full table rebuild (create
312/// new table, copy rows, drop old). `alter_column_properties_atomic`
313/// decides whether several column-property changes (rename, retype,
314/// `NOT NULL`, default) collapse into one statement or emit one per
315/// property. MySQL sets both to `true`; PostgreSQL alters in place
316/// but requires one statement per property.
317///
318/// Pre-built configurations: [`SQLITE`](Self::SQLITE),
319/// [`POSTGRESQL`](Self::POSTGRESQL), [`MYSQL`](Self::MYSQL),
320/// [`DYNAMODB`](Self::DYNAMODB).
321///
322/// # Examples
323///
324/// Access through [`Capability::schema_mutations`]:
325///
326/// ```
327/// use toasty_core::driver::Capability;
328///
329/// let cap = &Capability::POSTGRESQL;
330/// assert!(cap.schema_mutations.alter_column_type);
331/// assert!(!cap.schema_mutations.alter_column_properties_atomic);
332/// ```
333#[derive(Debug)]
334pub struct SchemaMutations {
335    /// Whether the database can change the type of an existing column.
336    pub alter_column_type: bool,
337
338    /// Whether the database can change name, type and constraints of a column all
339    /// withing a single statement.
340    pub alter_column_properties_atomic: bool,
341}

343impl Capability {
344    /// Validates the consistency of the capability configuration.
345    ///
346    /// This performs sanity checks to ensure the capability fields are
347    /// internally consistent. For example, if `native_varchar` is true,
348    /// then `storage_types.varchar` must be Some, and vice versa.
349    ///
350    /// Returns an error if any inconsistencies are found.
351    pub fn validate(&self) -> crate::Result<()> {
352        // Validate varchar consistency
353        if self.native_varchar && self.storage_types.varchar.is_none() {
354            return Err(crate::Error::invalid_driver_configuration(
355                "native_varchar is true but storage_types.varchar is None",
356            ));
357        }
358
359        if !self.native_varchar && self.storage_types.varchar.is_some() {
360            return Err(crate::Error::invalid_driver_configuration(
361                "native_varchar is false but storage_types.varchar is Some",
362            ));
363        }
364
365        Ok(())
366    }
367
368    /// Returns the default string length limit for this database.
369    ///
370    /// This is useful for tests and applications that need to respect
371    /// database-specific string length constraints.
372    pub fn default_string_max_length(&self) -> Option<u64> {
373        match &self.storage_types.default_string_type {
374            db::Type::VarChar(len) => Some(*len),
375            _ => None, // Handle other types gracefully
376        }
377    }
378
379    /// Returns the native database type for an application-level type.
380    ///
381    /// If the database supports the type natively, returns the same type.
382    /// Otherwise, returns the bridge/storage type that the application type
383    /// maps to in this database.
384    ///
385    /// This uses the existing `db::Type::bridge_type()` method to determine
386    /// the appropriate bridge type based on the database's storage capabilities.
387    pub fn native_type_for(&self, ty: &stmt::Type) -> stmt::Type {
388        match ty {
389            stmt::Type::Uuid => self.storage_types.default_uuid_type.bridge_type(ty),
390            #[cfg(feature = "jiff")]
391            stmt::Type::Timestamp => self.storage_types.default_timestamp_type.bridge_type(ty),
392            #[cfg(feature = "jiff")]
393            stmt::Type::Zoned => self.storage_types.default_zoned_type.bridge_type(ty),
394            #[cfg(feature = "jiff")]
395            stmt::Type::Date => self.storage_types.default_date_type.bridge_type(ty),
396            #[cfg(feature = "jiff")]
397            stmt::Type::Time => self.storage_types.default_time_type.bridge_type(ty),
398            #[cfg(feature = "jiff")]
399            stmt::Type::DateTime => self.storage_types.default_datetime_type.bridge_type(ty),
400            _ => ty.clone(),
401        }
402    }

617        vec_remove: false,
618        vec_pop: false,
619        vec_remove_at: false,
620    };
621}
622
623impl StorageTypes {
624    /// SQLite storage types
625    pub const SQLITE: StorageTypes = StorageTypes {
626        default_string_type: db::Type::Text,
627
628        // SQLite doesn't really enforce the "N" in VARCHAR(N) at all – it
629        // treats any type containing "CHAR", "CLOB", or "TEXT" as having TEXT
630        // affinity, and simply ignores the length specifier. In other words,
631        // whether you declare a column as VARCHAR(10), VARCHAR(1000000), or
632        // just TEXT, SQLite won't truncate or complain based on that number.
633        //
634        // Instead, the only hard limit on how big a string (or BLOB) can be is
635        // the SQLITE_MAX_LENGTH parameter, which is set to 1 billion by default.
636        varchar: Some(1_000_000_000),
637
638        // SQLite does not have an inbuilt UUID type. The binary blob type is more
639        // difficult to read than Text but likely has better performance characteristics.
640        default_uuid_type: db::Type::Blob,
641
642        default_bytes_type: db::Type::Blob,
643
644        // SQLite does not have a native decimal type. Store as TEXT.
645        default_decimal_type: db::Type::Text,
646        default_bigdecimal_type: db::Type::Text,
647
648        // SQLite does not have native date/time types. Store as TEXT in ISO 8601 format.
649        default_timestamp_type: db::Type::Text,
650        default_zoned_type: db::Type::Text,
651        default_date_type: db::Type::Text,
652        default_time_type: db::Type::Text,
653        default_datetime_type: db::Type::Text,
654
655        // SQLite INTEGER is a signed 64-bit integer, so unsigned integers
656        // are limited to i64::MAX to prevent overflow
657        max_unsigned_integer: Some(i64::MAX as u64),
658    };
659
660    /// PostgreSQL storage types.
661    pub const POSTGRESQL: StorageTypes = StorageTypes {
662        default_string_type: db::Type::Text,
663
664        // The maximum n you can specify is 10 485 760 characters. Attempts to
665        // declare varchar with a larger typmod will be rejected at
666        // table‐creation time.
667        varchar: Some(10_485_760),
668
669        default_uuid_type: db::Type::Uuid,
670
671        default_bytes_type: db::Type::Blob,
672
673        // PostgreSQL has native NUMERIC type for fixed and arbitrary-precision decimals.
674        default_decimal_type: db::Type::Numeric(None),
675        // TODO: PostgreSQL has native NUMERIC type for arbitrary-precision decimals,
676        // but the encoding is complicated and has to be done separately in the future.
677        default_bigdecimal_type: db::Type::Text,
678
679        // PostgreSQL has native support for temporal types with microsecond precision (6 digits)
680        default_timestamp_type: db::Type::Timestamp(6),
681        default_zoned_type: db::Type::Text,
682        default_date_type: db::Type::Date,
683        default_time_type: db::Type::Time(6),
684        default_datetime_type: db::Type::DateTime(6),
685
686        // PostgreSQL BIGINT is signed 64-bit, so unsigned integers are limited
687        // to i64::MAX. While NUMERIC could theoretically support larger values,
688        // we prefer explicit limits over implicit type switching.
689        max_unsigned_integer: Some(i64::MAX as u64),
690    };
691
692    /// MySQL storage types.
693    pub const MYSQL: StorageTypes = StorageTypes {
694        default_string_type: db::Type::VarChar(191),
695
696        // Values in VARCHAR columns are variable-length strings. The length can
697        // be specified as a value from 0 to 65,535. The effective maximum
698        // length of a VARCHAR is subject to the maximum row size (65,535 bytes,
699        // which is shared among all columns) and the character set used.
700        varchar: Some(65_535),
701
702        // MySQL does not have an inbuilt UUID type. The binary blob type is
703        // more difficult to read than Text but likely has better performance
704        // characteristics. However, limitations in the engine make it easier to
705        // use VarChar for now.
706        default_uuid_type: db::Type::VarChar(36),
707
708        default_bytes_type: db::Type::Blob,
709
710        // MySQL does not have an arbitrary-precision decimal type. The DECIMAL type
711        // requires a fixed precision and scale to be specified upfront. Store as TEXT.
712        default_decimal_type: db::Type::Text,
713        default_bigdecimal_type: db::Type::Text,
714
715        // MySQL has native support for temporal types with microsecond precision (6 digits)
716        // The `TIMESTAMP` time only supports a limited range (1970-2038), so we default to
717        // DATETIME and let Toasty do the UTC conversion.
718        default_timestamp_type: db::Type::DateTime(6),
719        default_zoned_type: db::Type::Text,
720        default_date_type: db::Type::Date,
721        default_time_type: db::Type::Time(6),
722        default_datetime_type: db::Type::DateTime(6),
723
724        // MySQL supports full u64 range via BIGINT UNSIGNED
725        max_unsigned_integer: None,
726    };
727
728    /// DynamoDB storage types.
729    pub const DYNAMODB: StorageTypes = StorageTypes {
730        default_string_type: db::Type::Text,
731
732        // DynamoDB does not support varchar types
733        varchar: None,
734
735        default_uuid_type: db::Type::Text,
736
737        default_bytes_type: db::Type::Blob,
738
739        // DynamoDB does not have a native decimal type. Store as TEXT.
740        default_decimal_type: db::Type::Text,
741        default_bigdecimal_type: db::Type::Text,
742
743        // DynamoDB does not have native date/time types. Store as TEXT (strings).
744        default_timestamp_type: db::Type::Text,
745        default_zoned_type: db::Type::Text,
746        default_date_type: db::Type::Text,
747        default_time_type: db::Type::Text,
748        default_datetime_type: db::Type::Text,
749
750        // DynamoDB supports full u64 range (numbers stored as strings)
751        max_unsigned_integer: None,
752    };
753}

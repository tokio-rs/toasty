3use crate::{
4    Result,
5    stmt::{Type, Value},
6};

8impl Type {
9    /// Casts a [`Value`] to this jiff temporal type, returning the converted value.
10    ///
11    /// Supports conversions between:
12    /// - `String` and any jiff type (parsing ISO 8601 format)
13    /// - Any jiff type and `String` (formatting with fixed 9-digit nanosecond precision)
14    /// - `Timestamp` and `Zoned` (via UTC timezone)
15    /// - `Timestamp`/`Zoned` and `DateTime` (via UTC timezone)
16    ///
17    /// Returns `Ok(None)` if the conversion is not supported for this
18    /// value/type combination. Returns `Err` if parsing fails.
19    pub fn cast_jiff(&self, value: &Value) -> Result<Option<Value>> {
20        Ok(Some(match (value, self) {
21            // String -> jiff
22            (Value::String(value), Type::Timestamp) => {
23                let v = value.clone();
24                Value::Timestamp(
25                    value.parse().map_err(|_| {
26                        crate::Error::type_conversion(Value::String(v), "Timestamp")
27                    })?,
28                )
29            }
30            (Value::String(value), Type::Zoned) => {
31                let v = value.clone();
32                Value::Zoned(
33                    value
34                        .parse()
35                        .map_err(|_| crate::Error::type_conversion(Value::String(v), "Zoned"))?,
36                )
37            }
38            (Value::String(value), Type::Date) => {
39                let v = value.clone();
40                Value::Date(
41                    value
42                        .parse()
43                        .map_err(|_| crate::Error::type_conversion(Value::String(v), "Date"))?,
44                )
45            }
46            (Value::String(value), Type::Time) => {
47                let v = value.clone();
48                Value::Time(
49                    value
50                        .parse()
51                        .map_err(|_| crate::Error::type_conversion(Value::String(v), "Time"))?,
52                )
53            }
54            (Value::String(value), Type::DateTime) => {
55                let v = value.clone();
56                Value::DateTime(
57                    value
58                        .parse()
59                        .map_err(|_| crate::Error::type_conversion(Value::String(v), "DateTime"))?,
60                )
61            }
62
63            // Identity
64            (value @ Value::Timestamp(_), Type::Timestamp) => value.clone(),
65            (value @ Value::Zoned(_), Type::Zoned) => value.clone(),
66            (value @ Value::Date(_), Type::Date) => value.clone(),
67            (value @ Value::Time(_), Type::Time) => value.clone(),
68            (value @ Value::DateTime(_), Type::DateTime) => value.clone(),
69
70            // jiff -> String
71            //
72            // Types with sub-second precision use fixed 9-digit nanosecond
73            // formatting so that the resulting strings sort lexicographically
74            // in chronological order (ISO 8601 guarantees this for
75            // uniform-precision representations).
76            (Value::Timestamp(value), Type::String) => Value::String(format!("{value:.9}")),
77            (Value::Zoned(value), Type::String) => Value::String(format!("{value:.9}")),
78            (Value::Date(value), Type::String) => Value::String(value.to_string()),
79            (Value::Time(value), Type::String) => Value::String(format!("{value:.9}")),
80            (Value::DateTime(value), Type::String) => Value::String(format!("{value:.9}")),
81
82            // UTC <-> Zoned
83            (Value::Timestamp(value), Type::Zoned) => Value::Zoned(value.to_zoned(TimeZone::UTC)),
84            (Value::Zoned(value), Type::Timestamp) => Value::Timestamp(value.into()),
85
86            // UTC <-> Civil
87            (Value::Timestamp(value), Type::DateTime) => {
88                Value::DateTime(value.to_zoned(TimeZone::UTC).into())
89            }
90            (Value::DateTime(value), Type::Timestamp) => Value::Timestamp(
91                value
92                    .to_zoned(TimeZone::UTC)
93                    .expect("value was too close to minimum or maximum DateTime")
94                    .into(),
95            ),
96
97            // Zoned <-> Civil
98            (Value::Zoned(value), Type::DateTime) => Value::DateTime(value.into()),
99            (Value::DateTime(value), Type::Zoned) => Value::Zoned(
100                value
101                    .to_zoned(TimeZone::UTC)
102                    .expect("value was too close to minimum or maximum DateTime"),
103            ),
104
105            _ => return Ok(None),
106        }))
107    }
108}

19    /// Converts a DynamoDB AttributeValue to a Toasty value.
20    pub fn from_ddb(ty: &stmt::Type, val: &AttributeValue) -> Self {
21        use AttributeValue as AV;
22        use stmt::Type;
23
24        let core_value = match (ty, val) {
25            (Type::Bool, AV::Bool(val)) => stmt::Value::from(*val),
26            (Type::String, AV::S(val)) => stmt::Value::from(val.clone()),
27            (Type::I8, AV::N(val)) => stmt::Value::from(val.parse::<i8>().unwrap()),
28            (Type::I16, AV::N(val)) => stmt::Value::from(val.parse::<i16>().unwrap()),
29            (Type::I32, AV::N(val)) => stmt::Value::from(val.parse::<i32>().unwrap()),
30            (Type::I64, AV::N(val)) => stmt::Value::from(val.parse::<i64>().unwrap()),
31            (Type::U8, AV::N(val)) => stmt::Value::from(val.parse::<u8>().unwrap()),
32            (Type::U16, AV::N(val)) => stmt::Value::from(val.parse::<u16>().unwrap()),
33            (Type::U32, AV::N(val)) => stmt::Value::from(val.parse::<u32>().unwrap()),
34            (Type::U64, AV::N(val)) => stmt::Value::from(val.parse::<u64>().unwrap()),
35            (Type::F32, AV::N(val)) => stmt::Value::from(val.parse::<f32>().unwrap()),
36            (Type::F64, AV::N(val)) => stmt::Value::from(val.parse::<f64>().unwrap()),
37            (Type::Bytes, AV::B(val)) => stmt::Value::Bytes(val.clone().into_inner()),
38            (Type::Uuid, AV::S(val)) => stmt::Value::from(val.parse::<uuid::Uuid>().unwrap()),
39            (Type::List(elem), AV::L(items)) => {
40                let items = items
41                    .iter()
42                    .map(|item| Value::from_ddb(elem, item).into_inner())
43                    .collect();
44                stmt::Value::List(items)
45            }
46            _ => todo!("ty={:#?}; value={:#?}", ty, val),
47        };
48
49        Value(core_value)
50    }

52    /// Converts this value to a DynamoDB AttributeValue.
53    pub fn to_ddb(&self) -> AttributeValue {
54        use AttributeValue as AV;
55
56        match &self.0 {
57            stmt::Value::Bool(val) => AV::Bool(*val),
58            stmt::Value::String(val) => AV::S(val.to_string()),
59            stmt::Value::I8(val) => AV::N(val.to_string()),
60            stmt::Value::I16(val) => AV::N(val.to_string()),
61            stmt::Value::I32(val) => AV::N(val.to_string()),
62            stmt::Value::I64(val) => AV::N(val.to_string()),
63            stmt::Value::U8(val) => AV::N(val.to_string()),
64            stmt::Value::U16(val) => AV::N(val.to_string()),
65            stmt::Value::U32(val) => AV::N(val.to_string()),
66            stmt::Value::U64(val) => AV::N(val.to_string()),
67            stmt::Value::F32(val) => AV::N(val.to_string()),
68            stmt::Value::F64(val) => AV::N(val.to_string()),
69            stmt::Value::Bytes(val) => AV::B(val.clone().into()),
70            stmt::Value::Uuid(val) => AV::S(val.to_string()),
71            stmt::Value::List(vals) => {
72                let items = vals
73                    .iter()
74                    .map(|val| Value(val.clone()).to_ddb())
75                    .collect::<Vec<_>>();
76                AV::L(items)
77            }
78            stmt::Value::Null => AV::Null(true),
79            _ => todo!("{:#?}", self.0),
80        }
81    }

11/// Input is the key to delete
12#[derive(Debug)]
13pub(crate) struct DeleteByKey {
14    /// How to access input from the variable table.
15    pub input: VarId,
16
17    /// Where to store the output (impacted row count)
18    pub output: Output,
19
20    /// Which model to get
21    pub table: TableId,
22
23    /// Only delete keys that match the filter
24    pub filter: Option<stmt::Expr>,
25
26    /// Condition for optimistic locking (e.g., version check).
27    pub condition: Option<stmt::Expr>,
28}
29
30impl Exec<'_> {
31    pub(super) async fn action_delete_by_key(&mut self, action: &DeleteByKey) -> Result<()> {
32        let keys = self
33            .vars
34            .load(action.input)
35            .await?
36            .values
37            .collect_as_value()
38            .await?
39            .into_list_unwrap();
40
41        let res = if keys.is_empty() {
42            Rows::Count(0)
43        } else {
44            let mut total_count = 0u64;
45
46            for key in keys {
47                let op = operation::DeleteByKey {
48                    table: action.table,
49                    keys: vec![key],
50                    filter: action.filter.clone(),
51                    condition: action.condition.clone(),
52                };
53
54                let res = self.connection.exec(&self.engine.schema, op.into()).await?;
55
56                match res.values {
57                    Rows::Count(n) => total_count += n,
58                    _ => panic!("expected Count from DeleteByKey"),
59                }
60            }
61
62            Rows::Count(total_count)
63        };
64
65        self.vars.store(
66            action.output.var,
67            action.output.num_uses,
68            ExecResponse::from_rows(res),
69        );
70
71        Ok(())
72    }
73}
74
75impl From<DeleteByKey> for Action {
76    fn from(src: DeleteByKey) -> Self {
77        Self::DeleteByKey(src)
78    }
79}

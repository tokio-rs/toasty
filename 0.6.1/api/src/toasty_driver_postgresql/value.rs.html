9/// Wrapper for reading string values from PostgreSQL enum columns.
10///
11/// The standard `String::FromSql::accepts()` rejects custom enum types.
12/// This wrapper accepts `Kind::Enum` types and reads the value as UTF-8 text.
13struct EnumString(String);
14
15impl<'a> postgres_types::FromSql<'a> for EnumString {
16    fn from_sql(
17        _ty: &Type,
18        raw: &'a [u8],
19    ) -> std::result::Result<Self, Box<dyn std::error::Error + Sync + Send>> {
20        Ok(EnumString(
21            std::str::from_utf8(raw)
22                .map_err(|e| Box::new(e) as Box<dyn std::error::Error + Sync + Send>)?
23                .to_string(),
24        ))
25    }
26
27    fn accepts(ty: &Type) -> bool {
28        matches!(ty.kind(), Kind::Enum(_))
29    }
30}

32/// Captures the raw wire bytes of a column without interpreting them.
33/// Used by the array decode path to hand the column body to
34/// [`array_from_sql`] directly, bypassing tokio-postgres's
35/// `Vec<Option<T>>` element decoder.
36struct RawBytes<'a>(&'a [u8]);
37
38impl<'a> postgres_types::FromSql<'a> for RawBytes<'a> {
39    fn from_sql(
40        _ty: &Type,
41        raw: &'a [u8],
42    ) -> std::result::Result<Self, Box<dyn std::error::Error + Sync + Send>> {
43        Ok(RawBytes(raw))
44    }
45
46    fn accepts(_ty: &Type) -> bool {
47        true
48    }
49}

61    /// Converts this PostgreSQL driver value into the core Toasty value.
62    pub fn into_inner(self) -> CoreValue {

66    /// Converts a PostgreSQL value within a row to a Toasty value.
67    pub fn from_sql(index: usize, row: &Row, column: &Column, expected_ty: &stmt::Type) -> Self {
68        // Gets the value from the row as Option<T> and return stmt::Value::Null if the Option is
69        // None.
70        macro_rules! get_or_return_null {
71            ($ty:ty) => {{
72                match row.get::<usize, Option<$ty>>(index) {
73                    Some(inner) => inner,
74                    None => return Self(stmt::Value::Null),
75                }
76            }};
77        }
78
79        // NOTE: unfortunately, the inner representation of the PostgreSQL type enum is not
80        // accessible, so we must manually match each type like so.
81        let core_value = if column.type_() == &Type::TEXT || column.type_() == &Type::VARCHAR {
82            text_to_value(get_or_return_null!(String), expected_ty)
83        } else if column.type_() == &Type::BOOL {
84            stmt::Value::Bool(get_or_return_null!(bool))
85        } else if column.type_() == &Type::INT2 {
86            int2_to_value(get_or_return_null!(i16), expected_ty)
87        } else if column.type_() == &Type::INT4 {
88            int4_to_value(get_or_return_null!(i32), expected_ty)
89        } else if column.type_() == &Type::INT8 {
90            int8_to_value(get_or_return_null!(i64), expected_ty)
91        } else if column.type_() == &Type::UUID {
92            let v = get_or_return_null!(uuid::Uuid);
93            match expected_ty {
94                stmt::Type::Uuid => stmt::Value::Uuid(v),
95                stmt::Type::String => stmt::Value::String(v.to_string()),
96                _ => stmt::Value::Uuid(v),
97            }
98        } else if column.type_() == &Type::BYTEA {
99            let v = get_or_return_null!(Vec<u8>);
100            match expected_ty {
101                stmt::Type::Uuid => stmt::Value::Uuid(v.try_into().expect("invalid uuid bytes")),
102                stmt::Type::Bytes => stmt::Value::Bytes(v),
103                _ => todo!(
104                    "unsupported conversion from {:#?} to {expected_ty:?}",
105                    column.type_()
106                ),
107            }
108        } else if column.type_() == &Type::TIMESTAMPTZ {
109            #[cfg(feature = "jiff")]
110            {
111                stmt::Value::Timestamp(get_or_return_null!(jiff::Timestamp))
112            }
113            #[cfg(not(feature = "jiff"))]
114            {
115                panic!("TIMESTAMPTZ requires jiff feature to be enabled")
116            }
117        } else if column.type_() == &Type::TIMESTAMP {
118            #[cfg(feature = "jiff")]
119            {
120                stmt::Value::DateTime(get_or_return_null!(jiff::civil::DateTime))
121            }
122            #[cfg(not(feature = "jiff"))]
123            {
124                panic!("TIMESTAMP requires jiff feature to be enabled")
125            }
126        } else if column.type_() == &Type::DATE {
127            #[cfg(feature = "jiff")]
128            {
129                stmt::Value::Date(get_or_return_null!(jiff::civil::Date))
130            }
131            #[cfg(not(feature = "jiff"))]
132            {
133                panic!("DATE requires jiff feature to be enabled")
134            }
135        } else if column.type_() == &Type::TIME {
136            #[cfg(feature = "jiff")]
137            {
138                stmt::Value::Time(get_or_return_null!(jiff::civil::Time))
139            }
140            #[cfg(not(feature = "jiff"))]
141            {
142                panic!("TIME requires jiff feature to be enabled")
143            }
144        } else if column.type_() == &Type::FLOAT4 {
145            float4_to_value(get_or_return_null!(f32), expected_ty)
146        } else if column.type_() == &Type::FLOAT8 {
147            float8_to_value(get_or_return_null!(f64), expected_ty)
148        } else if column.type_() == &Type::NUMERIC {
149            #[cfg(feature = "rust_decimal")]
150            {
151                stmt::Value::Decimal(get_or_return_null!(rust_decimal::Decimal))
152            }
153            #[cfg(not(feature = "rust_decimal"))]
154            {
155                panic!("NUMERIC requires rust_decimal feature to be enabled")
156            }
157        } else if matches!(column.type_().kind(), Kind::Enum(_)) {
158            // Native database enum types (CREATE TYPE ... AS ENUM) are read as strings.
159            // We use EnumString instead of String because String::FromSql::accepts()
160            // rejects custom enum types.
161            match row.get::<usize, Option<EnumString>>(index) {
162                Some(EnumString(v)) => stmt::Value::String(v),
163                None => return Self(stmt::Value::Null),
164            }
165        } else if let Kind::Array(_) = column.type_().kind() {
166            // Native array column (e.g. `text[]`, `int8[]`) — decoded
167            // directly into `Vec<stmt::Value>` via `array_from_sql`. The
168            // element type is taken from `expected_ty.as_list_unwrap()` so
169            // the per-element conversion mirrors a column of that scalar
170            // type.
171            let elem_ty = match expected_ty {
172                stmt::Type::List(elem) => elem.as_ref(),
173                other => panic!("array column expected stmt::Type::List, got {other:?}"),
174            };
175            let items = read_array_items(index, row, column, elem_ty);
176            match items {
177                Some(items) => stmt::Value::List(items),
178                None => return Self(stmt::Value::Null),
179            }
180        } else {
181            todo!(
182                "implement PostgreSQL to toasty conversion for `{:#?}`",
183                column.type_()
184            );
185        };
186
187        Value(core_value)
188    }
189}

191// ============================================================================
192// Per-primitive conversions
193// ----------------------------------------------------------------------------
194// These functions translate a single decoded PostgreSQL primitive (the value
195// you get from `Row::get` or from an array element) into a `stmt::Value`,
196// respecting Toasty's expected element type. Sharing them between the column
197// path ([`Value::from_sql`]) and the array path ([`read_array_items`]) keeps
198// the two reading paths consistent: a `text[]` element and a `text` column
199// decode through the same logic.
200// ============================================================================
201
202fn text_to_value(v: String, expected_ty: &stmt::Type) -> stmt::Value {
203    match expected_ty {
204        stmt::Type::String => stmt::Value::String(v),
205        stmt::Type::Uuid => stmt::Value::Uuid(
206            v.parse()
207                .unwrap_or_else(|_| panic!("uuid could not be parsed from text")),
208        ),
209        _ => stmt::Value::String(v),
210    }
211}
212
213fn int2_to_value(v: i16, expected_ty: &stmt::Type) -> stmt::Value {
214    match expected_ty {
215        stmt::Type::I8 => stmt::Value::I8(v as i8),
216        stmt::Type::I16 => stmt::Value::I16(v),
217        stmt::Type::U8 => stmt::Value::U8(
218            u8::try_from(v).unwrap_or_else(|_| panic!("u8 value out of range: {v}")),
219        ),
220        stmt::Type::U16 => stmt::Value::U16(v as u16),
221        _ => panic!("unexpected type for INT2: {expected_ty:#?}"),
222    }
223}
224
225fn int4_to_value(v: i32, expected_ty: &stmt::Type) -> stmt::Value {
226    match expected_ty {
227        stmt::Type::I32 => stmt::Value::I32(v),
228        stmt::Type::U16 => stmt::Value::U16(
229            u16::try_from(v).unwrap_or_else(|_| panic!("u16 value out of range: {v}")),
230        ),
231        stmt::Type::U32 => stmt::Value::U32(v as u32),
232        _ => stmt::Value::I32(v),
233    }
234}
235
236fn int8_to_value(v: i64, expected_ty: &stmt::Type) -> stmt::Value {
237    match expected_ty {
238        stmt::Type::I64 => stmt::Value::I64(v),
239        stmt::Type::U32 => stmt::Value::U32(
240            u32::try_from(v).unwrap_or_else(|_| panic!("u32 value out of range: {v}")),
241        ),
242        stmt::Type::U64 => stmt::Value::U64(
243            u64::try_from(v).unwrap_or_else(|_| panic!("u64 value out of range: {v}")),
244        ),
245        _ => stmt::Value::I64(v),
246    }
247}

249fn float4_to_value(v: f32, expected_ty: &stmt::Type) -> stmt::Value {
250    match expected_ty {
251        stmt::Type::F32 => stmt::Value::F32(v),
252        stmt::Type::F64 => stmt::Value::F64(v as f64),
253        _ => panic!("unexpected type for FLOAT4: {expected_ty:#?}"),
254    }
255}

257fn float8_to_value(v: f64, expected_ty: &stmt::Type) -> stmt::Value {
258    match expected_ty {
259        stmt::Type::F32 => stmt::Value::F32(v as f32),
260        stmt::Type::F64 => stmt::Value::F64(v),
261        _ => panic!("unexpected type for FLOAT8: {expected_ty:#?}"),
262    }
263}

265/// Decode a PostgreSQL array column into a list of Toasty values. Returns
266/// `None` for SQL NULL. The column body is parsed with [`array_from_sql`]
267/// and each element is decoded straight into a `stmt::Value` via the
268/// per-primitive helpers above — the same ones used by [`Value::from_sql`]
269/// for scalar columns. No intermediate `Vec<Option<T>>` is allocated.
270fn read_array_items(
271    index: usize,
272    row: &Row,
273    column: &Column,
274    elem_ty: &stmt::Type,
275) -> Option<Vec<stmt::Value>> {
276    let elem_pg_ty = match column.type_().kind() {
277        Kind::Array(elem) => elem,
278        _ => panic!(
279            "read_array_items called on non-array column: {:?}",
280            column.type_()
281        ),
282    };
283
284    let RawBytes(raw) = row.get::<usize, Option<RawBytes<'_>>>(index)?;
285    let array = array_from_sql(raw).expect("invalid PostgreSQL array wire format");
286
287    let ndims = array
288        .dimensions()
289        .count()
290        .expect("invalid PostgreSQL array dimensions header");
291    if ndims > 1 {
292        panic!(
293            "multi-dimensional PostgreSQL arrays are not supported \
294             (got {ndims} dimensions). See https://github.com/tokio-rs/toasty/issues/870"
295        );
296    }
297
298    let mut values = array.values();
299    let (cap, _) = values.size_hint();
300    let mut out = Vec::with_capacity(cap);
301    while let Some(elem) = values
302        .next()
303        .expect("invalid PostgreSQL array element framing")
304    {
305        out.push(match elem {
306            None => stmt::Value::Null,
307            Some(bytes) => decode_array_element(elem_pg_ty, bytes, elem_ty),
308        });
309    }
310    Some(out)
311}
312
313/// Decode a single PostgreSQL array element from its raw wire bytes and
314/// project it into the Toasty value space using `elem_ty`. Reuses the
315/// scalar-column `_to_value` helpers so a `text[]` element and a `text`
316/// column decode through identical logic.
317fn decode_array_element(elem_pg_ty: &Type, bytes: &[u8], elem_ty: &stmt::Type) -> stmt::Value {
318    if elem_pg_ty == &Type::TEXT || elem_pg_ty == &Type::VARCHAR {
319        text_to_value(
320            String::from_sql(elem_pg_ty, bytes).expect("decode TEXT array element"),
321            elem_ty,
322        )
323    } else if elem_pg_ty == &Type::BOOL {
324        stmt::Value::Bool(bool::from_sql(elem_pg_ty, bytes).expect("decode BOOL array element"))
325    } else if elem_pg_ty == &Type::INT2 {
326        int2_to_value(
327            i16::from_sql(elem_pg_ty, bytes).expect("decode INT2 array element"),
328            elem_ty,
329        )
330    } else if elem_pg_ty == &Type::INT4 {
331        int4_to_value(
332            i32::from_sql(elem_pg_ty, bytes).expect("decode INT4 array element"),
333            elem_ty,
334        )
335    } else if elem_pg_ty == &Type::INT8 {
336        int8_to_value(
337            i64::from_sql(elem_pg_ty, bytes).expect("decode INT8 array element"),
338            elem_ty,
339        )
340    } else if elem_pg_ty == &Type::FLOAT4 {
341        float4_to_value(
342            f32::from_sql(elem_pg_ty, bytes).expect("decode FLOAT4 array element"),
343            elem_ty,
344        )
345    } else if elem_pg_ty == &Type::FLOAT8 {
346        float8_to_value(
347            f64::from_sql(elem_pg_ty, bytes).expect("decode FLOAT8 array element"),
348            elem_ty,
349        )
350    } else if elem_pg_ty == &Type::UUID {
351        stmt::Value::Uuid(
352            uuid::Uuid::from_sql(elem_pg_ty, bytes).expect("decode UUID array element"),
353        )
354    } else {
355        todo!(
356            "implement PostgreSQL array decoding for element type `{:#?}`",
357            elem_pg_ty
358        )
359    }
360}

362impl ToSql for Value {
363    fn to_sql(
364        &self,
365        ty: &Type,
366        out: &mut BytesMut,
367    ) -> std::result::Result<IsNull, Box<dyn std::error::Error + Sync + Send>>
368    where
369        Self: Sized,
370    {
371        value_to_sql(&self.0, ty, out)
372    }
373
374    fn accepts(ty: &Type) -> bool {
375        matches!(
376            *ty,
377            Type::BOOL
378                | Type::INT2
379                | Type::INT4
380                | Type::INT8
381                | Type::TEXT
382                | Type::FLOAT4
383                | Type::FLOAT8
384                | Type::VARCHAR
385                | Type::BYTEA
386                | Type::UUID
387                | Type::NUMERIC
388                | Type::TIMESTAMP
389                | Type::TIMESTAMPTZ
390                | Type::DATE
391                | Type::TIME
392        ) || matches!(ty.kind(), Kind::Enum(_) | Kind::Array(_))
393    }
394    to_sql_checked!();
395}

397/// Free-fn form of `Value::to_sql` so the array-element closure can call it
398/// with a `&CoreValue` borrowed from the input slice — no per-element clone
399/// or wrapper construction.
400fn value_to_sql(
401    value: &CoreValue,
402    ty: &Type,
403    out: &mut BytesMut,
404) -> std::result::Result<IsNull, Box<dyn std::error::Error + Sync + Send>> {
405    match (value, ty) {
406        (stmt::Value::Bool(value), _) => value.to_sql(ty, out),
407        (stmt::Value::I8(value), &Type::INT2) => (*value as i16).to_sql(ty, out),
408        (stmt::Value::I8(value), &Type::INT4) => (*value as i32).to_sql(ty, out),
409        (stmt::Value::I8(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
410        (stmt::Value::I16(value), &Type::INT2) => value.to_sql(ty, out),
411        (stmt::Value::I16(value), &Type::INT4) => (*value as i32).to_sql(ty, out),
412        (stmt::Value::I16(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
413        (stmt::Value::I32(value), &Type::INT4) => value.to_sql(ty, out),
414        (stmt::Value::I32(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
415        (stmt::Value::I64(value), &Type::INT4) => (*value as i32).to_sql(ty, out),
416        (stmt::Value::I64(value), &Type::INT8) => value.to_sql(ty, out),
417        (stmt::Value::U8(value), &Type::INT2) => (*value as i16).to_sql(ty, out),
418        (stmt::Value::U8(value), &Type::INT4) => (*value as i32).to_sql(ty, out),
419        (stmt::Value::U8(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
420        (stmt::Value::U16(value), &Type::INT4) => (*value as i32).to_sql(ty, out),
421        (stmt::Value::U16(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
422        (stmt::Value::U32(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
423        (stmt::Value::U64(value), &Type::INT8) => {
424            if *value > i64::MAX as u64 {
425                return Err(Box::new(std::io::Error::new(
426                    std::io::ErrorKind::InvalidData,
427                    format!(
428                        "u64 value {} exceeds i64::MAX ({}), cannot store in PostgreSQL BIGINT",
429                        value,
430                        i64::MAX
431                    ),
432                )));
433            }
434            (*value as i64).to_sql(ty, out)
435        }
436        (stmt::Value::F32(value), &Type::FLOAT4) => value.to_sql(ty, out),
437        (stmt::Value::F32(value), &Type::FLOAT8) => (*value as f64).to_sql(ty, out),
438        (stmt::Value::F64(value), &Type::FLOAT4) => (*value as f32).to_sql(ty, out),
439        (stmt::Value::F64(value), &Type::FLOAT8) => value.to_sql(ty, out),
440        (stmt::Value::Null, _) => Ok(IsNull::Yes),
441        // PG enums are wire-encoded as plain UTF-8 text. `String::ToSql::accepts`
442        // rejects `Kind::Enum`, so write the bytes directly.
443        (stmt::Value::String(value), _) if matches!(ty.kind(), Kind::Enum(_)) => {
444            out.extend_from_slice(value.as_bytes());
445            Ok(IsNull::No)
446        }
447        (stmt::Value::String(value), _) => value.to_sql(ty, out),
448        (stmt::Value::Bytes(value), &Type::BYTEA) => value.to_sql(ty, out),
449        (stmt::Value::Uuid(value), &Type::UUID) => value.to_sql(ty, out),
450        #[cfg(feature = "rust_decimal")]
451        (stmt::Value::Decimal(value), _) => value.to_sql(ty, out),
452        #[cfg(feature = "jiff")]
453        (stmt::Value::Timestamp(value), _) => value.to_sql(ty, out),
454        #[cfg(feature = "jiff")]
455        (stmt::Value::Date(value), _) => value.to_sql(ty, out),
456        #[cfg(feature = "jiff")]
457        (stmt::Value::Time(value), _) => value.to_sql(ty, out),
458        #[cfg(feature = "jiff")]
459        (stmt::Value::DateTime(value), _) => value.to_sql(ty, out),
460        // List → bind as a PostgreSQL array via the streaming `array_to_sql`
461        // primitive: the closure runs per element and writes directly into
462        // `out`, so there's no intermediate `Vec<Option<T>>`. The element PG
463        // type (carried by the prepared statement, see `db::Type::List` →
464        // `to_postgres_type`) drives per-item conversion via this same fn.
465        (stmt::Value::List(items), _) => {
466            let Kind::Array(elem) = ty.kind() else {
467                return Err(format!("Value::List bound to non-array PG type {ty:?}").into());
468            };
469            let len = i32::try_from(items.len())
470                .map_err(|_| format!("array length {} exceeds i32::MAX", items.len()))?;
471            array_to_sql(
472                [ArrayDimension {
473                    len,
474                    lower_bound: 1,
475                }],
476                elem.oid(),
477                items.iter(),
478                |v, buf| match value_to_sql(v, elem, buf)? {
479                    IsNull::No => Ok(postgres_protocol::IsNull::No),
480                    IsNull::Yes => Ok(postgres_protocol::IsNull::Yes),
481                },
482                out,
483            )?;
484            Ok(IsNull::No)
485        }
486        (value, _) => todo!("unsupported Value for PostgreSQL type: {value:#?}, type: {ty:#?}"),
487    }

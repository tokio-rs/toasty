4/// Database-level storage types representing how values are stored in the target database.
5///
6/// `db::Type` represents the **external** types used by specific database systems
7/// (PostgreSQL, MySQL, SQLite, DynamoDB, etc.) to store column values. These are the actual
8/// storage types that appear in `CREATE TABLE` statements and database schemas.
9///
10/// # Type System Hierarchy
11///
12/// Toasty has two distinct type systems:
13///
14/// 1. **[`stmt::Type`](crate::stmt::Type)**: Application and query engine types (internal to Toasty)
15///    - Represents Rust types: `I8`, `I16`, `String`, etc.
16///    - Types of [`stmt::Value`] and [`stmt::Expr`]
17///    - Used throughout Toasty's query processing at both application and engine levels
18///
19/// 2. **`db::Type`** (this type): Database storage types (external)
20///    - Represents database column types: `Integer(n)`, `Text`, `VarChar(n)`, etc.
21///    - External representation specific to the target database
22///    - Specified in the schema, used by drivers
23///
24/// # Mapping from Application to Database Types
25///
26/// The mapping from [`stmt::Type`] to `db::Type` happens at the driver boundary:
27///
28/// ```text
29/// stmt::Type::String  →  db::Type::Text         (default for most databases)
30///                     →  db::Type::VarChar(255)  (if specified in schema)
31///
32/// stmt::Type::I64     →  db::Type::Integer(8)   (8-byte integer)
33/// stmt::Type::I32     →  db::Type::Integer(4)   (4-byte integer)
34/// stmt::Type::Bool    →  db::Type::Boolean
35/// ```
36///
37/// See [`Type::from_app`] for the complete mapping logic.
38///
39/// # Schema Usage
40///
41/// Each column in the database schema ([`Column`](crate::schema::db::Column)) stores both:
42/// - `column.ty: stmt::Type` - How Toasty views the column internally (application/engine type)
43/// - `column.storage_ty: Option<db::Type>` - How the database stores it externally (storage type)
44///
45/// When `storage_ty` is `None`, the driver uses default mappings from `stmt::Type`.
46/// When specified, it allows fine-grained control over storage (e.g., `VARCHAR(50)` vs `TEXT`).
47///
48/// # Database-Specific Behavior
49///
50/// Different databases support different storage types. The driver's capability
51/// structure ([`driver::Capability`]) describes what types are available:
52///
53/// - **PostgreSQL**: Supports `Text`, `VarChar`, `Integer`, `Boolean`, etc.
54/// - **SQLite**: Uses type affinity; most types map to `TEXT`, `INTEGER`, `REAL`, or `BLOB`
55/// - **DynamoDB**: Uses NoSQL types like `S` (string), `N` (number), `BOOL`, etc.
56///
57/// # See Also
58///
59/// - [`stmt::Type`](crate::stmt::Type) - Application and query engine type system with detailed flow documentation
60/// - [`Type::from_app`] - Mapping logic from statement types to database types
61/// - [`Column`](crate::schema::db::Column) - Schema representation with both type systems
62#[derive(Debug, Clone, PartialEq)]
63#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
64pub enum Type {
65    /// A boolean value
66    Boolean,
67
68    /// A signed integer of `n` bytes
69    Integer(u8),
70
71    /// An unsigned integer of `n` bytes
72    UnsignedInteger(u8),
73
74    /// A floating point number of `n` bytes
75    Float(u8),
76
77    /// Unconstrained text type
78    Text,
79
80    /// Text type with an explicit maximum length
81    VarChar(u64),
82
83    /// 128-bit universally unique identifier (UUID)
84    Uuid,
85
86    /// Decimal number with optional precision and scale.
87    /// - `None`: Arbitrary-precision decimal
88    /// - `Some((precision, scale))`: Fixed precision and scale
89    Numeric(Option<(u32, u32)>),
90
91    /// Unconstrained binary type
92    Blob,
93
94    /// Fixed-size binary type of `n` bytes
95    Binary(u8),
96
97    /// An instant in time with fractional seconds precision (0-9 digits).
98    Timestamp(u8),
99
100    /// A representation of a civil date in the Gregorian calendar.
101    Date,
102
103    /// A representation of civil "wall clock" time with fractional seconds precision (0-9 digits).
104    Time(u8),
105
106    /// A representation of a civil datetime in the Gregorian calendar with fractional seconds precision (0-9 digits).
107    DateTime(u8),
108
109    /// A database enum type. See [`TypeEnum`].
110    Enum(TypeEnum),
111
112    /// An array of `T`, e.g. PostgreSQL `INT8[]` or `TEXT[]`. Used both for
113    /// array column storage (where supported) and to type list-shaped bind
114    /// parameters that the engine sends as a single PG array operand.
115    List(Box<Type>),
116
117    /// User-specified unrecognized type
118    Custom(String),
119}

121impl Type {
122    /// Maps an application-level type to a database-level storage type.
123    pub fn from_app(
124        ty: &stmt::Type,
125        hint: Option<&Type>,
126        db: &driver::StorageTypes,
127    ) -> Result<Type> {
128        match hint {
129            Some(ty) => Ok(ty.clone()),
130            None => match ty {
131                stmt::Type::Bool => Ok(Type::Boolean),
132                stmt::Type::I8 => Ok(Type::Integer(1)),
133                stmt::Type::I16 => Ok(Type::Integer(2)),
134                stmt::Type::I32 => Ok(Type::Integer(4)),
135                stmt::Type::I64 => Ok(Type::Integer(8)),
136                // Map unsigned types to UnsignedInteger with appropriate byte width
137                stmt::Type::U8 => Ok(Type::UnsignedInteger(1)),
138                stmt::Type::U16 => Ok(Type::UnsignedInteger(2)),
139                stmt::Type::U32 => Ok(Type::UnsignedInteger(4)),
140                stmt::Type::U64 => Ok(Type::UnsignedInteger(8)),
141                stmt::Type::F32 => Ok(Type::Float(4)),
142                stmt::Type::F64 => Ok(Type::Float(8)),
143                stmt::Type::String => Ok(db.default_string_type.clone()),
144                stmt::Type::Uuid => Ok(db.default_uuid_type.clone()),
145                stmt::Type::Bytes => Ok(db.default_bytes_type.clone()),
146                // Decimal type
147                #[cfg(feature = "rust_decimal")]
148                stmt::Type::Decimal => Ok(db.default_decimal_type.clone()),
149                // BigDecimal type
150                #[cfg(feature = "bigdecimal")]
151                stmt::Type::BigDecimal => Ok(db.default_bigdecimal_type.clone()),
152                // Date/time types from jiff
153                #[cfg(feature = "jiff")]
154                stmt::Type::Timestamp => Ok(db.default_timestamp_type.clone()),
155                #[cfg(feature = "jiff")]
156                stmt::Type::Zoned => Ok(db.default_zoned_type.clone()),
157                #[cfg(feature = "jiff")]
158                stmt::Type::Date => Ok(db.default_date_type.clone()),
159                #[cfg(feature = "jiff")]
160                stmt::Type::Time => Ok(db.default_time_type.clone()),
161                #[cfg(feature = "jiff")]
162                stmt::Type::DateTime => Ok(db.default_datetime_type.clone()),
163                stmt::Type::List(elem) => Ok(Type::List(Box::new(Self::from_app(elem, None, db)?))),
164                _ => Err(crate::Error::unsupported_feature(format!(
165                    "type {:?} is not supported by this database",
166                    ty
167                ))),
168            },
169        }
170    }
171
172    /// Determines the [`stmt::Type`] closest to this [`db::Type`] that should be used
173    /// as an intermediate conversion step to lessen the work done by each individual driver.
174    pub fn bridge_type(&self, ty: &stmt::Type) -> stmt::Type {
175        match (self, ty) {
176            (Self::Blob | Self::Binary(_), stmt::Type::Uuid) => stmt::Type::Bytes,
177            (Self::Text | Self::VarChar(_), _) => stmt::Type::String,
178            // Enum values are always strings at the application level
179            (Self::Enum(_), _) => stmt::Type::String,
180            // Let engine handle UTC conversion
181            #[cfg(feature = "jiff")]
182            (Self::Timestamp(_) | Self::DateTime(_), stmt::Type::Zoned) => stmt::Type::Timestamp,
183            _ => ty.clone(),
184        }
185    }
186
187    pub(crate) fn verify(&self, db: &driver::Capability) -> Result<()> {
188        match *self {
189            Type::VarChar(size) => match db.storage_types.varchar {
190                Some(max) if size > max => Err(crate::Error::unsupported_feature(format!(
191                    "VARCHAR({}) exceeds database maximum of {}",
192                    size, max
193                ))),
194                None => Err(crate::Error::unsupported_feature(
195                    "VARCHAR type is not supported by this database",
196                )),
197                _ => Ok(()),
198            },
199            _ => Ok(()),
200        }
201    }

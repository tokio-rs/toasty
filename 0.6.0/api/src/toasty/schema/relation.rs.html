4use toasty_core::schema::Name;

7/// Describes how a model participates in associations.
8///
9/// This trait is implemented by `#[derive(Model)]` and provides the associated
10/// types that the generated `HasMany`, `HasOne`, and `BelongsTo` wrappers use
11/// to construct query builders, create builders, and field accessors for
12/// the relation target.
13///
14/// Users do not implement this trait manually.
15pub trait Relation: Load<Output = Self> {
16    /// The target model
17    type Model: Model;
18
19    /// The target expression (e.g. `Option<Model>`)
20    type Expr;
21
22    /// The query builder type for querying this relation's target.
23    type Query;
24
25    /// Create builder type for this relation's target model
26    type Create: Default + IntoInsert<Model = Self::Model> + IntoExpr<Self::Model>;
27
28    /// HasMany relation type
29    type Many;
30
31    /// The field accessor type used when this model appears as the "many" side
32    /// of a has-many relation, parameterized by the origin model.
33    type ManyField<Origin>;
34
35    /// The has-one relation wrapper type for this model.
36    type One;
37
38    /// The field accessor type used when this model appears as the "one" side
39    /// of a has-one relation, parameterized by the origin model.
40    type OneField<Origin>;
41
42    /// The optional has-one relation wrapper type. Used when the foreign key
43    /// is nullable, making the association optional.
44    type OptionOne;
45
46    /// Return a fresh, default-initialized create builder.
47    fn new_create() -> Self::Create {
48        Self::Create::default()
49    }
50
51    /// Construct a `ManyField` from a path targeting a list of the model.
52    fn new_many_field<Origin>(path: Path<Origin, List<Self::Model>>) -> Self::ManyField<Origin>;
53
54    /// Map a field name string to its [`FieldId`].
55    ///
56    /// Panics if `name` does not match any field on the model.
57    fn field_name_to_id(name: &str) -> FieldId;
58
59    /// Returns `true` if this relation target is nullable (i.e., wrapped in
60    /// `Option`). The default is `false`.
61    fn nullable() -> bool {
62        false
63    }
64
65    /// Build the [`FieldTy`] for a `BelongsTo` relation wrapper, given the
66    /// foreign key resolved from the field's `#[belongs_to(...)]` attribute.
67    ///
68    /// Only [`BelongsTo`](super::BelongsTo) overrides this; the default
69    /// panics so that misuse (e.g. applying `#[belongs_to]` to a field whose
70    /// type is not a `BelongsTo<T>`) fails loudly.
71    fn belongs_to_field_ty(_foreign_key: ForeignKey) -> FieldTy {
72        unimplemented!("not a BelongsTo relation wrapper")
73    }
74
75    /// Build the [`FieldTy`] for a `HasMany` relation wrapper, given the
76    /// singular name derived from the field identifier and an optional
77    /// paired `BelongsTo` field on the target model resolved from
78    /// `#[has_many(pair = <field>)]`. When `None`, the linker selects the
79    /// pair by searching the target for a unique `BelongsTo` back to the
80    /// source.
81    ///
82    /// Only [`HasMany`](super::HasMany) overrides this.
83    fn has_many_field_ty(_singular: Name, _pair: Option<FieldId>) -> FieldTy {
84        unimplemented!("not a HasMany relation wrapper")
85    }
86
87    /// Build the [`FieldTy`] for a `HasOne` relation wrapper, given an
88    /// optional paired `BelongsTo` field on the target model resolved
89    /// from `#[has_one(pair = <field>)]`. When `None`, the linker selects
90    /// the pair by searching the target for a unique `BelongsTo` back to
91    /// the source.
92    ///
93    /// Only [`HasOne`](super::HasOne) overrides this.
94    fn has_one_field_ty(_pair: Option<FieldId>) -> FieldTy {
95        unimplemented!("not a HasOne relation wrapper")
96    }
97}

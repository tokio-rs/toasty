30/// The query execution engine.
31///
32/// [`Engine`] orchestrates the multi-phase compilation pipeline that transforms
33/// user queries into database operations. It owns the schema and capability
34/// reference, and provides the main entry point ([`exec`](Self::exec)) for
35/// executing statements.
36///
37/// The execution pipeline follows this process:
38///
39/// 1. **Verification.** Validate statement structure and reject AST shapes
40///    the driver does not support.
41/// 2. **Lowering.** Convert to HIR with dependency tracking.
42/// 3. **Planning.** Build MIR operation graph.
43/// 4. **Execution.** Run actions against the database driver.
44#[derive(Debug, Clone)]
45pub(crate) struct Engine {
46    /// The schema being managed by this database instance.
47    pub(crate) schema: Arc<Schema>,
48
49    /// Driver capabilities, used during planning.
50    pub(crate) capability: &'static Capability,
51}
52
53impl Engine {
54    /// Creates a new [`Engine`] with the given schema and capability.
55    pub(crate) fn new(schema: Arc<Schema>, capability: &'static Capability) -> Engine {
56        Engine { schema, capability }
57    }
58
59    /// Returns the driver's capabilities.
60    pub(crate) fn capability(&self) -> &Capability {
61        self.capability
62    }
63
64    /// Executes a statement and returns the full response including pagination metadata.
65    ///
66    /// The statement passes through the full compilation pipeline
67    /// (lowering -> planning -> execution) before being sent to the database
68    /// driver via the provided connection.
69    pub(crate) async fn exec(
70        &self,
71        connection: &mut dyn Connection,
72        stmt: Statement,
73        in_transaction: bool,
74    ) -> Result<toasty_core::driver::ExecResponse> {
75        tracing::debug!(stmt.kind = stmt.name(), "executing statement");
76
77        self.verify(&stmt)?;
78
79        if let stmt::Statement::Insert(stmt) = &stmt {
80            assert!(matches!(
81                stmt.returning,
82                Some(stmt::Returning::Model { .. })
83            ));
84        }
85
86        // Lower the statement to High-level intermediate representation
87        let hir = self.lower_stmt(stmt)?;
88
89        // Translate the optimized statement into a series of driver operations.
90        let plan = self.plan_hir_statement(hir)?;
91
92        tracing::trace!(
93            actions = plan.actions.len(),
94            needs_transaction = plan.needs_transaction,
95            "execution plan ready"
96        );
97
98        // The plan is called once (single entry record stream) with no arguments
99        // (empty record).
100        self.exec_plan(connection, plan, in_transaction).await
101    }
102
103    /// Returns a new [`ExprContext`](stmt::ExprContext) for a specific target.
104    fn expr_cx_for<'a>(&'a self, target: impl stmt::IntoExprTarget<'a>) -> stmt::ExprContext<'a> {
105        stmt::ExprContext::new_with_target(&self.schema, target)
106    }
107}

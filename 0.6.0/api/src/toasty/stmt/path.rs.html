9/// A typed path from a root model `T` to a field of type `U`.
10///
11/// `Path` represents a traversal through a model's fields and relations. The
12/// type parameter `T` is the root model the path starts from, and `U` is the
13/// type of the value at the end of the path.
14///
15/// Paths are the primary way to reference model fields in queries. Generated
16/// code provides accessor methods (e.g., `User::fields().name()`) that return
17/// paths, which you then use with comparison methods to build filter
18/// expressions:
19///
20/// ```
21/// # #[derive(Debug, toasty::Model)]
22/// # struct User {
23/// #     #[key]
24/// #     id: i64,
25/// #     name: String,
26/// # }
27/// // Path<User, String> — the "name" field on User
28/// let path = User::fields().name();
29///
30/// // Expr<bool> — a filter expression
31/// let filter = path.eq("Alice");
32/// ```
33///
34/// Paths can also be used to construct order-by clauses via
35/// [`asc`](Path::asc) and [`desc`](Path::desc), and can be chained to
36/// navigate through relations with [`chain`](Path::chain).
37pub struct Path<T, U> {
38    pub(super) untyped: stmt::Path,
39    _p: PhantomData<(T, U)>,
40}

45    /// # Examples

131    /// Append `other` to this path, producing a new path from `T` to `V`.
132    ///
133    /// The origin of `other` is left unconstrained because list field structs
134    /// store `Path<Origin, List<M>>` while chaining segments rooted at `M`
135    /// (not `List<M>`). The untyped path concatenation is always correct; the
136    /// generic parameters serve only as compile-time markers.
137    ///
138    /// # Examples
139    ///
140    /// ```
141    /// # #[derive(Debug, toasty::Model)]
142    /// # struct User {
143    /// #     #[key]
144    /// #     id: i64,
145    /// #     name: String,
146    /// # }
147    /// use toasty::stmt::Path;
148    ///
149    /// let user_path = Path::<User, User>::root();
150    /// let name_path = Path::<User, String>::from_field_index(1);
151    /// let _chained: Path<User, String> = user_path.chain(name_path);
152    /// ```
153    pub fn chain<X, V>(mut self, other: impl Into<Path<X, V>>) -> Path<T, V> {
154        let other = other.into();
155        self.untyped.chain(&other.untyped);
156
157        Path {
158            untyped: self.untyped,
159            _p: PhantomData,
160        }
161    }

163    /// Build a filter `Expr<bool>` from this path, automatically wrapping
164    /// the body with an `is_variant(parent, variant_id)` AND-gate when the
165    /// path is variant-rooted.
166    ///
167    /// All boolean-producing methods on `Path` (`eq`, `ne`, `gt`, `is_none`,
168    /// `starts_with`, `any`, …) funnel through this so that filter-context
169    /// uses of a variant-rooted path implicitly require the variant to
170    /// match. Path-yielding contexts (`include`, `order_by`, `chain`)
171    /// bypass this helper and keep the bare path.
172    fn build_filter<F>(self, build_body: F) -> Expr<bool>
173    where
174        F: FnOnce(stmt::Expr) -> stmt::Expr,
175    {
176        let gate = match &self.untyped.root {
177            stmt::PathRoot::Variant { parent, variant_id } => {
178                let parent_stmt = parent.as_ref().clone().into_stmt();
179                Some(stmt::Expr::is_variant(parent_stmt, *variant_id))
180            }
181            _ => None,
182        };
183        let body = build_body(self.untyped.into_stmt());
184        let untyped = match gate {
185            Some(g) => stmt::Expr::and(g, body),
186            None => body,
187        };
188        Expr {
189            untyped,
190            _p: PhantomData,
191        }
192    }

195    ///
196    /// For a variant-rooted path (e.g. `contact().email().address()`), the
197    /// resulting filter implicitly requires the variant to match — it
198    /// expands to `is_email(contact) AND address == rhs`.
199    ///
200    /// # Examples
201    ///
202    /// ```
203    /// # #[derive(Debug, toasty::Model)]
204    /// # struct User {
205    /// #     #[key]
206    /// #     id: i64,
207    /// #     name: String,
208    /// # }
209    /// let filter = User::fields().name().eq("Alice");
210    /// ```
211    pub fn eq(self, rhs: impl IntoExpr<U>) -> Expr<bool> {
212        let rhs = rhs.into_expr().untyped;
213        self.build_filter(move |path| stmt::Expr::eq(path, rhs))
214    }

398impl<T, U> Path<T, List<U>> {
399    /// Build an `IN subquery` expression that tests whether **any** associated
400    /// record satisfies `filter`.
401    ///
402    /// The path must point to a `HasMany` (or similar collection) field on the
403    /// parent model. The returned expression can be used as a filter on the
404    /// parent query.
405    ///
406    /// # Examples
407    ///
408    /// ```
409    /// # #[derive(Debug, toasty::Model)]
410    /// # struct User {
411    /// #     #[key]
412    /// #     id: i64,
413    /// #     name: String,
414    /// # }
415    /// # #[derive(Debug, toasty::Model)]
416    /// # struct Todo {
417    /// #     #[key]
418    /// #     id: i64,
419    /// #     title: String,
420    /// # }
421    /// use toasty::stmt::{Path, List};
422    ///
423    /// // Find users that have at least one todo with "urgent" in the title
424    /// let todos_path = Path::<User, List<Todo>>::from_field_index(2);
425    /// let filter = todos_path.any(Todo::fields().title().eq("urgent"));
426    /// ```
427    pub fn any(self, filter: Expr<bool>) -> Expr<bool>
428    where
429        U: crate::schema::Model,
430    {
431        // Build a query on the child model filtered by `filter`
432        let child_query = super::Query::<List<U>>::filter(filter);
433        self.build_filter(move |path| stmt::Expr::in_subquery(path, child_query.untyped))
434    }
435
436    /// Build a `NOT IN subquery` expression that tests whether **all** associated
437    /// records satisfy `filter`.
438    ///
439    /// The path must point to a `HasMany` (or similar collection) field on the
440    /// parent model. Returns `true` when every associated record matches
441    /// `filter`, including the vacuous case where the parent has no associated
442    /// records (matching Rust's `[].iter().all()` semantics).
443    ///
444    /// # Examples
445    ///
446    /// ```
447    /// # #[derive(Debug, toasty::Model)]
448    /// # struct User {
449    /// #     #[key]
450    /// #     id: i64,
451    /// #     name: String,
452    /// # }
453    /// # #[derive(Debug, toasty::Model)]
454    /// # struct Todo {
455    /// #     #[key]
456    /// #     id: i64,
457    /// #     complete: bool,
458    /// # }
459    /// use toasty::stmt::{Path, List};
460    ///
461    /// // Find users whose todos are all complete
462    /// let todos_path = Path::<User, List<Todo>>::from_field_index(2);
463    /// let filter = todos_path.all(Todo::fields().complete().eq(true));
464    /// ```
465    pub fn all(self, filter: Expr<bool>) -> Expr<bool>
466    where
467        U: crate::schema::Model,
468    {
469        // parent NOT IN (SELECT child_fk FROM child WHERE NOT filter)
470        let child_query = super::Query::<List<U>>::filter(filter.not());
471        self.build_filter(move |path| {
472            stmt::Expr::not(stmt::Expr::in_subquery(path, child_query.untyped))
473        })
474    }
475}

477/// Container-style predicates on a `Vec<scalar>` model field. The path target
478/// is the [`List<U>`] marker (matching `Field::Path<Origin>` for `Vec<U>`),
479/// and the element type `U` is constrained to a path-target scalar so the
480/// same `IntoExpr` infrastructure that powers `eq`/`in_list` covers the
481/// right-hand side.
482impl<T, U> Path<T, List<U>>
483where
484    U: crate::schema::Scalar,
485{
486    /// Test whether the array contains `value`.
487    ///
488    /// Mirrors [`Vec::contains`]. Lowers to `value = ANY(col)` on PostgreSQL.
489    ///
490    /// # Examples
491    ///
492    /// ```ignore
493    /// let filter = User::fields().tags().contains("admin");
494    /// ```
495    pub fn contains(self, value: impl IntoExpr<U>) -> Expr<bool> {
496        let value = value.into_expr().untyped;
497        self.build_filter(move |path| stmt::Expr::any_op(value, stmt::BinaryOp::Eq, path))
498    }
499
500    /// Test whether the array contains every element of `values`.
501    ///
502    /// Mirrors [`HashSet::is_superset`](std::collections::HashSet::is_superset).
503    /// Lowers to `col @> values` (PostgreSQL `@>` operator).
504    pub fn is_superset(self, values: impl IntoExpr<List<U>>) -> Expr<bool> {
505        let values = values.into_expr().untyped;
506        self.build_filter(move |path| stmt::Expr::array_is_superset(path, values))
507    }
508
509    /// Test whether the array shares at least one element with `values`.
510    ///
511    /// Negation of [`HashSet::is_disjoint`](std::collections::HashSet::is_disjoint).
512    /// Lowers to `col && values` (PostgreSQL `&&` operator).
513    pub fn intersects(self, values: impl IntoExpr<List<U>>) -> Expr<bool> {
514        let values = values.into_expr().untyped;
515        self.build_filter(move |path| stmt::Expr::array_intersects(path, values))
516    }
517
518    /// Returns the array length.
519    ///
520    /// Mirrors [`Vec::len`]. Lowers to `cardinality(col)` on PostgreSQL.
521    pub fn len(self) -> Expr<i64> {
522        Expr::from_untyped(stmt::Expr::array_length(self.untyped.into_stmt()))
523    }
524
525    /// Returns `true` if the array is empty.
526    ///
527    /// Equivalent to `.len().eq(0)`. Mirrors [`Vec::is_empty`].
528    pub fn is_empty(self) -> Expr<bool> {
529        let untyped = stmt::Expr::eq(
530            stmt::Expr::array_length(self.untyped.into_stmt()),
531            stmt::Expr::Value(stmt::Value::I64(0)),
532        );
533        Expr::from_untyped(untyped)
534    }
535}

575impl<T, U> Path<T, U>
576where
577    U: Field<Inner = String>,
578{
579    /// Test whether this string field starts with `prefix`.
580    ///
581    /// Available on any string-valued field, including `String`,
582    /// `Option<String>`, and other wrappers whose `Field::Inner` is `String`.
583    /// For DynamoDB, this maps to `begins_with` in a `KeyConditionExpression`
584    /// (sort key) or `FilterExpression` (non-key attribute).
585    ///
586    /// # Examples
587    ///
588    /// ```
589    /// # #[derive(Debug, toasty::Model)]
590    /// # struct User {
591    /// #     #[key]
592    /// #     id: i64,
593    /// #     name: String,
594    /// #     nickname: Option<String>,
595    /// # }
596    /// let filter = User::fields().name().starts_with("Al".to_string());
597    /// let filter = User::fields().nickname().starts_with("Al".to_string());
598    /// ```
599    pub fn starts_with(self, prefix: impl IntoExpr<String>) -> Expr<bool> {
600        let prefix = prefix.into_expr().untyped;
601        self.build_filter(move |path| stmt::Expr::starts_with(path, prefix))
602    }
603
604    /// Test whether this string field matches a SQL `LIKE` pattern.
605    ///
606    /// Available on any string-valued field, including `String`,
607    /// `Option<String>`, and other wrappers whose `Field::Inner` is `String`.
608    /// The caller is responsible for including any `%` or `_` wildcard
609    /// characters in `pattern`. Not supported by the DynamoDB driver — use
610    /// [`starts_with`](Self::starts_with) instead.
611    ///
612    /// # Examples
613    ///
614    /// ```
615    /// # #[derive(Debug, toasty::Model)]
616    /// # struct User {
617    /// #     #[key]
618    /// #     id: i64,
619    /// #     name: String,
620    /// #     nickname: Option<String>,
621    /// # }
622    /// let filter = User::fields().name().like("Al%".to_string());
623    /// let filter = User::fields().nickname().like("Al%".to_string());
624    /// ```
625    pub fn like(self, pattern: impl IntoExpr<String>) -> Expr<bool> {
626        let pattern = pattern.into_expr().untyped;
627        self.build_filter(move |path| stmt::Expr::like(path, pattern))
628    }
629
630    /// Case-insensitive variant of [`like`](Self::like).
631    ///
632    /// On PostgreSQL this serializes to `ILIKE`. On SQLite and MySQL it
633    /// serializes to plain `LIKE`, since both engines are already
634    /// case-insensitive for ASCII by default — note that Unicode case-folding
635    /// behavior depends on locale (PostgreSQL) or column collation (MySQL),
636    /// and SQLite's `LIKE` is ASCII-only without the ICU extension. Not
637    /// supported by the DynamoDB driver.
638    ///
639    /// # Examples
640    ///
641    /// ```
642    /// # #[derive(Debug, toasty::Model)]
643    /// # struct User {
644    /// #     #[key]
645    /// #     id: i64,
646    /// #     name: String,
647    /// # }
648    /// // Matches "Alice", "ALICIA", and "alfred".
649    /// let filter = User::fields().name().ilike("al%".to_string());
650    /// ```
651    pub fn ilike(self, pattern: impl IntoExpr<String>) -> Expr<bool> {
652        Expr {
653            untyped: stmt::Expr::ilike(self.untyped.into_stmt(), pattern.into_expr().untyped),
654            _p: PhantomData,
655        }
656    }

32impl Engine {
33    pub(super) fn lower_stmt(&self, stmt: stmt::Statement) -> Result<HirStatement> {
34        let schema = &self.schema;
35
36        let mut state = LoweringState {
37            hir: HirStatement::new(),
38            scopes: IndexVec::new(),
39            engine: self,
40            relations: vec![],
41            errors: vec![],
42            dependencies: HashSet::new(),
43        };
44
45        state.lower_stmt(stmt::ExprContext::new(schema), None, stmt);
46
47        if let Some(err) = state.errors.into_iter().next() {
48            return Err(err);
49        }
50
51        Ok(state.hir)
52    }
53}

55impl LoweringState<'_> {
56    fn lower_stmt(
57        &mut self,
58        expr_cx: stmt::ExprContext,
59        row_index: Option<usize>,
60        mut stmt: stmt::Statement,
61    ) -> hir::StmtId {
62        // App-level rewrites that lowering depends on. `Source::Model { via }`
63        // must be converted to a WHERE filter before the lowering walk
64        // converts `Source::Model` into `Source::Table`.  The IN-subquery
65        // lift fires here too: code paths outside the lowering walk (e.g.
66        // `ApplyInsertScope::apply_expr`) see the already-lifted form.
67        // The eq/ne operand rewrite (model→PK, BelongsTo→FK) fires inside
68        // the lowering walk itself via `LowerStatement::visit_expr_binary_op_mut`.
69        association::RewriteVia::new(expr_cx).rewrite(&mut stmt);
70        lift_in_subquery::LiftInSubquery::new(expr_cx).rewrite(&mut stmt);
71
72        Simplify::with_context(expr_cx, self.engine.capability).visit_mut(&mut stmt);
73
74        let stmt_id = self.hir.new_statement_info(self.dependencies.clone());
75        let scope_id = self.scopes.push(Scope { stmt_id, row_index });
76        let mut collect_dependencies = None;
77
78        // Map the statement
79        LowerStatement {
80            state: self,
81            expr_cx,
82            scope_id,
83            cx: LoweringContext::Statement,
84            collect_dependencies: &mut collect_dependencies,
85        }
86        .visit_stmt_mut(&mut stmt);
87
88        self.engine.simplify_stmt(&mut stmt);
89
90        let stmt_info = &mut self.hir[stmt_id];
91        stmt_info.stmt = Some(Box::new(stmt));
92
93        self.scopes.pop();
94
95        debug_assert!(collect_dependencies.is_none());
96
97        stmt_id
98    }
99}

101struct LowerStatement<'a, 'b> {
102    /// Lowering state. This is the state that is constant throughout the entire
103    /// lowering process.
104    state: &'a mut LoweringState<'b>,
105
106    /// Expression context in which the statement is being lowered
107    expr_cx: stmt::ExprContext<'a>,
108
109    /// Identifier to the current scope (stored in `scopes` on LoweringState)
110    scope_id: ScopeId,
111
112    /// Current lowering context
113    cx: LoweringContext<'a>,
114
115    /// Track dependencies here.
116    collect_dependencies: &'a mut Option<HashSet<hir::StmtId>>,
117}

119#[derive(Debug)]
120struct LoweringState<'a> {
121    /// Database engine handle
122    engine: &'a Engine,
123
124    /// Statements to be executed by the database, though they may still be
125    /// broken down into multiple sub-statements.
126    hir: HirStatement,
127
128    /// Scope state
129    scopes: IndexVec<ScopeId, Scope>,

132    /// consistency. This field tracks the current relation edge being traversed

173/// A collection operator on a `Vec<scalar>` field.
174///
175/// `Append`, `Remove`, `Pop`, and `RemoveAt` are all collection operators:
176/// they target a single-column `Vec<scalar>` field and carry operator
177/// operands (a list of elements, a single element, an index, or nothing).
178/// Whether the operand is a list or a scalar is a per-operator encoding
179/// detail — the unifying property is that none carries a *field value*
180/// that decomposes across columns, so none needs the `model_to_table`
181/// substitution that `Set` flows through.
182enum CollectionOp<'a> {
183    Append(&'a mut stmt::Expr),
184    Remove(&'a mut stmt::Expr),
185    Pop,
186    RemoveAt(&'a mut stmt::Expr),
187}

189impl LowerStatement<'_, '_> {
190    /// Lower a `Set` assignment — the only assignment that carries a whole
191    /// *field value*.
192    ///
193    /// The field may span multiple columns (embedded structs decompose
194    /// across columns; enums into a discriminant plus variant columns), so
195    /// this resolves the field mapping and, for each impacted column,
196    /// substitutes the lowered value into that column's `model_to_table`
197    /// template before emitting the column-level write.
198    fn lower_set_assignment(
199        &mut self,
200        out: &mut stmt::Assignments,
201        mapping: &toasty_core::schema::mapping::Model,
202        projection: &stmt::Projection,
203        expr: &mut stmt::Expr,
204    ) {
205        self.visit_expr_mut(expr);
206
207        let Some(field) = mapping.resolve_field_mapping(projection) else {
208            self.state
209                .errors
210                .push(crate::Error::invalid_statement(format!(
211                    "invalid assignment projection: {projection:?}"
212                )));
213            return;
214        };
215
216        for (column, lowering_idx) in field.columns() {
217            let mut lowering_expr = mapping.model_to_table[lowering_idx].clone();
218
219            // Substitute the field reference in the column's lowering
220            // template with the lowered value.
221            lowering_expr.substitute(AssignmentInput {
222                assignment_projection: projection.clone(),
223                value: expr,
224            });
225
226            self.visit_expr_mut(&mut lowering_expr);
227
228            out.set(column, lowering_expr);
229        }
230    }
231
232    /// Lower a collection operator (`Append` / `Remove` / `Pop` /
233    /// `RemoveAt`) on a `Vec<scalar>` field.
234    ///
235    /// `Vec<scalar>` fields always resolve to a single primitive column, so
236    /// this skips the `model_to_table` substitution that `lower_set_assignment`
237    /// performs — the operands are operator arguments, not a field value,
238    /// and for a single-column field `model_to_table` is identity anyway.
239    ///
240    /// `Append` is supported on every backend; the removal operators are
241    /// gated by per-backend capability flags and emit a clear error where
242    /// the native form is not available.
243    fn lower_collection_op(
244        &mut self,
245        out: &mut stmt::Assignments,
246        mapping: &toasty_core::schema::mapping::Model,
247        projection: &stmt::Projection,
248        op: CollectionOp,
249    ) {
250        let Some(field) = mapping.resolve_field_mapping(projection) else {
251            self.state
252                .errors
253                .push(crate::Error::invalid_statement(format!(
254                    "invalid assignment projection: {projection:?}"
255                )));
256            return;
257        };
258
259        let Some(prim) = field.as_primitive() else {
260            self.state
261                .errors
262                .push(crate::Error::invalid_statement(format!(
263                    "collection operator on non-primitive field: {projection:?}"
264                )));
265            return;
266        };
267
268        // `Append` is universally supported; the removal operators are
269        // gated per backend.
270        let cap = self.capability();
271        let unsupported = match &op {
272            CollectionOp::Append(_) => None,
273            CollectionOp::Remove(_) if !cap.vec_remove => Some("stmt::remove"),
274            CollectionOp::Pop if !cap.vec_pop => Some("stmt::pop"),
275            CollectionOp::RemoveAt(_) if !cap.vec_remove_at => Some("stmt::remove_at"),
276            _ => None,
277        };
278
279        if let Some(op_name) = unsupported {
280            self.state
281                .errors
282                .push(crate::Error::invalid_statement(format!(
283                    "{op_name} is not yet supported on this backend"
284                )));
285            return;
286        }
287
288        match op {
289            CollectionOp::Append(expr) => {
290                self.visit_expr_mut(expr);
291                out.append(prim.column, expr.take());
292            }
293            CollectionOp::Remove(expr) => {
294                self.visit_expr_mut(expr);
295                out.remove(prim.column, expr.take());
296            }
297            CollectionOp::Pop => {
298                out.pop(prim.column);
299            }
300            CollectionOp::RemoveAt(expr) => {
301                self.visit_expr_mut(expr);
302                out.remove_at(prim.column, expr.take());
303            }
304        }
305    }
306}

308impl LowerStatement<'_, '_> {
309    fn new_dependency(&mut self, stmt: impl Into<stmt::Statement>) -> hir::StmtId {
310        let row_index = match self.cx {
311            LoweringContext::Insert(_, row_index) => row_index,
312            LoweringContext::Returning(row_index) => row_index,
313            _ => None,
314        };
315
316        let stmt_id = self.state.lower_stmt(self.expr_cx, row_index, stmt.into());
317
318        if let Some(dependencies) = &mut self.collect_dependencies {
319            dependencies.insert(stmt_id);
320        }
321
322        self.curr_stmt_info().deps.insert(stmt_id);
323
324        stmt_id
325    }
326
327    fn collect_dependencies(
328        &mut self,
329        f: impl FnOnce(&mut LowerStatement<'_, '_>),
330    ) -> HashSet<hir::StmtId> {
331        let old = self.collect_dependencies.replace(HashSet::new());
332        f(self);
333        std::mem::replace(self.collect_dependencies, old).unwrap()
334    }
335
336    fn track_dependency(&mut self, dependency: hir::StmtId) {
337        self.curr_stmt_info().deps.insert(dependency);
338    }
339
340    fn with_dependencies(
341        &mut self,
342        mut dependencies: HashSet<hir::StmtId>,
343        f: impl FnOnce(&mut LowerStatement<'_, '_>),
344    ) {
345        // Dependencies should stack
346        dependencies.extend(&self.state.dependencies);
347
348        let old = std::mem::replace(&mut self.state.dependencies, dependencies);
349        f(self);
350        self.state.dependencies = old;
351    }
352}

354impl visit_mut::VisitMut for LowerStatement<'_, '_> {
355    fn visit_order_by_expr_mut(&mut self, node: &mut stmt::OrderByExpr) {
356        // First, run the default visitor to lower sub-expressions
357        self.visit_expr_mut(&mut node.expr);
358
359        // Reuse binary-op lowering: synthesize `expr == expr` so that
360        // cast conversions are applied, then keep the LHS result.
361        let mut lhs = node.expr.clone();
362        let mut rhs = node.expr.take();
363        self.lower_expr_binary_op(stmt::BinaryOp::Eq, &mut lhs, &mut rhs);
364        node.expr = lhs;
365    }
366
367    fn visit_assignments_mut(&mut self, i: &mut stmt::Assignments) {
368        let mut lowered = stmt::Assignments::default();
369        let mapping = self.mapping_unwrap();
370
371        // Two lowering paths, divided by what the assignment carries:
372        //
373        // - `Set` carries a whole *field value*. It is the only assignment
374        //   that can target a multi-column field (embedded structs decompose
375        //   across columns, enums into a discriminant plus variants), so it
376        //   is the only one that needs the `model_to_table` substitution.
377        //
378        // - `Append` / `Remove` / `Pop` / `RemoveAt` are *collection
379        //   operators*. They carry operator operands (a list, an element,
380        //   an index, or nothing) and always target a single-column
381        //   `Vec<scalar>` field, so none needs `model_to_table`.
382        for (projection, assignment) in &mut *i {
383            match assignment {
384                stmt::Assignment::Set(expr) => {
385                    self.lower_set_assignment(&mut lowered, mapping, projection, expr);
386                }
387                stmt::Assignment::Append(expr) => {
388                    self.lower_collection_op(
389                        &mut lowered,
390                        mapping,
391                        projection,
392                        CollectionOp::Append(expr),
393                    );
394                }
395                stmt::Assignment::Remove(expr) => {
396                    self.lower_collection_op(
397                        &mut lowered,
398                        mapping,
399                        projection,
400                        CollectionOp::Remove(expr),
401                    );
402                }
403                stmt::Assignment::Pop => {
404                    self.lower_collection_op(&mut lowered, mapping, projection, CollectionOp::Pop);
405                }
406                stmt::Assignment::RemoveAt(expr) => {
407                    self.lower_collection_op(
408                        &mut lowered,
409                        mapping,
410                        projection,
411                        CollectionOp::RemoveAt(expr),
412                    );
413                }
414                stmt::Assignment::Insert(_) | stmt::Assignment::Batch(_) => {
415                    todo!(
416                        "Insert / Batch assignments are not produced for table lowering; got {assignment:#?}"
417                    )
418                }
419            }
420        }
421
422        *i = lowered;
423    }

429    fn visit_expr_binary_op_mut(&mut self, i: &mut stmt::ExprBinaryOp) {
430        // App-level operand rewrite for `eq` / `ne` (`Reference::Model` →
431        // primary-key field, `BelongsTo` → foreign-key field).  Must fire
432        // before the operands are walked, since walking lowers the field
433        // references into column references and the rewrite has nothing to
434        // match on.  Nested binary ops are handled by the recursive walk
435        // through `visit_expr_mut`.
436        if i.op.is_eq() || i.op.is_ne() {
437            self.rewrite_eq_operand(&mut i.lhs);
438            self.rewrite_eq_operand(&mut i.rhs);
439        }
440
441        stmt::visit_mut::visit_expr_binary_op_mut(self, i);
442    }
443
444    fn visit_expr_in_list_mut(&mut self, i: &mut stmt::ExprInList) {
445        // App-level operand rewrite: `Reference::Model { nesting } IN list`
446        // becomes `<pk_field> IN list`.  Must fire before the LHS is walked,
447        // since walking lowers the model reference into something the rewrite
448        // can no longer match on.
449        self.rewrite_in_list_model_operand(i);
450
451        stmt::visit_mut::visit_expr_in_list_mut(self, i);
452    }

454    fn visit_expr_mut(&mut self, expr: &mut stmt::Expr) {
455        match expr {
456            stmt::Expr::BinaryOp(e) => {
457                self.visit_expr_binary_op_mut(e);
458
459                if let Some(lowered) = self.lower_expr_binary_op(e.op, &mut e.lhs, &mut e.rhs) {
460                    *expr = lowered;
461                }
462            }
463            // App-level rewrite: an OR covering every variant of an enum
464            // via `IsVariant` is a tautology.  Must fire before the children
465            // walk lowers the `IsVariant` nodes into discriminant
466            // comparisons, since the rewrite pattern-matches on
467            // `Expr::IsVariant`.
468            stmt::Expr::Or(e) if expr_or::is_variant_tautology_or(self.expr_cx.schema(), e) => {
469                *expr = true.into();
470            }
471            stmt::Expr::InList(e) => {
472                self.visit_expr_in_list_mut(e);
473
474                if let Some(lowered) = self.lower_expr_in_list(&mut e.expr, &mut e.list) {
475                    *expr = lowered;
476                }
477
478                // PostgreSQL-style: `x IN (a, b, c)` → `x = ANY($1)` with the
479                // list bound as a single array param.
480                if let stmt::Expr::InList(e) = expr
481                    && self.supports_any_rewrite()
482                    && in_list_is_value_list(e)
483                {
484                    let stmt::Expr::InList(e) = expr.take() else {
485                        unreachable!()
486                    };
487                    *expr = stmt::Expr::any_op(*e.expr, stmt::BinaryOp::Eq, *e.list);
488                }
489            }
490            stmt::Expr::Not(e) if matches!(*e.expr, stmt::Expr::InList(_)) => {
491                // Recurse into inner first so the InList itself is lowered.
492                self.visit_expr_not_mut(e);
493
494                // If the inner is still an InList (not yet rewritten because
495                // the gate is off or the list shape doesn't qualify), leave
496                // `Not(InList)` alone and let the SQL serializer render
497                // `NOT IN`. Otherwise, rewrite `NOT (x = ANY(arr))` into the
498                // canonical `x <> ALL(arr)` form.
499                if self.supports_any_rewrite()
500                    && let stmt::Expr::AnyOp(any) = e.expr.as_mut()
501                    && any.op == stmt::BinaryOp::Eq
502                {
503                    let stmt::Expr::Not(not) = expr.take() else {
504                        unreachable!()
505                    };
506                    let stmt::Expr::AnyOp(any) = *not.expr else {
507                        unreachable!()
508                    };
509                    *expr = stmt::Expr::all_op(*any.lhs, stmt::BinaryOp::Ne, *any.rhs);
510                }
511            }
512            stmt::Expr::InSubquery(e) => {
513                if self.capability().sql {
514                    self.visit_expr_in_subquery_mut(e);
515
516                    let maybe_res = self.lower_expr_binary_op(
517                        stmt::BinaryOp::Eq,
518                        &mut e.expr,
519                        e.query.returning_mut_unwrap().as_project_mut_unwrap(),
520                    );
521
522                    assert!(maybe_res.is_none(), "TODO");
523
524                    let returning = e.query.returning_mut_unwrap().as_project_mut_unwrap();
525
526                    if !returning.is_record() {
527                        *returning = stmt::Expr::record([returning.take()]);
528                    }
529                } else {
530                    self.visit_expr_mut(&mut e.expr);
531
532                    let source_id = self.scope_stmt_id();
533                    let target_id = self.scope_statement(|child| {
534                        child.visit_stmt_query_mut(&mut e.query);
535                    });
536
537                    // For now, we wonly support independent sub-queries. I.e.
538                    // the subquery must be able to be executed without any
539                    // context from the parent query.
540                    let target_stmt_info = &self.state.hir[target_id];
541                    debug_assert!(target_stmt_info.args.is_empty(), "TODO");
542                    debug_assert!(target_stmt_info.back_refs.is_empty(), "TODO");
543
544                    self.track_dependency(target_id);
545
546                    let maybe_res = self.lower_expr_binary_op(
547                        stmt::BinaryOp::Eq,
548                        &mut e.expr,
549                        e.query.returning_mut_unwrap().as_project_mut_unwrap(),
550                    );
551
552                    assert!(maybe_res.is_none(), "TODO");
553
554                    let stmt::Expr::InSubquery(e) = expr.take() else {
555                        panic!()
556                    };
557
558                    let arg =
559                        self.new_sub_statement(source_id, target_id, Box::new((*e.query).into()));
560
561                    *expr = stmt::ExprInList {
562                        expr: e.expr,
563                        list: Box::new(arg),
564                    }
565                    .into();
566                }
567            }
568            stmt::Expr::IsVariant(e) => {
569                // Look up the enum model and variant directly via VariantId
570                let enum_model = self
571                    .schema()
572                    .app
573                    .model(e.variant.model)
574                    .as_embedded_enum_unwrap();
575                let has_data = enum_model.has_data_variants();
576                let disc_value = enum_model.variants[e.variant.index].discriminant.clone();
577
578                // Lower the inner expression
579                self.visit_expr_mut(&mut e.expr);
580
581                let lowered_expr = e.expr.take();
582
583                // Emit the appropriate comparison
584                if has_data {
585                    // Data-carrying: project([0]) to extract discriminant from Record
586                    *expr = stmt::Expr::eq(
587                        stmt::Expr::project(lowered_expr, [0usize]),
588                        stmt::Expr::Value(disc_value),
589                    );
590                } else {
591                    // Unit-only: compare directly
592                    *expr = stmt::Expr::eq(lowered_expr, stmt::Expr::Value(disc_value));
593                }
594            }
595            stmt::Expr::Reference(expr_reference) => {
596                match expr_reference {
597                    // A reference to a relation field inside a Returning
598                    // clause becomes a subquery that loads the related
599                    // model(s).  This is the `.select(rel_field)` path; it
600                    // mirrors the include-subquery machinery that
601                    // `.include(...)` uses for `Returning::Model`.
602                    stmt::ExprReference::Field { nesting: 0, index }
603                        if matches!(self.cx, LoweringContext::Returning(_))
604                            && self.model_unwrap().fields[*index].ty.is_relation() =>
605                    {
606                        *expr = self.build_relation_subquery(*index, &[]);
607                    }
608                    stmt::ExprReference::Field { nesting, index } => {
609                        *expr = self.lower_expr_field(*nesting, *index);
610                        self.visit_expr_mut(expr);
611                    }
612                    stmt::ExprReference::Model { .. } => todo!(),
613                    stmt::ExprReference::Column(expr_column) => {
614                        if expr_column.nesting > 0 {
615                            let source_id = self.scope_stmt_id();
616                            let target_id = self.resolve_stmt_id(expr_column.nesting);
617
618                            // the current scope ID should also be the top of the stack
619                            debug_assert_eq!(self.state.scopes.len(), self.scope_id + 1);
620
621                            // The statement is not independent. Walk up the
622                            // scope stack until the referened target statement
623                            // and flag any intermediate statements as also not
624                            // indepdnendent.
625                            for scope in self.state.scopes.iter().rev() {
626                                if scope.stmt_id == target_id {
627                                    break;
628                                }
629
630                                self.state.hir[scope.stmt_id].independent = false;
631                            }
632
633                            let position = self.new_ref(source_id, target_id, *expr_reference);
634
635                            // Using ExprArg as a placeholder. It will be rewritten
636                            // later.
637                            *expr = stmt::Expr::arg(position);
638                        }
639                    }
640                }
641            }
642            stmt::Expr::Stmt(_) => {
643                let stmt::Expr::Stmt(mut expr_stmt) = expr.take() else {
644                    panic!()
645                };
646
647                // Expr::Stmt subqueries are valid in returning expressions (e.g.,
648                // INCLUDE preloading) and in VALUES bodies of batch queries.
649                debug_assert!(
650                    self.cx.is_returning() || matches!(self.cx, LoweringContext::Statement),
651                    "cx={:#?}",
652                    self.cx,
653                );
654
655                // For now, we assume nested sub-statements cannot be executed on the
656                // target database. Eventually, we will need to make this smarter.
657                let source_id = self.scope_stmt_id();
658                let target_id = self.scope_statement(|child| {
659                    visit_mut::visit_expr_stmt_mut(child, &mut expr_stmt);
660                });
661
662                // Cheap canonicalization is enough here: the parent statement's
663                // post-lowering simplify will recursively visit this embedded
664                // sub-statement and apply the heavyweight rules.
665                fold::fold_stmt(&mut *expr_stmt.stmt);
666
667                *expr = self.new_sub_statement(source_id, target_id, expr_stmt.stmt);
668
669                if self.state.hir[target_id].independent {
670                    self.curr_stmt_info().deps.insert(target_id);
671                }
672            }
673            stmt::Expr::StartsWith(_)
674                if self.capability().sql && !self.capability().native_starts_with =>
675            {
676                self.lower_expr_starts_with(expr);
677            }
678            stmt::Expr::Exists(_) if !self.capability().sql => {
679                let stmt::Expr::Exists(mut expr_exists) = expr.take() else {
680                    panic!()
681                };
682
683                // Extract the EXISTS subquery into a sub-statement so the
684                // executor can evaluate the condition in memory.
685                let source_id = self.scope_stmt_id();
686                let target_id = self.scope_statement(|child| {
687                    child.visit_stmt_query_mut(&mut expr_exists.subquery);
688                });
689
690                let mut stmt = stmt::Statement::Query(*expr_exists.subquery);
691                // Cheap canonicalization is enough here: the parent statement's
692                // post-lowering simplify will recursively visit this embedded
693                // sub-statement and apply the heavyweight rules.
694                fold::fold_stmt(&mut stmt);
695
696                let arg = self.new_sub_statement(source_id, target_id, Box::new(stmt));
697
698                if self.state.hir[target_id].independent {
699                    self.curr_stmt_info().deps.insert(target_id);
700                }
701
702                // The sub-statement result is a list of rows. Wrap it in
703                // a single-row VALUES query so the evaluator unwraps the
704                // outer list, letting EXISTS check the inner row count.
705                let mut subquery = stmt::Query::values(arg);
706                subquery.single = true;
707                *expr = stmt::Expr::Exists(stmt::ExprExists {
708                    subquery: Box::new(subquery),
709                });
710            }
711            _ => {
712                // Recurse down the statement tree
713                stmt::visit_mut::visit_expr_mut(self, expr);
714            }
715        }
716    }

750    fn visit_returning_mut(&mut self, i: &mut stmt::Returning) {
751        if let stmt::Returning::Model { include } = i {
752            // Start from the schema's pre-computed default returning — every
753            // `#[deferred]` field, top-level or nested, is already `Null`.
754            // `process_top_level_includes` then splices loaded forms in for
755            // the fields named by include paths (and for every deferred field
756            // when this is an `INSERT ... RETURNING`).
757            let mut returning = self.mapping_unwrap().default_returning.clone();
758            let include_paths = std::mem::take(include);
759            let is_insert = self.cx.is_insert();

762
763            *i = stmt::Returning::Project(returning);
764        }

766        // For multi-row INSERT returning, visit each row with its row index so
767        // that sub-statements (e.g., child INSERTs for HasOne relations) capture
768        // the correct parent row index via scope_statement.
769        if matches!(&self.cx, LoweringContext::Insert(..))
770            && let stmt::Returning::Expr(stmt::Expr::List(list)) = i
771        {
772            for (index, item) in list.items.iter_mut().enumerate() {
773                self.lower_returning_for_row(index).visit_expr_mut(item);
774            }
775            return;
776        }

778        stmt::visit_mut::visit_returning_mut(&mut self.lower_returning(), i);
779    }

781    fn visit_stmt_delete_mut(&mut self, stmt: &mut stmt::Delete) {
782        // Create a new expr scope for the statement, and lower all parts
783        // *except* the source field (since it is borrowed).
784        let mut lower = self.scope_expr(&stmt.from);
785
786        // Before lowering, handle cascading deletes
787        lower.plan_stmt_delete_relations(stmt);
788
789        lower.visit_filter_mut(&mut stmt.filter);
790
791        if let Some(expr) = &mut stmt.condition.expr {
792            lower.visit_expr_mut(expr);
793        }
794
795        if let Some(returning) = &mut stmt.returning {
796            lower.visit_returning_mut(returning);
797        }
798
799        lower.apply_lowering_filter_constraint(&mut stmt.filter);
800
801        self.visit_source_mut(&mut stmt.from);
802    }

804    fn visit_stmt_insert_mut(&mut self, stmt: &mut stmt::Insert) {
805        // First, if an insertion scope is specified, lower the scope to be just "model"
806        self.apply_insert_scope(&mut stmt.target, &mut stmt.source);
807
808        // Create a new expr scope for the statement, and lower all parts
809        // *except* the target field (since it is borrowed).
810        let mut lower = self.lower_insert(&stmt.target);
811
812        if let Some(returning) = &mut stmt.returning {
813            lower.visit_returning_mut(returning);
814        }
815
816        // Preprocess the insertion source (values usually)
817        lower.preprocess_insert_values(&mut stmt.source, &mut stmt.returning);
818
819        // Lower the insertion source
820        lower.visit_stmt_query_mut(&mut stmt.source);
821
822        if let Some(returning) = &mut stmt.returning {
823            lower.visit_returning_mut(returning);
824            lower.constantize_insert_returning(returning, &stmt.source);
825
826            if stmt.source.single
827                && let stmt::Returning::Expr(expr) = &returning
828            {
829                // Not strictly true, but there is nothing that needs to
830                // return a list at this point for a "single" query. If this
831                // is ever needed, remove the assertion.
832                debug_assert!(!expr.is_list());
833            }
834        }
835
836        self.visit_insert_target_mut(&mut stmt.target);
837    }

842        if let Some(with) = &mut stmt.with {

861

863        lower.visit_returning_mut(&mut stmt.returning);
864        lower.apply_lowering_filter_constraint(&mut stmt.filter);

869    fn visit_stmt_update_mut(&mut self, stmt: &mut stmt::Update) {
870        let mut lower = self.scope_expr(&stmt.target);
871
872        let mut returning_changed = false;
873
874        // Before lowering children, convert the "Changed" returning statement
875        // to an expression referencing changed fields.
876        if let Some(returning) = &mut stmt.returning
877            && returning.is_changed()
878        {
879            returning_changed = true;
880
881            if let Some(model) = lower.model() {
882                let mapping = lower.mapping_unwrap();
883
884                // Step 1 — build a mask of all primitives being changed by
885                // OR-ing each assigned field's coverage mask together.
886                let mut changed_bits = stmt::PathFieldSet::new();
887                for projection in stmt.assignments.keys() {
888                    if let Some(mf) = mapping.resolve_field_mapping(projection) {
889                        changed_bits |= mf.field_mask();
890                    }
891                }
892
893                // Step 2 — build the returning expression.
894                *returning = stmt::Returning::Project(build_update_returning(
895                    model.id,
896                    None,
897                    &mapping.fields,
898                    &changed_bits,
899                ));
900            }
901        }
902
903        // Plan relations
904        lower.plan_stmt_update_relations(
905            &mut stmt.assignments,
906            &stmt.filter,
907            &mut stmt.returning,
908            returning_changed,
909        );
910
911        lower.visit_assignments_mut(&mut stmt.assignments);
912        lower.visit_filter_mut(&mut stmt.filter);
913
914        if let Some(expr) = &mut stmt.condition.expr {
915            lower.visit_expr_mut(expr);
916        }
917
918        if let Some(returning) = &mut stmt.returning {
919            lower.visit_returning_mut(returning);
920            // Use the lowered assignments (which are now column-indexed)
921            lower.constantize_update_returning(returning, &stmt.assignments);
922        }
923
924        self.visit_update_target_mut(&mut stmt.target);
925    }

930
931            let table_id = self.schema().table_id_for(source_model.id);
932            *stmt = stmt::Source::table(table_id);

956    /// App-level operand rewrite for `eq` and `ne` binary ops.
957    ///
958    /// `Reference::Model { nesting }` becomes a reference to the model's
959    /// primary-key field; a `BelongsTo` field reference becomes a reference
960    /// to the relation's foreign-key field.  Both rewrites only match on
961    /// app-level shapes; after the surrounding lowering walk converts the
962    /// references to columns, this method has nothing to rewrite.
963    ///
964    /// Must fire before the operand's children are visited, since lowering
965    /// otherwise replaces the field reference with a column reference and
966    /// the rewrite has no app-level shape to match on.
967    fn rewrite_eq_operand(&self, operand: &mut stmt::Expr) {
968        if let stmt::Expr::Reference(expr_reference) = operand {
969            match &*expr_reference {
970                stmt::ExprReference::Model { nesting } => {
971                    let model = self
972                        .expr_cx
973                        .resolve_expr_reference(expr_reference)
974                        .as_model_unwrap();
975
976                    let [pk_field] = &model.primary_key.fields[..] else {
977                        todo!("handle composite keys");
978                    };
979
980                    *operand = stmt::Expr::ref_field(*nesting, pk_field);
981                }
982                stmt::ExprReference::Field { .. } => {
983                    let field = self
984                        .expr_cx
985                        .resolve_expr_reference(expr_reference)
986                        .as_field_unwrap();
987
988                    match &field.ty {
989                        app::FieldTy::Primitive(_) | app::FieldTy::Embedded(_) => {}
990                        app::FieldTy::HasMany(_) | app::FieldTy::HasOne(_) => todo!(),
991                        app::FieldTy::BelongsTo(rel) => {
992                            let [fk_field] = &rel.foreign_key.fields[..] else {
993                                todo!("handle composite keys");
994                            };
995
996                            let stmt::ExprReference::Field { index, .. } = expr_reference else {
997                                panic!()
998                            };
999                            *index = fk_field.source.index;
1000                        }
1001                    }
1002                }
1003                _ => {}
1004            }
1005        }
1006    }

1008    /// App-level rewrite for the LHS of an `IN`-list expression:
1009    /// `Reference::Model { nesting } IN list` becomes `<pk_field> IN list`.
1010    ///
1011    /// Must fire before the LHS is walked, since walking lowers the model
1012    /// reference into a column reference and the rewrite has nothing to
1013    /// match on.
1014    fn rewrite_in_list_model_operand(&self, expr: &mut stmt::ExprInList) {
1015        let (nesting, pk_field_id) = {
1016            let stmt::Expr::Reference(expr_ref @ stmt::ExprReference::Model { nesting }) =
1017                &*expr.expr
1018            else {
1019                return;
1020            };
1021            let nesting = *nesting;
1022            let model = self
1023                .expr_cx
1024                .resolve_expr_reference(expr_ref)
1025                .as_model_unwrap();
1026            let [pk_field_id] = &model.primary_key.fields[..] else {
1027                todo!()
1028            };
1029            (nesting, *pk_field_id)
1030        };
1031
1032        let pk = self.expr_cx.schema().app.field(pk_field_id);
1033
1034        // Sanity-check the RHS shape against the PK type.
1035        match &mut *expr.list {
1036            stmt::Expr::List(expr_list) => {
1037                for item in &mut expr_list.items {
1038                    match item {
1039                        stmt::Expr::Value(value) => {
1040                            assert!(value.is_a(&pk.ty.as_primitive_unwrap().ty));
1041                        }
1042                        _ => todo!("{item:#?}"),
1043                    }
1044                }
1045            }
1046            stmt::Expr::Value(stmt::Value::List(values)) => {
1047                for value in values {
1048                    assert!(value.is_a(&pk.ty.as_primitive_unwrap().ty));
1049                }
1050            }
1051            _ => todo!("expr={expr:#?}"),
1052        }
1053
1054        *expr.expr = stmt::Expr::ref_field(nesting, pk.id());
1055    }

1057    fn lower_expr_binary_op(
1058        &mut self,
1059        op: stmt::BinaryOp,
1060        lhs: &mut stmt::Expr,
1061        rhs: &mut stmt::Expr,
1062    ) -> Option<stmt::Expr> {
1063        match (&mut *lhs, &mut *rhs) {
1064            (stmt::Expr::Value(value), other) | (other, stmt::Expr::Value(value))
1065                if value.is_null() =>
1066            {
1067                let other = other.take();
1068
1069                Some(match op {
1070                    stmt::BinaryOp::Eq => stmt::Expr::is_null(other),
1071                    stmt::BinaryOp::Ne => stmt::Expr::is_not_null(other),
1072                    _ => todo!(),
1073                })
1074            }
1075            // Record-vs-record decomposition for eq/ne. Embedded fields lower
1076            // to record expressions, so a comparison like
1077            // `Record([cast(col, T)]) == Record([val])` only exposes the
1078            // cast (and the cast-stripping rule below) once the record is
1079            // split per-element. Recurse into each pair so cast handling
1080            // fires.
1081            //
1082            // Mirrors the two record shapes handled in `simplify_expr_binary_op`:
1083            // both sides `Expr::Record`, or one side folded to
1084            // `Expr::Value(Value::Record)`.
1085            (stmt::Expr::Record(lhs_rec), stmt::Expr::Record(rhs_rec))
1086                if (op.is_eq() || op.is_ne()) && lhs_rec.len() == rhs_rec.len() =>
1087            {
1088                Some(self.combine_record_op(
1089                    op,
1090                    std::mem::take(&mut lhs_rec.fields),
1091                    std::mem::take(&mut rhs_rec.fields),
1092                ))
1093            }
1094            (stmt::Expr::Record(rec), stmt::Expr::Value(stmt::Value::Record(val_rec)))
1095            | (stmt::Expr::Value(stmt::Value::Record(val_rec)), stmt::Expr::Record(rec))
1096                if (op.is_eq() || op.is_ne()) && rec.len() == val_rec.len() =>
1097            {
1098                let val_exprs = std::mem::take(&mut val_rec.fields)
1099                    .into_iter()
1100                    .map(stmt::Expr::Value)
1101                    .collect();
1102                Some(self.combine_record_op(op, std::mem::take(&mut rec.fields), val_exprs))
1103            }
1104            (stmt::Expr::Cast(expr_cast), _) | (_, stmt::Expr::Cast(expr_cast)) => {
1105                let target_ty = self.capability().native_type_for(&expr_cast.ty);
1106                self.cast_expr(lhs, &target_ty);
1107                self.cast_expr(rhs, &target_ty);
1108                None
1109            }
1110            _ => None,
1111        }
1112    }

1114    /// Combines per-element comparisons of two record-shaped operands into a
1115    /// single boolean expression: AND for `eq`, OR for `ne`. Each pair is
1116    /// recursed through `lower_expr_binary_op` so any inner cast handling
1117    /// fires as if the comparison had been written elementwise.
1118    fn combine_record_op(
1119        &mut self,
1120        op: stmt::BinaryOp,
1121        lhs_fields: Vec<stmt::Expr>,
1122        rhs_fields: Vec<stmt::Expr>,
1123    ) -> stmt::Expr {
1124        let comparisons: Vec<_> = lhs_fields
1125            .into_iter()
1126            .zip(rhs_fields)
1127            .map(|(mut l, mut r)| {
1128                self.lower_expr_binary_op(op, &mut l, &mut r)
1129                    .unwrap_or_else(|| stmt::Expr::binary_op(l, op, r))
1130            })
1131            .collect();
1132
1133        if op.is_eq() {
1134            stmt::Expr::and_from_vec(comparisons)
1135        } else {
1136            stmt::Expr::or_from_vec(comparisons)
1137        }
1138    }

1213        }

1238    /// Returns the ArgId for the new reference
1239    fn new_ref(
1240        &mut self,
1241        source_id: hir::StmtId,
1242        target_id: hir::StmtId,
1243        mut expr_reference: stmt::ExprReference,
1244    ) -> usize {
1245        let stmt::ExprReference::Column(expr_column) = &mut expr_reference else {
1246            todo!()
1247        };
1248
1249        // First, get the nesting so we can resolve the target statmeent
1250        let nesting = expr_column.nesting;
1251
1252        // We only track references that point to statements being executed by
1253        // separate operations. References within the same operation are handled
1254        // by the target database.
1255        debug_assert!(nesting != 0, "expr_reference={expr_reference:#?}");
1256
1257        // Set the nesting to zero as the stored ExprReference will be used from
1258        // the context of the *target* statement.
1259        expr_column.nesting = 0;
1260
1261        let target = &mut self.state.hir[target_id];
1262
1263        // The `batch_load_index` is the index for this reference in the row
1264        // returned from the target statement's ExecStatement operation. This
1265        // ExecStatement operation batch loads all records needed to execute
1266        // the full root statement.
1267        target
1268            .back_refs
1269            .entry(source_id)
1270            .or_default()
1271            .exprs
1272            .insert_full(expr_reference);
1273
1274        // Create an argument for inputing the expr reference's value into the statement.
1275        let source = &mut self.state.hir[source_id];
1276
1277        // See if an arg already exists
1278        for (i, arg) in source.args.iter().enumerate() {
1279            let hir::Arg::Ref {
1280                target_expr_ref, ..
1281            } = arg
1282            else {
1283                continue;
1284            };
1285
1286            if *target_expr_ref == expr_reference {
1287                return i;
1288            }
1289        }
1290
1291        let arg = source.args.len();
1292
1293        source.args.push(hir::Arg::Ref {
1294            target_expr_ref: expr_reference,
1295            stmt_id: target_id,
1296            nesting,

1342    /// Run the canonical pipeline (pre-lower simplify, lowering walk, post-lower
1343    /// simplify) on a synthesized sub-statement and stitch it onto the parent
1344    /// as an `Expr::Arg`.
1345    ///
1346    /// Used at sites where lowering itself synthesizes a new statement to embed
1347    /// in the parent's `Returning` (include subqueries, child inserts for
1348    /// relation planning).  Equivalent to a recursive `lower_stmt` call that
1349    /// passes through the `Expr::Stmt` arm in `visit_expr_mut`, but expressed
1350    /// directly so the synthesized statement does not have to round-trip
1351    /// through an `Expr::Stmt` placeholder.
1352    fn lower_sub_stmt(&mut self, stmt: stmt::Statement) -> stmt::Expr {
1353        let source_id = self.scope_stmt_id();
1354        let mut stmt = Box::new(stmt);
1355
1356        let target_id = self.scope_statement(|child| {
1357            // Via-association rewrite: `Source::Model { via }` becomes an
1358            // explicit WHERE filter so the lowering walk only sees rewritten
1359            // sources.  The IN-subquery lift fires next so non-walk code
1360            // paths see the lifted form.  The eq/ne operand rewrite
1361            // (model→PK, BelongsTo→FK) fires inside the lowering walk via
1362            // `LowerStatement::visit_expr_binary_op_mut`.
1363            association::RewriteVia::new(child.expr_cx).rewrite(&mut stmt);
1364            lift_in_subquery::LiftInSubquery::new(child.expr_cx).rewrite(&mut stmt);
1365            // Pre-lower simplify: remaining heavyweight rules the lowering
1366            // visitor expects to have already fired.
1367            Simplify::with_context(child.expr_cx, child.state.engine.capability)
1368                .visit_mut(&mut *stmt);
1369            // Lowering walk.
1370            child.visit_stmt_mut(&mut stmt);
1371            // Post-lower simplify: heavyweight rules on the lowered tree.
1372            child.state.engine.simplify_stmt(&mut *stmt);
1373        });
1374
1375        // Sub-statements built via this helper always live in the parent's
1376        // Returning clause (include subqueries, child inserts for relation
1377        // planning).  Force the new `Arg::Sub`'s `returning` flag accordingly:
1378        // `plan_nested_merge` keys off `returning: true` to discover include
1379        // subqueries, and the line-399 `Expr::Stmt` arm gets the same flag
1380        // because it fires during the parent's Returning walk.
1381        let saved_cx = std::mem::replace(&mut self.cx, LoweringContext::Returning(None));
1382        let arg = self.new_sub_statement(source_id, target_id, stmt);
1383        self.cx = saved_cx;
1384
1385        if self.state.hir[target_id].independent {
1386            self.curr_stmt_info().deps.insert(target_id);
1387        }
1388
1389        arg
1390    }

1606/// Input implementation for assignment substitution.
1607///
1608/// Provides assignment values when substituting field references in `model_to_table`
1609/// expressions. Handles projections automatically for embedded fields.
1610struct AssignmentInput<'a> {
1611    assignment_projection: stmt::Projection,
1612    value: &'a stmt::Expr,
1613}
1614
1615impl stmt::Input for AssignmentInput<'_> {
1616    fn resolve_ref(
1617        &mut self,
1618        expr_reference: &stmt::ExprReference,
1619        expr_projection: &stmt::Projection,
1620    ) -> Option<stmt::Expr> {
1621        let stmt::ExprReference::Field { nesting: 0, index } = expr_reference else {
1622            return None;
1623        };
1624
1625        let assignment_steps = self.assignment_projection.as_slice();
1626
1627        if *index != assignment_steps[0] {
1628            return None;
1629        }
1630
1631        let remaining_steps = &assignment_steps[1..];
1632
1633        if expr_projection.as_slice() == remaining_steps {
1634            Some(self.value.clone())
1635        } else {
1636            self.value.entry(expr_projection).map(|e| e.to_expr())
1637        }
1638    }
1639}

1641/// Builds the returning expression for a `Returning::Changed` update.
1642///
1643/// Iterates `mapping_fields` using `changed_bits` to determine what to include:
1644///
1645/// - Relation fields → null placeholder (populated during relation planning)
1646/// - Field whose `field_mask` is fully covered by `changed_bits` → emit
1647///   `project(ref_self_field, sub_projection)` (constantized later; project
1648///   omitted when `sub_projection` is identity)
1649/// - Field only partially covered → recurse to build a nested `SparseRecord`
1650///
1651/// `root_field_id` controls how the self-field reference is constructed:
1652/// - `None` at the top level: derived per-iteration as `{ model_id, i }`, since
1653///   each model field is its own root.
1654/// - `Some(id)` when recursing into an embedded type: fixed throughout, because
1655///   all sub-field expressions project from the same top-level ancestor field.
1656fn build_update_returning(
1657    model_id: app::ModelId,
1658    root_field_id: Option<app::FieldId>,
1659    mapping_fields: &[mapping::Field],
1660    changed_bits: &stmt::PathFieldSet,
1661) -> stmt::Expr {
1662    let mut exprs = vec![];
1663    let mut field_set = stmt::PathFieldSet::new();
1664
1665    for (i, mf) in mapping_fields.iter().enumerate() {
1666        let intersection = changed_bits.clone() & mf.field_mask();
1667
1668        if intersection.is_empty() {
1669            continue;
1670        }
1671
1672        field_set.insert(i);
1673
1674        if mf.is_relation() {
1675            // Relation field: null placeholder for the relation planner to fill
1676            // in via set_returning_field.
1677            exprs.push(stmt::Expr::null());
1678        } else {
1679            let root_field_id = root_field_id.unwrap_or(app::FieldId {
1680                model: model_id,
1681                index: i,
1682            });
1683
1684            if intersection == mf.field_mask() {
1685                // Full coverage: all primitives in this field are being updated.
1686                // Emit a projected field reference; the lowering + constantize
1687                // pipeline will substitute the assignment value.
1688                let base = stmt::Expr::ref_self_field(root_field_id);
1689                let expr = if mf.sub_projection().is_identity() {
1690                    base
1691                } else {
1692                    stmt::Expr::project(base, mf.sub_projection().clone())
1693                };
1694                exprs.push(expr);
1695            } else {
1696                // Partial embedded update: only some sub-fields are changing.
1697                // Recurse to build a nested SparseRecord. The lowering pipeline
1698                // resolves each reference to the correct column, and the simplifier
1699                // folds project(record([...]), [i]) → column_ref.
1700                let emb_mapping = mf.as_struct().unwrap();
1701                exprs.push(build_update_returning(
1702                    model_id,
1703                    Some(root_field_id),
1704                    &emb_mapping.fields,
1705                    &intersection,
1706                ));
1707            }
1708        }
1709    }
1710
1711    stmt::Expr::cast(
1712        stmt::ExprRecord::from_vec(exprs),
1713        stmt::Type::SparseRecord(field_set),
1714    )
1715}

1717/// True when an `IN` list is a candidate for the `= ANY($1)` rewrite:
1718///
1719/// - The lhs is scalar (not a `Record`). Composite-key `IN` would need a PG
1720///   row-array bind which the current driver doesn't support.
1721/// - The list is a constant collection of scalar values. Lists containing
1722///   references, sub-statements, or record values stay as expanded `IN`.
1723fn in_list_is_value_list(e: &stmt::ExprInList) -> bool {
1724    if matches!(*e.expr, stmt::Expr::Record(_)) {
1725        return false;
1726    }
1727    let scalar = |v: &stmt::Value| !matches!(v, stmt::Value::Record(_) | stmt::Value::List(_));
1728    match &*e.list {
1729        stmt::Expr::Value(stmt::Value::List(items)) => items.iter().all(scalar),
1730        stmt::Expr::List(list) => list
1731            .items
1732            .iter()
1733            .all(|i| matches!(i, stmt::Expr::Value(v) if scalar(v))),
1734        _ => false,
1735    }

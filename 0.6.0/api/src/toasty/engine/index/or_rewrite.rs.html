3/// Converts an index filter to canonical `ANY(MAP(...))` form for backends that do
4/// not support OR in key conditions (e.g. DynamoDB).
5///
6/// Steps:
7///   1. If the expression is already an `InList(col, [v1, v2, ...])`, expand it
8///      directly to `ANY(MAP([v1, v2, ...], col = arg(0)))` — equivalent but in
9///      the canonical fan-out form.
10///   2. Flatten the expression to Disjunctive Normal Form (DNF) — a flat list of OR
11///      branches, each branch being a single predicate or AND of predicates.
12///   3. Group branches by their structural shape (predicate with literal values
13///      replaced by `arg(i)`).
14///   4. Unify each same-shape group into `ANY(MAP(Value::List([v1, v2, ...]), shape))`.
15pub(super) fn index_filter_to_any_map(expr: stmt::Expr) -> stmt::Expr {
16    // `col IN [v1, v2, ...]` is equivalent to `v1 = col OR v2 = col OR ...`.
17    // Expand it directly to the canonical fan-out form so the DNF step doesn't
18    // need to handle InList as a leaf.
19    if let stmt::Expr::InList(ref in_list) = expr
20        && matches!(*in_list.list, stmt::Expr::Value(stmt::Value::List(_)))
21    {
22        let shape = stmt::Expr::from(stmt::ExprBinaryOp {
23            lhs: in_list.expr.clone(),
24            op: stmt::BinaryOp::Eq,
25            rhs: Box::new(stmt::Expr::arg(0)),
26        });
27        return stmt::Expr::any(stmt::Expr::map(*in_list.list.clone(), shape));
28    }
29
30    let branches = flatten_to_dnf(expr);
31    unify_dnf_branches(branches)
32}

34/// Flatten an expression to a list of OR branches in Disjunctive Normal Form
35/// using an iterative work queue.
36///
37/// - `Or`: extend the queue with each operand.
38/// - `And`: find the first `Or` operand and distribute over its branches,
39///   re-queuing each resulting `And` for further processing. When no `Or`
40///   operands remain the `And` is in final form.
41/// - Leaf predicate: emit directly as a final branch.
42fn flatten_to_dnf(expr: stmt::Expr) -> Vec<stmt::Expr> {
43    let mut branches: Vec<stmt::Expr> = Vec::new();
44    let mut queue: Vec<stmt::Expr> = vec![expr];
45
46    while let Some(expr) = queue.pop() {
47        match expr {
48            stmt::Expr::Or(or) => queue.extend(or.operands.into_iter().rev()),
49            stmt::Expr::And(and) => process_and(and, &mut queue, &mut branches),
50            leaf => branches.push(leaf),
51        }
52    }
53
54    // Validate that no branch contains an Or inside an Any(Map(...)). If it
55    // does, the distribution logic above has a bug.
56    for branch in &branches {
57        assert_no_or_in_any(branch);
58    }
59
60    branches
61}

63/// Process one `And` expression from the DNF work queue.
64///
65/// Priority:
66///   1. If any operand is `Or`, distribute AND over it and re-queue each branch.
67///   2. If any operand is `Any(Map(...))`, distribute the remaining operands
68///      into the map predicate and re-queue the resulting `Any`.
69///   3. No `Or` or `Any` operands: emit as a final DNF conjunction.
70fn process_and(and: stmt::ExprAnd, queue: &mut Vec<stmt::Expr>, branches: &mut Vec<stmt::Expr>) {
71    if let Some(pos) = and
72        .operands
73        .iter()
74        .position(|op| matches!(op, stmt::Expr::Or(_)))
75    {
76        return distribute_over_or(and, pos, queue);
77    }
78
79    if let Some(pos) = and
80        .operands
81        .iter()
82        .position(|op| matches!(op, stmt::Expr::Any(_)))
83    {
84        return distribute_into_any(and, pos, queue);
85    }
86
87    branches.push(stmt::Expr::And(and));
88}

90/// Distribute AND over an `Or` operand at `pos`, re-queuing one `And` per branch.
91///
92/// `(p AND (a OR b) AND q)` → `(p AND a AND q)` and `(p AND b AND q)` on the queue.
93fn distribute_over_or(and: stmt::ExprAnd, pos: usize, queue: &mut Vec<stmt::Expr>) {
94    let mut operands = and.operands;
95    let stmt::Expr::Or(or) = operands.remove(pos) else {
96        unreachable!()
97    };
98
99    for branch in or.operands.into_iter().rev() {
100        let mut new_operands = operands.clone();
101        new_operands.insert(pos, branch);
102        queue.push(
103            stmt::ExprAnd {
104                operands: new_operands,
105            }
106            .into(),
107        );
108    }
109}

111/// Distribute the non-`Any` operands of `and` into an `Any(Map(...))` at `pos`.
112///
113/// `AND(p, ANY(MAP(base, pred)))` → `ANY(MAP(base, AND(pred, p)))`.
114///
115/// This is valid because the non-Any operands do not reference the map's arg variable.
116fn distribute_into_any(and: stmt::ExprAnd, pos: usize, queue: &mut Vec<stmt::Expr>) {
117    let mut operands = and.operands;
118    let stmt::Expr::Any(any) = operands.remove(pos) else {
119        unreachable!()
120    };
121    let stmt::Expr::Map(map) = *any.expr else {
122        todo!("Any with non-Map expr in AND distribution");
123    };
124
125    // Keep the original map predicate first, then the distributed And operands.
126    let mut inner_operands = vec![*map.map];
127    inner_operands.extend(operands);
128    let inner: stmt::Expr = if inner_operands.len() == 1 {
129        inner_operands.into_iter().next().unwrap()
130    } else {
131        stmt::ExprAnd {
132            operands: inner_operands,
133        }
134        .into()
135    };
136
137    queue.push(
138        stmt::ExprAny {
139            expr: Box::new(stmt::Expr::Map(stmt::ExprMap {
140                base: map.base,
141                map: Box::new(inner),
142            })),
143        }
144        .into(),
145    );
146}

148/// Group DNF branches by shape; unify each group into `ANY(MAP(...))`.
149/// If there is only a single branch (no OR), returns it unchanged.
150fn unify_dnf_branches(branches: Vec<stmt::Expr>) -> stmt::Expr {
151    if branches.len() == 1 {
152        return branches.into_iter().next().unwrap();
153    }
154
155    // Each group: (shape, per-branch scalar-or-record values).
156    let mut groups: Vec<(stmt::Expr, Vec<stmt::Value>)> = vec![];
157
158    for branch in branches {
159        let (shape, value) = extract_shape(branch);
160        if let Some((_, values)) = groups.iter_mut().find(|(s, _)| *s == shape) {
161            values.push(value);
162        } else {
163            groups.push((shape, vec![value]));
164        }
165    }
166
167    if groups.len() > 1 {
168        todo!(
169            "OR index filter with multiple distinct branch shapes is not yet implemented; \
170             shapes: {:#?}",
171            groups.iter().map(|(s, _)| s).collect::<Vec<_>>()
172        );
173    }
174
175    let (shape, values) = groups.into_iter().next().unwrap();
176
177    stmt::Expr::any(stmt::Expr::map(
178        stmt::Expr::Value(stmt::Value::List(values)),
179        shape,
180    ))
181}

183/// Extract the per-call predicate template (shape) and single value for one DNF branch.
184///
185/// - `col op literal` → shape `col op arg(0)`, value `literal`
186/// - `col1 op1 v1 AND col2 op2 v2 AND ...` → shape with `arg(i)` per column,
187///   value `Value::Record([v1, v2, ...])` — composite key fan-out (TODO)
188fn extract_shape(branch: stmt::Expr) -> (stmt::Expr, stmt::Value) {
189    match branch {
190        stmt::Expr::BinaryOp(b) => {
191            let stmt::Expr::Value(v) = *b.rhs else {
192                todo!("non-literal value in OR branch rhs: {:#?}", b.rhs);
193            };
194            let shape: stmt::Expr = stmt::ExprBinaryOp {
195                lhs: b.lhs,
196                op: b.op,
197                rhs: Box::new(stmt::Expr::arg(0)),
198            }
199            .into();
200            (shape, v)
201        }
202        // Composite key: (col1 = t1 AND col2 >= s1) OR (col1 = t2 AND col2 >= s2)
203        // → ANY(MAP([(t1,s1),(t2,s2)], col1=arg(0) AND col2>=arg(1)))
204        stmt::Expr::And(and) => {
205            let mut values = vec![];
206            let mut shape_operands = vec![];
207
208            for (i, operand) in and.operands.into_iter().enumerate() {
209                let stmt::Expr::BinaryOp(b) = operand else {
210                    todo!(
211                        "non-BinaryOp operand in composite AND branch: {:#?}",
212                        operand
213                    );
214                };
215                let stmt::Expr::Value(v) = *b.rhs else {
216                    todo!(
217                        "non-literal value in composite AND branch rhs: {:#?}",
218                        b.rhs
219                    );
220                };
221                values.push(v);
222                shape_operands.push(stmt::Expr::from(stmt::ExprBinaryOp {
223                    lhs: b.lhs,
224                    op: b.op,
225                    rhs: Box::new(stmt::Expr::arg(i)),
226                }));
227            }
228
229            let shape = stmt::Expr::from(stmt::ExprAnd {
230                operands: shape_operands,
231            });
232            let record = stmt::Value::Record(stmt::ValueRecord::from_vec(values));
233            (shape, record)
234        }
235        _ => todo!("unsupported branch type in OR index filter: {branch:#?}"),
236    }
237}

239/// Asserts that no `Any(Map(...))` in `expr` contains an `Or` anywhere in its
240/// predicate sub-tree. This catches bugs where OR distribution was incomplete.
241fn assert_no_or_in_any(expr: &stmt::Expr) {
242    match expr {
243        stmt::Expr::Any(any) => {
244            assert!(
245                !contains_or(&any.expr),
246                "Any(Map(...)) contains an Or expression after DNF distribution; \
247                 this is a bug in flatten_to_dnf: {:#?}",
248                any.expr
249            );
250        }
251        stmt::Expr::And(and) => and.operands.iter().for_each(assert_no_or_in_any),
252        _ => {}
253    }
254}
255
256/// Returns true if `expr` contains an `Expr::Or` anywhere in its sub-tree.
257fn contains_or(expr: &stmt::Expr) -> bool {
258    match expr {
259        stmt::Expr::Or(_) => true,
260        stmt::Expr::And(and) => and.operands.iter().any(contains_or),
261        stmt::Expr::Any(a) => contains_or(&a.expr),
262        stmt::Expr::Map(m) => contains_or(&m.base) || contains_or(&m.map),
263        stmt::Expr::BinaryOp(b) => contains_or(&b.lhs) || contains_or(&b.rhs),
264        stmt::Expr::Not(n) => contains_or(&n.expr),
265        stmt::Expr::IsNull(n) => contains_or(&n.expr),
266        // Leaf nodes (Arg, Reference, Value, Default, Type, etc.) contain no sub-expressions.
267        _ => false,
268    }

17/// Configures the schema and driver for a [`Db`] instance.
18///
19/// Provide model types with [`models`](Self::models) (using the
20/// [`models!`](crate::models!) macro), optionally set a table name prefix with
21/// [`table_name_prefix`](Self::table_name_prefix), then call
22/// [`connect`](Self::connect) (URL-based) or [`build`](Self::build) (driver
23/// instance) to open the database.
24///
25/// # Examples
26///
27/// ```
28/// # tokio::runtime::Runtime::new().unwrap().block_on(async {
29/// # #[derive(Debug, toasty::Model)]
30/// # struct User {
31/// #     #[key]
32/// #     id: i64,
33/// #     name: String,
34/// # }
35/// let driver = toasty_driver_sqlite::Sqlite::in_memory();
36/// let db = toasty::Db::builder()
37///     .models(toasty::models!(User))
38///     .build(driver)
39///     .await
40///     .unwrap();
41/// # });
42/// ```
43#[derive(Default)]
44pub struct Builder {
45    /// Model definitions from macro
46    ///
47    /// TODO: move this into `core::schema::Builder` after old schema file
48    /// implementatin is removed.
49    models: app::ModelSet,
50
51    /// Schema builder
52    core: schema::Builder,
53
54    /// Connection pool configuration
55    pool: PoolConfig,
56}
57
58impl Builder {
59    /// Set the models whose schemas will be included when the database is
60    /// built. Related and embedded models are discovered automatically through
61    /// field traversal, so you only need to list your entry-point models.
62    ///
63    /// Use the [`models!`](crate::models!) macro to build a [`ModelSet`] from
64    /// your `#[derive(Model)]` types.
65    ///
66    /// # Examples
67    ///
68    /// ```
69    /// # #[derive(Debug, toasty::Model)]
70    /// # struct User {
71    /// #     #[key]
72    /// #     id: i64,
73    /// #     name: String,
74    /// # }
75    /// let mut builder = toasty::Db::builder();
76    /// builder.models(toasty::models!(User));
77    /// ```
78    pub fn models(&mut self, models: ModelSet) -> &mut Self {
79        self.models = models;
80        self
81    }
82
83    /// Set the table name prefix for all tables
84    pub fn table_name_prefix(&mut self, prefix: &str) -> &mut Self {
85        self.core.table_name_prefix(prefix);
86        self
87    }

89    /// Set the maximum number of connections the pool will maintain.
90    ///
91    /// Defaults to `num_cpus * 2`.
92    ///
93    /// Drivers may cap this below the requested value. For example, an
94    /// in-memory SQLite database forces a single connection. When that
95    /// happens a warning is emitted and the driver cap is used.
96    pub fn max_pool_size(&mut self, max_size: usize) -> &mut Self {
97        self.pool.max_size = max_size;
98        self
99    }
100
101    /// Set the maximum time to wait for a free connection from the pool
102    /// before returning an error. Passing `None` disables the timeout,
103    /// which is the default.
104    pub fn pool_wait_timeout(&mut self, timeout: Option<Duration>) -> &mut Self {
105        self.pool.timeouts.wait = timeout;
106        self
107    }
108
109    /// Set the maximum time allowed for establishing a new database
110    /// connection. Passing `None` disables the timeout, which is the
111    /// default.
112    pub fn pool_create_timeout(&mut self, timeout: Option<Duration>) -> &mut Self {
113        self.pool.timeouts.create = timeout;
114        self
115    }
116
117    /// Configure how often the pool's background sweep pings an idle
118    /// connection to detect a silently-broken backend (a database
119    /// restart, a load-balancer-closed socket, a session timeout). On
120    /// success the connection is returned as most-recently-used; on
121    /// failure the pool eagerly pings every other idle connection and
122    /// drops the ones that fail, so a single bad result drains every
123    /// dead connection in one pass.
124    ///
125    /// The same eager sweep also fires when a user query observes
126    /// [`Error::is_connection_lost`](toasty_core::Error::is_connection_lost)
127    /// — so a restart usually costs at most one failed query rather
128    /// than one per pooled connection.
129    ///
130    /// Defaults to `Some(60s)`. Pass `None` to disable the sweep and
131    /// rely on passive error-driven recovery only.
132    pub fn pool_health_check_interval(&mut self, interval: Option<Duration>) -> &mut Self {
133        self.pool.health_check_interval = interval;
134        self
135    }
136
137    /// Validate every connection with an active ping before handing it
138    /// to the caller. Useful for deployments that cannot tolerate even
139    /// one failed user query — a public API behind a flaky network, an
140    /// idempotent worker that does not implement retry. A failing ping
141    /// evicts the connection and the pool reuses another idle slot or
142    /// opens a fresh one; the caller sees either a healthy connection
143    /// or a clean `connection_pool` error if no slot can be opened
144    /// within [`pool_create_timeout`](Self::pool_create_timeout).
145    ///
146    /// The trade-off is one round-trip per checkout. Combine with a
147    /// larger [`max_pool_size`](Self::max_pool_size) if the extra
148    /// latency starts queueing requests. Independent of the background
149    /// sweep — most deployments want one or the other, but enabling
150    /// both is safe.
151    ///
152    /// Defaults to `false`.
153    pub fn pool_pre_ping(&mut self, pre_ping: bool) -> &mut Self {
154        self.pool.pre_ping = pre_ping;
155        self
156    }
157
158    /// Evict any pooled connection older than this duration when the
159    /// pool considers reusing it. Useful when an idle timeout on the
160    /// server, a load balancer, or a NAT in front of the database
161    /// silently closes long-lived sockets — capping the lifetime
162    /// bounds how long a connection can survive past such a close.
163    ///
164    /// The check runs in `recycle` (when the pool considers handing
165    /// the connection back out), not in the background. A query that
166    /// holds a connection past the cap is allowed to finish; the
167    /// connection is evicted on its next return.
168    ///
169    /// Recommended for any deployment that talks to a remote database:
170    /// pick a duration shorter than every idle timeout in the path
171    /// (server, load balancer, NAT). 30 minutes works for most clouds.
172    ///
173    /// Defaults to `None` (no cap).
174    pub fn pool_max_connection_lifetime(&mut self, lifetime: Option<Duration>) -> &mut Self {
175        self.pool.max_connection_lifetime = lifetime;
176        self
177    }
178
179    /// Evict any pooled connection that has been sitting unused for
180    /// longer than this duration. Complements
181    /// [`pool_max_connection_lifetime`](Self::pool_max_connection_lifetime):
182    /// the idle cap targets connections specifically held idle past a
183    /// known timeout, the lifetime cap targets all connections
184    /// regardless of recent use.
185    ///
186    /// Checked in `recycle`, not in the background.
187    ///
188    /// Defaults to `None` (no cap).
189    pub fn pool_max_connection_idle_time(&mut self, idle: Option<Duration>) -> &mut Self {
190        self.pool.max_connection_idle_time = idle;
191        self
192    }
193
194    /// Build and return the app-level schema from the registered models
195    /// without opening a database connection.
196    ///
197    /// This is useful for tooling that needs the schema without a running
198    /// database (e.g., migration generators).
199    pub fn build_app_schema(&self) -> Result<app::Schema> {
200        app::Schema::from_macro(self.models.clone())
201    }
202
203    /// Open a database connection using a URL string.
204    ///
205    /// The URL scheme selects the driver (`sqlite://`, `postgresql://`,
206    /// `mysql://`, `dynamodb://`). The corresponding feature flag must be
207    /// enabled.
208    ///
209    /// # Errors
210    ///
211    /// Returns an error if the URL cannot be parsed, the scheme is
212    /// unsupported or its feature flag is not enabled, or the initial
213    /// connection fails.
214    ///
215    /// # Examples
216    ///
217    /// ```
218    /// # tokio::runtime::Runtime::new().unwrap().block_on(async {
219    /// # #[derive(Debug, toasty::Model)]
220    /// # struct User {
221    /// #     #[key]
222    /// #     id: i64,
223    /// #     name: String,
224    /// # }
225    /// let db = toasty::Db::builder()
226    ///     .models(toasty::models!(User))
227    ///     .connect("sqlite://memory")
228    ///     .await
229    ///     .unwrap();
230    /// # });
231    /// ```
232    pub async fn connect(&mut self, url: &str) -> Result<Db> {
233        let con = Connect::new(url).await?;
234        self.build(con).await
235    }

237    /// Build a [`Db`] from an already-constructed driver instance.
238    ///
239    /// Use this instead of [`connect`](Self::connect) when you need to
240    /// configure the driver yourself (e.g., an in-memory SQLite database for
241    /// tests).
242    ///
243    /// # Errors
244    ///
245    /// Returns an error if the driver reports invalid capabilities or the
246    /// initial connection cannot be acquired.
247    ///
248    /// # Examples
249    ///
250    /// ```
251    /// # tokio::runtime::Runtime::new().unwrap().block_on(async {
252    /// # #[derive(Debug, toasty::Model)]
253    /// # struct User {
254    /// #     #[key]
255    /// #     id: i64,
256    /// #     name: String,
257    /// # }
258    /// let driver = toasty_driver_sqlite::Sqlite::in_memory();
259    /// let db = toasty::Db::builder()
260    ///     .models(toasty::models!(User))
261    ///     .build(driver)
262    ///     .await
263    ///     .unwrap();
264    /// # });
265    /// ```
266    pub async fn build(&mut self, driver: impl Driver) -> Result<Db> {
267        tracing::info!(models = self.models.len(), "building database schema");
268        let capability = driver.capability();
269        capability.validate()?;
270        let schema = self.core.build(self.build_app_schema()?, capability)?;
271
272        tracing::info!(tables = schema.db.tables.len(), "schema built successfully");
273
274        let engine = Engine::new(Arc::new(schema), capability);
275        let pool = Pool::new(driver, engine.clone(), self.pool.clone())?;
276
277        let shared = Arc::new(Shared { engine, pool });
278
279        // see if we're able to acquire a valid connection
280        let conn = shared.pool.get(shared.clone()).await?;
281        std::mem::drop(conn);
282
283        tracing::info!("database ready");
284        Ok(Db { shared })
285    }

1use crate::serializer::ExprContext;
2
3use super::{Flavor, ToSql};
4
5use toasty_core::schema::db;
6
7impl ToSql for &db::Type {
8    fn to_sql(self, cx: &ExprContext<'_>, f: &mut super::Formatter<'_>) {
9        match self {
10            db::Type::Boolean => fmt!(cx, f, "BOOLEAN"),
11            db::Type::Integer(1..=2) => fmt!(cx, f, "SMALLINT"),
12            db::Type::Integer(3..=4) => fmt!(cx, f, "INTEGER"),
13            db::Type::Integer(5..=8) => fmt!(cx, f, "BIGINT"),
14            db::Type::Integer(_) => todo!(),
15            db::Type::UnsignedInteger(size) => {
16                match f.serializer.flavor {
17                    Flavor::Mysql => match size {
18                        1 => fmt!(cx, f, "TINYINT UNSIGNED"),
19                        2 => fmt!(cx, f, "SMALLINT UNSIGNED"),
20                        3..=4 => fmt!(cx, f, "INT UNSIGNED"),
21                        5..=8 => fmt!(cx, f, "BIGINT UNSIGNED"),
22                        _ => todo!("Unsupported unsigned integer size: {}", size),
23                    },
24                    Flavor::Postgresql => {
25                        match size {
26                            1 => fmt!(cx, f, "SMALLINT"),   // u8 -> SMALLINT (i16)
27                            2 => fmt!(cx, f, "INTEGER"),    // u16 -> INTEGER (i32)
28                            3..=4 => fmt!(cx, f, "BIGINT"), // u32 -> BIGINT (i64)
29                            5..=8 => fmt!(cx, f, "BIGINT"), // u64 -> BIGINT (i64) with capability limits
30                            _ => todo!("Unsupported unsigned integer size: {}", size),
31                        }
32                    }
33                    Flavor::Sqlite => {
34                        // SQLite uses INTEGER for all integer types
35                        fmt!(cx, f, "INTEGER")
36                    }
37                }
38            }
39            db::Type::Float(size) => match f.serializer.flavor {
40                Flavor::Sqlite => fmt!(cx, f, "REAL"),
41                Flavor::Postgresql => {
42                    if *size <= 4 {
43                        fmt!(cx, f, "REAL")
44                    } else {
45                        fmt!(cx, f, "DOUBLE PRECISION")
46                    }
47                }
48                Flavor::Mysql => {
49                    if *size <= 4 {
50                        fmt!(cx, f, "FLOAT")
51                    } else {
52                        fmt!(cx, f, "DOUBLE")
53                    }
54                }
55            },
56            db::Type::Text => fmt!(cx, f, "TEXT"),
57            db::Type::VarChar(size) => fmt!(cx, f, "VARCHAR(" size ")"),
58            db::Type::Uuid => {
59                fmt!(
60                    cx,
61                    f,
62                    match f.serializer.flavor {
63                        Flavor::Postgresql => "UUID",
64                        _ => todo!("Unsupported type UUID"),
65                    }
66                );
67            }
68            db::Type::Numeric(None) => match f.serializer.flavor {
69                Flavor::Postgresql => fmt!(cx, f, "NUMERIC"),
70                Flavor::Mysql => todo!(
71                    "MySQL does not support arbitrary-precision NUMERIC; precision and scale must be specified"
72                ),
73                Flavor::Sqlite => todo!("SQLite does not support NUMERIC type"),
74            },
75            db::Type::Numeric(Some((precision, scale))) => match f.serializer.flavor {
76                Flavor::Postgresql => fmt!(cx, f, "NUMERIC(" precision ", " scale ")"),
77                Flavor::Mysql => fmt!(cx, f, "DECIMAL(" precision ", " scale ")"),
78                Flavor::Sqlite => todo!("SQLite does not support NUMERIC type"),
79            },
80            db::Type::Binary(size) => match f.serializer.flavor {
81                Flavor::Mysql => fmt!(cx, f, "BINARY(" size ")"),
82                _ => todo!("Unsupported fixed size binary type"),
83            },
84            db::Type::Blob => match f.serializer.flavor {
85                Flavor::Postgresql => fmt!(cx, f, "BYTEA"),
86                Flavor::Mysql => fmt!(cx, f, "BLOB"),
87                Flavor::Sqlite => fmt!(cx, f, "BLOB"),
88            },
89            db::Type::Timestamp(precision) => match f.serializer.flavor {
90                Flavor::Postgresql => fmt!(cx, f, "TIMESTAMPTZ(" precision ")"),
91                Flavor::Mysql => fmt!(cx, f, "TIMESTAMP(" precision ")"),
92                Flavor::Sqlite => todo!("SQLite does not support Timestamp"),
93            },
94            db::Type::Date => match f.serializer.flavor {
95                Flavor::Postgresql | Flavor::Mysql => fmt!(cx, f, "DATE"),
96                Flavor::Sqlite => todo!("SQLite does not support Date"),
97            },
98            db::Type::Time(precision) => match f.serializer.flavor {
99                Flavor::Postgresql => fmt!(cx, f, "TIME(" precision ")"),
100                Flavor::Mysql => fmt!(cx, f, "TIME(" precision ")"),
101                Flavor::Sqlite => todo!("SQLite does not support Time"),
102            },
103            db::Type::DateTime(precision) => match f.serializer.flavor {
104                Flavor::Postgresql => fmt!(cx, f, "TIMESTAMP(" precision ")"),
105                Flavor::Mysql => fmt!(cx, f, "DATETIME(" precision ")"),
106                Flavor::Sqlite => todo!("SQLite does not support DateTime"),
107            },
108            db::Type::Enum(type_enum) => match f.serializer.flavor {
109                // PostgreSQL: reference the named enum type created with CREATE TYPE.
110                Flavor::Postgresql => {
111                    let name = type_enum
112                        .name
113                        .as_deref()
114                        .expect("PostgreSQL enums require a type name");
115                    fmt!(cx, f, name);
116                }
117                // MySQL: inline ENUM('label1', 'label2', ...) column type.
118                Flavor::Mysql => {
119                    use toasty_core::stmt::Value;
120
121                    f.dst.push_str("ENUM(");
122                    for (i, variant) in type_enum.variants.iter().enumerate() {
123                        if i > 0 {
124                            f.dst.push_str(", ");
125                        }
126                        Value::String(variant.name.clone()).to_sql(cx, f);
127                    }
128                    f.dst.push(')');
129                }
130                // SQLite: TEXT column (CHECK constraint added in ColumnDef).
131                Flavor::Sqlite => fmt!(cx, f, "TEXT"),
132            },
133            db::Type::List(elem) => match f.serializer.flavor {
134                Flavor::Postgresql => fmt!(cx, f, elem.as_ref() "[]"),
135                // MySQL stores `Vec<scalar>` as a JSON document; SQLite uses
136                // TEXT (JSON1 functions operate on either, but TEXT is the
137                // idiomatic affinity). The element type is tracked by the
138                // engine — it doesn't surface in the column DDL.
139                Flavor::Mysql => fmt!(cx, f, "JSON"),
140                Flavor::Sqlite => fmt!(cx, f, "TEXT"),
141            },
142            db::Type::Custom(custom) => fmt!(cx, f, custom.as_str()),
143        }
144    }
145}

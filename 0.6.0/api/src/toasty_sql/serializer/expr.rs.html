10impl ToSql for &stmt::Expr {
11    fn to_sql(self, cx: &ExprContext<'_>, f: &mut super::Formatter<'_>) {
12        match self {
13            stmt::Expr::And(expr) => {
14                fmt!(cx, f, Delimited(&expr.operands, " AND "));
15            }
16            stmt::Expr::BinaryOp(expr) => {
17                assert!(!expr.lhs.is_value_null());
18                assert!(!expr.rhs.is_value_null());
19
20                fmt!(cx, f, expr.lhs " " expr.op " " expr.rhs);
21            }
22            stmt::Expr::Exists(expr) => {
23                f.depth += 1;
24                fmt!(cx, f, "EXISTS (" expr.subquery ")");
25                f.depth -= 1;
26            }
27            stmt::Expr::Func(stmt::ExprFunc::Count(func)) => match (&func.arg, &func.filter) {
28                (None, None) => fmt!(cx, f, "COUNT(*)"),
29                // Mysql does not support filters, so translate it to an expression
30                (None, Some(expr)) if f.serializer.is_mysql() => {
31                    fmt!(cx, f, "COUNT(CASE WHEN " expr " THEN 1 END)")
32                }
33                (None, Some(expr)) => fmt!(cx, f, "COUNT(*) FILTER (WHERE " expr ")"),
34                _ => todo!("func={func:#?}"),
35            },
36            stmt::Expr::Func(stmt::ExprFunc::LastInsertId(_)) => {
37                fmt!(cx, f, "LAST_INSERT_ID()")
38            }
39            stmt::Expr::IsSuperset(e) => match f.serializer.flavor {
40                Flavor::Postgresql => fmt!(cx, f, e.lhs.as_ref() " @> " e.rhs.as_ref()),
41                // The rhs Value::List is bound as one JSON string. MySQL's
42                // `JSON_CONTAINS(target, candidate)` matches when every
43                // element of `candidate` appears in `target`.
44                Flavor::Mysql => {
45                    fmt!(cx, f, "JSON_CONTAINS(" e.lhs.as_ref() ", " e.rhs.as_ref() ")")
46                }
47                // SQLite has no direct superset operator; emulate via
48                // `NOT EXISTS (rhs element with no match in lhs)`.
49                Flavor::Sqlite => fmt!(
50                    cx, f,
51                    "NOT EXISTS (SELECT 1 FROM json_each(" e.rhs.as_ref()
52                    ") AS r WHERE r.value NOT IN (SELECT l.value FROM json_each("
53                    e.lhs.as_ref() ") AS l))"
54                ),
55            },
56            stmt::Expr::Intersects(e) => match f.serializer.flavor {
57                Flavor::Postgresql => fmt!(cx, f, e.lhs.as_ref() " && " e.rhs.as_ref()),
58                Flavor::Mysql => {
59                    fmt!(cx, f, "JSON_OVERLAPS(" e.lhs.as_ref() ", " e.rhs.as_ref() ")")
60                }
61                Flavor::Sqlite => fmt!(
62                    cx, f,
63                    "EXISTS (SELECT 1 FROM json_each(" e.rhs.as_ref()
64                    ") AS r WHERE r.value IN (SELECT l.value FROM json_each("
65                    e.lhs.as_ref() ") AS l))"
66                ),
67            },
68            stmt::Expr::Length(e) => match f.serializer.flavor {
69                Flavor::Postgresql => fmt!(cx, f, "cardinality(" e.expr.as_ref() ")"),
70                Flavor::Mysql => fmt!(cx, f, "JSON_LENGTH(" e.expr.as_ref() ")"),
71                Flavor::Sqlite => fmt!(cx, f, "json_array_length(" e.expr.as_ref() ")"),
72            },
73            stmt::Expr::Ident(name) => {
74                fmt!(cx, f, Ident(name));
75            }
76            stmt::Expr::InList(expr) => {
77                fmt!(cx, f, expr.expr " IN " expr.list);
78            }
79            stmt::Expr::AnyOp(expr) => match f.serializer.flavor {
80                // `value = ANY(col)` — PostgreSQL's array membership operator.
81                // Drives `Path::contains` for native-array columns and the
82                // IN-list rewrite.
83                Flavor::Postgresql => {
84                    fmt!(cx, f, expr.lhs " " expr.op " ANY(" expr.rhs ")");
85                }
86                // MySQL's `value MEMBER OF (json_array)` (8.0.17+). Only the
87                // equality form makes sense; `Path::contains` is the only
88                // current emitter and the lowering pass never produces
89                // ANY on MySQL since `predicate_match_any` is false.
90                Flavor::Mysql if matches!(expr.op, stmt::BinaryOp::Eq) => {
91                    fmt!(cx, f, expr.lhs " MEMBER OF (" expr.rhs ")");
92                }
93                Flavor::Mysql => unreachable!("AnyOp with non-Eq operator on MySQL: {expr:?}"),
94                // SQLite renders `value = ANY(col)` (i.e. `Path::contains`)
95                // as `value IN (SELECT value FROM json_each(col))`.
96                Flavor::Sqlite if matches!(expr.op, stmt::BinaryOp::Eq) => {
97                    fmt!(
98                        cx, f,
99                        expr.lhs " IN (SELECT value FROM json_each(" expr.rhs "))"
100                    );
101                }
102                Flavor::Sqlite => unreachable!("AnyOp with non-Eq operator on SQLite: {expr:?}"),
103            },
104            stmt::Expr::AllOp(expr) => {
105                fmt!(cx, f, expr.lhs " " expr.op " ALL(" expr.rhs ")");
106            }
107            stmt::Expr::InSubquery(expr) => {
108                fmt!(cx, f, expr.expr " IN (" expr.query ")");
109            }
110            stmt::Expr::IsNull(expr) => {
111                fmt!(cx, f, expr.expr " IS NULL");
112            }
113            stmt::Expr::Like(expr) => {
114                let op =
115                    if expr.case_insensitive && matches!(f.serializer.flavor, Flavor::Postgresql) {
116                        " ILIKE "
117                    } else {
118                        " LIKE "
119                    };
120                fmt!(cx, f, expr.expr op expr.pattern);
121                if let Some(escape) = expr.escape {
122                    let escape = &stmt::Value::String(escape.to_string());
123                    fmt!(cx, f, " ESCAPE " escape);
124                }
125            }
126            stmt::Expr::StartsWith(expr) => {
127                // The lowering pass leaves `StartsWith` in place when
128                // `Capability::native_prefix_match_op` is true. PostgreSQL
129                // is the only such SQL flavor today (`^@` operator).
130                match f.serializer.flavor {
131                    Flavor::Postgresql => {
132                        fmt!(cx, f, expr.expr " ^@ " expr.prefix);
133                    }
134                    Flavor::Sqlite | Flavor::Mysql => {
135                        unreachable!(
136                            "StartsWith should have been lowered to LIKE for non-PostgreSQL flavors"
137                        );
138                    }
139                }
140            }
141            stmt::Expr::Not(expr) => {
142                fmt!(cx, f, "NOT (" expr.expr ")");
143            }
144            stmt::Expr::Or(expr) => {
145                fmt!(cx, f, Delimited(&expr.operands, " OR "));
146            }
147            stmt::Expr::Record(expr) => {
148                let fields = Comma(expr.fields.iter());
149                fmt!(cx, f, "(" fields ")");
150            }
151            stmt::Expr::Reference(expr_reference @ stmt::ExprReference::Column(expr_column)) => {
152                if f.alias {
153                    let depth = f.depth - expr_column.nesting;
154
155                    match cx.resolve_expr_reference(expr_reference) {
156                        ResolvedRef::Column(column) => {
157                            let name = Ident(&column.name);
158                            fmt!(cx, f, "tbl_" depth "_" expr_column.table "." name)
159                        }
160                        ResolvedRef::Cte { .. } | ResolvedRef::Derived(_) => {
161                            fmt!(cx, f, "tbl_" depth "_" expr_column.table "." ColumnAlias(expr_column.column))
162                        }
163                        ResolvedRef::Model(model) => {
164                            panic!("Model references cannot be serialized to SQL; model={model:?}")
165                        }
166                        ResolvedRef::Field(field) => {
167                            panic!("Field references cannot be serialized to SQL; field={field:?}")
168                        }
169                    }
170                } else {
171                    let column = cx.resolve_expr_reference(expr_reference).as_column_unwrap();
172                    fmt!(cx, f, Ident(&column.name))
173                }
174            }
175            stmt::Expr::Stmt(expr) => {
176                let stmt = &*expr.stmt;
177                fmt!(cx, f, "(" stmt ")");
178            }
179            stmt::Expr::List(expr) => {
180                let items = Comma(expr.items.iter());
181                fmt!(cx, f, "(" items ")");
182            }
183            stmt::Expr::Value(expr) => expr.to_sql(cx, f),
184            stmt::Expr::Arg(arg) => {
185                // Pre-extracted bind parameter placeholder — render as a
186                // positional parameter. The arg position is 0-based; the
187                // placeholder is 1-based.
188                f.arg_positions.push(arg.position);
189                let placeholder = super::Placeholder(arg.position + 1);
190                fmt!(cx, f, placeholder);
191            }
192            stmt::Expr::Default => match f.serializer.flavor {
193                Flavor::Postgresql | Flavor::Mysql => fmt!(cx, f, "DEFAULT"),
194                // SQLite does not support the DEFAULT keyword but NULL acts similarly.
195                Flavor::Sqlite => fmt!(cx, f, "NULL"),
196            },
197            _ => todo!("expr={:#?}", self),
198        }
199    }
200}

202impl ToSql for &stmt::BinaryOp {
203    fn to_sql(self, _cx: &ExprContext<'_>, f: &mut super::Formatter<'_>) {
204        f.dst.push_str(match self {
205            stmt::BinaryOp::Eq => "=",
206            stmt::BinaryOp::Gt => ">",
207            stmt::BinaryOp::Ge => ">=",
208            stmt::BinaryOp::Lt => "<",
209            stmt::BinaryOp::Le => "<=",
210            stmt::BinaryOp::Ne => "<>",
211        })
212    }

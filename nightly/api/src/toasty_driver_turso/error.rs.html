1use toasty_core::Error;
2use turso::Error as TursoError;
3
4/// Classifies a [`turso::Error`] into a Toasty [`Error`].
5///
6/// * `Busy` and `BusySnapshot` — what the engine returns when a
7///   `BEGIN CONCURRENT` transaction conflicts on commit, or when a writer
8///   would have blocked. Both are retryable; map to
9///   [`Error::serialization_failure`].
10/// * `Error(msg)` containing the substring `"conflict"` — the current
11///   `turso` crate (0.6) sometimes surfaces MVCC commit conflicts on this
12///   generic variant rather than as `Busy*`. Its own
13///   `examples/concurrent_writes.rs` checks the message text the same way;
14///   treat it as retryable until upstream normalizes the variant.
15/// * `Readonly` — the database refused a write because the connection is
16///   in read-only mode. Map to [`Error::read_only_transaction`].
17/// * `IoError` — a low-level I/O fault on the storage layer. Map to
18///   [`Error::connection_lost`] so the pool evicts the slot.
19/// * Everything else carries an opaque message; map to
20///   [`Error::driver_operation_failed`].
21pub(crate) fn classify_turso_error(err: TursoError) -> Error {
22    match err {
23        TursoError::Busy(msg) | TursoError::BusySnapshot(msg) => Error::serialization_failure(msg),
24        TursoError::Error(msg) if msg.contains("conflict") => Error::serialization_failure(msg),
25        TursoError::Readonly(msg) => Error::read_only_transaction(msg),
26        TursoError::IoError(_, _) => Error::connection_lost(err),
27        _ => Error::driver_operation_failed(err),
28    }
29}

1//! Value-conversion between Toasty's `stmt::Value` and `turso::Value`.
2//!
3//! Scope is intentionally limited to SQLite's classic storage classes
4//! (integer, real, text, blob, null) plus the `Vec<scalar>` JSON
5//! round-trip the SQLite driver already does. Turso's richer typing —
6//! `experimental_custom_types`, decimal, native date/time — is still
7//! moving upstream, so we leave those cases as `todo!()` for now and
8//! revisit once the upstream surface settles. The current shape lets
9//! Turso ride on the SQLite-flavored serializer without an
10//! independent type system.
11
12use toasty_core::stmt::{self, Value as CoreValue};
13use turso::Value as TursoValue;
14
15/// Converts a [`toasty_core::stmt::Value`] to a [`turso::Value`].
16pub(crate) fn to_turso(value: &CoreValue) -> TursoValue {
17    match value {
18        CoreValue::Bool(true) => TursoValue::Integer(1),
19        CoreValue::Bool(false) => TursoValue::Integer(0),
20        CoreValue::I8(v) => TursoValue::Integer(*v as i64),
21        CoreValue::I16(v) => TursoValue::Integer(*v as i64),
22        CoreValue::I32(v) => TursoValue::Integer(*v as i64),
23        CoreValue::I64(v) => TursoValue::Integer(*v),
24        CoreValue::U8(v) => TursoValue::Integer(*v as i64),
25        CoreValue::U16(v) => TursoValue::Integer(*v as i64),
26        CoreValue::U32(v) => TursoValue::Integer(*v as i64),
27        CoreValue::U64(v) => TursoValue::Integer(*v as i64),
28        CoreValue::F32(v) => TursoValue::Real(*v as f64),
29        CoreValue::F64(v) => TursoValue::Real(*v),
30        CoreValue::String(v) => TursoValue::Text(v.clone()),
31        CoreValue::Bytes(v) => TursoValue::Blob(v.clone()),
32        CoreValue::Null => TursoValue::Null,
33        CoreValue::List(_) => TursoValue::Text(value_list_to_json_text(value)),
34        _ => todo!("to_turso: value = {value:#?}"),
35    }
36}
37
38/// Converts a [`turso::Value`] back to a [`toasty_core::stmt::Value`] using
39/// the expected return type to interpret the raw SQLite value correctly.
40pub(crate) fn from_turso(value: TursoValue, ty: &stmt::Type) -> CoreValue {
41    match value {
42        TursoValue::Null => CoreValue::Null,
43        TursoValue::Integer(v) => match ty {
44            stmt::Type::Bool => CoreValue::Bool(v != 0),
45            stmt::Type::I8 => CoreValue::I8(v as i8),
46            stmt::Type::I16 => CoreValue::I16(v as i16),
47            stmt::Type::I32 => CoreValue::I32(v as i32),
48            stmt::Type::I64 => CoreValue::I64(v),
49            stmt::Type::U8 => CoreValue::U8(v as u8),
50            stmt::Type::U16 => CoreValue::U16(v as u16),
51            stmt::Type::U32 => CoreValue::U32(v as u32),
52            stmt::Type::U64 => CoreValue::U64(v as u64),
53            _ => todo!("from_turso integer: ty={ty:#?}"),
54        },
55        TursoValue::Real(v) => match ty {
56            stmt::Type::F32 => CoreValue::F32(v as f32),
57            stmt::Type::F64 => CoreValue::F64(v),
58            _ => todo!("from_turso real: ty={ty:#?}"),
59        },
60        TursoValue::Text(v) => match ty {
61            stmt::Type::Uuid => CoreValue::Uuid(v.parse().expect("text is a valid uuid")),
62            stmt::Type::List(elem) => json_text_to_value_list(&v, elem),
63            _ => CoreValue::String(v),
64        },
65        TursoValue::Blob(v) => match ty {
66            stmt::Type::Bytes => CoreValue::Bytes(v),
67            _ => todo!("from_turso blob: value={v:#?}"),
68        },
69    }
70}

72fn value_list_to_json_text(value: &CoreValue) -> String {
73    let json = toasty_sql::value_json::value_list_to_json(value);
74    serde_json::to_string(&json).expect("serialize Vec<scalar> to JSON")
75}
76
77fn json_text_to_value_list(text: &str, elem_ty: &stmt::Type) -> CoreValue {
78    let json: serde_json::Value =
79        serde_json::from_str(text).expect("Turso returned non-JSON for a Vec<scalar> column");
80    toasty_sql::value_json::value_list_from_json(json, elem_ty)
81}

7pub(crate) fn expand(item: &CreateItem) -> TokenStream {
8    match item {
9        CreateItem::Typed { path, fields } => {
10            let span = path.span();
11            let fields_path = quote_spanned! { span=> #path::fields() };
12            let field_calls = expand_field_set(fields, &fields_path);
13
14            quote_spanned! { span=>
15                #path::create() #(#field_calls)*
16            }
17        }
18        CreateItem::Scoped { expr, fields } => expand_scoped(expr, fields),
19        CreateItem::TypedBatch { path, items } => {
20            let batch = expand_typed_batch(path, items);
21            quote! { toasty::batch(#batch) }
22        }
23        CreateItem::Tuple { items } => {
24            let elements: Vec<_> = items.iter().map(expand_as_element).collect();
25            quote! { toasty::batch(( #( #elements, )* )) }
26        }
27    }
28}

30/// Expand a scoped creation (`in expr { fields }`).
31fn expand_scoped(expr: &syn::Expr, fields: &FieldSet) -> TokenStream {
32    let span = expr.span();
33    let fields_path = quote! { __scope_fields };
34    let field_calls = expand_field_set(fields, &fields_path);
35
36    let scope_fields_call =
37        quote_spanned! { span=> toasty::codegen_support::scope_fields(&__scope) };
38    let create_call = quote_spanned! { span=> toasty::codegen_support::create_in_scope(__scope) };
39
40    quote! {
41        {
42            let __scope = #expr;
43            let __scope_fields = #scope_fields_call;
44            #create_call #(#field_calls)*
45        }
46    }
47}

49/// Expand a `CreateItem` as an element inside a tuple batch.
50///
51/// For `TypedBatch`, this produces a plain array `[b1, b2, ...]` (which
52/// implements `IntoStatement` with `Returning = List<T>`) rather than
53/// wrapping in `toasty::batch()`.
54fn expand_as_element(item: &CreateItem) -> TokenStream {
55    match item {
56        CreateItem::TypedBatch { path, items } => expand_typed_batch(path, items),
57        // All other forms expand identically to top-level.
58        other => expand(other),
59    }
60}

62/// Expand a `TypedBatch` into `[ builder1, builder2, ... ]`.
63fn expand_typed_batch(path: &syn::Path, items: &[FieldSet]) -> TokenStream {
64    let span = path.span();
65    let fields_path = quote_spanned! { span=> #path::fields() };
66
67    let builders: Vec<_> = items
68        .iter()
69        .map(|fields| {
70            let field_calls = expand_field_set(fields, &fields_path);
71
72            quote_spanned! { span=>
73                #path::create() #(#field_calls)*
74            }
75        })
76        .collect();
77    quote! { [ #( #builders, )* ] }
78}

80/// Expand a `FieldSet` into method calls.
81fn expand_field_set(fields: &FieldSet, path: &TokenStream) -> Vec<TokenStream> {
82    fields.0.iter().map(|f| expand_field(f, path)).collect()
83}

85/// Expand a single field entry into a method call token stream.
86fn expand_field(field: &FieldEntry, path: &TokenStream) -> TokenStream {
87    let name = &field.name;
88    let span = name.span();
89
90    match &field.value {
91        FieldValue::Expr(expr) => {
92            quote_spanned! { span=> .#name(#expr) }
93        }
94        FieldValue::Single(sub_fields) => {
95            let nested_path = quote! { #path.#name() };
96            let sub_calls = expand_field_set(sub_fields, &nested_path);
97            quote_spanned! { span=> .#name(#path.#name().create() #(#sub_calls)*) }
98        }
99        FieldValue::List(items) => {
100            let nested_path = quote! { #path.#name() };
101            let item_builders: Vec<_> = items
102                .iter()
103                .map(|item| expand_nested_item(item, path, name, span, &nested_path))
104                .collect();
105            quote_spanned! { span=> .#name([#(#item_builders),*]) }
106        }
107    }
108}

110/// Expand a single item within a field-level list using path-based builders.
111///
112/// `parent_path` is the current fields path (e.g., `User::fields()`).
113/// `field_name` is the field identifier (e.g., `todos`).
114/// `span` is the span of the field name for error reporting.
115/// `nested_path` is `parent_path.field_name()` for deeper nesting.
116fn expand_nested_item(
117    item: &FieldValue,
118    parent_path: &TokenStream,
119    field_name: &syn::Ident,
120    span: Span,
121    nested_path: &TokenStream,
122) -> TokenStream {
123    match item {
124        FieldValue::Single(fields) => {
125            let sub_calls = expand_field_set(fields, nested_path);
126            quote_spanned! { span=> #parent_path.#field_name().create() #(#sub_calls)* }
127        }
128        FieldValue::Expr(e) => {
129            quote! { #e }
130        }
131        FieldValue::List(_) => {
132            quote! { compile_error!("nested lists are not supported in create!") }
133        }
134    }
135}

31    /// Generates tokens for an `is_variant(path, variant_id)` expression.
32    /// Reused by `is_{variant}()` methods and the `matches()` method.
33    fn expand_is_variant_expr(&self, variant_idx: &TokenStream) -> TokenStream {
34        let toasty = &self.toasty;
35        let model_ident = &self.model.ident;
36        quote! {
37            {
38                let path_stmt: #toasty::core::stmt::Expr = {
39                    let p: #toasty::core::stmt::Path = self.path().into();
40                    p.into_stmt()
41                };
42                let variant_id = #toasty::core::schema::app::VariantId {
43                    model: <#model_ident as #toasty::Embed>::id(),
44                    index: #variant_idx,
45                };
46                #toasty::stmt::Expr::from_untyped(
47                    #toasty::core::stmt::Expr::is_variant(path_stmt, variant_id)
48                )
49            }
50        }
51    }

53    /// Generates delegated comparison methods (`eq`, `ne`, `in_list`) that
54    /// forward to `self.path()`. Ordered comparisons (`gt`, `ge`, `lt`, `le`)
55    /// are intentionally excluded because enums have no meaningful ordering.
56    fn expand_comparison_methods(&self) -> TokenStream {
57        let toasty = &self.toasty;
58        let vis = &self.model.vis;
59        let model_ident = &self.model.ident;
60
61        let methods = ["eq", "ne"].iter().map(|name| {
62            let method_ident = syn::Ident::new(name, proc_macro2::Span::call_site());
63            quote! {
64                #vis fn #method_ident(&self, rhs: impl #toasty::stmt::IntoExpr<#model_ident>) -> #toasty::stmt::Expr<bool> {
65                    self.path().#method_ident(rhs)
66                }
67            }
68        });
69
70        quote! {
71            #( #methods )*
72
73            #vis fn in_list(&self, rhs: impl #toasty::stmt::IntoExpr<#toasty::List<#model_ident>>) -> #toasty::stmt::Expr<bool> {
74                self.path().in_list(rhs)
75            }
76        }
77    }

79    /// Generates the `{Enum}Fields` struct for embedded enums with
80    /// `is_{variant}()` methods, variant accessor methods, and delegated
81    /// comparison methods. Also generates per-variant field structs for
82    /// data-carrying variants (e.g., `ContactInfoEmailFields`).
83    pub(super) fn expand_enum_field_struct(&self) -> TokenStream {
84        let toasty = &self.toasty;
85        let vis = &self.model.vis;
86        let model_ident = &self.model.ident;
87        let embedded_enum = self.model.kind.as_embedded_enum_unwrap();
88        let field_struct_ident = &embedded_enum.field_struct_ident;
89
90        let is_variant_methods: Vec<_> = embedded_enum
91            .variants
92            .iter()
93            .enumerate()
94            .map(|(variant_index, variant)| {
95                let method_name = &variant.is_method_ident;
96                let variant_idx = util::int(variant_index);
97                let is_variant_check = self.expand_is_variant_expr(&variant_idx);
98
99                quote! {
100                    #vis fn #method_name(&self) -> #toasty::stmt::Expr<bool> {
101                        #is_variant_check
102                    }
103                }
104            })
105            .collect();
106
107        // Generate variant accessor methods for data-carrying variants.
108        let variant_accessor_methods: Vec<_> = embedded_enum
109            .variants
110            .iter()
111            .filter(|v| v.variant_handle_ident.is_some())
112            .map(|variant| {
113                let method_name = &variant.name.ident;
114                let variant_handle_ident = variant.variant_handle_ident.as_ref().unwrap();
115
116                quote! {
117                    #vis fn #method_name(&self) -> #variant_handle_ident<__Origin> {
118                        #variant_handle_ident {
119                            path: self.path()
120                        }
121                    }
122                }
123            })
124            .collect();
125
126        // Generate one struct per data-carrying variant. The same struct is
127        // both the entry point returned by `email()` (used for `.matches(...)`
128        // filters and direct `.eq()`/include path access) and the namespace
129        // for variant-field accessors. The struct stores the parent path
130        // (the path to the embed enum field) and exposes:
131        //
132        // - `parent_path()` — model-rooted path to the enum field. Used by
133        //   `matches()` to build the `is_variant` gate and by every filter
134        //   method on a variant-rooted Path to inject the implicit gate.
135        // - `path()` — variant-rooted path. Field accessors chain off this
136        //   so that the produced Path knows its variant context.
137        let variant_field_structs: Vec<_> = embedded_enum
138            .variants
139            .iter()
140            .enumerate()
141            .filter(|(_, v)| v.variant_handle_ident.is_some())
142            .map(|(variant_index, variant)| {
143                let variant_handle_ident = variant.variant_handle_ident.as_ref().unwrap();
144                let variant_idx = util::int(variant_index);
145
146                let field_methods: Vec<_> = self
147                    .variant_fields(variant_index)
148                    .iter()
149                    .enumerate()
150                    .map(|(field_index, field)| {
151                        let field_ident = &field.name.ident;
152                        let field_ty = primitive_ty_unwrap(field);
153                        let field_offset = util::int(field_index);
154                        self.expand_primitive_field_method(field_ident, field_ty, &field_offset)
155                    })
156                    .collect();
157
158                quote! {
159                    #vis struct #variant_handle_ident<__Origin> {
160                        path: #toasty::Path<__Origin, #model_ident>,
161                    }
162
163                    impl<__Origin> #variant_handle_ident<__Origin> {
164                        fn parent_path(&self) -> #toasty::Path<__Origin, #model_ident> {
165                            self.path.clone()
166                        }
167
168                        fn path(&self) -> #toasty::Path<__Origin, #model_ident> {
169                            let variant_id = #toasty::core::schema::app::VariantId {
170                                model: <#model_ident as #toasty::Embed>::id(),
171                                index: #variant_idx,
172                            };
173                            self.parent_path().into_variant(variant_id)
174                        }
175
176                        #vis fn matches(
177                            self,
178                            f: impl FnOnce(Self) -> #toasty::stmt::Expr<bool>,
179                        ) -> #toasty::stmt::Expr<bool> {
180                            let parent_stmt: #toasty::core::stmt::Expr = {
181                                let p: #toasty::core::stmt::Path = self.parent_path().into();
182                                p.into_stmt()
183                            };
184                            let variant_id = #toasty::core::schema::app::VariantId {
185                                model: <#model_ident as #toasty::Embed>::id(),
186                                index: #variant_idx,
187                            };
188                            let is_var = #toasty::stmt::Expr::from_untyped(
189                                #toasty::core::stmt::Expr::is_variant(parent_stmt, variant_id)
190                            );
191                            let body = f(self);
192                            is_var.and(body)
193                        }
194
195                        #( #field_methods )*
196                    }
197                }
198            })
199            .collect();
200
201        let comparison_methods = self.expand_comparison_methods();
202
203        quote! {
204            #vis struct #field_struct_ident<__Origin> {
205                path: #toasty::Path<__Origin, #model_ident>,
206            }
207
208            impl<__Origin> #field_struct_ident<__Origin> {
209                fn path(&self) -> #toasty::Path<__Origin, #model_ident> {
210                    self.path.clone()
211                }
212
213                #( #is_variant_methods )*
214
215                #( #variant_accessor_methods )*
216
217                #comparison_methods
218            }
219
220            impl<__Origin> Into<#toasty::Path<__Origin, #model_ident>> for #field_struct_ident<__Origin> {
221                fn into(self) -> #toasty::Path<__Origin, #model_ident> {
222                    self.path
223                }
224            }
225
226            impl<__Origin> #toasty::IntoExpr<#model_ident> for #field_struct_ident<__Origin> {
227                fn into_expr(self) -> #toasty::stmt::Expr<#model_ident> {
228                    use #toasty::IntoExpr;
229                    self.path.into_expr()
230                }
231
232                fn by_ref(&self) -> #toasty::stmt::Expr<#model_ident> {
233                    use #toasty::IntoExpr;
234                    self.path.by_ref()
235                }
236            }
237
238            #( #variant_field_structs )*
239        }
240    }

291    /// Generates the flat list of `Field` schema tokens for all variant fields,
292    /// with each field tagged with its `VariantId`.
293    pub(super) fn expand_enum_schema_fields(&self) -> Vec<TokenStream> {
294        let toasty = &self.toasty;
295
296        self.model
297            .fields
298            .iter()
299            .map(|field| {
300                let index = util::int(field.id);
301                let app_name = field.name.as_str();
302                // A `#[column("name")]` on a variant field overrides its
303                // database column name. When two variants give a field the same
304                // column name, the schema builder coalesces them into one shared
305                // column (see `BuildMapping::map_field_primitive`).
306                let storage_name = match field
307                    .attrs
308                    .column
309                    .as_ref()
310                    .and_then(|column| column.name.as_ref())
311                {
312                    Some(name) => quote! { Some(#name.to_string()) },
313                    None => quote! { None },
314                };
315                let ty = primitive_ty_unwrap(field);
316                let variant_index = field.variant.expect("enum field must have variant");
317                let variant_idx = util::int(variant_index);
318                quote! {
319                    #toasty::core::schema::app::Field {
320                        id: #toasty::core::schema::app::FieldId {
321                            model: id,
322                            index: #index,
323                        },
324                        name: #toasty::core::schema::app::FieldName {
325                            app: Some(#app_name.to_string()),
326                            storage: #storage_name,
327                        },
328                        ty: <#ty as #toasty::Field>::field_ty(None),
329                        nullable: <#ty as #toasty::Field>::NULLABLE,
330                        primary_key: false,
331                        auto: None,
332                        versionable: false,
333                        deferred: <#ty as #toasty::Field>::DEFERRED,
334                        constraints: vec![],
335                        variant: Some(#toasty::core::schema::app::VariantId {
336                            model: id,
337                            index: #variant_idx,
338                        }),
339                    }
340                }
341            })
342            .collect()
343    }

345    /// Emits compile-time checks for variant fields that share a column via
346    /// `#[column("name")]`.
347    ///
348    /// Two variants that give a field the same `#[column("name")]` coalesce into
349    /// one shared column, which can hold only one storage type. This emits two
350    /// obligations rather than deferring to a schema-build error:
351    ///
352    /// - **Same-variant collision:** two fields in the *same* variant mapped to
353    ///   one column is invalid — both are active for one discriminant, so the
354    ///   column can't hold both. Emit a `compile_error!` pinned to the second.
355    /// - **Type agreement:** across variants, every shared field must have the
356    ///   same type. Emit a `SameColumnType` bound between each field's type and
357    ///   the first in its group; unequal types fail to compile.
358    ///
359    /// Only fields with an explicit `#[column("name")]` are grouped. A field
360    /// without one is not necessarily single-column (an embedded struct flattens
361    /// to several columns named after its own fields), so predicting a collision
362    /// from the field name alone would reject legitimate schemas. Those cases
363    /// fall back to the schema-build check.
364    pub(super) fn expand_shared_column_checks(&self) -> TokenStream {
365        let toasty = &self.toasty;
366
367        // Group variant fields by their explicit column name, preserving
368        // declaration order within each group.
369        let mut groups: Vec<(String, Vec<&crate::model::schema::Field>)> = Vec::new();
370        for field in &self.model.fields {
371            let Some(name) = field
372                .attrs
373                .column
374                .as_ref()
375                .and_then(|column| column.name.as_ref())
376                .map(|name| name.value())
377            else {
378                continue;
379            };
380
381            match groups.iter_mut().find(|(existing, _)| *existing == name) {
382                Some((_, fields)) => fields.push(field),
383                None => groups.push((name, vec![field])),
384            }
385        }
386
387        let mut checks = Vec::new();
388        for (name, fields) in groups.iter().filter(|(_, fields)| fields.len() > 1) {
389            // Reject a second field in the same variant landing on this column.
390            let mut seen_variants = Vec::new();
391            let mut same_variant = false;
392            for field in fields {
393                let variant = field
394                    .variant
395                    .expect("enum variant field must have a variant");
396                if seen_variants.contains(&variant) {
397                    let ident = &field.name.ident;
398                    checks.push(
399                        syn::Error::new_spanned(
400                            ident,
401                            format!(
402                                "two fields in the same variant map to column `{name}`; \
403                                 a column can be shared only across different variants"
404                            ),
405                        )
406                        .to_compile_error(),
407                    );
408                    same_variant = true;
409                } else {
410                    seen_variants.push(variant);
411                }
412            }
413
414            // A malformed group can't be meaningfully type-checked; the
415            // collision error above is the actionable one.
416            if same_variant {
417                continue;
418            }
419
420            let first = primitive_ty_unwrap(fields[0]);
421            for field in &fields[1..] {
422                let ty = primitive_ty_unwrap(field);
423                checks.push(quote_spanned! { ty.span()=>
424                    const _: () = {
425                        fn check<A, B>()
426                        where
427                            A: #toasty::shared_column::SameColumnType<B>,
428                        {
429                        }
430                        let _ = check::<#ty, #first>;
431                    };
432                });
433            }
434        }
435
436        quote! { #( #checks )* }
437    }

439    /// Generates the full `impl Load for Enum { ... }` block, adapting
440    /// discriminant matching based on whether integer or string discriminants are used.
441    pub(super) fn expand_enum_load_impl(&self) -> TokenStream {
442        let toasty = &self.toasty;
443        let model_ident = &self.model.ident;
444        let ty_expr = self.expand_enum_primitive_ty();
445
446        let unit_load_arms = self.expand_enum_load_arms(true);
447        let data_load_arms = self.expand_enum_load_arms(false);
448
449        // Generate discriminant-specific match tokens. The outer match uses
450        // `ref` so `value` stays available for the error path. The inner
451        // (record) match owns the taken element.
452        let (ref_pattern, owned_pattern, match_expr) = if self.uses_string_discriminants() {
453            (
454                quote! { #toasty::core::stmt::Value::String(ref d) },
455                quote! { #toasty::core::stmt::Value::String(d) },
456                quote! { d.as_str() },
457            )
458        } else {
459            (
460                quote! { #toasty::core::stmt::Value::I64(ref d) },
461                quote! { #toasty::core::stmt::Value::I64(d) },
462                quote! { d },
463            )
464        };
465
466        quote! {
467            impl #toasty::Load for #model_ident {
468                type Output = Self;
469
470                fn ty() -> #toasty::core::stmt::Type {
471                    #ty_expr
472                }
473
474                fn load(value: #toasty::core::stmt::Value) -> #toasty::Result<Self> {
475                    match value {
476                        #ref_pattern => match #match_expr {
477                            #( #unit_load_arms )*
478                            _ => Err(#toasty::Error::type_conversion(
479                                value,
480                                stringify!(#model_ident),
481                            )),
482                        },
483                        #toasty::core::stmt::Value::Record(mut record) => match record[0].take() {
484                            #owned_pattern => match #match_expr {
485                                #( #data_load_arms )*
486                                _ => Err(#toasty::Error::type_conversion(
487                                    #owned_pattern,
488                                    stringify!(#model_ident),
489                                )),
490                            },
491                            other => Err(#toasty::Error::type_conversion(
492                                other,
493                                stringify!(#model_ident),
494                            )),
495                        },
496                        value => Err(#toasty::Error::type_conversion(value, stringify!(#model_ident))),
497                    }
498                }
499
500                fn reload(target: &mut Self, value: #toasty::core::stmt::Value) -> #toasty::Result<()> {
501                    *target = Self::load(value)?;
502                    Ok(())
503                }
504            }
505        }
506    }

508    /// Generates match arms for Load. When `unit_only` is true, emits arms for
509    /// unit variants; otherwise emits arms for data-carrying variants.
510    fn expand_enum_load_arms(&self, unit_only: bool) -> Vec<TokenStream> {
511        let toasty = &self.toasty;
512        let model_ident = &self.model.ident;
513        let embedded_enum = self.model.kind.as_embedded_enum_unwrap();
514
515        embedded_enum
516            .variants
517            .iter()
518            .enumerate()
519            .filter(|(variant_index, _)| {
520                self.variant_fields(*variant_index).is_empty() == unit_only
521            })
522            .map(|(variant_index, variant)| {
523                let ident = &variant.ident;
524                let pattern = expand_discriminant_match_pattern(&variant.attrs.discriminant);
525
526                let construction = if unit_only {
527                    quote! { #model_ident::#ident }
528                } else {
529                    let fields = self.variant_fields(variant_index);
530                    let field_loads: Vec<_> = fields
531                        .iter()
532                        .enumerate()
533                        .map(|(i, field)| {
534                            let field_ident = &field.name.ident;
535                            let ty = primitive_ty_unwrap(field);
536                            let record_pos = util::int(i + 1);
537                            let load = quote! {
538                                <#ty as #toasty::Load>::load(record[#record_pos].take())?
539                            };
540                            if variant.fields_named {
541                                quote! { #field_ident: #load, }
542                            } else {
543                                quote! { #load, }
544                            }
545                        })
546                        .collect();
547
548                    if variant.fields_named {
549                        quote! { #model_ident::#ident { #( #field_loads )* } }
550                    } else {
551                        quote! { #model_ident::#ident( #( #field_loads )* ) }
552                    }
553                };
554
555                quote! { #pattern => Ok(#construction), }
556            })
557            .collect()
558    }

560    /// Generates match arms for `IntoExpr::into_expr` and `IntoExpr::by_ref`.
561    /// Unit variants emit `Value::I64(discriminant)`. Data variants emit
562    /// `Value::Record([I64(disc), field_exprs...])`. The same arms work for both
563    /// methods: for `by_ref(&self)` match ergonomics bind field names as `&T`,
564    /// and `IntoExpr` is implemented for both `T` and `&T`.
565    pub(super) fn expand_enum_into_expr_arms(&self) -> Vec<TokenStream> {
566        let toasty = &self.toasty;
567        let model_ident = &self.model.ident;
568        let embedded_enum = self.model.kind.as_embedded_enum_unwrap();
569
570        embedded_enum
571            .variants
572            .iter()
573            .enumerate()
574            .map(|(variant_index, variant)| {
575                let ident = &variant.ident;
576                let fields = self.variant_fields(variant_index);
577
578                let discriminant_expr =
579                    self.expand_discriminant_value_expr(&variant.attrs.discriminant);
580
581                if fields.is_empty() {
582                    // In a mixed enum (has_data_variants), the model value is always a
583                    // Record so that `project([0])` uniformly extracts the discriminant
584                    // for the `model_to_table` mapping.  Unit-only enums keep the plain
585                    // I64 form (cheaper and the load path expects it).
586                    if self.expand_enum_has_data_variants() {
587                        quote! {
588                            #model_ident::#ident => #toasty::stmt::Expr::from_untyped(
589                                #toasty::core::stmt::Expr::record([#discriminant_expr])
590                            ),
591                        }
592                    } else {
593                        quote! {
594                            #model_ident::#ident => #toasty::stmt::Expr::from_untyped(
595                                #discriminant_expr
596                            ),
597                        }
598                    }
599                } else {
600                    let field_idents: Vec<_> = fields.iter().map(|f| &f.name.ident).collect();
601
602                    let pattern = if variant.fields_named {
603                        quote! { #model_ident::#ident { #( #field_idents ),* } }
604                    } else {
605                        quote! { #model_ident::#ident( #( #field_idents ),* ) }
606                    };
607
608                    let field_exprs = fields.iter().map(|field| {
609                        let field_ident = &field.name.ident;
610                        let ty = primitive_ty_unwrap(field);
611                        quote!(#toasty::into_untyped_expr::<FieldExprTarget<#ty>, _>(#field_ident))
612                    });
613
614                    quote! {
615                        #pattern =>
616                            #toasty::stmt::Expr::from_untyped(
617                                #toasty::core::stmt::Expr::record([
618                                    #discriminant_expr,
619                                    #( #field_exprs ),*
620                                ])
621                            ),
622                    }
623                }
624            })
625            .collect()
626    }

628    /// Generates the `Field::ty()` return expression. Unit-only enums map to
629    /// `Type::I64` or `Type::String`; enums with at least one data variant map to `Type::Model`.
630    pub(super) fn expand_enum_primitive_ty(&self) -> TokenStream {
631        let toasty = &self.toasty;
632        if self.expand_enum_has_data_variants() {
633            quote! { #toasty::core::stmt::Type::Model(<Self as #toasty::Embed>::id()) }
634        } else if self.uses_string_discriminants() {
635            quote! { #toasty::core::stmt::Type::String }
636        } else {
637            quote! { #toasty::core::stmt::Type::I64 }
638        }
639    }

641    /// Generates the `storage_ty` token for the discriminant `FieldPrimitive`.
642    ///
643    /// - Native enum: `Some(db::Type::Enum(TypeEnum { ... }))`
644    /// - Plain string (`#[column(type = text)]`): `Some(db::Type::Text)`
645    /// - Integer discriminants: `None`
646    pub(super) fn expand_enum_storage_ty(&self) -> TokenStream {
647        let toasty = &self.toasty;
648        let embedded_enum = self.model.kind.as_embedded_enum_unwrap();
649
650        match &embedded_enum.storage_strategy {
651            Some(EnumStorageStrategy::NativeEnum(custom_name)) => {
652                // Determine the type name: custom name or default snake_case of enum ident.
653                let type_name = match custom_name {
654                    Some(name) => name.clone(),
655                    None => {
656                        use heck::ToSnakeCase;
657                        self.model.ident.to_string().to_snake_case()
658                    }
659                };
660
661                // Collect variant names in declaration order.
662                let variant_names: Vec<&str> = embedded_enum
663                    .variants
664                    .iter()
665                    .map(|v| match &v.attrs.discriminant {
666                        VariantValue::String(s) => s.as_str(),
667                        _ => unreachable!("native enum requires string discriminants"),
668                    })
669                    .collect();
670
671                quote! {
672                    ::std::option::Option::Some(
673                        #toasty::core::schema::db::Type::Enum(
674                            #toasty::core::schema::db::TypeEnum {
675                                name: ::std::option::Option::Some(#type_name.to_string()),
676                                variants: vec![
677                                    #( #toasty::core::schema::db::EnumVariant {
678                                        name: #variant_names.to_string(),
679                                    } ),*
680                                ],
681                            }
682                        )
683                    )
684                }
685            }
686            Some(EnumStorageStrategy::PlainString(col_ty)) => {
687                let ty_tokens = col_ty.expand_with(toasty);
688                quote! { ::std::option::Option::Some(#ty_tokens) }
689            }
690            None => {
691                // Integer discriminants: no storage type hint.
692                quote! { ::std::option::Option::None }
693            }
694        }
695    }
696}
697
698fn primitive_ty_unwrap(field: &crate::model::schema::Field) -> &syn::Type {
699    match &field.ty {
700        FieldTy::Primitive(ty) => ty,
701        _ => panic!("expected primitive field type for enum variant field"),
702    }
703}
704
705/// Generates the match pattern token for a discriminant value:
706/// a string literal for `VariantValue::String`, an integer literal for `VariantValue::Integer`.
707fn expand_discriminant_match_pattern(value: &VariantValue) -> TokenStream {
708    match value {
709        VariantValue::String(s) => quote! { #s },
710        VariantValue::Integer(n) => quote! { #n },
711    }

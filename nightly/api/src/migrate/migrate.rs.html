1//! service-ops (migrate binary): a project-local migration CLI built from `toasty-cli`. The
2//! tool is YOUR binary — not a global command — because diffing the schema needs your compiled
3//! model types. Run it from this crate's directory (so it finds `Toasty.toml`):
4//!
5//!   cargo run -p example-service-ops --bin migrate -- migration generate --name initial
6//!   cargo run -p example-service-ops --bin migrate -- migration apply
7//!
8//! `generate` diffs the models against the stored snapshot and writes incremental SQL under
9//! `toasty/`; `apply` runs the pending files and records them in a `__toasty_migrations` table.
10
11use toasty_cli::{Config, ToastyCli};
12
13#[tokio::main]
14async fn main() -> anyhow::Result<()> {
15    // Reads Toasty.toml ([migration] path = "toasty").
16    let config = Config::load()?;
17
18    // Same models and connection as the server, via the shared library.
19    let db = example_service_ops::build_db().await?;
20
21    // ToastyCli exposes the `migration generate/apply/...` subcommands over your models.
22    ToastyCli::with_config(db, config).parse_and_run().await?;
23
24    Ok(())
25}

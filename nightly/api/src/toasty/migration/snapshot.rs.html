1use crate::{Error, Result, schema::db::Schema};
2use serde::{Deserialize, Serialize};
3use std::{fmt, path::Path, str::FromStr};
4use toml_edit::{DocumentMut, Item};
5
6const SNAPSHOT_VERSION: u32 = 1;
7
8/// A TOML-serializable snapshot of a database schema.
9///
10/// Snapshots capture the schema after a generated migration. The next
11/// generation compares the most recent snapshot against the current schema to
12/// build a diff.
13#[derive(Debug, Clone, Serialize, Deserialize)]
14pub struct Snapshot {
15    /// Snapshot format version.
16    version: u32,
17
18    /// The database schema captured by this snapshot.
19    pub schema: Schema,
20}
21
22impl Snapshot {
23    /// Create a new snapshot for `schema`.
24    pub fn new(schema: Schema) -> Self {
25        Self {
26            version: SNAPSHOT_VERSION,
27            schema,
28        }
29    }
30
31    /// Load a snapshot from a TOML file.
32    pub fn load(path: impl AsRef<Path>) -> Result<Self> {
33        let contents = std::fs::read_to_string(path.as_ref())?;
34        contents.parse()
35    }
36
37    /// Save the snapshot to a TOML file.
38    pub fn save(&self, path: impl AsRef<Path>) -> Result<()> {
39        std::fs::write(path.as_ref(), self.to_string())?;
40        Ok(())
41    }
42
43    fn to_toml_document(&self) -> Result<DocumentMut> {
44        let mut doc = toml_edit::ser::to_document(self)
45            .map_err(|err| Error::from_args(format_args!("{err}")))?;
46
47        for (_key, item) in doc.as_table_mut().iter_mut() {
48            if item.is_inline_table() {
49                let mut placeholder = Item::None;
50                std::mem::swap(item, &mut placeholder);
51                let mut table = placeholder.into_table().unwrap();
52
53                for (_key, item) in table.iter_mut() {
54                    if item.is_array() {
55                        let mut placeholder = Item::None;
56                        std::mem::swap(item, &mut placeholder);
57                        let mut array = placeholder.into_array_of_tables().unwrap();
58
59                        for table in array.iter_mut() {
60                            for (_key, item) in table.iter_mut() {
61                                if item.is_array() {
62                                    let mut placeholder = Item::None;
63                                    std::mem::swap(item, &mut placeholder);
64                                    let array = placeholder.into_array_of_tables().unwrap();
65                                    *item = array.into();
66                                }
67                            }
68                        }
69
70                        *item = array.into();
71                    }
72                }
73
74                *item = table.into();
75            }
76        }
77
78        Ok(doc)
79    }
80}
81
82impl FromStr for Snapshot {
83    type Err = Error;
84
85    fn from_str(s: &str) -> Result<Self> {
86        let snapshot: Snapshot =
87            toml::from_str(s).map_err(|err| Error::from_args(format_args!("{err}")))?;
88
89        if snapshot.version != SNAPSHOT_VERSION {
90            return Err(Error::from_args(format_args!(
91                "unsupported snapshot version: {}. Expected version {}",
92                snapshot.version, SNAPSHOT_VERSION
93            )));
94        }
95
96        Ok(snapshot)
97    }
98}
99
100impl fmt::Display for Snapshot {
101    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
102        let doc = self.to_toml_document().map_err(|_| fmt::Error)?;
103        write!(f, "{doc}")
104    }

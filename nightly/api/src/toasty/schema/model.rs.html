6/// Trait for root models that map to database tables and can be queried.
7///
8/// Root models have primary keys, can be queried independently, and support
9/// full CRUD operations. They carry queryability and deserialization
10/// capabilities along with the relation-target metadata
11/// that the [`RelationManyField`](super::RelationManyField) and
12/// [`RelationOneField`](super::RelationOneField) traits project through when
13/// describing a field that references this model.
14pub trait Model: Load<Output = Self> + Sized {
15    /// Query builder type for this model
16    type Query;
17
18    /// Create builder type for this model
19    type Create: Default + IntoInsert<Model = Self> + IntoExpr<Self>;
20
21    /// Update builder type for this model
22    type Update<'a>;
23
24    /// Update by query builder type for this model
25    type UpdateQuery;
26
27    /// A typed path from `Origin` into this model.
28    type Path<Origin>;
29
30    /// The model's primary key type.
31    ///
32    /// For a single-column key, this is the column's Rust type (e.g.
33    /// `Uuid`, `i64`). For a composite key, this is a tuple of the
34    /// column types in declaration order.
35    ///
36    /// Generic code can bind on this to write functions that accept any
37    /// model identified by a particular PK type — for example, a request
38    /// extractor for any `M: Model<PrimaryKey = Uuid>`.
39    type PrimaryKey;
40
41    /// The has-many relation wrapper type for this model.
42    type Many;
43
44    /// The has-many relation wrapper type for multi-step scopes.
45    type ViaMany;
46
47    /// The field accessor type used when this model appears as the "many" side
48    /// of a has-many relation, parameterized by the origin model.
49    type ManyField<Origin>;
50
51    /// The has-one relation wrapper type for this model (non-nullable).
52    type One;
53
54    /// The has-one relation wrapper type for multi-step scopes (non-nullable).
55    type ViaOne;
56
57    /// The field accessor type used when this model appears as the "one" side
58    /// of a has-one relation, parameterized by the origin model.
59    type OneField<Origin>;
60
61    /// The optional has-one relation wrapper type, used when the foreign key
62    /// is nullable so the association is optional.
63    type OptionOne;
64
65    /// The optional has-one relation wrapper type for multi-step scopes.
66    type ViaOptionOne;
67
68    /// Unique identifier for this model within the schema.
69    ///
70    /// Identifiers are *not* unique across schemas.
71    fn id() -> ModelId;
72
73    /// Returns the schema definition for this model.
74    fn schema() -> app::Model;
75
76    /// Register this model and all models reachable through its fields into
77    /// the given [`ModelSet`].
78    ///
79    /// If this model is already present in the set (checked via
80    /// [`ModelSet::contains`]), the method returns immediately. Otherwise it
81    /// inserts the model and recursively registers any models referenced by
82    /// embedded or relation fields. This is the entry point used by
83    /// [`models!`](crate::models) to register root models; embedded types are
84    /// discovered transitively through their containing fields.
85    fn register(model_set: &mut ModelSet);
86
87    /// Construct a model path from a [`Path`] targeting this model.
88    fn new_path<Origin>(path: Path<Origin, Self>) -> Self::Path<Origin>;
89
90    /// Construct a path rooted at this model.
91    fn new_root_path() -> Self::Path<Self> {
92        Self::new_path(Self::path_root())
93    }
94
95    /// An identity [`Path`] rooted at this model.
96    ///
97    /// This is how `Path` recovers a model's [`ModelId`] without a dedicated
98    /// registration trait: the model supplies its own id via [`id`](Self::id).
99    fn path_root() -> Path<Self, Self> {
100        Path::from_model_id(Self::id())
101    }
102
103    /// A [`Path`] from this model to the field at `index`.
104    fn path_field<U>(index: usize) -> Path<Self, U> {
105        Path::field_at(Self::id(), index)
106    }
107
108    /// An identity [`Path`] rooted at a list of this model, used as the root of
109    /// has-many relation scopes.
110    fn path_model_list() -> Path<List<Self>, List<Self>> {
111        Path::from_model_id(Self::id())
112    }
113
114    /// Return a fresh, default-initialized create builder.
115    fn new_create() -> Self::Create {
116        Self::Create::default()
117    }
118
119    /// Construct a [`ManyField`](Self::ManyField) from a path targeting a list
120    /// of this model.
121    fn new_many_field<Origin>(path: Path<Origin, List<Self>>) -> Self::ManyField<Origin>;
122
123    /// Map a field name string to its [`FieldId`].
124    ///
125    /// Panics if `name` does not match any field on the model.
126    fn field_name_to_id(name: &str) -> FieldId;
127
128    /// Build a query that filters this model by its primary key.
129    ///
130    /// `id` takes any expression that evaluates to the model's
131    /// [`PrimaryKey`](Self::PrimaryKey) type — bare PK values via their
132    /// `IntoExpr` impl, subqueries that return a PK, or any other
133    /// [`Expr`] of the correct type.
134    fn find_by_primary_key(id: Expr<Self::PrimaryKey>) -> Self::Query;
135}

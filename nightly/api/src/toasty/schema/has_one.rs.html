8/// A lazily-loaded has-one association.
9///
10/// `HasOne<T>` wraps an optional `T` that is populated when the association
11/// is eagerly loaded (via `include`) or accessed through a generated relation
12/// accessor. Before loading, calling [`get`](HasOne::get) panics.
13///
14/// This type appears as a field on model structs for has-one relations.
15#[derive(Clone)]
16pub struct HasOne<T> {
17    value: Option<Box<T>>,
18}
19
20impl<T: Relation> Load for HasOne<T> {
21    type Output = Self;
22
23    fn ty() -> stmt::Type {
24        T::ty_relation()
25    }
26
27    fn load(input: Value) -> crate::Result<Self> {
28        Ok(match input {
29            Value::Null => Self::default(),
30            value => Self {
31                value: Some(Box::new(T::load_relation(value)?)),
32            },
33        })
34    }
35}
36
37impl<T: Relation> HasOne<T> {
38    /// Returns a reference to the loaded associated record.
39    ///
40    /// # Panics
41    ///
42    /// Panics if the association has not been loaded.
43    #[track_caller]
44    pub fn get(&self) -> &T {
45        self.value.as_ref().expect("association not loaded")
46    }
47
48    /// Returns `true` if the association has not been loaded yet.
49    pub fn is_unloaded(&self) -> bool {
50        self.value.is_none()
51    }
52
53    /// Clear the loaded value, returning this association to the unloaded
54    /// state.
55    pub fn unload(&mut self) {
56        self.value = None;
57    }
58}

77    fn field_name_to_id(name: &str) -> toasty_core::schema::app::FieldId {
78        T::field_name_to_id(name)
79    }
80
81    fn nullable() -> bool {
82        T::nullable()
83    }
84
85    fn has_one_field_ty(pair: Option<FieldId>) -> FieldTy {
86        FieldTy::HasOne(app::HasOne {
87            target: <T::Model as Register>::id(),
88            expr_ty: stmt::Type::Model(<T::Model as Register>::id()),
89            // If unresolved, the pair is populated by the schema linker.
90            pair: pair.unwrap_or(FieldId {
91                model: ModelId(usize::MAX),
92                index: usize::MAX,
93            }),
94        })
95    }

104impl<T: fmt::Debug> fmt::Debug for HasOne<T> {
105    fn fmt(&self, fmt: &mut fmt::Formatter<'_>) -> fmt::Result {
106        match self.value.as_ref() {
107            Some(t) => t.fmt(fmt),
108            None => {
109                write!(fmt, "<not loaded>")?;
110                Ok(())
111            }
112        }
113    }
114}

8/// Placeholder via target for a scalar terminal, overwritten when `toasty-core`
9/// resolves the relation chain while linking. It must never reach a query.
10const UNRESOLVED_TARGET: ModelId = ModelId(usize::MAX);
11
12/// Implemented by a `#[has_many(via = …)]` field's Rust type, exposing the
13/// via's [`Target`](Self::Target) — the terminal type the path projects.
14///
15/// Blanket-implemented over `Vec<E>` and [`Deferred<Vec<E>>`](Deferred) for
16/// every `E`, so the derive can route a via field through [`ViaTarget`]
17/// without requiring `E: Model`. A single blanket impl per outer shape keeps
18/// it coherent with the per-target `ViaTarget` impls.
19///
20/// A join-model many-to-many field is a model-terminal via. For example,
21/// `#[has_many(via = memberships.group)] groups: Deferred<Vec<Group>>`
22/// follows the declaring model's `memberships` relation and then each
23/// membership's `group` relation.
24pub trait ViaManyField {
25    /// The via's terminal type — the model or scalar the path projects.
26    type Target: ViaTarget;
27
28    /// Whether the field stores its value in a deferred load slot.
29    const DEFERRED: bool;
30}
31
32impl<E: ViaTarget> ViaManyField for Vec<E> {
33    type Target = E;
34    const DEFERRED: bool = false;
35}
36
37impl<E: ViaTarget> ViaManyField for Deferred<Vec<E>> {
38    type Target = E;
39    const DEFERRED: bool = true;
40}

42/// The query builder a `#[has_many(via = …)]` navigation method returns for a
43/// field of type `F` (e.g. `Vec<String>`, `Deferred<Vec<Article>>`).
44///
45/// Resolves to the terminal type's [`ViaTarget::Query`]: `QueryMany<M>` for a
46/// model terminal, `Query<List<scalar>>` for a scalar. Mirrors
47/// [`QueryMany`](super::QueryMany), which aliases a model's query builder the
48/// same way.
49pub type ViaMany<F> = <<F as ViaManyField>::Target as ViaTarget>::Query;
50
51/// The typed path handle a `#[has_many(via = …)]` field's accessor returns for
52/// a field of type `F` (e.g. `Vec<String>`, `Deferred<Vec<Article>>`), rooted
53/// at `Origin`.
54///
55/// Resolves to the terminal type's [`ViaTarget::Path`]: the model's chainable
56/// `ManyField` for a model terminal, `Path<Origin, List<scalar>>` for a scalar.
57/// Mirrors [`ViaMany`], which aliases the navigation query the same way.
58pub type ViaPath<F, Origin> = <<F as ViaManyField>::Target as ViaTarget>::Path<Origin>;
59
60/// A type that can be the terminal element of a `#[has_many(via = …)]`
61/// relation.
62///
63/// A `via` relation reaches its terminal by following a path of existing
64/// relations. The terminal is usually another model (`via = comments.article`),
65/// but it may also be a **scalar field** (`via = todos.tags.name`), which
66/// projects that field across the relation path to yield `Vec<String>`.
67///
68/// This trait is keyed on the *terminal element type*, with disjoint concrete
69/// impls so they never overlap:
70///
71/// - The `#[derive(Model)]` macro emits an impl for each model, setting
72///   [`Query`](Self::Query) to that model's generated query builder (so a
73///   relation-terminal via keeps its rich `QueryMany<M>` API).
74/// - This module emits an impl for each scalar primitive, setting
75///   [`Query`](Self::Query) to the plain [`Query<List<Self>>`].
76pub trait ViaTarget {
77    /// The query builder returned by the generated has-many-via navigation
78    /// method.
79    type Query;
80
81    /// The typed path handle a `#[has_many(via = …)]` field's accessor returns,
82    /// parameterized by the origin model. Mirrors [`Model::Path`] for the
83    /// via-many case.
84    ///
85    /// A **model** terminal returns that model's chainable
86    /// [`ManyField`](crate::schema::Model::ManyField) — a `Path` wrapper that
87    /// adds field accessors, so navigation can continue *through* a via step
88    /// (e.g. `organizations.todos.title`, where `organizations.todos` is itself
89    /// a via). A **scalar** terminal returns a plain list [`Path`]: a leaf with
90    /// nothing further to chain.
91    ///
92    /// [`Model::Path`]: crate::schema::Model::Path
93    type Path<Origin>;
94
95    /// Build the [`FieldTy::Via`] for a `#[has_many(via = …)]` field whose
96    /// terminal element type is `Self`.
97    ///
98    /// `path` is the fully-resolved field path (rooted at the declaring
99    /// model), including the terminal step. The via target is `Self::id()` for
100    /// a model terminal; for a scalar terminal it is the model the relation
101    /// chain reaches, which the path alone can't name here — so the scalar
102    /// impls leave it unset and `toasty-core` fills it in while linking.
103    fn via_field_ty(singular: Name, path: stmt::Path) -> FieldTy;
104
105    /// Build the [`Path`](Self::Path) handle from the typed `path` reaching this
106    /// terminal. A model terminal wraps the path in its chainable `ManyField`; a
107    /// scalar terminal is a leaf, so the path is the handle.
108    fn new_path<Origin>(path: Path<Origin, List<Self>>) -> Self::Path<Origin>
109    where
110        Self: Sized;
111
112    /// Wrap a via-field association into the navigation query.
113    ///
114    /// A scalar terminal yields a query whose source model and terminal
115    /// projection are filled in during lowering ([`RewriteVia`]) from the
116    /// core-resolved schema, so neither is needed here.
117    ///
118    /// [`RewriteVia`]: ../../engine/lower/association/struct.RewriteVia.html
119    fn make_via_query(assoc: Association<List<Self>>) -> Self::Query
120    where
121        Self: Sized;
122}
123
124/// Emit `ViaTarget` for scalar primitives. Each impl is concrete, so it
125/// stays disjoint from the per-model impls the derive macro generates.
126macro_rules! impl_via_many_scalar {
127    ( $( $t:ty ),* $(,)? ) => {
128        $(
129            impl ViaTarget for $t {
130                type Query = Query<List<$t>>;
131                type Path<Origin> = Path<Origin, List<$t>>;
132
133                fn new_path<Origin>(path: Path<Origin, List<$t>>) -> Self::Path<Origin> {
134                    // A scalar terminal is a leaf — the path is the handle.
135                    path
136                }
137
138                fn via_field_ty(singular: Name, path: stmt::Path) -> FieldTy {
139                    // The terminal scalar field is the path's last step.
140                    let terminal = *path
141                        .projection
142                        .as_slice()
143                        .last()
144                        .expect("via path has at least one step");
145                    let expr_ty = stmt::Type::List(Box::new(<$t as Load>::ty()));
146                    // `target` (the model the relation chain reaches) is resolved
147                    // in core's link phase; leave a placeholder it overwrites.
148                    FieldTy::Via(Via::new(
149                        UNRESOLVED_TARGET,
150                        expr_ty,
151                        app::Cardinality::Many { singular },
152                        path,
153                        Some(terminal),
154                    ))
155                }
156
157                fn make_via_query(assoc: Association<List<$t>>) -> Self::Query {
158                    // A via-as-source query whose source model and terminal
159                    // projection RewriteVia fills in from the resolved schema
160                    // (see its scalar branch). The placeholder source id is
161                    // overwritten there. `distinct` collapses duplicate terminal
162                    // values, matching the include path.
163                    let mut query = stmt::Query::builder(stmt::SourceModel {
164                        id: UNRESOLVED_TARGET,
165                        via: Some(assoc.untyped),
166                    })
167                    .build();
168                    query.body.as_select_mut_unwrap().distinct = true;
169                    Query::from_untyped(query)
170                }
171            }
172        )*
173    };
174}

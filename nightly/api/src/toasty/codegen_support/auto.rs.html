1//! Compile-time validation for explicit `#[auto(...)]` strategies.
2//!
3//! When `#[derive(Model)]` sees an explicit strategy on a field
4//! (`#[auto(uuid)]`, `#[auto(uuid(v4))]`, `#[auto(uuid(v7))]`,
5//! `#[auto(increment)]`), the generated code emits a one-shot obligation that
6//! the field's Rust type implements `AutoCompatible<Tag>` for the matching
7//! marker `Tag`. The bare `#[auto]` form already gets a `T: Auto` obligation
8//! via the `STRATEGY` const lookup, so it does not need this trait.
9//!
10//! Mirrors the pattern in [`crate::codegen_support::storage`]: tag types are
11//! zero-sized markers, the trait carries a `#[diagnostic::on_unimplemented]`
12//! message, and the macro names tags by absolute path so user-side type
13//! aliases or shadowing cannot defeat the check.
14
15/// Marker types for each `#[auto(...)]` strategy.
16///
17/// These types are zero-sized and only ever appear as the `Strategy`
18/// parameter to [`AutoCompatible`]. Macro-generated code names them by
19/// absolute path.
20#[allow(missing_docs)]
21pub mod tag {
22    pub struct Uuid;
23    pub struct Increment;
24}
25
26/// Asserts that a Rust field type can carry an auto-generated value of the
27/// given strategy.
28///
29/// Macro-generated code emits a `_check::<FieldType>()` call where `_check`
30/// is bounded by `AutoCompatible<Tag>`. The bound is checked against the
31/// resolved field type, so type aliases and newtype embeds compose naturally.
32#[diagnostic::on_unimplemented(
33    message = "field type `{Self}` is not compatible with the requested auto strategy `{Strategy}`",
34    label = "incompatible Rust type for `#[auto(...)]`",
35    note = "see `toasty::codegen_support::auto` for the compatibility table"
36)]
37pub trait AutoCompatible<Strategy> {}
38
39// `Option<T>` is compatible with the same strategies as `T`. Auto-populated
40// fields are typically non-nullable, but the schema tracks nullability
41// separately so the strategy check is unaffected by the option wrapper.
42impl<T, S> AutoCompatible<S> for Option<T> where T: AutoCompatible<S> {}
43
44impl AutoCompatible<tag::Increment> for i8 {}
45impl AutoCompatible<tag::Increment> for i16 {}
46impl AutoCompatible<tag::Increment> for i32 {}
47impl AutoCompatible<tag::Increment> for i64 {}
48impl AutoCompatible<tag::Increment> for u8 {}
49impl AutoCompatible<tag::Increment> for u16 {}
50impl AutoCompatible<tag::Increment> for u32 {}
51impl AutoCompatible<tag::Increment> for u64 {}
52impl AutoCompatible<tag::Increment> for isize {}
53impl AutoCompatible<tag::Increment> for usize {}
54
55impl AutoCompatible<tag::Uuid> for uuid::Uuid {}

57// A tuple-newtype embed is compatible with whatever its inner type is
58// compatible with. Mirrors the `Auto` blanket in
59// [`crate::codegen_support::newtype`].
60//
61// `do_not_recommend` keeps the blanket out of suggestion lists so the
62// `AutoCompatible` `#[diagnostic::on_unimplemented]` message wins.
63#[diagnostic::do_not_recommend]
64impl<T, S> AutoCompatible<S> for T
65where
66    T: super::newtype::NewtypeOf,
67    <T as super::newtype::NewtypeOf>::Inner: AutoCompatible<S>,
68{

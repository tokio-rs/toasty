1//! Marker trait emitted by `#[derive(Embed)]` for tuple-newtype embeds.
2//!
3//! The companion blanket impl turns any newtype embed whose inner type
4//! implements [`Auto`] into an [`Auto`] type itself, so a user does not need
5//! to repeat the strategy on the wrapper:
6//!
7//! ```ignore
8//! #[derive(toasty::Embed)]
9//! struct UserId(uuid::Uuid);
10//!
11//! // Automatically: `UserId: Auto` because `Uuid: Auto`.
12//! ```
13//!
14//! Newtypes around non-`Auto` types stay non-`Auto` — the blanket's
15//! `T::Inner: Auto` bound is on a generic associated type, so it is properly
16//! conditional and doesn't error at impl declaration.
17//!
18//! `NewtypeOf` is local to toasty. Combined with the orphan rule, that keeps
19//! the blanket from overlapping with the concrete `impl Auto for u64`,
20//! `impl Auto for uuid::Uuid`, etc.: no downstream crate can ever add
21//! `impl NewtypeOf for u64`, and toasty itself never will.
22
23use crate::schema::{Auto, Field};
24use toasty_core::schema::app::AutoStrategy;
25
26/// Marker for a tuple-newtype embedded type, carrying its single inner field
27/// type as an associated type. Emitted by `#[derive(Embed)]` for tuple
28/// structs with exactly one field.
29pub trait NewtypeOf {
30    /// The inner field's type.
31    type Inner;
32}
33
34// `do_not_recommend` keeps the blanket out of error suggestions so the
35// `Auto` trait's `#[diagnostic::on_unimplemented]` message wins for users
36// who hit a missing-`Auto` error on an embed.
37#[diagnostic::do_not_recommend]
38impl<T> Auto for T
39where
40    T: NewtypeOf + Field,
41    <T as NewtypeOf>::Inner: Auto,
42{
43    const STRATEGY: AutoStrategy = <<T as NewtypeOf>::Inner as Auto>::STRATEGY;
44}

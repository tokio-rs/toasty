1//! Compile-time validation for `#[column(type = ...)]`.
2//!
3//! When `#[derive(Model)]` sees an explicit storage type on a field, the
4//! generated code emits a one-shot obligation that the field's Rust type
5//! implements `CompatibleWith<Tag>` for the matching marker `Tag`. The Rust
6//! type checker — not the macro — resolves the field type, so type aliases
7//! and generic parameters are handled correctly.
8
9/// Marker types corresponding to each `#[column(type = ...)]` storage tag.
10///
11/// These types are zero-sized and only ever appear as the `S` parameter to
12/// [`CompatibleWith`]. Macro-generated code names them by absolute path.
13#[allow(missing_docs, non_camel_case_types)]
14pub mod tag {
15    pub struct Boolean;
16
17    /// Signed integer storage with a non-standard byte width.
18    pub struct Int<const BYTES: u8>;
19    /// Unsigned integer storage with a non-standard byte width.
20    pub struct UInt<const BYTES: u8>;
21
22    pub struct I8;
23    pub struct I16;
24    pub struct I32;
25    pub struct I64;
26
27    pub struct U8;
28    pub struct U16;
29    pub struct U32;
30    pub struct U64;
31
32    pub struct F32;
33    pub struct F64;
34
35    pub struct Text;
36    pub struct VarChar;
37    pub struct Json;
38    pub struct Jsonb;
39
40    pub struct Binary;
41    pub struct Blob;
42
43    pub struct Timestamp;
44    pub struct Date;
45    pub struct Time;
46    pub struct DateTime;
47}

49/// Asserts that a Rust field type can be stored as the given storage tag.
50///
51/// Macro-generated code emits a `_check::<FieldType>()` call where `_check`
52/// is bounded by `CompatibleWith<Tag>`. Trait resolution happens after type
53/// alias substitution, so `type String = u32;` cannot defeat the check the
54/// way it would defeat any analysis of the raw `String` token in the macro.
55#[diagnostic::on_unimplemented(
56    message = "field type `{Self}` is not compatible with the requested column storage `{Storage}`",
57    label = "incompatible Rust type for `#[column(type = ...)]`",
58    note = "see `toasty::codegen_support::storage` for the compatibility table"
59)]
60pub trait CompatibleWith<Storage> {}
61
62/// Marker implemented by the integer storage tags accepted for embedded enum
63/// discriminant columns.
64pub trait IntegerStorage {}

67impl IntegerStorage for tag::I16 {}

69impl IntegerStorage for tag::I64 {}

74impl<const BYTES: u8> IntegerStorage for tag::Int<BYTES> {}
75impl<const BYTES: u8> IntegerStorage for tag::UInt<BYTES> {}

77// `Option<T>` is compatible with the same storage tags as `T`. The schema
78// tracks nullability separately, so the storage type is unaffected by the
79// option wrapper.
80impl<T, S> CompatibleWith<S> for Option<T> where T: CompatibleWith<S> {}

88// A unit enum is both `Embed` and `Scalar`. Integer storage selected on a
89// `Vec<unit-enum>` field describes each discriminant, not the collection as a
90// whole. Keep these implementations limited to enum collections so `Vec<u8>`
91// retains its binary-blob meaning.
92macro_rules! impl_integer_enum_collection_compat {
93    ($($storage:ty),* $(,)?) => {
94        $(
95            impl<T> CompatibleWith<$storage> for Vec<T>
96            where
97                T: crate::schema::Embed
98                    + crate::schema::Scalar
99                    + CompatibleWith<$storage>,
100            {}
101        )*
102    };
103}
104
105impl_integer_enum_collection_compat!(
106    tag::I8,
107    tag::I16,
108    tag::I32,
109    tag::I64,
110    tag::U8,
111    tag::U16,
112    tag::U32,
113    tag::U64,
114);

138// `int(N)` and `uint(N)` have no exact Rust counterpart when N is not a
139// power of two. Preserve the integer-family check while the engine performs
140// the value-level checked cast.
141macro_rules! impl_custom_integer_compat {
142    ($storage:ident => $($ty:ty),* $(,)?) => {
143        $(
144            #[diagnostic::do_not_recommend]
145            impl<const BYTES: u8> CompatibleWith<tag::$storage<BYTES>> for $ty {}
146        )*
147    };
148}
149
150impl_custom_integer_compat!(Int => i8, i16, i32, i64);
151impl_custom_integer_compat!(UInt => u8, u16, u32, u64);

1//! Per-connection worker task and its message protocol.
2//!
3//! Each pooled connection is owned by a [`ConnectionTask`] running in a
4//! dedicated tokio task; callers interact with it through a
5//! [`ConnectionHandle`] by sending [`ConnectionOperation`] messages over
6//! an mpsc channel and awaiting a oneshot response. This serializes
7//! access to the underlying driver connection — drivers are not
8//! required to be `Sync` on `exec` — and gives the connection its own
9//! lifecycle independent of the pool object that holds the handle.
10
11use std::sync::Arc;
12
13use toasty_core::driver::operation::RawSql;
14use toasty_core::driver::{Connection, Rows};
15use toasty_core::stmt::Value;
16use tokio::{
17    sync::{mpsc, oneshot},
18    task::JoinHandle,
19};
20use tracing::Instrument;
21
22use super::pool::SweepWaker;
23use crate::engine::Engine;

25/// Operations sent to the connection task.
26///
27/// Each variant carries the caller's `tracing::Span`, captured at send time.
28/// The worker task runs on its own tokio task, so span context does not
29/// propagate to it implicitly; the task enters the carried span while
30/// handling the operation so engine and driver events land inside the
31/// caller's span tree.
32pub(crate) enum ConnectionOperation {
33    /// Execute a statement (compile + run on the connection).
34    ExecStatement {
35        stmt: Box<toasty_core::stmt::Statement>,
36        in_transaction: bool,
37        span: tracing::Span,
38        tx: oneshot::Sender<crate::Result<toasty_core::driver::ExecResponse>>,
39    },
40    ExecOperation {
41        operation: Box<toasty_core::driver::operation::Operation>,
42        span: tracing::Span,
43        tx: oneshot::Sender<crate::Result<toasty_core::driver::ExecResponse>>,
44    },
45    /// Execute user-authored SQL through the engine's SQL capability guard.
46    ExecRawSql {
47        raw: Box<RawSql>,
48        span: tracing::Span,
49        tx: oneshot::Sender<crate::Result<toasty_core::driver::ExecResponse>>,
50    },
51    /// Push schema to the database.
52    PushSchema {
53        span: tracing::Span,
54        tx: oneshot::Sender<crate::Result<()>>,
55    },
56    /// Active liveness probe issued by the pool's background sweep
57    /// (and, eventually, per-acquire pre-ping). Routes through the
58    /// connection task so the ping is serialized against any in-flight
59    /// `exec` on the same connection.
60    Ping {
61        span: tracing::Span,
62        tx: oneshot::Sender<crate::Result<()>>,
63    },
64}
65
66/// Handle to a dedicated connection task.
67///
68/// When dropped, `in_tx` closes the channel, causing the background task to
69/// finish processing remaining messages and exit gracefully.
70pub(crate) struct ConnectionHandle {
71    pub(crate) in_tx: mpsc::UnboundedSender<ConnectionOperation>,
72    join_handle: JoinHandle<()>,
73}
74
75impl ConnectionHandle {
76    /// Spawn a worker task that owns `connection` and serializes
77    /// operations against it. `sweep_waker` is woken whenever the task
78    /// observes that the connection went bad so the pool's health-check
79    /// sweep can escalate immediately rather than waiting for its next
80    /// periodic tick.
81    pub(crate) fn spawn(
82        connection: Box<dyn Connection>,
83        engine: Engine,
84        sweep_waker: Arc<SweepWaker>,
85    ) -> Self {
86        let (in_tx, in_rx) = mpsc::unbounded_channel::<ConnectionOperation>();
87        let task = ConnectionTask {
88            connection,
89            engine,
90            in_rx,
91            sweep_waker,
92        };
93        let join_handle = tokio::spawn(task.run());
94        Self { in_tx, join_handle }
95    }
96
97    /// Returns true once the worker task has exited. Used by the pool's
98    /// `recycle` to detect dead slots.
99    pub(crate) fn is_finished(&self) -> bool {
100        self.join_handle.is_finished()
101    }
102}
103
104impl std::fmt::Debug for ConnectionHandle {
105    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
106        f.debug_struct("ConnectionHandle")
107            .field("channel_closed", &self.in_tx.is_closed())
108            .field("task_finished", &self.join_handle.is_finished())
109            .finish()
110    }
111}

113/// Owns one database connection and serializes operations against it.
114/// Spawned by [`ConnectionHandle::spawn`]; runs until `in_rx` closes
115/// (sender side dropped) or the connection reports `is_valid() == false`.
116struct ConnectionTask {
117    connection: Box<dyn Connection>,
118    engine: Engine,
119    in_rx: mpsc::UnboundedReceiver<ConnectionOperation>,
120    sweep_waker: Arc<SweepWaker>,
121}
122
123impl ConnectionTask {
124    async fn run(mut self) {
125        while let Some(op) = self.in_rx.recv().await {
126            if !self.handle(op).await {
127                return;
128            }
129        }
130    }
131
132    /// Dispatch one operation, entering the caller's span for its duration.
133    /// Returns `false` if the connection went bad during it and the task
134    /// should exit.
135    async fn handle(&mut self, op: ConnectionOperation) -> bool {
136        match op {
137            ConnectionOperation::ExecStatement {
138                stmt,
139                in_transaction,
140                span,
141                tx,
142            } => {
143                let result = self
144                    .exec_statement(*stmt, in_transaction)
145                    .instrument(span)
146                    .await;
147                self.respond(tx, result)
148            }
149            ConnectionOperation::ExecOperation {
150                operation,
151                span,
152                tx,
153            } => {
154                let result = self
155                    .connection
156                    .exec(&self.engine.schema, *operation)
157                    .instrument(span)
158                    .await;
159                self.respond(tx, result)
160            }
161            ConnectionOperation::ExecRawSql { raw, span, tx } => {
162                let result = self
163                    .engine
164                    .exec_raw_sql(&mut *self.connection, *raw)
165                    .instrument(span)
166                    .await;
167                self.respond(tx, result)
168            }
169            ConnectionOperation::PushSchema { span, tx } => {
170                let result = self
171                    .connection
172                    .push_schema(&self.engine.schema)
173                    .instrument(span)
174                    .await;
175                self.respond(tx, result)
176            }
177            ConnectionOperation::Ping { span, tx } => {
178                let result = self.connection.ping().instrument(span).await;
179                self.respond(tx, result)
180            }
181        }
182    }
183
184    async fn exec_statement(
185        &mut self,
186        stmt: toasty_core::stmt::Statement,
187        in_transaction: bool,
188    ) -> crate::Result<toasty_core::driver::ExecResponse> {
189        let single = stmt.is_single();
190        let mut response = self
191            .engine
192            .exec(&mut *self.connection, stmt, in_transaction)
193            .await?;
194        response.values.buffer().await?;
195
196        if single {
197            let Rows::Value(Value::List(mut items)) = response.values else {
198                unreachable!()
199            };
200            assert!(
201                items.len() <= 1,
202                "expected at most 1 row for single statement, got {}",
203                items.len()
204            );
205            response.values = Rows::Value(items.pop().unwrap_or(Value::Null));
206        }
207
208        Ok(response)
209    }
210
211    /// Send `result` to the caller, but if the connection reported
212    /// invalid during the op, close `in_rx` *first* so the mpsc transitions
213    /// to closed synchronously. A consumer woken by the response that then
214    /// re-enters the pool sees `in_tx.is_closed() == true` in
215    /// `Manager::recycle` and the slot is evicted — no race with this
216    /// task's exit. Returns whether the task should keep running.
217    ///
218    /// Also wakes the pool's background sweep so it can escalate to a
219    /// full idle-pool ping pass without waiting for the next periodic
220    /// tick. A real connection-lost error usually means more than one
221    /// connection in the pool is affected (database restart, network
222    /// event), and catching the rest eagerly turns a multi-second
223    /// recovery window into one round-trip per idle slot.
224    fn respond<T>(&mut self, tx: oneshot::Sender<T>, result: T) -> bool {
225        if self.connection.is_valid() {
226            let _ = tx.send(result);
227            true
228        } else {
229            tracing::debug!("connection reported invalid; closing channel and exiting");
230            self.in_rx.close();
231            self.sweep_waker.wake();
232            let _ = tx.send(result);
233            false
234        }
235    }
236}

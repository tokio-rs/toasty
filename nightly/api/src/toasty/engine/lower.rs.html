34/// Wrap a nullable single-relation subquery so a missing row passes through as
35/// `Null`, while a present row is transformed by `present`. Used when lowering
36/// `.include()`/`.select()` of nullable single (`Deferred<Option<_>>` /
37/// `Deferred<Option<_>>`) relations that need to project out of the subquery
38/// result:
39///
40/// ```text
41///   Let {
42///     binding: subquery,
43///     body: Match { subject: arg(0), arms: [Null → Null], else: present },
44///   }
45/// ```
46///
47/// The subquery result is bound as `arg(0)`; `present` is the expression for a
48/// non-null row and may reference that binding (e.g. project a field out of it).
49fn map_nullable_single(subquery: stmt::Expr, present: stmt::Expr) -> stmt::Expr {
50    stmt::Expr::Let(stmt::ExprLet {
51        bindings: vec![subquery],
52        body: Box::new(stmt::Expr::match_expr(
53            stmt::Expr::arg(0),
54            vec![stmt::MatchArm {
55                pattern: stmt::Value::Null,
56                expr: stmt::Expr::Value(stmt::Value::Null),
57            }],
58            present,
59        )),
60    })
61}

63impl Engine {
64    pub(super) fn lower_stmt(&self, stmt: stmt::Statement) -> Result<HirStatement> {
65        let schema = &self.schema;
66
67        let mut state = LoweringState {
68            hir: HirStatement::new(),
69            scopes: IndexVec::new(),
70            engine: self,
71            relations: vec![],
72            errors: vec![],
73            dependencies: HashSet::new(),
74        };
75
76        state.lower_stmt(stmt::ExprContext::new(schema), None, stmt);
77
78        if let Some(err) = state.errors.into_iter().next() {
79            return Err(err);
80        }
81
82        Ok(state.hir)
83    }
84}

86impl LoweringState<'_> {
87    fn lower_stmt(
88        &mut self,
89        expr_cx: stmt::ExprContext,
90        row_index: Option<usize>,
91        mut stmt: stmt::Statement,
92    ) -> hir::StmtId {
93        // App-level rewrites that lowering depends on. `Source::Model { via }`
94        // must be converted to a WHERE filter before the lowering walk
95        // converts `Source::Model` into `Source::Table`.  The IN-subquery
96        // lift fires here too: code paths outside the lowering walk (e.g.
97        // `ApplyInsertScope::apply_expr`) see the already-lifted form.
98        // `UpdateTarget::Query` is lifted into `UpdateTarget::Model` with
99        // the inner query's filter merged onto the outer update before the
100        // walk sees the target.  The eq/ne operand rewrite (model→PK,
101        // BelongsTo→FK) fires inside the lowering walk itself via
102        // `LowerStatement::visit_expr_binary_op_mut`.
103        association::RewriteVia::new(expr_cx).rewrite(&mut stmt);
104        lift_in_subquery::LiftInSubquery::new(expr_cx).rewrite(&mut stmt);
105        lift_update_query::LiftUpdateQuery::new().rewrite(&mut stmt);
106
107        Simplify::with_context(expr_cx, self.engine.capability).visit_mut(&mut stmt);
108
109        let stmt_id = self.hir.new_statement_info(self.dependencies.clone());
110        let scope_id = self.scopes.push(Scope { stmt_id, row_index });
111        let mut collect_dependencies = None;
112
113        // Map the statement
114        LowerStatement {
115            state: self,
116            expr_cx,
117            scope_id,
118            cx: LoweringContext::Statement,
119            collect_dependencies: &mut collect_dependencies,
120        }
121        .visit_stmt_mut(&mut stmt);
122
123        self.engine.simplify_stmt(&mut stmt);
124
125        let stmt_info = &mut self.hir[stmt_id];
126        stmt_info.stmt = Some(Box::new(stmt));
127
128        self.scopes.pop();
129
130        debug_assert!(collect_dependencies.is_none());
131
132        stmt_id
133    }
134}

208/// A collection operator on a `Vec<scalar>` field.
209///
210/// `Append`, `Remove`, `Pop`, and `RemoveAt` are all collection operators:
211/// they target a single-column `Vec<scalar>` field and carry operator
212/// operands (a list of elements, a single element, an index, or nothing).
213/// Whether the operand is a list or a scalar is a per-operator encoding
214/// detail — the unifying property is that none carries a *field value*
215/// that decomposes across columns, so none needs the `model_to_table`
216/// substitution that `Set` flows through.
217enum CollectionOp<'a> {
218    Append(&'a mut stmt::Expr),
219    Remove(&'a mut stmt::Expr),
220    Pop,
221    RemoveAt(&'a mut stmt::Expr),
222}

224/// Arithmetic assignment operator on a numeric scalar field.
225///
226/// `Add` and `Subtract` carry a value to combine atomically with the
227/// current column value (`col = col ± expr`). Like collection ops, they
228/// target a single-column primitive field and skip the `model_to_table`
229/// substitution.
230enum ArithmeticOp {
231    Add,
232    Subtract,
233}

235impl LowerStatement<'_, '_> {
236    /// Lower a `Set` assignment — the only assignment that carries a whole
237    /// *field value*.
238    ///
239    /// The field may span multiple columns (embedded structs decompose
240    /// across columns; enums into a discriminant plus variant columns), so
241    /// this resolves the field mapping and, for each impacted column,
242    /// substitutes the lowered value into that column's `model_to_table`
243    /// template before emitting the column-level write.
244    fn lower_set_assignment(
245        &mut self,
246        out: &mut stmt::Assignments,
247        mapping: &mapping::Model,
248        projection: &stmt::Projection,
249        expr: &mut stmt::Expr,
250    ) {
251        self.visit_expr_mut(expr);
252
253        let Some(field) = mapping.resolve_field_mapping(projection) else {
254            self.state
255                .errors
256                .push(crate::Error::invalid_statement(format!(
257                    "invalid assignment projection: {projection:?}"
258                )));
259            return;
260        };
261
262        for (column, lowering_idx) in field.columns() {
263            let mut lowering_expr = mapping.model_to_table[lowering_idx].clone();
264
265            // Substitute the field reference in the column's lowering
266            // template with the lowered value.
267            lowering_expr.substitute(AssignmentInput {
268                assignment_projection: projection.clone(),
269                value: expr,
270            });
271
272            self.visit_expr_mut(&mut lowering_expr);
273
274            out.set(column, lowering_expr);
275        }
276    }

278    /// Lower a collection operator (`Append` / `Remove` / `Pop` /
279    /// `RemoveAt`) on a `Vec<scalar>` field.
280    ///
281    /// `Vec<scalar>` fields always resolve to a single primitive column, so
282    /// this skips the `model_to_table` substitution that `lower_set_assignment`
283    /// performs — the operands are operator arguments, not a field value,
284    /// and for a single-column field `model_to_table` is identity anyway.
285    ///
286    /// `Append` is supported on every backend; the removal operators are
287    /// gated by per-backend capability flags and emit a clear error where
288    /// the native form is not available. On document collections the
289    /// removal operators are rejected outright: the per-backend renderings
290    /// are native-array forms that do not apply to a document column.
291    fn lower_collection_op(
292        &mut self,
293        out: &mut stmt::Assignments,
294        mapping: &mapping::Model,
295        projection: &stmt::Projection,
296        op: CollectionOp,
297    ) {
298        let Some(field) = mapping.resolve_field_mapping(projection) else {
299            self.state
300                .errors
301                .push(crate::Error::invalid_statement(format!(
302                    "invalid assignment projection: {projection:?}"
303                )));
304            return;
305        };
306
307        let Some(prim) = field.as_primitive() else {
308            self.state
309                .errors
310                .push(crate::Error::invalid_statement(format!(
311                    "collection operator on non-primitive field: {projection:?}"
312                )));
313            return;
314        };
315
316        // `Append` is universally supported; the removal operators are
317        // gated per backend and rejected on document collections, where
318        // the capability flags speak for the native-array renderings only.
319        let cap = self.capability();
320        let is_document = self
321            .expr_cx
322            .schema()
323            .mapping
324            .document_column_ty(prim.column)
325            .is_some();
326        let unsupported = match &op {
327            CollectionOp::Append(_) => None,
328            CollectionOp::Remove(_) if is_document || !cap.vec_remove => Some("stmt::remove"),
329            CollectionOp::Pop if is_document || !cap.vec_pop => Some("stmt::pop"),
330            CollectionOp::RemoveAt(_) if is_document || !cap.vec_remove_at => {
331                Some("stmt::remove_at")
332            }
333            _ => None,
334        };
335
336        if let Some(op_name) = unsupported {
337            let target = if is_document {
338                "document collections"
339            } else {
340                "this backend"
341            };
342            self.state
343                .errors
344                .push(crate::Error::invalid_statement(format!(
345                    "{op_name} is not yet supported on {target}"
346                )));
347            return;
348        }
349
350        match op {
351            CollectionOp::Append(expr) => {
352                self.visit_expr_mut(expr);
353                let mut value = expr.take();
354
355                // A document collection's append operand (a list of elements,
356                // matching the column's document type) lowers through the
357                // same schema-directed cast a `Set` picks up from
358                // `model_to_table`; `Append` bypasses that template, so the
359                // cast is applied here.
360                let schema = self.expr_cx.schema();
361                if let Some(doc_ty) = schema.mapping.document_column_ty(prim.column) {
362                    let column_ty = &schema.db.column(prim.column).ty;
363                    value = stmt::Expr::cast_from(value, doc_ty, column_ty);
364                }
365
366                out.append(prim.column, value);
367            }
368            CollectionOp::Remove(expr) => {
369                self.visit_expr_mut(expr);
370                out.remove(prim.column, expr.take());
371            }
372            CollectionOp::Pop => {
373                out.pop(prim.column);
374            }
375            CollectionOp::RemoveAt(expr) => {
376                self.visit_expr_mut(expr);
377                out.remove_at(prim.column, expr.take());
378            }
379        }
380    }

382    /// Lower an arithmetic assignment operator (`Add` / `Subtract`) on a
383    /// scalar numeric field. Mirrors `lower_collection_op`: single primitive
384    /// column, no `model_to_table` substitution needed.
385    fn lower_arithmetic_op(
386        &mut self,
387        out: &mut stmt::Assignments,
388        mapping: &mapping::Model,
389        projection: &stmt::Projection,
390        op: ArithmeticOp,
391        expr: &mut stmt::Expr,
392    ) {
393        let Some(field) = mapping.resolve_field_mapping(projection) else {
394            self.state
395                .errors
396                .push(crate::Error::invalid_statement(format!(
397                    "invalid assignment projection: {projection:?}"
398                )));
399            return;
400        };
401
402        let Some(prim) = field.as_primitive() else {
403            self.state
404                .errors
405                .push(crate::Error::invalid_statement(format!(
406                    "arithmetic operator on non-primitive field: {projection:?}"
407                )));
408            return;
409        };
410
411        self.visit_expr_mut(expr);
412        let value = expr.take();
413        match op {
414            ArithmeticOp::Add => out.add(prim.column, value),
415            ArithmeticOp::Subtract => out.subtract(prim.column, value),
416        }
417    }

419    /// Dispatch a single owned assignment to the matching per-kind lowering
420    /// helper. `Batch` first folds via [`fold_batch`] and then recurses on
421    /// the folded result. A residual `Batch` (from a non-composable pair)
422    /// is unsupported today; see #717.
423    fn lower_owned_assignment(
424        &mut self,
425        out: &mut stmt::Assignments,
426        mapping: &mapping::Model,
427        projection: &stmt::Projection,
428        assignment: stmt::Assignment,
429    ) {
430        match assignment {
431            stmt::Assignment::Set(mut expr) => {
432                self.lower_set_assignment(out, mapping, projection, &mut expr);
433            }
434            stmt::Assignment::Append(mut expr) => {
435                self.lower_collection_op(out, mapping, projection, CollectionOp::Append(&mut expr));
436            }
437            stmt::Assignment::Remove(mut expr) => {
438                self.lower_collection_op(out, mapping, projection, CollectionOp::Remove(&mut expr));
439            }
440            stmt::Assignment::Pop => {
441                self.lower_collection_op(out, mapping, projection, CollectionOp::Pop);
442            }
443            stmt::Assignment::RemoveAt(mut expr) => {
444                self.lower_collection_op(
445                    out,
446                    mapping,
447                    projection,
448                    CollectionOp::RemoveAt(&mut expr),
449                );
450            }
451            stmt::Assignment::Add(mut expr) => {
452                self.lower_arithmetic_op(out, mapping, projection, ArithmeticOp::Add, &mut expr);
453            }
454            stmt::Assignment::Subtract(mut expr) => {
455                self.lower_arithmetic_op(
456                    out,
457                    mapping,
458                    projection,
459                    ArithmeticOp::Subtract,
460                    &mut expr,
461                );
462            }
463            stmt::Assignment::Batch(entries) => {
464                let folded = fold_batch(entries);
465                if matches!(folded, stmt::Assignment::Batch(_)) {
466                    todo!("non-composable batch shape on {projection:?}: {folded:#?}")
467                }
468                self.lower_owned_assignment(out, mapping, projection, folded);
469            }
470            stmt::Assignment::Insert(_) => {
471                // `Insert` only applies to has-many relation fields, which
472                // `plan_stmt_update_relations` consumes before table-level
473                // lowering runs.
474                todo!("Insert assignment is not produced for table lowering; got {assignment:#?}")
475            }
476        }
477    }

480/// Fold a `Batch` of column-level ops on the same field into a single
481/// equivalent assignment, applied left-to-right via [`compose_assignment`].
482///
483/// The result is a non-`Batch` assignment when every pair composes; if any
484/// pair is not algebraically reducible (today: `Pop`/`Remove`/`RemoveAt`
485/// interleaved with other ops, or non-literal `Append` concat), the
486/// residual is returned as a `Batch` so the caller can error with the full
487/// shape for diagnostics.
488fn fold_batch(entries: Vec<stmt::Assignment>) -> stmt::Assignment {
489    let mut iter = entries.into_iter();
490    let mut acc = iter.next().expect("batch is non-empty");
491    for next in iter {
492        acc = compose_assignment(acc, next);
493    }
494    acc
495}

497/// Compose two assignments on the same field into a single equivalent
498/// assignment, in execution order. Returns the un-composed pair as a
499/// `Batch` when the combination is not algebraically reducible — the
500/// caller treats a residual `Batch` as an unsupported shape.
501fn compose_assignment(acc: stmt::Assignment, next: stmt::Assignment) -> stmt::Assignment {
502    use stmt::Assignment::*;
503    match (acc, next) {
504        // `Set` clobbers everything that came before it.
505        (_, Set(v)) => Set(v),
506
507        // Arithmetic following `Set` rewrites the stored value.
508        (Set(v), Add(x)) => Set(stmt::Expr::add(v, x)),
509        (Set(v), Subtract(x)) => Set(stmt::Expr::sub(v, x)),
510
511        // Arithmetic on arithmetic: combine operands. With `Subtract` as
512        // the outer op, subsequent operand signs flip
513        // (col - a + b = col - (a - b); col - a - b = col - (a + b)).
514        (Add(a), Add(x)) => Add(stmt::Expr::add(a, x)),
515        (Add(a), Subtract(x)) => Add(stmt::Expr::sub(a, x)),
516        (Subtract(a), Add(x)) => Subtract(stmt::Expr::sub(a, x)),
517        (Subtract(a), Subtract(x)) => Subtract(stmt::Expr::add(a, x)),
518
519        // Two literal-list `Append`s concatenate. Non-literal Append
520        // shapes fall through to a residual `Batch`.
521        (Append(a), Append(b)) => try_concat_list_literals_to_assignment(a, b),
522
523        // A `Batch` accumulator absorbs the next entry as a tail. Arises
524        // when a previous compose failed to reduce a pair; collecting the
525        // rest preserves the full shape for the caller's error message.
526        (Batch(mut tail), next) => {
527            tail.push(next);
528            Batch(tail)
529        }
530
531        // Not composable today — return as a residual `Batch`.
532        (acc, next) => Batch(vec![acc, next]),
533    }
534}

536/// Concatenate two list-shaped expressions if both are literals (either
537/// `Expr::List` or `Expr::Value(Value::List)`). Returns the original pair
538/// on failure so the caller can preserve the un-folded shape.
539fn try_concat_list_literals_to_assignment(a: stmt::Expr, b: stmt::Expr) -> stmt::Assignment {
540    use stmt::Assignment::{Append, Batch};
541    if !is_list_literal(&a) || !is_list_literal(&b) {
542        return Batch(vec![Append(a), Append(b)]);
543    }
544    let mut items = take_list_items(a).expect("checked is_list_literal");
545    items.extend(take_list_items(b).expect("checked is_list_literal"));
546    Append(stmt::Expr::list_from_vec(items))
547}
548
549fn is_list_literal(e: &stmt::Expr) -> bool {
550    matches!(
551        e,
552        stmt::Expr::List(_) | stmt::Expr::Value(stmt::Value::List(_))
553    )
554}
555
556fn take_list_items(e: stmt::Expr) -> Option<Vec<stmt::Expr>> {
557    match e {
558        stmt::Expr::List(list) => Some(list.items),
559        stmt::Expr::Value(stmt::Value::List(values)) => {
560            Some(values.into_iter().map(stmt::Expr::from).collect())
561        }
562        _ => None,
563    }
564}

566impl LowerStatement<'_, '_> {
567    fn new_dependency(&mut self, stmt: impl Into<stmt::Statement>) -> hir::StmtId {
568        let row_index = match self.cx {
569            LoweringContext::Insert(_, row_index) => row_index,
570            LoweringContext::Returning(row_index) => row_index,
571            _ => None,
572        };
573
574        let stmt_id = self.state.lower_stmt(self.expr_cx, row_index, stmt.into());
575
576        if let Some(dependencies) = &mut self.collect_dependencies {
577            dependencies.insert(stmt_id);
578        }
579
580        self.curr_stmt_info().deps.insert(stmt_id);
581
582        stmt_id
583    }
584
585    fn collect_dependencies(
586        &mut self,
587        f: impl FnOnce(&mut LowerStatement<'_, '_>),
588    ) -> HashSet<hir::StmtId> {
589        let old = self.collect_dependencies.replace(HashSet::new());
590        f(self);
591        std::mem::replace(self.collect_dependencies, old).unwrap()
592    }
593
594    fn track_dependency(&mut self, dependency: hir::StmtId) {
595        self.curr_stmt_info().deps.insert(dependency);
596    }
597
598    fn with_dependencies(
599        &mut self,
600        mut dependencies: HashSet<hir::StmtId>,
601        f: impl FnOnce(&mut LowerStatement<'_, '_>),
602    ) {
603        // Dependencies should stack
604        dependencies.extend(&self.state.dependencies);
605
606        let old = std::mem::replace(&mut self.state.dependencies, dependencies);
607        f(self);
608        self.state.dependencies = old;
609    }
610}

612impl visit_mut::VisitMut for LowerStatement<'_, '_> {
613    fn visit_order_by_expr_mut(&mut self, node: &mut stmt::OrderByExpr) {
614        // First, run the default visitor to lower sub-expressions
615        self.visit_expr_mut(&mut node.expr);
616
617        // Reuse binary-op lowering: synthesize `expr == expr` so that
618        // cast conversions are applied, then keep the LHS result.
619        let mut lhs = node.expr.clone();
620        let mut rhs = node.expr.take();
621        self.lower_expr_binary_op(stmt::BinaryOp::Eq, &mut lhs, &mut rhs);
622        node.expr = lhs;
623    }
624
625    fn visit_assignments_mut(&mut self, i: &mut stmt::Assignments) {
626        let mut lowered = stmt::Assignments::default();
627        let mapping = self.mapping_unwrap();
628        let assignments = std::mem::take(i);
629        for (projection, assignment) in assignments {
630            self.lower_owned_assignment(&mut lowered, mapping, &projection, assignment);
631        }
632        *i = lowered;
633    }
634
635    fn visit_expr_set_op_mut(&mut self, i: &mut stmt::ExprSetOp) {
636        todo!("stmt={i:#?}");
637    }
638
639    fn visit_expr_binary_op_mut(&mut self, i: &mut stmt::ExprBinaryOp) {
640        // App-level operand rewrite for `eq` / `ne` (`Reference::Model` →
641        // primary-key field, `BelongsTo` → foreign-key field).  Must fire
642        // before the operands are walked, since walking lowers the field
643        // references into column references and the rewrite has nothing to
644        // match on.  Nested binary ops are handled by the recursive walk
645        // through `visit_expr_mut`.
646        if i.op.is_eq() || i.op.is_ne() {
647            self.rewrite_eq_operand(&mut i.lhs);
648            self.rewrite_eq_operand(&mut i.rhs);
649        }
650
651        stmt::visit_mut::visit_expr_binary_op_mut(self, i);
652    }
653
654    fn visit_expr_in_list_mut(&mut self, i: &mut stmt::ExprInList) {
655        // App-level operand rewrite: `Reference::Model { nesting } IN list`
656        // becomes `<pk_field> IN list`.  Must fire before the LHS is walked,
657        // since walking lowers the model reference into something the rewrite
658        // can no longer match on.
659        self.rewrite_in_list_model_operand(i);
660
661        stmt::visit_mut::visit_expr_in_list_mut(self, i);
662    }
663
664    fn visit_expr_mut(&mut self, expr: &mut stmt::Expr) {
665        match expr {
666            stmt::Expr::BinaryOp(e) => {
667                self.visit_expr_binary_op_mut(e);
668
669                if let Some(lowered) = self.lower_expr_binary_op(e.op, &mut e.lhs, &mut e.rhs) {
670                    *expr = lowered;
671                }
672            }
673            // App-level rewrite: an OR covering every variant of an enum
674            // via `IsVariant` is a tautology.  Must fire before the children
675            // walk lowers the `IsVariant` nodes into discriminant
676            // comparisons, since the rewrite pattern-matches on
677            // `Expr::IsVariant`.
678            stmt::Expr::Or(e) if expr_or::is_variant_tautology_or(self.expr_cx.schema(), e) => {
679                *expr = true.into();
680            }
681            stmt::Expr::InList(e) => {
682                self.visit_expr_in_list_mut(e);
683
684                if let Some(lowered) = self.lower_expr_in_list(&mut e.expr, &mut e.list) {
685                    *expr = lowered;
686                }
687
688                // PostgreSQL-style: `x IN (a, b, c)` → `x = ANY($1)` with the
689                // list bound as a single array param.
690                if let stmt::Expr::InList(e) = expr
691                    && self.supports_any_rewrite()
692                    && in_list_is_value_list(e)
693                {
694                    let stmt::Expr::InList(e) = expr.take() else {
695                        unreachable!()
696                    };
697                    *expr = stmt::Expr::any_op(*e.expr, stmt::BinaryOp::Eq, *e.list);
698                }
699            }
700            stmt::Expr::Not(e) if matches!(*e.expr, stmt::Expr::InList(_)) => {
701                // Recurse into inner first so the InList itself is lowered.
702                self.visit_expr_not_mut(e);
703
704                // If the inner is still an InList (not yet rewritten because
705                // the gate is off or the list shape doesn't qualify), leave
706                // `Not(InList)` alone and let the SQL serializer render
707                // `NOT IN`. Otherwise, rewrite `NOT (x = ANY(arr))` into the
708                // canonical `x <> ALL(arr)` form.
709                if self.supports_any_rewrite()
710                    && let stmt::Expr::AnyOp(any) = e.expr.as_mut()
711                    && any.op == stmt::BinaryOp::Eq
712                {
713                    let stmt::Expr::Not(not) = expr.take() else {
714                        unreachable!()
715                    };
716                    let stmt::Expr::AnyOp(any) = *not.expr else {
717                        unreachable!()
718                    };
719                    *expr = stmt::Expr::all_op(*any.lhs, stmt::BinaryOp::Ne, *any.rhs);
720                }
721            }
722            stmt::Expr::InSubquery(e) => {
723                if self.capability().sql {
724                    self.visit_expr_in_subquery_mut(e);
725
726                    self.lower_in_subquery_operands(
727                        &mut e.expr,
728                        e.query.returning_mut_unwrap().as_project_mut_unwrap(),
729                    );
730
731                    let returning = e.query.returning_mut_unwrap().as_project_mut_unwrap();
732
733                    if !returning.is_record() {
734                        *returning = stmt::Expr::record([returning.take()]);
735                    }
736                } else {
737                    self.visit_expr_mut(&mut e.expr);
738
739                    let source_id = self.scope_stmt_id();
740                    let target_id = self.scope_statement(|child| {
741                        child.visit_stmt_query_mut(&mut e.query);
742                    });
743
744                    // The subquery must be executable without parent context:
745                    // no `Arg::Ref` (which would reach back into a parent
746                    // scope) and no `back_refs` (columns the parent must
747                    // batch-load on its behalf). `Arg::Sub` is fine — it's a
748                    // forward dependency on the subquery's own child, which
749                    // the planner chains in execution order.
750                    let target_stmt_info = &self.state.hir[target_id];
751                    debug_assert!(
752                        target_stmt_info
753                            .args
754                            .iter()
755                            .all(|arg| matches!(arg, hir::Arg::Sub { .. })),
756                        "TODO: sub-statement references parent scope"
757                    );
758                    debug_assert!(target_stmt_info.back_refs.is_empty(), "TODO");
759
760                    self.track_dependency(target_id);
761
762                    self.lower_in_subquery_operands(
763                        &mut e.expr,
764                        e.query.returning_mut_unwrap().as_project_mut_unwrap(),
765                    );
766
767                    let stmt::Expr::InSubquery(e) = expr.take() else {
768                        panic!()
769                    };
770
771                    let arg =
772                        self.new_sub_statement(source_id, target_id, Box::new((*e.query).into()));
773
774                    *expr = stmt::ExprInList {
775                        expr: e.expr,
776                        list: Box::new(arg),
777                    }
778                    .into();
779                }
780            }
781            stmt::Expr::IsVariant(e) => {
782                // Look up the enum model and variant directly via VariantId
783                let enum_model = self
784                    .schema()
785                    .app
786                    .model(e.variant.model)
787                    .as_embedded_enum_unwrap();
788                let has_data = enum_model.has_data_variants();
789                let disc_value = enum_model.variants[e.variant.index].discriminant.clone();
790
791                // Lower the inner expression
792                self.visit_expr_mut(&mut e.expr);
793
794                let lowered_expr = e.expr.take();
795
796                // Emit the appropriate comparison
797                if has_data {
798                    // Data-carrying: project([0]) to extract discriminant from Record
799                    *expr = stmt::Expr::eq(
800                        stmt::Expr::project(lowered_expr, [0usize]),
801                        stmt::Expr::Value(disc_value),
802                    );
803                } else {
804                    // Unit-only: compare directly
805                    *expr = stmt::Expr::eq(lowered_expr, stmt::Expr::Value(disc_value));
806                }
807            }
808            stmt::Expr::Project(project)
809                if let stmt::Expr::Incoming(stmt::ExprIncoming::Model(model)) =
810                    project.base.as_ref() =>
811            {
812                let (table, column) = {
813                    let mapping = self.schema().mapping_for(*model);
814                    let mapped = mapping
815                        .resolve_field_mapping(&project.projection)
816                        .expect("incoming upsert field must map to a column");
817                    let mut columns = mapped.columns();
818                    let (column, _) = columns
819                        .next()
820                        .expect("incoming upsert field has no database column");
821                    assert!(
822                        columns.next().is_none(),
823                        "incoming() does not yet support column-expanded embedded fields"
824                    );
825                    (mapping.table, column)
826                };
827                debug_assert_eq!(table, column.table);
828                *project.base = stmt::ExprIncoming::table(table).into();
829                project.projection = stmt::Projection::from_index(column.index);
830            }
831            // A null-check on an `Option<Embed>` field reduces to a null-check on
832            // the embed's head column instead of distributing the check over
833            // every flattened column. The head column is `NULL` for `None`, so
834            // for `Account::fields().contact().is_none()` (an `Option<enum>`)
835            // this emits `contact IS NULL` rather than checking each variant
836            // column. `.is_some()` arrives as `Not(IsNull(..))`, so the
837            // surrounding `Not` yields `contact IS NOT NULL`. Scalar `Option`
838            // and non-nullable fields fall through to the normal reference
839            // lowering below.
840            //
841            // Only fires for WHERE-clause predicates (`Statement` context),
842            // where a field reference lowers to a column. The `model_to_table`
843            // encode wraps `is_not_null(field)` over the head column too; in the
844            // `InsertRow` encode context the field reference must resolve to the
845            // row value (and `is_null` then folds), so this rewrite must not
846            // intercept it. (The UPDATE encode substitutes the value before
847            // re-visiting, so the inner is no longer a field reference there.)
848            stmt::Expr::IsNull(e) => {
849                let head_column = match &*e.expr {
850                    stmt::Expr::Reference(stmt::ExprReference::Field { nesting, index })
851                        if self.cx.is_statement() =>
852                    {
853                        // The head column whose null-ness is the option's
854                        // none-ness: a struct embed's presence column, or an
855                        // enum embed's (nullable) discriminant column. A
856                        // non-nullable embed's `is_null` is already folded to
857                        // `false` in simplify, so reaching here implies nullable.
858                        self.mapping_at_unwrap(*nesting)
859                            .fields
860                            .get(*index)
861                            .and_then(|f| match f {
862                                mapping::Field::Struct(fs) => fs.presence.map(|p| p.index),
863                                mapping::Field::Enum(fe) => Some(fe.discriminant.column.index),
864                                _ => None,
865                            })
866                            .map(|column| (*nesting, column))
867                    }
868                    _ => None,
869                };
870
871                if let Some((nesting, column)) = head_column {
872                    let mut col_ref = stmt::Expr::column(stmt::ExprColumn {
873                        nesting,
874                        table: 0,
875                        column,
876                    });
877                    self.visit_expr_mut(&mut col_ref);
878                    *expr = stmt::Expr::is_null(col_ref);
879                } else {
880                    stmt::visit_mut::visit_expr_mut(self, expr);
881                }
882            }
883            stmt::Expr::Reference(expr_reference) => {
884                match expr_reference {
885                    // A reference to a relation field inside a Returning
886                    // clause becomes a subquery that loads the related
887                    // model(s).  This is the `.select(rel_field)` path; it
888                    // mirrors the include-subquery machinery that
889                    // `.include(...)` uses for `Returning::Model`.
890                    stmt::ExprReference::Field { nesting: 0, index }
891                        if matches!(self.cx, LoweringContext::Returning(_))
892                            && self.model_unwrap().fields[*index].ty.is_relation() =>
893                    {
894                        *expr = self.build_relation_subquery(*index, &[]);
895                    }
896                    stmt::ExprReference::Field { nesting, index } => {
897                        *expr = self.lower_expr_field(*nesting, *index);
898                        self.visit_expr_mut(expr);
899                    }
900                    stmt::ExprReference::Model { .. } => todo!(),
901                    stmt::ExprReference::Column(expr_column) => {
902                        if expr_column.nesting > 0 {
903                            let source_id = self.scope_stmt_id();
904                            let target_id = self.resolve_stmt_id(expr_column.nesting);
905
906                            // the current scope ID should also be the top of the stack
907                            debug_assert_eq!(self.state.scopes.len(), self.scope_id + 1);
908
909                            // The statement is not independent. Walk up the
910                            // scope stack until the referened target statement
911                            // and flag any intermediate statements as also not
912                            // indepdnendent.
913                            for scope in self.state.scopes.iter().rev() {
914                                if scope.stmt_id == target_id {
915                                    break;
916                                }
917
918                                self.state.hir[scope.stmt_id].independent = false;
919                            }
920
921                            let position = self.new_ref(source_id, target_id, *expr_reference);
922
923                            // Using ExprArg as a placeholder. It will be rewritten
924                            // later.
925                            *expr = stmt::Expr::arg(position);
926                        }
927                    }
928                }
929            }
930            stmt::Expr::Stmt(_) => {
931                let stmt::Expr::Stmt(mut expr_stmt) = expr.take() else {
932                    panic!()
933                };
934
935                // Expr::Stmt subqueries are valid in returning expressions (e.g.,
936                // INCLUDE preloading) and in VALUES bodies of batch queries.
937                debug_assert!(
938                    self.cx.is_returning() || matches!(self.cx, LoweringContext::Statement),
939                    "cx={:#?}",
940                    self.cx,
941                );
942
943                // For now, we assume nested sub-statements cannot be executed on the
944                // target database. Eventually, we will need to make this smarter.
945                let source_id = self.scope_stmt_id();
946                let target_id = self.scope_statement(|child| {
947                    visit_mut::visit_expr_stmt_mut(child, &mut expr_stmt);
948                });
949
950                // Post-lower simplify. The sub-statement detaches into its own
951                // HIR entry (`new_sub_statement`), so the parent statement's
952                // post-lowering simplify never sees it — the heavyweight rules
953                // (e.g. the schema-directed `#[document]` cast folding) must
954                // run here.
955                self.state.engine.simplify_stmt(&mut *expr_stmt.stmt);
956
957                *expr = self.new_sub_statement(source_id, target_id, expr_stmt.stmt);
958
959                if self.state.hir[target_id].independent {
960                    self.curr_stmt_info().deps.insert(target_id);
961                }
962            }
963            stmt::Expr::Exists(_) if !self.capability().sql => {
964                let stmt::Expr::Exists(mut expr_exists) = expr.take() else {
965                    panic!()
966                };
967
968                // Extract the EXISTS subquery into a sub-statement so the
969                // executor can evaluate the condition in memory.
970                let source_id = self.scope_stmt_id();
971                let target_id = self.scope_statement(|child| {
972                    child.visit_stmt_query_mut(&mut expr_exists.subquery);
973                });
974
975                let mut stmt = stmt::Statement::Query(*expr_exists.subquery);
976                // Post-lower simplify. The sub-statement detaches into its own
977                // HIR entry (`new_sub_statement`), so the parent statement's
978                // post-lowering simplify never sees it — the heavyweight rules
979                // must run here.
980                self.state.engine.simplify_stmt(&mut stmt);
981
982                let arg = self.new_sub_statement(source_id, target_id, Box::new(stmt));
983
984                if self.state.hir[target_id].independent {
985                    self.curr_stmt_info().deps.insert(target_id);
986                }
987
988                // The sub-statement result is a list of rows. Wrap it in
989                // a single-row VALUES query so the evaluator unwraps the
990                // outer list, letting EXISTS check the inner row count.
991                let mut subquery = stmt::Query::values(arg);
992                subquery.single = true;
993                *expr = stmt::Expr::Exists(stmt::ExprExists {
994                    subquery: Box::new(subquery),
995                });
996            }
997            _ => {
998                // Recurse down the statement tree
999                stmt::visit_mut::visit_expr_mut(self, expr);
1000            }
1001        }
1002    }

1036    fn visit_returning_mut(&mut self, i: &mut stmt::Returning) {
1037        if let stmt::Returning::Model { include } = i {
1038            // Start from the schema's pre-computed default returning — every
1039            // Deferred fields, top-level or nested, are already `Null`.
1040            // `process_top_level_includes` then splices loaded forms in for
1041            // the fields named by include paths (and for every deferred field
1042            // when this is an `INSERT ... RETURNING`).
1043            let mut returning = self.mapping_unwrap().default_returning.clone();
1044            let mut include_paths = std::mem::take(include);
1045            let is_insert = self.cx.is_insert();
1046
1047            self.prepare_model_returning_for_context(&mut returning, &mut include_paths, is_insert);
1048            self.process_top_level_includes(&mut returning, &include_paths, is_insert);
1049
1050            *i = stmt::Returning::Project(returning);
1051        }
1052
1053        // For multi-row INSERT returning, visit each row with its row index so
1054        // that sub-statements (e.g., child INSERTs for HasOne relations) capture
1055        // the correct parent row index via scope_statement.
1056        if matches!(&self.cx, LoweringContext::Insert(..))
1057            && let stmt::Returning::Expr(stmt::Expr::List(list)) = i
1058        {
1059            for (index, item) in list.items.iter_mut().enumerate() {
1060                self.lower_returning_for_row(index).visit_expr_mut(item);
1061            }
1062            return;
1063        }
1064
1065        stmt::visit_mut::visit_returning_mut(&mut self.lower_returning(), i);
1066    }

1070        // *except* the source field (since it is borrowed).

1072
1073        // Before lowering, handle cascading deletes
1074        lower.plan_stmt_delete_relations(stmt);

1077
1078        if let Some(expr) = &mut stmt.condition.expr {

1082        if let Some(returning) = &mut stmt.returning {

1091    fn visit_stmt_insert_mut(&mut self, stmt: &mut stmt::Insert) {
1092        // First, if an insertion scope is specified, lower the scope to be just "model"
1093        self.apply_insert_scope(&mut stmt.target, &mut stmt.source);
1094
1095        let sql = self.state.engine.capability.sql;
1096        if let Err(err) = upsert::normalize(stmt, !sql) {
1097            self.state.errors.push(err);
1098            return;
1099        }
1100
1101        // Create a new expr scope for the statement, and lower all parts
1102        // *except* the target field (since it is borrowed).
1103        let mut lower = self.lower_insert(&stmt.target);
1104
1105        if let Some(upsert) = &mut stmt.upsert {
1106            if let stmt::UpsertTarget::Fields(fields) = &upsert.target {
1107                let mapping = lower.mapping_unwrap();
1108                let mut columns = Vec::new();
1109                for field in fields {
1110                    let mapped = mapping
1111                        .resolve_field_mapping(field)
1112                        .expect("upsert target must map to database columns");
1113                    columns.extend(mapped.columns().map(|(column, _)| column));
1114                }
1115                upsert.target = stmt::UpsertTarget::Columns(columns);
1116            }
1117            if upsert.action == stmt::UpsertAction::Update {
1118                let version = lower.inject_version_increment(&mut upsert.shared);
1119                if !sql && let Some(projection) = version {
1120                    upsert.defaults.set(projection, stmt::Value::U64(0));
1121                }
1122            }
1123            lower.visit_assignments_mut(&mut upsert.shared);
1124            lower.visit_assignments_mut(&mut upsert.defaults);
1125            lower.visit_assignments_mut(&mut upsert.update_defaults);
1126        }
1127
1128        if let Some(returning) = &mut stmt.returning {
1129            lower.visit_returning_mut(returning);
1130        }
1131
1132        // Preprocess the insertion source (values usually)
1133        // `DO NOTHING RETURNING` legitimately produces zero rows. Keep its
1134        // returning clause as a projection over the database result so the
1135        // normal single-statement boundary can turn an empty list into
1136        // `Value::Null` (and therefore `Option::None`). Converting it to an
1137        // expression would project row zero eagerly and panic on conflict.
1138        let preserve_returning_projection = stmt
1139            .upsert
1140            .as_ref()
1141            .is_some_and(|upsert| upsert.action == stmt::UpsertAction::Ignore);
1142        lower.preprocess_insert_values(
1143            &mut stmt.source,
1144            &mut stmt.returning,
1145            preserve_returning_projection,
1146        );
1147
1148        // Lower the insertion source
1149        lower.visit_stmt_query_mut(&mut stmt.source);
1150
1151        if let Some(returning) = &mut stmt.returning {
1152            lower.visit_returning_mut(returning);
1153            if stmt.upsert.is_none() {
1154                lower.constantize_insert_returning(returning, &stmt.source);
1155            }
1156
1157            if stmt.source.single
1158                && let stmt::Returning::Expr(expr) = &returning
1159            {
1160                // Not strictly true, but there is nothing that needs to
1161                // return a list at this point for a "single" query. If this
1162                // is ever needed, remove the assertion.
1163                debug_assert!(!expr.is_list());
1164            }
1165        }
1166
1167        self.visit_insert_target_mut(&mut stmt.target);
1168    }

1200    fn visit_stmt_update_mut(&mut self, stmt: &mut stmt::Update) {
1201        let mut lower = self.scope_expr(&stmt.target);
1202
1203        let mut returning_changed = false;
1204
1205        // Before lowering children, convert the "Changed" returning statement
1206        // to an expression referencing changed fields.
1207        if let Some(returning) = &mut stmt.returning
1208            && returning.is_changed()
1209        {
1210            returning_changed = true;
1211
1212            if let Some(model) = lower.model() {
1213                let mapping = lower.mapping_unwrap();
1214
1215                // Step 1 — build a mask of all primitives being changed by
1216                // OR-ing each assigned field's coverage mask together.
1217                let mut changed_bits = stmt::PathFieldSet::new();
1218                for projection in stmt.assignments.keys() {
1219                    if let Some(mf) = mapping.resolve_field_mapping(projection) {
1220                        changed_bits |= mf.field_mask();
1221                    }
1222                }
1223
1224                // Step 2 — build the returning expression.
1225                *returning = stmt::Returning::Project(build_update_returning(
1226                    model.id,
1227                    None,
1228                    &mapping.fields,
1229                    &changed_bits,
1230                ));
1231            }
1232        }
1233
1234        // Plan relations
1235        lower.plan_stmt_update_relations(
1236            &mut stmt.assignments,
1237            &stmt.filter,
1238            &mut stmt.returning,
1239            returning_changed,
1240        );
1241
1242        // A versionable model bumps its OCC counter on every update. The
1243        // instance path sets the counter explicitly in generated code,
1244        // alongside the `version == <last read>` condition — only it knows the
1245        // value the caller last loaded. A query-based update leaves the counter
1246        // unset, so the engine injects the relative `version = version + 1`
1247        // here. Upsert conflict updates use the same injection helper from the
1248        // insert lowering path.
1249        //
1250        // Injecting after the `Changed` returning projection is built above
1251        // keeps the engine-managed counter out of the returning clause. A
1252        // relative `+1` is not constantizable; leaving it in the returning
1253        // would force a per-row read-back that the multi-row DynamoDB transact
1254        // path cannot serve. The counter still advances in the database — a
1255        // query update just doesn't read it back (it discards its result).
1256        let _ = lower.inject_version_increment(&mut stmt.assignments);
1257
1258        lower.visit_assignments_mut(&mut stmt.assignments);
1259        lower.visit_filter_mut(&mut stmt.filter);
1260
1261        if let Some(expr) = &mut stmt.condition.expr {
1262            lower.visit_expr_mut(expr);
1263        }
1264
1265        if let Some(returning) = &mut stmt.returning {
1266            lower.visit_returning_mut(returning);
1267            // Use the lowered assignments (which are now column-indexed)
1268            returning::constantize_update_returning(lower.expr_cx, returning, &stmt.assignments);
1269        }
1270
1271        self.visit_update_target_mut(&mut stmt.target);
1272    }
1273
1274    fn visit_source_mut(&mut self, stmt: &mut stmt::Source) {
1275        if let stmt::Source::Model(source_model) = stmt {
1276            debug_assert!(source_model.via.is_none(), "TODO");
1277
1278            let table_id = self.schema().table_id_for(source_model.id);
1279            *stmt = stmt::Source::table(table_id);
1280        }
1281    }
1282
1283    fn visit_values_mut(&mut self, stmt: &mut stmt::Values) {
1284        if self.cx.is_insert()
1285            && let Some(mapping) = self.mapping()
1286        {
1287            for row in &mut stmt.rows {
1288                let mut lowered = mapping.model_to_table.clone();
1289                self.lower_insert_row(row)
1290                    .visit_expr_record_mut(&mut lowered);
1291
1292                *row = lowered.into();
1293            }
1294
1295            return;
1296        }
1297
1298        visit_mut::visit_values_mut(self, stmt);
1299    }

1303    /// App-level operand rewrite for `eq` and `ne` binary ops.
1304    ///
1305    /// `Reference::Model { nesting }` becomes a reference to the model's
1306    /// primary-key field; a `BelongsTo` field reference becomes a reference
1307    /// to the relation's foreign-key field.  Both rewrites only match on
1308    /// app-level shapes; after the surrounding lowering walk converts the
1309    /// references to columns, this method has nothing to rewrite.
1310    ///
1311    /// Must fire before the operand's children are visited, since lowering
1312    /// otherwise replaces the field reference with a column reference and
1313    /// the rewrite has no app-level shape to match on.
1314    fn rewrite_eq_operand(&self, operand: &mut stmt::Expr) {
1315        if let stmt::Expr::Reference(expr_reference) = operand {
1316            match &*expr_reference {
1317                stmt::ExprReference::Model { nesting } => {
1318                    let nesting = *nesting;
1319                    let model = self
1320                        .expr_cx
1321                        .resolve_expr_reference(expr_reference)
1322                        .as_model_unwrap();
1323
1324                    *operand = key_field_refs(nesting, model.primary_key.fields.iter().copied());
1325                }
1326                stmt::ExprReference::Field { nesting, .. } => {
1327                    let nesting = *nesting;
1328                    let field = self
1329                        .expr_cx
1330                        .resolve_expr_reference(expr_reference)
1331                        .as_field_unwrap();
1332
1333                    match &field.ty {
1334                        app::FieldTy::Primitive(_) | app::FieldTy::Embedded(_) => {}
1335                        app::FieldTy::Has(_) | app::FieldTy::Via(_) => todo!(),
1336                        app::FieldTy::BelongsTo(rel) => {
1337                            *operand = key_field_refs(
1338                                nesting,
1339                                rel.foreign_key.fields.iter().map(|fk| fk.source),
1340                            );
1341                        }
1342                    }
1343                }
1344                _ => {}
1345            }
1346        }
1347    }

1350    /// `Reference::Model { nesting } IN list` becomes `<pk_field> IN list`.

1356        let (nesting, pk_field_id) = {
1357            let stmt::Expr::Reference(expr_ref @ stmt::ExprReference::Model { nesting }) =
1358                &*expr.expr
1359            else {
1360                return;

1364                .expr_cx
1365                .resolve_expr_reference(expr_ref)
1366                .as_model_unwrap();
1367            let [pk_field_id] = &model.primary_key.fields[..] else {
1368                todo!()
1369            };
1370            (nesting, *pk_field_id)
1371        };
1372
1373        let schema = self.expr_cx.schema();
1374        let pk = schema.app.field(pk_field_id);
1375
1376        // Sanity-check the RHS shape against the PK type.
1377        match &mut *expr.list {
1378            stmt::Expr::List(expr_list) => {
1379                for item in &mut expr_list.items {
1380                    match item {
1381                        stmt::Expr::Value(value) => {
1382                            assert!(value.is_a(&schema.app, &pk.ty.as_primitive_unwrap().ty));
1383                        }
1384                        _ => todo!("{item:#?}"),
1385                    }
1386                }
1387            }
1388            stmt::Expr::Value(stmt::Value::List(values)) => {
1389                for value in values {
1390                    assert!(value.is_a(&schema.app, &pk.ty.as_primitive_unwrap().ty));
1391                }
1392            }
1393            _ => todo!("expr={expr:#?}"),
1394        }

1399    fn lower_expr_binary_op(
1400        &mut self,
1401        op: stmt::BinaryOp,
1402        lhs: &mut stmt::Expr,
1403        rhs: &mut stmt::Expr,
1404    ) -> Option<stmt::Expr> {
1405        match (&mut *lhs, &mut *rhs) {
1406            (stmt::Expr::Value(value), other) | (other, stmt::Expr::Value(value))
1407                if value.is_null() =>
1408            {
1409                let other = other.take();
1410
1411                Some(match op {
1412                    stmt::BinaryOp::Eq => stmt::Expr::is_null(other),
1413                    stmt::BinaryOp::Ne => stmt::Expr::is_not_null(other),
1414                    _ => todo!(),
1415                })
1416            }
1417            // Record-vs-record decomposition for eq/ne. Embedded fields lower
1418            // to record expressions, so a comparison like
1419            // `Record([cast(col, T)]) == Record([val])` only exposes the
1420            // cast (and the cast-stripping rule below) once the record is
1421            // split per-element. Recurse into each pair so cast handling
1422            // fires.
1423            //
1424            // Mirrors the two record shapes handled in `simplify_expr_binary_op`:
1425            // both sides `Expr::Record`, or one side folded to
1426            // `Expr::Value(Value::Record)`.
1427            (stmt::Expr::Record(lhs_rec), stmt::Expr::Record(rhs_rec))
1428                if (op.is_eq() || op.is_ne()) && lhs_rec.len() == rhs_rec.len() =>
1429            {
1430                Some(self.combine_record_op(
1431                    op,
1432                    std::mem::take(&mut lhs_rec.fields),
1433                    std::mem::take(&mut rhs_rec.fields),
1434                ))
1435            }
1436            (stmt::Expr::Record(rec), stmt::Expr::Value(stmt::Value::Record(val_rec)))
1437            | (stmt::Expr::Value(stmt::Value::Record(val_rec)), stmt::Expr::Record(rec))
1438                if (op.is_eq() || op.is_ne()) && rec.len() == val_rec.len() =>
1439            {
1440                let val_exprs = std::mem::take(&mut val_rec.fields)
1441                    .into_iter()
1442                    .map(stmt::Expr::Value)
1443                    .collect();
1444                Some(self.combine_record_op(op, std::mem::take(&mut rec.fields), val_exprs))
1445            }
1446            (stmt::Expr::Cast(expr_cast), _) | (_, stmt::Expr::Cast(expr_cast)) => {
1447                let target_ty = self.capability().native_type_for(&expr_cast.ty);
1448                self.cast_expr(lhs, &target_ty);
1449                self.cast_expr(rhs, &target_ty);
1450                None
1451            }
1452            _ => None,
1453        }
1454    }

1456    /// Combines per-element comparisons of two record-shaped operands into a
1457    /// single boolean expression: AND for `eq`, OR for `ne`. Each pair is
1458    /// recursed through `lower_expr_binary_op` so any inner cast handling
1459    /// fires as if the comparison had been written elementwise.
1460    fn combine_record_op(
1461        &mut self,
1462        op: stmt::BinaryOp,
1463        lhs_fields: Vec<stmt::Expr>,
1464        rhs_fields: Vec<stmt::Expr>,
1465    ) -> stmt::Expr {
1466        let comparisons: Vec<_> = lhs_fields
1467            .into_iter()
1468            .zip(rhs_fields)
1469            .map(|(mut l, mut r)| {
1470                self.lower_expr_binary_op(op, &mut l, &mut r)
1471                    .unwrap_or_else(|| stmt::Expr::binary_op(l, op, r))
1472            })
1473            .collect();
1474
1475        if op.is_eq() {
1476            stmt::Expr::and_from_vec(comparisons)
1477        } else {
1478            stmt::Expr::or_from_vec(comparisons)
1479        }
1480    }

1620    /// Returns the ArgId for the new reference
1621    fn new_ref(
1622        &mut self,
1623        source_id: hir::StmtId,
1624        target_id: hir::StmtId,
1625        mut expr_reference: stmt::ExprReference,
1626    ) -> usize {
1627        let stmt::ExprReference::Column(expr_column) = &mut expr_reference else {
1628            todo!()
1629        };
1630
1631        // First, get the nesting so we can resolve the target statmeent
1632        let nesting = expr_column.nesting;
1633
1634        // We only track references that point to statements being executed by
1635        // separate operations. References within the same operation are handled
1636        // by the target database.
1637        debug_assert!(nesting != 0, "expr_reference={expr_reference:#?}");
1638
1639        // Set the nesting to zero as the stored ExprReference will be used from
1640        // the context of the *target* statement.
1641        expr_column.nesting = 0;
1642
1643        let target = &mut self.state.hir[target_id];
1644
1645        // The `batch_load_index` is the index for this reference in the row
1646        // returned from the target statement's ExecStatement operation. This
1647        // ExecStatement operation batch loads all records needed to execute
1648        // the full root statement.
1649        target
1650            .back_refs
1651            .entry(source_id)
1652            .or_default()
1653            .exprs
1654            .insert_full(expr_reference);
1655
1656        // Create an argument for inputing the expr reference's value into the statement.
1657        let source = &mut self.state.hir[source_id];
1658
1659        // See if an arg already exists
1660        for (i, arg) in source.args.iter().enumerate() {
1661            let hir::Arg::Ref {
1662                target_expr_ref, ..
1663            } = arg
1664            else {
1665                continue;
1666            };
1667
1668            if *target_expr_ref == expr_reference {
1669                return i;
1670            }
1671        }
1672
1673        let arg = source.args.len();
1674
1675        source.args.push(hir::Arg::Ref {
1676            target_expr_ref: expr_reference,
1677            stmt_id: target_id,
1678            nesting,
1679            data_load_input: Cell::new(None),
1680            returning_input: Cell::new(None),
1681            batch_load_index: if let Some(row_index) = self.state.scopes[self.scope_id].row_index {
1682                debug_assert_eq!(1, nesting, "TODO");
1683                Cell::new(Some(row_index))
1684            } else {
1685                Cell::new(None)
1686            },
1687        });

1689        arg
1690    }

1724    /// Run the canonical pipeline (pre-lower simplify, lowering walk, post-lower
1725    /// simplify) on a synthesized sub-statement and stitch it onto the parent
1726    /// as an `Expr::Arg`.
1727    ///
1728    /// Used at sites where lowering itself synthesizes a new statement to embed
1729    /// in the parent's `Returning` (include subqueries, child inserts for
1730    /// relation planning).  Equivalent to a recursive `lower_stmt` call that
1731    /// passes through the `Expr::Stmt` arm in `visit_expr_mut`, but expressed
1732    /// directly so the synthesized statement does not have to round-trip
1733    /// through an `Expr::Stmt` placeholder.
1734    fn lower_sub_stmt(&mut self, stmt: stmt::Statement) -> stmt::Expr {
1735        let source_id = self.scope_stmt_id();
1736        let mut stmt = Box::new(stmt);
1737
1738        let target_id = self.scope_statement(|child| {
1739            // Via-association rewrite: `Source::Model { via }` becomes an
1740            // explicit WHERE filter so the lowering walk only sees rewritten
1741            // sources.  The IN-subquery lift fires next so non-walk code
1742            // paths see the lifted form.  The eq/ne operand rewrite
1743            // (model→PK, BelongsTo→FK) fires inside the lowering walk via
1744            // `LowerStatement::visit_expr_binary_op_mut`.
1745            association::RewriteVia::new(child.expr_cx).rewrite(&mut stmt);
1746            lift_in_subquery::LiftInSubquery::new(child.expr_cx).rewrite(&mut stmt);
1747            // Pre-lower simplify: remaining heavyweight rules the lowering
1748            // visitor expects to have already fired.
1749            Simplify::with_context(child.expr_cx, child.state.engine.capability)
1750                .visit_mut(&mut *stmt);
1751            // Lowering walk.
1752            child.visit_stmt_mut(&mut stmt);
1753            // Post-lower simplify: heavyweight rules on the lowered tree.
1754            child.state.engine.simplify_stmt(&mut *stmt);
1755        });
1756
1757        // Sub-statements built via this helper always live in the parent's
1758        // Returning clause (include subqueries, child inserts for relation
1759        // planning).  Force the new `Arg::Sub`'s `returning` flag accordingly:
1760        // `plan_nested_merge` keys off `returning: true` to discover include
1761        // subqueries, and the line-399 `Expr::Stmt` arm gets the same flag
1762        // because it fires during the parent's Returning walk.
1763        let saved_cx = std::mem::replace(&mut self.cx, LoweringContext::Returning(None));
1764        let arg = self.new_sub_statement(source_id, target_id, stmt);
1765        self.cx = saved_cx;
1766
1767        if self.state.hir[target_id].independent {
1768            self.curr_stmt_info().deps.insert(target_id);
1769        }
1770
1771        arg
1772    }

1995/// (primary or foreign): a single `ref_field` for a single-column key, a
1996/// `Record` of `ref_field`s for a composite key. Used both for the

2006        stmt::Expr::ref_field(nesting, fields.next().unwrap())

2012/// Input implementation for assignment substitution.
2013///
2014/// Provides assignment values when substituting field references in `model_to_table`
2015/// expressions. Handles projections automatically for embedded fields.
2016struct AssignmentInput<'a> {
2017    assignment_projection: stmt::Projection,
2018    value: &'a stmt::Expr,
2019}
2020
2021impl stmt::Input for AssignmentInput<'_> {
2022    fn resolve_ref(
2023        &mut self,
2024        expr_reference: &stmt::ExprReference,
2025        expr_projection: &stmt::Projection,
2026    ) -> Option<stmt::Expr> {
2027        let stmt::ExprReference::Field { nesting: 0, index } = expr_reference else {
2028            return None;
2029        };
2030
2031        let assignment_steps = self.assignment_projection.as_slice();
2032
2033        if *index != assignment_steps[0] {
2034            return None;
2035        }
2036
2037        let remaining_steps = &assignment_steps[1..];
2038
2039        if expr_projection.as_slice() == remaining_steps {
2040            Some(self.value.clone())
2041        } else {
2042            // The column's encode template projects into variant-field
2043            // positions of the assignment value. When a mixed enum is updated
2044            // to a unit (or otherwise shorter) variant, the value record is
2045            // narrower than a sibling variant's column expects, so the
2046            // projection falls out of bounds and `entry` returns `None`. Those
2047            // references live inside a discriminant-guarded match arm that is
2048            // unreachable for this value, so resolving them to `null` is safe —
2049            // the simplifier drops the dead arm.
2050            Some(
2051                self.value
2052                    .entry(expr_projection)
2053                    .map(|e| e.to_expr())
2054                    .unwrap_or_else(stmt::Expr::null),
2055            )
2056        }
2057    }
2058}

2060/// Builds the returning expression for a `Returning::Changed` update.
2061///
2062/// Iterates `mapping_fields` using `changed_bits` to determine what to include:
2063///
2064/// - Relation fields → null placeholder (populated during relation planning)
2065/// - Field whose `field_mask` is fully covered by `changed_bits` → emit
2066///   `project(ref_self_field, sub_projection)` (constantized later; project
2067///   omitted when `sub_projection` is identity)
2068/// - Field only partially covered → recurse to build a nested `SparseRecord`
2069///
2070/// `root_field_id` controls how the self-field reference is constructed:
2071/// - `None` at the top level: derived per-iteration as `{ model_id, i }`, since
2072///   each model field is its own root.
2073/// - `Some(id)` when recursing into an embedded type: fixed throughout, because
2074///   all sub-field expressions project from the same top-level ancestor field.
2075fn build_update_returning(
2076    model_id: app::ModelId,
2077    root_field_id: Option<app::FieldId>,
2078    mapping_fields: &[mapping::Field],
2079    changed_bits: &stmt::PathFieldSet,
2080) -> stmt::Expr {
2081    let mut exprs = vec![];
2082    let mut field_set = stmt::PathFieldSet::new();
2083
2084    for (i, mf) in mapping_fields.iter().enumerate() {
2085        let intersection = changed_bits.clone() & mf.field_mask();
2086
2087        if intersection.is_empty() {
2088            continue;
2089        }
2090
2091        field_set.insert(i);
2092
2093        if mf.is_relation() {
2094            // Relation field: null placeholder for the relation planner to fill
2095            // in via set_returning_field.
2096            exprs.push(stmt::Expr::null());
2097        } else {
2098            let root_field_id = root_field_id.unwrap_or(app::FieldId {
2099                model: model_id,
2100                index: i,
2101            });
2102
2103            if intersection == mf.field_mask() {
2104                // Full coverage: all primitives in this field are being updated.
2105                // Emit a projected field reference; the lowering + constantize
2106                // pipeline will substitute the assignment value.
2107                let base = stmt::Expr::ref_self_field(root_field_id);
2108                let expr = if mf.sub_projection().is_identity() {
2109                    base
2110                } else {
2111                    stmt::Expr::project(base, mf.sub_projection().clone())
2112                };
2113                exprs.push(expr);
2114            } else {
2115                // Partial embedded update: only some sub-fields are changing.
2116                // Recurse to build a nested SparseRecord. The lowering pipeline
2117                // resolves each reference to the correct column, and the simplifier
2118                // folds project(record([...]), [i]) → column_ref.
2119                let emb_mapping = mf.as_struct().unwrap();
2120                exprs.push(build_update_returning(
2121                    model_id,
2122                    Some(root_field_id),
2123                    &emb_mapping.fields,
2124                    &intersection,
2125                ));
2126            }
2127        }
2128    }
2129
2130    stmt::Expr::cast(
2131        stmt::ExprRecord::from_vec(exprs),
2132        stmt::Type::SparseRecord(field_set),
2133    )
2134}

2136/// True when an `IN` list is a candidate for the `= ANY($1)` rewrite:
2137///
2138/// - The lhs is scalar (not a `Record`). Composite-key `IN` would need a PG
2139///   row-array bind which the current driver doesn't support.
2140/// - The list is a constant collection of scalar values. Lists containing
2141///   references, sub-statements, or record values stay as expanded `IN`.
2142fn in_list_is_value_list(e: &stmt::ExprInList) -> bool {
2143    if matches!(*e.expr, stmt::Expr::Record(_)) {
2144        return false;
2145    }
2146    let scalar = |v: &stmt::Value| !matches!(v, stmt::Value::Record(_) | stmt::Value::List(_));
2147    match &*e.list {
2148        stmt::Expr::Value(stmt::Value::List(items)) => items.iter().all(scalar),
2149        stmt::Expr::List(list) => list
2150            .items
2151            .iter()
2152            .all(|i| matches!(i, stmt::Expr::Value(v) if scalar(v))),
2153        _ => false,
2154    }

24/// Statement and expression simplifier.
25///
26/// [`Simplify`] implements the [`VisitMut`] trait to traverse and transform
27/// statement ASTs. It applies optimization and normalization rules defined in
28/// submodules of [`engine::simplify`](self).
29///
30/// Simplification runs twice during query compilation: once before lowering
31/// (to normalize the input) and once after (to clean up generated expressions).
32pub(crate) struct Simplify<'a> {
33    /// Expression context providing schema access and type information.
34    cx: stmt::ExprContext<'a>,
35    /// Driver capabilities, consulted by passes that emit driver-specific shapes.
36    capability: &'a Capability,
37}
38
39impl Engine {
40    /// Simplifies a statement or expression in place.
41    pub(crate) fn simplify_stmt<T: Node>(&self, stmt: &mut T) {
42        Simplify::new(&self.schema, self.capability).visit_mut(stmt);
43    }
44}
45
46/// Simplifies an expression in place using the given context and capability.
47pub(crate) fn simplify_expr(
48    cx: stmt::ExprContext<'_>,
49    capability: &Capability,
50    expr: &mut stmt::Expr,
51) {
52    Simplify { cx, capability }.visit_expr_mut(expr);
53}
54
55impl VisitMut for Simplify<'_> {
56    fn visit_expr_mut(&mut self, i: &mut stmt::Expr) {
57        // Recurse into children first.
58        stmt::visit_mut::visit_expr_mut(self, i);
59
60        // Fold this node bottom-up so heavyweight rules see canonical input.
61        // Children are already canonical (post-order recursion), so this is
62        // effectively local.
63        fold::fold_stmt(i);
64
65        let maybe_expr = match i {
66            Expr::Any(expr) => self.simplify_expr_any(expr),
67            Expr::And(expr) => self.simplify_expr_and(expr),
68            Expr::BinaryOp(expr) => {
69                self.simplify_expr_binary_op(expr.op, &mut expr.lhs, &mut expr.rhs)
70            }
71            Expr::Cast(expr) => self.simplify_expr_cast(expr),
72            Expr::Exists(expr) => self.simplify_expr_exists(expr),
73            Expr::Intersects(expr) => self.simplify_expr_intersects(expr),
74            Expr::IsSuperset(expr) => self.simplify_expr_is_superset(expr),
75            Expr::Let(expr) => self.simplify_expr_let(expr),
76            Expr::List(expr) => self.simplify_expr_list(expr),
77            Expr::Map(_) => self.simplify_expr_map(i),
78            Expr::Or(expr) => self.simplify_expr_or(expr),
79            Expr::IsNull(expr) => self.simplify_expr_is_null(expr),
80            Expr::Project(expr) => self.simplify_expr_project(expr),
81            _ => None,
82        };
83
84        if let Some(mut expr) = maybe_expr {
85            // Heavyweight rules may emit new fold-eligible structure
86            // (e.g., match elimination produces ANDs containing constants
87            // that need short-circuiting).
88            fold::fold_stmt(&mut expr);
89            *i = expr;
90        }
91    }
92
93    fn visit_expr_match_mut(&mut self, i: &mut stmt::ExprMatch) {
94        // Simplify the subject first.
95        self.visit_expr_mut(&mut i.subject);
96
97        // If the subject simplified to a constant, only simplify the matching arm.
98        // Skipping dead-code arms avoids panics on expressions like
99        // `project([1], Record([I64(disc)]))` that would be invalid to evaluate.
100        if let Expr::Value(ref value) = *i.subject {
101            let value = value.clone();
102            for arm in &mut i.arms {
103                if arm.pattern == value {
104                    self.visit_expr_mut(&mut arm.expr);
105                    return;
106                }
107            }
108        } else {
109            for arm in &mut i.arms {
110                self.visit_expr_mut(&mut arm.expr);
111            }
112        }
113    }
114
115    fn visit_expr_set_mut(&mut self, i: &mut stmt::ExprSet) {
116        match i {
117            stmt::ExprSet::SetOp(expr_set_op) if expr_set_op.operands.is_empty() => {
118                todo!("is there anything we do here?");
119            }
120            stmt::ExprSet::SetOp(expr_set_op) if expr_set_op.operands.len() == 1 => {
121                let operand = expr_set_op.operands.drain(..).next().unwrap();
122                *i = operand;
123            }
124            stmt::ExprSet::SetOp(expr_set_op) if expr_set_op.is_union() => {
125                // First, simplify each sub-query in the union, then rewrite the
126                // query as a single disjuntive query.
127                let mut operands = vec![];
128
129                Self::flatten_nested_unions(expr_set_op, &mut operands);
130
131                expr_set_op.operands = operands;
132            }
133            _ => {}
134        }
135
136        stmt::visit_mut::visit_expr_set_mut(self, i);
137    }
138
139    fn visit_stmt_delete_mut(&mut self, stmt: &mut stmt::Delete) {
140        // Visit and simplify source first before pushing a new scope
141        self.visit_source_mut(&mut stmt.from);
142
143        let mut s = self.scope(&stmt.from);
144
145        s.visit_filter_mut(&mut stmt.filter);
146
147        if let Some(returning) = &mut stmt.returning {
148            s.visit_returning_mut(returning);
149        }
150    }
151
152    fn visit_stmt_insert_mut(&mut self, stmt: &mut stmt::Insert) {
153        // Visit target first before pushing a new scope.
154        self.visit_insert_target_mut(&mut stmt.target);
155
156        // Create a new scope for the insert target
157        let mut s = self.scope(&stmt.target);
158
159        // First, simplify the source
160        s.visit_stmt_query_mut(&mut stmt.source);
161
162        if let Some(returning) = &mut stmt.returning {
163            s.visit_returning_mut(returning);
164        }
165    }
166
167    fn visit_stmt_query_mut(&mut self, stmt: &mut stmt::Query) {
168        stmt::visit_mut::visit_stmt_query_mut(self, stmt);
169
170        self.simplify_stmt_query_when_empty(stmt);
171    }
172
173    fn visit_stmt_select_mut(&mut self, stmt: &mut stmt::Select) {
174        if let stmt::Source::Model(model) = &mut stmt.source
175            && let Some(via) = model.via.take()
176        {
177            todo!("via={via:#?}");
178        }
179
180        // Simplify the source first
181        self.visit_source_mut(&mut stmt.source);
182
183        // Create a new scope for the insert target
184        let mut s = self.scope(&stmt.source);
185
186        s.visit_filter_mut(&mut stmt.filter);
187        s.visit_returning_mut(&mut stmt.returning);
188    }
189
190    fn visit_stmt_update_mut(&mut self, stmt: &mut stmt::Update) {
191        // If the update target is a query, start by simplifying the query, then
192        // rewriting it to be a filter.
193        if let stmt::UpdateTarget::Query(query) = &mut stmt.target {
194            self.visit_stmt_query_mut(query);
195
196            let stmt::ExprSet::Select(select) = &mut query.body else {
197                todo!()
198            };
199
200            assert!(select.returning.is_model());
201
202            stmt.filter.add_filter(select.filter.take());
203
204            stmt.target = stmt::UpdateTarget::Model(select.source.model_id_unwrap());
205        }
206
207        self.visit_update_target_mut(&mut stmt.target);
208
209        let mut s = self.scope(&stmt.target);
210        s.visit_assignments_mut(&mut stmt.assignments);
211
212        s.visit_filter_mut(&mut stmt.filter);
213
214        if let Some(expr) = &mut stmt.condition.expr {
215            s.visit_expr_mut(expr);
216        }
217
218        if let Some(returning) = &mut stmt.returning {
219            s.visit_returning_mut(returning);
220        }
221    }
222}

224impl<'a> Simplify<'a> {
225    pub(crate) fn new(schema: &'a Schema, capability: &'a Capability) -> Self {
226        Simplify::with_context(stmt::ExprContext::new(schema), capability)
227    }
228
229    pub(crate) fn with_context(cx: stmt::ExprContext<'a>, capability: &'a Capability) -> Self {
230        Simplify { cx, capability }
231    }
232
233    /// Return a new `Simplify` instance that operates on a nested scope
234    /// targeting the provided relation.
235    pub(crate) fn scope<'scope>(
236        &'scope self,
237        target: impl IntoExprTarget<'scope>,
238    ) -> Simplify<'scope> {
239        Simplify {
240            cx: self.cx.scope(target),
241            capability: self.capability,
242        }
243    }
244
245    /// Returns the source model
246    fn flatten_nested_unions(expr_set_op: &mut stmt::ExprSetOp, operands: &mut Vec<stmt::ExprSet>) {
247        assert!(expr_set_op.is_union());
248
249        for expr_set in &mut expr_set_op.operands {
250            match expr_set {
251                stmt::ExprSet::SetOp(nested_set_op) if nested_set_op.is_union() => {
252                    Self::flatten_nested_unions(nested_set_op, operands)
253                }
254                // Just drop empty values
255                stmt::ExprSet::Values(values) if values.is_empty() => {}
256                stmt::ExprSet::Select(select) => {
257                    if let Some(stmt::ExprSet::Select(tail)) = operands.last_mut() {
258                        todo!("merge select={:#?} tail={:#?}", select, tail);
259                    }
260
261                    operands.push(std::mem::take(expr_set));
262                }
263                stmt::ExprSet::Values(values) => {
264                    if let Some(stmt::ExprSet::Values(tail)) = operands.last_mut() {
265                        tail.rows.append(&mut values.rows);
266                        continue;
267                    }
268
269                    operands.push(std::mem::take(expr_set));
270                }
271                _ => todo!("expr={:#?}", expr_set),
272            }
273        }
274    }

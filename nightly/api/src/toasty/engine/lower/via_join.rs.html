1//! Lowering for `.include()` / `.select()` of multi-step (`via`) relations.
2//!
3//! A `via` relation reaches its target through a path of existing relations.
4//! [`ViaJoin`] resolves that path into a single JOIN from the target back to
5//! the root so the engine can issue one query per include and group children
6//! with their parent in `NestedMerge`. This relies on the database executing
7//! the join, so it is SQL-only — a key-value backend would need a cascade of
8//! per-step queries instead.
9
10use toasty_core::{
11    schema::{app, mapping},
12    stmt,
13};
14
15use crate::engine::lower::LowerStatement;
16
17impl LowerStatement<'_, '_> {
18    /// Build the include subquery for a multi-step (`via`) relation.
19    ///
20    /// The child query is rooted at the via target and joins back to the root
21    /// through every intermediate model (see [`ViaJoin`]), projecting the
22    /// linking foreign-key column so `NestedMerge` can group children by
23    /// parent. It is emitted in fully-lowered form, so the standard lowering
24    /// walk only has to rewrite the cross-statement parent-key reference into
25    /// an `Arg::Ref`. Each child row is `[link_key, target_record]`; the
26    /// trailing projection drops the link key so the parent sees only the
27    /// targets (a `Map` over the list for `has_many`, a direct project for a
28    /// single `has_one`).
29    pub(super) fn build_via_include_subquery(
30        &mut self,
31        field_index: usize,
32        via: &app::Via,
33        nested: &[stmt::Projection],
34    ) -> stmt::Expr {
35        if !nested.is_empty() {
36            todo!("nested `.include()` through a multi-step `via` relation");
37        }
38
39        let schema = self.schema();
40        let model = self.model_unwrap();
41        let single = match &model.fields[field_index].ty {
42            app::FieldTy::Via(via) => via.is_one(),
43            _ => unreachable!("build_via_include_subquery called on non-via field"),
44        };
45        let nullable = model.fields[field_index].nullable();
46        let join = ViaJoin::resolve(schema, model.id, via);
47
48        // WHERE: the linking column (on the root-adjacent model) equals the
49        // parent's referenced key. Use the field's model-level expression
50        // (column ref + any storage→model cast) so it lines up type-wise with
51        // the parent's lowered key both here and in the `NestedMerge` group
52        // key.
53        let (link_field, parent_key_field) = join.link();
54        let link_col = model_level_column_expr(schema, link_field, join.slot(1));
55        let filter = stmt::Expr::eq(link_col.clone(), stmt::Expr::ref_field(1, parent_key_field));
56
57        // RETURNING `[link_col, target_record]`. The link column lands in
58        // `load_data_select_items` so the qualification resolves to a
59        // `SortLookup`; `target_record` is the schema's `default_returning`,
60        // whose column refs already point at table slot 0 (the target).
61        let target_record = schema.mapping_for(join.target()).default_returning.clone();
62        let returning = stmt::Expr::record_from_vec(vec![link_col, target_record]);
63
64        // `DISTINCT` collapses duplicate targets produced when the path fans
65        // out (e.g. two comments on the same article) — matching a direct via
66        // query's distinct-target semantics.
67        let mut select = stmt::Select::new(join.build_source(schema), filter);
68        select.returning = stmt::Returning::Project(returning);
69        select.distinct = true;
70
71        let mut query = stmt::Query::builder(select).build();
72        query.single = single;
73
74        // The query is already lowered, so this is mostly a no-op beyond
75        // rewriting the parent-key `Reference::Field { nesting: 1 }` into an
76        // `Arg::Ref`.
77        let sub_expr = self.lower_sub_stmt(stmt::Statement::Query(query));
78
79        // Drop the link key from each `[link_key, target_record]` row; the
80        // parent wants only the target.
81        if !single {
82            // A `has_many` via yields a list, so map over it (`arg(0)` is the
83            // item) and project the target out of each row.
84            return stmt::Expr::map(sub_expr, stmt::Expr::project(stmt::Expr::arg(0), [1usize]));
85        }
86
87        // A single (`has_one`) via yields one `[link_key, target_record]`
88        // record; project the target out. A nullable single relation, though,
89        // produces `Null` when the `INNER JOIN` matched nothing, and projecting
90        // into `Null` would panic — so strip the link key only on the non-null
91        // branch.
92        if nullable {
93            super::map_nullable_single(sub_expr, stmt::Expr::project(stmt::Expr::arg(0), [1usize]))
94        } else {
95            stmt::Expr::project(sub_expr, [1usize])
96        }
97    }
98}

100/// A multi-step (`via`) relation resolved into a JOIN from the target back to
101/// the root.
102///
103/// `models` is the path `[root, …intermediates, target]`; `edges[i]` is the
104/// foreign key joining `models[i]` and `models[i + 1]`.
105///
106/// The child query lays its tables out target-first so the target's pre-built
107/// `default_returning` (whose column refs point at slot 0) is reused verbatim:
108///
109/// ```text
110///   slot 0       FROM   target
111///   slot 1..     JOIN   intermediates, target-adjacent first
112/// ```
113///
114/// The root is not a table — it is the parent query, reached through the WHERE
115/// filter on `edges[0]`.
116struct ViaJoin {
117    models: Vec<app::ModelId>,
118    edges: Vec<Edge>,
119}
120
121impl ViaJoin {
122    fn resolve(schema: &toasty_core::Schema, root: app::ModelId, via: &app::Via) -> ViaJoin {
123        let steps = flatten_via_steps(schema, root, via.path.projection.as_slice());
124        assert!(
125            !steps.is_empty(),
126            "via path must have at least one step (validated at schema build time)"
127        );
128
129        let mut models = Vec::with_capacity(steps.len() + 1);
130        let mut edges = Vec::with_capacity(steps.len());
131        models.push(root);
132        for &field_id in &steps {
133            debug_assert_eq!(field_id.model, *models.last().unwrap());
134            models.push(schema.app.field(field_id).relation_target_id().unwrap());
135            edges.push(Edge::resolve(schema, field_id));
136        }
137
138        ViaJoin { models, edges }
139    }
140
141    /// The via target — the model whose rows the include loads.
142    fn target(&self) -> app::ModelId {
143        *self.models.last().unwrap()
144    }
145
146    /// Table slot for the model at chain position `pos` (`1..=edges.len()`).
147    /// Tables are target-first, so the target (highest position) is slot 0 and
148    /// the root-adjacent model (position 1) is the last slot.
149    fn slot(&self, pos: usize) -> usize {
150        self.models.len() - 1 - pos
151    }
152
153    /// The `(link, parent_key)` field pair: `link` lives on the root-adjacent
154    /// model and matches `parent_key` on the root in the WHERE filter.
155    fn link(&self) -> (app::FieldId, app::FieldId) {
156        let root_edge = &self.edges[0];
157        (root_edge.target_side, root_edge.root_side)
158    }
159
160    /// The `FROM target JOIN …intermediates` source.
161    fn build_source(&self, schema: &toasty_core::Schema) -> stmt::Source {
162        let mut tables = Vec::with_capacity(self.edges.len());
163        let mut joins = Vec::with_capacity(self.edges.len().saturating_sub(1));
164
165        tables.push(stmt::TableRef::Table(schema.table_id_for(self.target())));
166
167        // Walk intermediates target-adjacent first so table slots increase
168        // toward the root. `edges[pos]` joins this intermediate (`models[pos]`)
169        // to its already-placed neighbour (`models[pos + 1]`).
170        for pos in (1..self.edges.len()).rev() {
171            tables.push(stmt::TableRef::Table(schema.table_id_for(self.models[pos])));
172
173            let edge = &self.edges[pos];
174            joins.push(stmt::Join {
175                table: stmt::SourceTableId(self.slot(pos)),
176                constraint: stmt::JoinOp::Inner(stmt::Expr::eq(
177                    raw_column(schema, self.slot(pos), edge.root_side),
178                    raw_column(schema, self.slot(pos + 1), edge.target_side),
179                )),
180            });
181        }
182
183        stmt::Source::Table(stmt::SourceTable {
184            tables,
185            from: vec![stmt::TableWithJoins {
186                relation: stmt::TableFactor::Table(stmt::SourceTableId(0)),
187                joins,
188            }],
189        })
190    }
191}

193/// A foreign-key edge between two adjacent models on a via path, with the FK
194/// field on each side resolved regardless of which side declared the relation.
195struct Edge {
196    /// FK field on the model nearer the root.
197    root_side: app::FieldId,
198    /// FK field on the model nearer the target.
199    target_side: app::FieldId,
200}
201
202impl Edge {
203    fn resolve(schema: &toasty_core::Schema, field_id: app::FieldId) -> Edge {
204        let field = schema.app.field(field_id);
205
206        // A has-relation is reached through its paired `BelongsTo`, which lives
207        // on the target-side model and owns the FK; a `BelongsTo` step is the
208        // edge itself, on the root-side model.
209        let (belongs_to, owner_is_target_side) = match &field.ty {
210            app::FieldTy::Has(_) => {
211                let pair = field
212                    .pair()
213                    .expect("via paths are unfolded into direct steps before edge resolution");
214                (schema.app.field(pair).ty.as_belongs_to_unwrap(), true)
215            }
216            app::FieldTy::BelongsTo(belongs_to) => (belongs_to, false),
217            _ => unreachable!("via step is not a relation field"),
218        };
219
220        // The FK source lives on the BT owner, the target on the model it
221        // references; map those onto root/target sides via the owner's side.
222        let [fk] = &belongs_to.foreign_key.fields[..] else {
223            todo!("composite foreign keys in via include path");
224        };
225        if owner_is_target_side {
226            Edge {
227                root_side: fk.target,
228                target_side: fk.source,
229            }
230        } else {
231            Edge {
232                root_side: fk.source,
233                target_side: fk.target,
234            }
235        }
236    }
237}

239/// Walk a via path, inlining any `via` field's own resolved path so the
240/// result is a flat sequence of direct relation `FieldId`s.
241fn flatten_via_steps(
242    schema: &toasty_core::Schema,
243    source_model_id: app::ModelId,
244    initial_steps: &[usize],
245) -> Vec<app::FieldId> {
246    let mut result = Vec::with_capacity(initial_steps.len());
247    let mut current_model = source_model_id;
248    let mut queue: Vec<usize> = initial_steps.to_vec();
249    queue.reverse(); // pop from the back
250
251    while let Some(idx) = queue.pop() {
252        let field = &schema.app.model(current_model).as_root_unwrap().fields[idx];
253        let field_id = app::FieldId {
254            model: current_model,
255            index: idx,
256        };
257
258        // If this step itself names a `via` relation, splice the nested
259        // path in place of it and continue (handles via-of-via naturally).
260        let nested_via = match &field.ty {
261            app::FieldTy::Via(via) => Some(via),
262            _ => None,
263        };
264        if let Some(via) = nested_via {
265            for step in via.path.projection.as_slice().iter().rev() {
266                queue.push(*step);
267            }
268            continue;
269        }
270
271        current_model = field
272            .relation_target_id()
273            .expect("via path step is a relation");
274        result.push(field_id);
275    }
276
277    result
278}

280/// The single-column mapping for a foreign-key field. Via include paths only
281/// resolve FK source/target fields, which the schema guarantees are primitive.
282fn fk_primitive(schema: &toasty_core::Schema, field_id: app::FieldId) -> &mapping::FieldPrimitive {
283    schema.mapping_for(field_id.model).fields[field_id.index]
284        .as_primitive()
285        .expect("FK field maps to a single column")
286}

288/// A raw (storage-level) column reference at table `slot` for a FK field.
289/// Used in JOIN constraints, which compare stored values directly and so need
290/// no storage→model cast.
291fn raw_column(schema: &toasty_core::Schema, slot: usize, field_id: app::FieldId) -> stmt::Expr {
292    stmt::Expr::column(stmt::ExprReference::column(
293        slot,
294        fk_primitive(schema, field_id).column.index,
295    ))
296}

298/// The model-level expression for a FK field — its column reference wrapped in
299/// the storage→model cast when the storage type differs (e.g. `Uuid` stored as
300/// `Bytes`) — re-pointed at table `slot`.
301///
302/// `FieldPrimitive::column_expr` is the schema's pre-built table→model
303/// expression at slot 0; we rewrite each column ref's slot.
304fn model_level_column_expr(
305    schema: &toasty_core::Schema,
306    field_id: app::FieldId,
307    slot: usize,
308) -> stmt::Expr {
309    let mut expr = fk_primitive(schema, field_id).column_expr.clone();
310
311    stmt::visit_mut::for_each_expr_mut(&mut expr, |e| {
312        if let stmt::Expr::Reference(stmt::ExprReference::Column(col)) = e {
313            col.table = slot;
314        }
315    });
316
317    expr

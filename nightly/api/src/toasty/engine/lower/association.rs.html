6/// Pre-lowering pass that rewrites `Source::Model { via: Some(_) }` into
7/// an explicit WHERE filter on the surrounding statement.
8///
9/// `via` associations are an app-level construct. They appear when a
10/// query is built from a relation traversal, e.g.
11/// `user.todos().delete(...)`, and the lowering walk converts
12/// `Source::Model` into `Source::Table` once the association has been
13/// rewritten as a filter. This pass runs before the lowering walk so
14/// that the walk only sees the rewritten form.
15pub(super) struct RewriteVia<'a> {
16    cx: ExprContext<'a>,
17}
18
19impl<'a> RewriteVia<'a> {
20    pub(super) fn new(cx: ExprContext<'a>) -> Self {
21        Self { cx }
22    }
23
24    /// Walk a statement and apply the via-association rewrite to every
25    /// Delete, Insert, and Query node it contains.
26    pub(super) fn rewrite(&mut self, stmt: &mut stmt::Statement) {
27        self.visit_mut(stmt);
28    }
29
30    fn schema(&self) -> &'a toasty_core::Schema {
31        self.cx.schema()
32    }
33
34    fn scope<'scope>(&'scope self, target: impl IntoExprTarget<'scope>) -> RewriteVia<'scope> {
35        RewriteVia {
36            cx: self.cx.scope(target),
37        }
38    }

40    pub(super) fn rewrite_via_for_delete(&mut self, stmt: &mut stmt::Delete) {
41        if let stmt::Source::Model(model) = &mut stmt.from
42            && let Some(via) = model.via.take()
43        {
44            // Create a new scope to indicate we are operating in the
45            // context of stmt.from
46            let mut s = self.scope(&stmt.from);
47
48            let filter = s.rewrite_association_as_filter(via);
49            stmt.filter = stmt::Filter::and(stmt.filter.take(), filter);
50        }
51    }
52
53    pub(super) fn rewrite_via_for_insert(&mut self, stmt: &mut stmt::Insert) {
54        if let stmt::InsertTarget::Scope(scope) = &mut stmt.target {
55            self.rewrite_via_for_query(scope);
56        }
57    }
58
59    pub(super) fn rewrite_via_for_query(&mut self, stmt: &mut stmt::Query) {
60        if let stmt::ExprSet::Select(select) = &mut stmt.body
61            && let stmt::Source::Model(model) = &mut select.source
62            && let Some(via) = model.via.take()
63        {
64            // Complete a scalar-terminal via used as a query source. For
65            // `#[has_many(via = todos.tags.name)]`, `user.tag_names()` selects
66            // the `name` column from `Tag` (the model the chain reaches), not
67            // whole `Tag` records:
68            //
69            //     SELECT DISTINCT tag.name      -- returning: project Tag.name
70            //     FROM tag                      -- source model: Tag (the via target)
71            //     WHERE <tag reachable from user>  -- the chain, unfolded below
72            //
73            // The navigation method can't build this: it runs in user code with
74            // no linked schema, so it can't name the target model and emits a
75            // placeholder source id with no projection. The resolved `Via` is
76            // available here, so set the source model (the FROM table) and the
77            // terminal projection (the RETURNING) before unfolding the chain
78            // into the WHERE filter.
79            //
80            // A model-terminal via has `terminal == None` — its source already
81            // selects whole target records — so this is a no-op there.
82            let scalar_terminal =
83                self.schema()
84                    .app
85                    .resolve_field_path(&via.path)
86                    .and_then(|field| match &field.ty {
87                        app::FieldTy::Via(v) => v.terminal.map(|terminal| (v.target, terminal)),
88                        _ => None,
89                    });
90
91            if let Some((target, _)) = scalar_terminal {
92                model.id = target;
93            }
94            if let Some((target, terminal)) = scalar_terminal {
95                select.returning =
96                    stmt::Returning::Project(stmt::Path::field(target, terminal).into_stmt());
97            }
98
99            // Create a new scope to indicate we are operating in the
100            // context of stmt.target
101            let mut s = self.scope(&select.source);
102
103            let filter = s.rewrite_association_as_filter(via);
104            select.filter = stmt::Filter::and(select.filter.take(), filter);
105        }
106    }

108    pub(super) fn rewrite_association_as_filter(
109        &mut self,
110        association: stmt::Association,
111    ) -> stmt::Filter {
112        assert!(
113            !association.path.projection.is_empty(),
114            "via path must have at least one step"
115        );
116
117        // Resolve every via in the path and unfold the chain into nested
118        // single-step `Source::Model { via }` wrappers. After this the path
119        // is one step and the terminal field is guaranteed not to be a via.
120        let mut association = self.unfold_path(association);
121
122        // Run the visitor's overridden `visit_stmt_query_mut` on the source
123        // so any `Source::Model { via: Some(_) }` introduced by unfolding is
124        // rewritten on its own merits before the outer single-step filter is
125        // built. The free-function walker would skip the override on the
126        // source query itself.
127        self.visit_stmt_query_mut(&mut association.source);
128
129        let Some(field) = self.schema().app.resolve_field_path(&association.path) else {
130            todo!()
131        };
132
133        match &field.ty {
134            app::FieldTy::BelongsTo(rel) => {
135                self.rewrite_association_belongs_to_as_filter(rel, association)
136            }
137            // Direct has-one / has-many: filter the target by its paired
138            // `BelongsTo` against the source query. Via relations were
139            // already unfolded, so only direct kinds reach this arm.
140            app::FieldTy::Has(has) => stmt::Expr::in_subquery(
141                stmt::Expr::ref_self_field(has.pair_id),
142                *association.source,
143            )
144            .into(),
145            _ => todo!("field={field:#?}"),
146        }
147    }
148
149    /// Resolve the association path into direct relation fields, then wrap
150    /// each intermediate field in a nested `Source::Model { via }` query.
151    /// Returns an association whose path contains one direct relation step.
152    fn unfold_path(&self, association: stmt::Association) -> stmt::Association {
153        let stmt::Association { source, path } = association;
154        let source_model_id = source.body.as_select_unwrap().source.model_id_unwrap();
155        let fields = super::relation_path::flatten_relation_path(
156            self.schema(),
157            source_model_id,
158            path.projection.as_slice(),
159        );
160
161        self.unfold_fields(source, &fields)
162    }
163
164    fn unfold_fields(
165        &self,
166        source: Box<stmt::Query>,
167        fields: &[app::FieldId],
168    ) -> stmt::Association {
169        let [field_id, rest @ ..] = fields else {
170            unreachable!("unfold_fields called with empty fields")
171        };
172        let field = self.schema().app.field(*field_id);
173
174        // Base case: a single direct relation step stays on the outer
175        // association.
176        if rest.is_empty() {
177            return stmt::Association {
178                source,
179                path: stmt::Path::from_index(field_id.model, field_id.index),
180            };
181        }
182
183        let next_model_id = field
184            .relation_target_id()
185            .expect("unfolded association path field is not a relation");
186
187        let inner = stmt::Association {
188            source,
189            path: stmt::Path::from_index(field_id.model, field_id.index),
190        };
191        let new_source = Box::new(stmt::Query::new_select(
192            stmt::Source::Model(stmt::SourceModel {
193                id: next_model_id,
194                via: Some(inner),
195            }),
196            stmt::Expr::Value(stmt::Value::Bool(true)),
197        ));
198
199        debug_assert_eq!(rest[0].model, next_model_id);
200        self.unfold_fields(new_source, rest)
201    }
202
203    fn rewrite_association_belongs_to_as_filter(
204        &mut self,
205        rel: &app::BelongsTo,
206        association: stmt::Association,
207    ) -> stmt::Filter {
208        // The FK lives on the source model; the target model carries the
209        // referenced fields. Filter is `<fk.target...> IN (SELECT
210        // <fk.source...> FROM <source>)` — a single field reference on each
211        // side for single-column FKs, a record of references for composite
212        // FKs (lowered to a tuple-style IN by the SQL serializer).
213        let target = super::key_field_refs(0, rel.foreign_key.fields.iter().map(|fk| fk.target));
214        let returning = super::key_field_refs(0, rel.foreign_key.fields.iter().map(|fk| fk.source));
215
216        let mut source = *association.source;
217        source.body.as_select_mut_unwrap().returning = stmt::Returning::Project(returning);
218
219        stmt::Expr::in_subquery(target, source).into()
220    }
221}

223impl VisitMut for RewriteVia<'_> {
224    fn visit_stmt_delete_mut(&mut self, i: &mut stmt::Delete) {
225        self.rewrite_via_for_delete(i);
226        stmt::visit_mut::visit_stmt_delete_mut(self, i);
227    }
228
229    fn visit_stmt_insert_mut(&mut self, i: &mut stmt::Insert) {
230        self.rewrite_via_for_insert(i);
231        stmt::visit_mut::visit_stmt_insert_mut(self, i);
232    }
233
234    fn visit_stmt_query_mut(&mut self, i: &mut stmt::Query) {
235        self.rewrite_via_for_query(i);
236        stmt::visit_mut::visit_stmt_query_mut(self, i);
237    }

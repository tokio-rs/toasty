6/// Pre-lowering pass that rewrites `Source::Model { via: Some(_) }` into
7/// an explicit WHERE filter on the surrounding statement.
8///
9/// `via` associations are an app-level construct. They appear when a
10/// query is built from a relation traversal, e.g.
11/// `user.todos().delete(...)`, and the lowering walk converts
12/// `Source::Model` into `Source::Table` once the association has been
13/// rewritten as a filter. This pass runs before the lowering walk so
14/// that the walk only sees the rewritten form.
15pub(super) struct RewriteVia<'a> {
16    cx: ExprContext<'a>,
17}
18
19impl<'a> RewriteVia<'a> {
20    pub(super) fn new(cx: ExprContext<'a>) -> Self {
21        Self { cx }
22    }
23
24    /// Walk a statement and apply the via-association rewrite to every
25    /// Delete, Insert, and Query node it contains.
26    pub(super) fn rewrite(&mut self, stmt: &mut stmt::Statement) {
27        self.visit_mut(stmt);
28    }
29
30    fn schema(&self) -> &'a toasty_core::Schema {
31        self.cx.schema()
32    }
33
34    fn scope<'scope>(&'scope self, target: impl IntoExprTarget<'scope>) -> RewriteVia<'scope> {
35        RewriteVia {
36            cx: self.cx.scope(target),
37        }
38    }
39
40    pub(super) fn rewrite_via_for_delete(&mut self, stmt: &mut stmt::Delete) {
41        if let stmt::Source::Model(model) = &mut stmt.from
42            && let Some(via) = model.via.take()
43        {
44            // Create a new scope to indicate we are operating in the
45            // context of stmt.from
46            let mut s = self.scope(&stmt.from);
47
48            let filter = s.rewrite_association_as_filter(via);
49            stmt.filter = stmt::Filter::and(stmt.filter.take(), filter);
50        }
51    }
52
53    pub(super) fn rewrite_via_for_insert(&mut self, stmt: &mut stmt::Insert) {
54        if let stmt::InsertTarget::Scope(scope) = &mut stmt.target {
55            self.rewrite_via_for_query(scope);
56        }
57    }
58
59    pub(super) fn rewrite_via_for_query(&mut self, stmt: &mut stmt::Query) {
60        if let stmt::ExprSet::Select(select) = &mut stmt.body
61            && let stmt::Source::Model(model) = &mut select.source
62            && let Some(via) = model.via.take()
63        {
64            // Create a new scope to indicate we are operating in the
65            // context of stmt.target
66            let mut s = self.scope(&select.source);
67
68            let filter = s.rewrite_association_as_filter(via);
69            select.filter = stmt::Filter::and(select.filter.take(), filter);
70        }
71    }
72
73    pub(super) fn rewrite_association_as_filter(
74        &mut self,
75        mut association: stmt::Association,
76    ) -> stmt::Filter {
77        assert!(
78            !association.path.projection.is_empty(),
79            "via path must have at least one step"
80        );
81
82        // Unfold any multi-step path into a chain of nested single-step
83        // `Source::Model { via }` wrappers via a single recursive descent.
84        // The recursion threads the source query through by value and borrows
85        // path steps as a slice — no per-step `Vec` rebuilds.
86        if association.path.projection.len() > 1 {
87            let stmt::Association { source, path } = association;
88            let source_model_id = source.body.as_select_unwrap().source.model_id_unwrap();
89            association = self.unfold_path(source, source_model_id, path.projection.as_slice());
90        }
91
92        // Run the visitor's overridden `visit_stmt_query_mut` on the source
93        // so any `Source::Model { via: Some(_) }` introduced by unfolding is
94        // rewritten on its own merits before the outer single-step filter is
95        // built. The free-function walker would skip the override on the
96        // source query itself.
97        self.visit_stmt_query_mut(&mut association.source);
98
99        let Some(field) = self.schema().app.resolve_field_path(&association.path) else {
100            todo!()
101        };
102
103        match &field.ty {
104            app::FieldTy::BelongsTo(rel) => {
105                self.rewrite_association_belongs_to_as_filter(rel, association)
106            }
107            app::FieldTy::HasOne(rel) => {
108                stmt::Expr::in_subquery(stmt::Expr::ref_self_field(rel.pair), *association.source)
109                    .into()
110            }
111            app::FieldTy::HasMany(rel) => {
112                stmt::Expr::in_subquery(stmt::Expr::ref_self_field(rel.pair), *association.source)
113                    .into()
114            }
115            _ => todo!("field={field:#?}"),
116        }
117    }
118
119    /// Recursively wrap every step but the last into a nested
120    /// `Source::Model { via }`, threading the source query through by value.
121    /// Returns the outer single-step association the caller filters against.
122    fn unfold_path(
123        &self,
124        source: Box<stmt::Query>,
125        source_model_id: app::ModelId,
126        steps: &[usize],
127    ) -> stmt::Association {
128        let [first, rest @ ..] = steps else {
129            unreachable!("unfold_path called with empty steps")
130        };
131
132        // Base case: the last step stays on the outer association.
133        if rest.is_empty() {
134            return stmt::Association {
135                source,
136                path: stmt::Path::from_index(source_model_id, *first),
137            };
138        }
139
140        let source_model = self.schema().app.model(source_model_id).as_root_unwrap();
141        let next_model_id = match &source_model.fields[*first].ty {
142            app::FieldTy::HasMany(rel) => rel.target,
143            app::FieldTy::HasOne(rel) => rel.target,
144            app::FieldTy::BelongsTo(rel) => rel.target,
145            other => todo!("non-relation field in via path: {other:#?}"),
146        };
147
148        let inner = stmt::Association {
149            source,
150            path: stmt::Path::from_index(source_model_id, *first),
151        };
152        let new_source = Box::new(stmt::Query::new_select(
153            stmt::Source::Model(stmt::SourceModel {
154                id: next_model_id,
155                via: Some(inner),
156            }),
157            stmt::Expr::Value(stmt::Value::Bool(true)),
158        ));
159
160        self.unfold_path(new_source, next_model_id, rest)
161    }
162
163    fn rewrite_association_belongs_to_as_filter(
164        &mut self,
165        rel: &app::BelongsTo,
166        association: stmt::Association,
167    ) -> stmt::Filter {
168        // The FK lives on the source model; the target model carries the
169        // referenced fields. Filter is `<fk.target...> IN (SELECT
170        // <fk.source...> FROM <source>)` — a single field reference on each
171        // side for single-column FKs, a record of references for composite
172        // FKs (lowered to a tuple-style IN by the SQL serializer).
173        let target = super::key_field_refs(0, rel.foreign_key.fields.iter().map(|fk| fk.target));
174        let returning = super::key_field_refs(0, rel.foreign_key.fields.iter().map(|fk| fk.source));
175
176        let mut source = *association.source;
177        source.body.as_select_mut_unwrap().returning = stmt::Returning::Project(returning);
178
179        stmt::Expr::in_subquery(target, source).into()
180    }
181}

183impl VisitMut for RewriteVia<'_> {
184    fn visit_stmt_delete_mut(&mut self, i: &mut stmt::Delete) {
185        self.rewrite_via_for_delete(i);
186        stmt::visit_mut::visit_stmt_delete_mut(self, i);
187    }
188
189    fn visit_stmt_insert_mut(&mut self, i: &mut stmt::Insert) {
190        self.rewrite_via_for_insert(i);
191        stmt::visit_mut::visit_stmt_insert_mut(self, i);
192    }
193
194    fn visit_stmt_query_mut(&mut self, i: &mut stmt::Query) {
195        self.rewrite_via_for_query(i);
196        stmt::visit_mut::visit_stmt_query_mut(self, i);
197    }

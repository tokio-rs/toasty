1//! App-level rewrites that lift relation references out of `IN` subqueries
2//! into direct foreign-key comparisons.
3//!
4//! [`LiftInSubquery`] runs as a whole-statement pre-pass before the main
5//! lowering walk.  The visitor overrides `visit_expr_mut` to fire two
6//! rewrites pre-children:
7//!
8//! - [`lift_in_subquery`] fires on `Expr::InSubquery` where the LHS is a
9//!   relation field reference (`BelongsTo`/`HasOne`/`HasMany`).  For
10//!   `BelongsTo` it tries to lift the subquery's filter into FK comparisons
11//!   on the parent, falling back to a re-targeted IN subquery on the
12//!   foreign-key column.  For `HasOne`/`HasMany` it always rewrites to a
13//!   foreign-key IN subquery against the related table.
14//!
15//! - [`try_lift_relation_path_comparison`] fires on `Expr::BinaryOp` where
16//!   one side is `project(ref_self_field(relation_field), [idx, ...])`.
17//!   It synthesises a subquery on the target model with the comparison
18//!   re-rooted there, then defers to [`lift_in_subquery`].
19//!
20//! A pre-pass is necessary (rather than folding into
21//! `LowerStatement::visit_expr_mut` per #823's pattern) because not every
22//! expression that contains an `IN` subquery flows through
23//! `LowerStatement::visit_expr_mut`; `ApplyInsertScope::apply_expr` in
24//! particular walks insert-scope constraint expressions through its own
25//! recursion and would panic on an unlifted relation `IN` subquery.
26//!
27//! The free functions [`lift_in_subquery`] and
28//! [`try_lift_relation_path_comparison`] are exposed on `&ExprContext` so
29//! the visitor and the unit tests can both call them without constructing
30//! a `LiftInSubquery`.
31
32use toasty_core::{
33    schema::app::{self, BelongsTo, FieldId, FieldTy, ModelId},
34    stmt::{self, Expr, ExprContext, IntoExprTarget, ResolvedRef, Visit, VisitMut},
35};

37/// Pre-lowering pass that lifts relation references out of `IN`-subquery
38/// and projection comparisons into direct foreign-key forms.  Runs as a
39/// whole-statement visitor before the main lowering walk: code paths that
40/// process expressions outside the lowering walk (notably
41/// `ApplyInsertScope::apply_expr`) see the already-lifted form.
42pub(super) struct LiftInSubquery<'a> {
43    cx: ExprContext<'a>,
44}
45
46impl<'a> LiftInSubquery<'a> {
47    pub(super) fn new(cx: ExprContext<'a>) -> Self {
48        Self { cx }
49    }
50
51    pub(super) fn rewrite(&mut self, stmt: &mut stmt::Statement) {
52        self.visit_mut(stmt);
53    }
54
55    fn scope<'scope>(&'scope self, target: impl IntoExprTarget<'scope>) -> LiftInSubquery<'scope> {
56        LiftInSubquery {
57            cx: self.cx.scope(target),
58        }
59    }
60}

62impl VisitMut for LiftInSubquery<'_> {
63    fn visit_expr_mut(&mut self, expr: &mut stmt::Expr) {
64        // Apply lifts pre-children: the rewrites pattern-match on app-
65        // level `Reference::Field` to a relation, and the children walk
66        // would not introduce relation references that were not there.
67        match expr {
68            stmt::Expr::InSubquery(e) => {
69                if let Some(lifted) = lift_in_subquery(&self.cx, &e.expr, &e.query) {
70                    *expr = lifted;
71                }
72            }
73            stmt::Expr::BinaryOp(e) => {
74                if let Some(lifted) =
75                    try_lift_relation_path_comparison(&self.cx, e.op, &e.lhs, &e.rhs)
76                {
77                    *expr = lifted;
78                } else if let Some(commuted) = e.op.commute()
79                    && let Some(lifted) =
80                        try_lift_relation_path_comparison(&self.cx, commuted, &e.rhs, &e.lhs)
81                {
82                    *expr = lifted;
83                }
84            }
85            _ => {}
86        }
87
88        // Walk children (which may themselves be expressions needing
89        // lifts on subtrees).
90        stmt::visit_mut::visit_expr_mut(self, expr);
91    }
92
93    fn visit_stmt_delete_mut(&mut self, stmt: &mut stmt::Delete) {
94        self.visit_source_mut(&mut stmt.from);
95
96        let mut s = self.scope(&stmt.from);
97
98        s.visit_filter_mut(&mut stmt.filter);
99
100        if let Some(returning) = &mut stmt.returning {
101            s.visit_returning_mut(returning);
102        }
103    }
104
105    fn visit_stmt_insert_mut(&mut self, stmt: &mut stmt::Insert) {
106        self.visit_insert_target_mut(&mut stmt.target);
107
108        let mut s = self.scope(&stmt.target);
109
110        s.visit_stmt_query_mut(&mut stmt.source);
111
112        if let Some(returning) = &mut stmt.returning {
113            s.visit_returning_mut(returning);
114        }
115    }
116
117    fn visit_stmt_select_mut(&mut self, stmt: &mut stmt::Select) {
118        self.visit_source_mut(&mut stmt.source);
119
120        let mut s = self.scope(&stmt.source);
121
122        s.visit_filter_mut(&mut stmt.filter);
123        s.visit_returning_mut(&mut stmt.returning);
124    }
125
126    fn visit_stmt_update_mut(&mut self, stmt: &mut stmt::Update) {
127        self.visit_update_target_mut(&mut stmt.target);
128
129        let mut s = self.scope(&stmt.target);
130
131        s.visit_assignments_mut(&mut stmt.assignments);
132        s.visit_filter_mut(&mut stmt.filter);
133
134        if let Some(expr) = &mut stmt.condition.expr {
135            s.visit_expr_mut(expr);
136        }
137
138        if let Some(returning) = &mut stmt.returning {
139            s.visit_returning_mut(returning);
140        }
141    }
142}

144struct LiftBelongsTo<'a> {
145    cx: ExprContext<'a>,
146    belongs_to: &'a BelongsTo,
147    // TODO: switch to bit field set
148    fk_field_matches: Vec<bool>,
149    fail: bool,
150    operands: Vec<stmt::Expr>,
151}

153/// Lift `expr IN (subquery)` into a foreign-key-based comparison when
154/// `expr` is a relation field reference and `subquery` targets the
155/// relation's target model.
156///
157/// Returns `None` when the LHS is not a relation field reference or when
158/// the lift cannot apply.  When the lift succeeds, the returned
159/// expression may itself contain unlowered references and the caller is
160/// expected to re-visit it through the lowering walk.
161pub(super) fn lift_in_subquery(
162    cx: &ExprContext,
163    expr: &stmt::Expr,
164    query: &stmt::Query,
165) -> Option<stmt::Expr> {
166    // The expression is a path expression referencing a relation.
167    let field = match expr {
168        // `Project(Ref(rel), [head, ...tail])` — the path traverses through a
169        // relation field (`rel`) before reaching the relation the subquery
170        // targets. Re-root the path at `rel`'s target model and rebuild as a
171        // nested IN-subquery on that model, then recurse to lift the outer
172        // `rel` hop.
173        stmt::Expr::Project(project_expr) => {
174            return lift_projection_in_subquery(cx, project_expr, query);
175        }
176        stmt::Expr::Reference(expr_reference @ stmt::ExprReference::Field { .. }) => {
177            cx.resolve_expr_reference(expr_reference).as_field_unwrap()
178        }
179        _ => {
180            return None;
181        }
182    };
183
184    // If the field is not a relation, abort. Multi-step (`via`) relations
185    // have no paired BelongsTo to lift through — they are already rewritten
186    // into nested IN subqueries by `RewriteVia` earlier in the lowering
187    // pipeline, so any reference reaching here that still names one is not
188    // something this pass can handle.
189    match &field.ty {
190        FieldTy::BelongsTo(belongs_to) => lift_belongs_to_in_subquery(cx, belongs_to, query),
191        FieldTy::Has(has) => lift_has_n_in_subquery(has.target, has.pair(&cx.schema().app), query),
192        _ => None,
193    }
194}

196/// Lifts an `IN`-subquery whose left side is a path through a relation field.
197///
198/// `.any()` on a relation chain — for example, `Release.project.topics.any(name
199/// == "rust")` — builds an `InSubquery` whose LHS projects through the
200/// chain:
201///
202/// ```text
203/// InSubquery {
204///     expr:  Project(Ref(Release.project), [Project.topics_idx]),
205///     query: SELECT FROM Topic WHERE name == "rust",
206/// }
207/// ```
208///
209/// Two paths handle this:
210///
211/// **Fused.** [`try_fuse_paired_relations`] recognizes the common
212/// `BelongsTo → Has` chain over a shared primary key and emits a single
213/// FK-on-FK `IN` that bypasses the intermediate model:
214///
215/// ```text
216/// Release.project_id IN (SELECT Topic.project_id FROM Topic WHERE name == "rust")
217/// ```
218///
219/// **General.** For chains the fast path doesn't match (multi-hop
220/// projections, non-paired relations), re-root the projection at the
221/// relation's target model and wrap the original subquery in a nested `IN`
222/// against that model, then recurse on the outer hop. For the same input,
223/// the fallback produces:
224///
225/// ```text
226/// Release.project IN (
227///     SELECT FROM Project
228///     WHERE Project.topics IN (SELECT FROM Topic WHERE name == "rust")
229/// )
230/// ```
231///
232/// The recursive call lifts the outer `Release.project` hop via the standard
233/// `BelongsTo` branch; the inner `Project.topics` hop is lifted on the next
234/// visitor pass when [`LiftInSubquery`]'s children walk reaches it.
235fn lift_projection_in_subquery(
236    cx: &ExprContext,
237    project_expr: &stmt::ExprProject,
238    query: &stmt::Query,
239) -> Option<stmt::Expr> {
240    let Expr::Reference(expr_ref) = &*project_expr.base else {
241        return None;
242    };
243    let ResolvedRef::Field(field) = cx.resolve_expr_reference(expr_ref) else {
244        return None;
245    };
246
247    let target_model_id = match &field.ty {
248        FieldTy::Has(rel) => rel.target,
249        FieldTy::Via(rel) => rel.target,
250        FieldTy::BelongsTo(rel) => rel.target,
251        _ => return None,
252    };
253
254    let (head_idx, tail) = project_expr.projection.as_slice().split_first()?;
255
256    if tail.is_empty()
257        && let Some(direct) =
258            try_fuse_paired_relations(cx, field, target_model_id, *head_idx, query)
259    {
260        return Some(direct);
261    }
262
263    let target_field = Expr::ref_self_field(FieldId {
264        model: target_model_id,
265        index: *head_idx,
266    });
267    let inner_lhs = if tail.is_empty() {
268        target_field
269    } else {
270        Expr::project(target_field, stmt::Projection::from(tail))
271    };
272
273    let new_subquery = stmt::Query::new_select(
274        stmt::Source::from(target_model_id),
275        Expr::in_subquery(inner_lhs, query.clone()),
276    );
277
278    lift_in_subquery(cx, &project_expr.base, &new_subquery)
279}

281/// Fuses a `BelongsTo → Has` chain into a single FK-on-FK `IN` when both
282/// relations meet at the same primary key.
283///
284/// # Example
285///
286/// Given the schema:
287///
288/// ```text
289/// Todo     { category_id: Category.id, ... }   // BelongsTo  Todo.category
290/// Category { todos: HasMany Todo, ... }        // Has, paired with Todo.category
291/// ```
292///
293/// The path `Todo.category.todos` lifts as follows:
294///
295/// ```text
296/// // Input
297/// InSubquery {
298///     expr:  Project(Ref(Todo.category), [Category.todos_idx]),
299///     query: SELECT FROM Todo WHERE title == "salad",
300/// }
301///
302/// // Output
303/// InSubquery {
304///     expr:  Ref(Todo.category_id),
305///     query: SELECT Todo.category_id FROM Todo WHERE title == "salad",
306/// }
307/// ```
308///
309/// The outer relation's FK source columns become the LHS; the inner
310/// relation's paired-BelongsTo FK source columns become the subquery's
311/// returning list. The user's original filter is preserved verbatim.
312///
313/// # Why the fusion is sound
314///
315/// Composing the two hops without fusion routes through the intermediate
316/// model:
317///
318/// ```text
319/// Todo.category_id IN (
320///     SELECT Category.id FROM Category
321///     WHERE Category.id IN (SELECT Todo.category_id FROM Todo WHERE title == "salad")
322/// )
323/// ```
324///
325/// Both FKs target `Category.id`, so every value the innermost subquery
326/// returns exists in `Category.id` under FK integrity (which the engine
327/// already assumes). The middle filter therefore admits every row the inner
328/// returns, and the middle `SELECT Category.id` simply re-emits them. The
329/// `Category` scan is a no-op and can be dropped.
330///
331/// Returns `None` when the chain doesn't match the pattern — outer isn't a
332/// `BelongsTo`, inner isn't a `Has`, or the FK columns don't line up.
333fn try_fuse_paired_relations(
334    cx: &ExprContext,
335    outer_field: &app::Field,
336    target_model_id: ModelId,
337    head_idx: usize,
338    query: &stmt::Query,
339) -> Option<stmt::Expr> {
340    let outer_belongs_to = match &outer_field.ty {
341        FieldTy::BelongsTo(rel) => rel,
342        _ => return None,
343    };
344
345    let target_model = cx.schema().app.model(target_model_id).as_root_unwrap();
346    let head_field = target_model.fields.get(head_idx)?;
347    let inner_has = match &head_field.ty {
348        FieldTy::Has(has) => has,
349        _ => return None,
350    };
351    let inner_pair = inner_has.pair(&cx.schema().app);
352
353    // Both FKs must reference the same PK columns in the same order.
354    if outer_belongs_to.foreign_key.fields.len() != inner_pair.foreign_key.fields.len() {
355        return None;
356    }
357    for (outer_fk, inner_fk) in outer_belongs_to
358        .foreign_key
359        .fields
360        .iter()
361        .zip(inner_pair.foreign_key.fields.iter())
362    {
363        if outer_fk.target != inner_fk.target {
364            return None;
365        }
366    }
367
368    let mut fused_subquery = query.clone();
369    fused_subquery.body.as_select_mut_unwrap().returning = stmt::Returning::Project(
370        super::key_field_refs(0, inner_pair.foreign_key.fields.iter().map(|fk| fk.source)),
371    );
372
373    Some(stmt::Expr::in_subquery(
374        super::key_field_refs(
375            0,
376            outer_belongs_to
377                .foreign_key
378                .fields
379                .iter()
380                .map(|fk| fk.source),
381        ),
382        fused_subquery,
383    ))
384}

386/// Lift `project(ref_self_field(rel), [idx, ...]) op other` into a
387/// foreign-key-based comparison by synthesising an IN subquery on the
388/// relation's target model and deferring to [`lift_in_subquery`].
389///
390/// Returns `None` when `project_side` is not a project through a
391/// relation field reference.
392pub(super) fn try_lift_relation_path_comparison(
393    cx: &ExprContext,
394    op: stmt::BinaryOp,
395    project_side: &stmt::Expr,
396    other_side: &stmt::Expr,
397) -> Option<stmt::Expr> {
398    let Expr::Project(project_expr) = project_side else {
399        return None;
400    };
401    let Expr::Reference(expr_ref) = &*project_expr.base else {
402        return None;
403    };
404    let ResolvedRef::Field(field) = cx.resolve_expr_reference(expr_ref) else {
405        return None;
406    };
407
408    let target_model_id = match &field.ty {
409        FieldTy::Has(rel) => rel.target,
410        FieldTy::Via(rel) => rel.target,
411        FieldTy::BelongsTo(rel) => rel.target,
412        _ => return None,
413    };
414
415    // Re-root the projection at the target model: the leading index
416    // points at the relation field itself, the rest indexes into the
417    // related model's fields.
418    let (head_idx, tail) = project_expr.projection.as_slice().split_first()?;
419    let target_field = Expr::ref_self_field(FieldId {
420        model: target_model_id,
421        index: *head_idx,
422    });
423    let target_lhs = if tail.is_empty() {
424        target_field
425    } else {
426        Expr::project(target_field, stmt::Projection::from(tail))
427    };
428
429    let subquery = stmt::Query::new_select(
430        stmt::Source::from(target_model_id),
431        Expr::binary_op(target_lhs, op, other_side.clone()),
432    );
433
434    lift_in_subquery(cx, &project_expr.base, &subquery)
435}

437/// BelongsTo branch: try to lift the subquery's filter into direct FK
438/// comparisons.  When the filter references only FK-mapped fields, return
439/// the AND of per-FK equalities.  Otherwise, fall back to an IN subquery
440/// on the foreign-key column(s) — a tuple-form IN for composite FKs.
441///
442/// Returns `None` when the subquery does not target the BelongsTo's
443/// target model.
444fn lift_belongs_to_in_subquery(
445    cx: &ExprContext,
446    belongs_to: &BelongsTo,
447    query: &stmt::Query,
448) -> Option<stmt::Expr> {
449    if belongs_to.target != query.body.as_select_unwrap().source.model_id_unwrap() {
450        return None;
451    }
452
453    let select = query.body.as_select_unwrap();
454
455    let mut lift = LiftBelongsTo {
456        cx: cx.scope(&select.source),
457        belongs_to,
458        fk_field_matches: vec![false; belongs_to.foreign_key.fields.len()],
459        operands: vec![],
460        fail: false,
461    };
462
463    lift.visit_filter(&select.filter);
464
465    // Fall back to the IN-subquery form whenever we couldn't account for
466    // every FK column with a direct equality from the filter. This covers
467    // both `fail=true` (a binary op referenced something that isn't on the
468    // FK) and the case where the filter contained no liftable binary ops at
469    // all — e.g. when it's a nested `InSubquery` that the LiftBelongsTo
470    // visitor deliberately skips (see `visit_expr_in_subquery`).
471    let all_fks_matched = lift.fk_field_matches.iter().all(|m| *m);
472
473    if lift.fail || !all_fks_matched {
474        let mut subquery = query.clone();
475
476        subquery.body.as_select_mut_unwrap().returning = stmt::Returning::Project(
477            super::key_field_refs(0, belongs_to.foreign_key.fields.iter().map(|fk| fk.target)),
478        );
479
480        Some(stmt::Expr::in_subquery(
481            super::key_field_refs(0, belongs_to.foreign_key.fields.iter().map(|fk| fk.source)),
482            subquery,
483        ))
484    } else {
485        Some(if lift.operands.len() == 1 {
486            lift.operands.into_iter().next().unwrap()
487        } else {
488            stmt::ExprAnd {
489                operands: lift.operands,
490            }
491            .into()
492        })
493    }
494}

496/// HasOne/HasMany branch: rewrite to a foreign-key IN subquery against
497/// the related table. Single-column FKs produce a scalar IN; composite
498/// FKs produce a tuple-form IN that the SQL serializer renders as
499/// `(a, b) IN (SELECT a, b FROM ...)`.
500fn lift_has_n_in_subquery(
501    target: ModelId,
502    pair: &BelongsTo,
503    query: &stmt::Query,
504) -> Option<stmt::Expr> {
505    if target != query.body.as_select_unwrap().source.model_id_unwrap() {
506        return None;
507    }
508
509    let mut subquery = query.clone();
510
511    match &mut subquery.body {
512        stmt::ExprSet::Select(select) => {
513            select.returning = stmt::Returning::Project(super::key_field_refs(
514                0,
515                pair.foreign_key.fields.iter().map(|fk| fk.source),
516            ));
517        }
518        _ => todo!(),
519    }
520
521    Some(
522        stmt::ExprInSubquery {
523            expr: Box::new(super::key_field_refs(
524                0,
525                pair.foreign_key.fields.iter().map(|fk| fk.target),
526            )),
527            query: Box::new(subquery),
528        }
529        .into(),
530    )
531}

533impl Visit for LiftBelongsTo<'_> {
534    fn visit_expr_in_subquery(&mut self, _i: &stmt::ExprInSubquery) {
535        // Stop the walk at a nested IN-subquery boundary. Field references
536        // inside resolve in the nested query's own scope, but the visitor's
537        // `cx` is scoped to the current (outer) subquery's source — recursing
538        // would misresolve inner refs as outer-model fields and produce
539        // bogus FK matches. The main LiftInSubquery walker handles the
540        // nested subquery on its own pass, with the correct scope.
541    }
542
543    fn visit_expr_binary_op(&mut self, i: &stmt::ExprBinaryOp) {
544        match (&*i.lhs, &*i.rhs) {
545            (stmt::Expr::Reference(expr_reference), other)
546            | (other, stmt::Expr::Reference(expr_reference)) => {
547                assert!(i.op.is_eq() || i.op.is_ne());
548
549                if i.op.is_eq() || i.op.is_ne() {
550                    let field = self
551                        .cx
552                        .resolve_expr_reference(expr_reference)
553                        .as_field_unwrap();
554
555                    self.lift_fk_constraint(field.id, i.op, other);
556                } else {
557                    self.fail = true;
558                }
559            }
560            // Constraints we can't lift to a direct FK comparison (e.g. a
561            // projection through an embedded field).  Bail to the IN-subquery
562            // form so the filter is preserved verbatim; without this, the
563            // empty `operands` list silently produced an empty AND (= true)
564            // and the subquery returned every row.
565            _ => {
566                self.fail = true;
567            }
568        }
569    }
570}
571
572impl LiftBelongsTo<'_> {
573    fn lift_fk_constraint(&mut self, field: FieldId, op: stmt::BinaryOp, expr: &stmt::Expr) {
574        for (i, fk_field) in self.belongs_to.foreign_key.fields.iter().enumerate() {
575            if fk_field.target == field {
576                if self.fk_field_matches[i] {
577                    todo!("not handled");
578                }
579
580                self.operands.push(stmt::Expr::binary_op(
581                    stmt::Expr::ref_self_field(fk_field.source),
582                    op,
583                    expr.clone(),
584                ));
585                self.fk_field_matches[i] = true;
586
587                return;
588            }
589        }
590
591        self.fail = true;
592    }
593}

1//! App-level rewrites that lift relation references out of `IN` subqueries
2//! into direct foreign-key comparisons.
3//!
4//! [`LiftInSubquery`] runs as a whole-statement pre-pass before the main
5//! lowering walk.  The visitor overrides `visit_expr_mut` to fire two
6//! rewrites pre-children:
7//!
8//! - [`lift_in_subquery`] fires on `Expr::InSubquery` where the LHS is a
9//!   relation field reference (`BelongsTo`/`HasOne`/`HasMany`).  For
10//!   `BelongsTo` it tries to lift the subquery's filter into FK comparisons
11//!   on the parent, falling back to a re-targeted IN subquery on the
12//!   foreign-key column.  For `HasOne`/`HasMany` it always rewrites to a
13//!   foreign-key IN subquery against the related table.
14//!
15//! - [`try_lift_relation_path_comparison`] fires on `Expr::BinaryOp` where
16//!   one side walks a relation field. For example, filtering profiles by
17//!   their user's name (`profile.user.name = 'alice'`) rewrites to:
18//!
19//!   ```text
20//!   Profile.user_id IN (SELECT User.id FROM User WHERE User.name = 'alice')
21//!   ```
22//!
23//! - [`try_lift_relation_path_like`] does the same for `LIKE`/`ILIKE`
24//!   (`profile.user.name LIKE 'al%'`). These are `Expr::Like`, not
25//!   `Expr::BinaryOp`, so they need their own entry point. Both rewrites
26//!   share [`lift_relation_path_predicate`].
27//!
28//! A pre-pass is necessary (rather than folding into
29//! `LowerStatement::visit_expr_mut` per #823's pattern) because not every
30//! expression that contains an `IN` subquery flows through
31//! `LowerStatement::visit_expr_mut`; `ApplyInsertScope::apply_expr` in
32//! particular walks insert-scope constraint expressions through its own
33//! recursion and would panic on an unlifted relation `IN` subquery.
34//!
35//! The free functions [`lift_in_subquery`] and
36//! [`try_lift_relation_path_comparison`] are exposed on `&ExprContext` so
37//! the visitor and the unit tests can both call them without constructing
38//! a `LiftInSubquery`.
39

41    schema::app::{self, BelongsTo, FieldId, FieldTy, ModelId},
42    stmt::{self, Expr, ExprContext, IntoExprTarget, ResolvedRef, Visit, VisitMut},
43};

45/// Pre-lowering pass that lifts relation references out of `IN`-subquery
46/// and projection comparisons into direct foreign-key forms.  Runs as a
47/// whole-statement visitor before the main lowering walk: code paths that
48/// process expressions outside the lowering walk (notably
49/// `ApplyInsertScope::apply_expr`) see the already-lifted form.
50pub(super) struct LiftInSubquery<'a> {
51    cx: ExprContext<'a>,
52}
53
54impl<'a> LiftInSubquery<'a> {
55    pub(super) fn new(cx: ExprContext<'a>) -> Self {
56        Self { cx }
57    }
58
59    pub(super) fn rewrite(&mut self, stmt: &mut stmt::Statement) {
60        self.visit_mut(stmt);
61    }
62
63    fn scope<'scope>(&'scope self, target: impl IntoExprTarget<'scope>) -> LiftInSubquery<'scope> {
64        LiftInSubquery {
65            cx: self.cx.scope(target),
66        }
67    }
68}

70impl VisitMut for LiftInSubquery<'_> {
71    fn visit_expr_mut(&mut self, expr: &mut stmt::Expr) {
72        // Apply lifts pre-children: the rewrites pattern-match on app-
73        // level `Reference::Field` to a relation, and the children walk
74        // would not introduce relation references that were not there.
75        match expr {
76            stmt::Expr::InSubquery(e) => {
77                if let Some(lifted) = lift_in_subquery(&self.cx, &e.expr, &e.query) {
78                    *expr = lifted;
79                }
80            }
81            stmt::Expr::BinaryOp(e) => {
82                if let Some(lifted) =
83                    try_lift_relation_path_comparison(&self.cx, e.op, &e.lhs, &e.rhs)
84                {
85                    *expr = lifted;
86                } else if let Some(commuted) = e.op.commute()
87                    && let Some(lifted) =
88                        try_lift_relation_path_comparison(&self.cx, commuted, &e.rhs, &e.lhs)
89                {
90                    *expr = lifted;
91                }
92            }
93            stmt::Expr::Like(e) => {
94                if let Some(lifted) = try_lift_relation_path_like(&self.cx, e) {
95                    *expr = lifted;
96                }
97            }
98            _ => {}
99        }
100
101        // Walk children (which may themselves be expressions needing
102        // lifts on subtrees).
103        stmt::visit_mut::visit_expr_mut(self, expr);
104    }
105
106    fn visit_stmt_delete_mut(&mut self, stmt: &mut stmt::Delete) {
107        self.visit_source_mut(&mut stmt.from);
108
109        let mut s = self.scope(&stmt.from);
110
111        s.visit_filter_mut(&mut stmt.filter);
112
113        if let Some(returning) = &mut stmt.returning {
114            s.visit_returning_mut(returning);
115        }
116    }
117
118    fn visit_stmt_insert_mut(&mut self, stmt: &mut stmt::Insert) {
119        self.visit_insert_target_mut(&mut stmt.target);
120
121        let mut s = self.scope(&stmt.target);
122
123        s.visit_stmt_query_mut(&mut stmt.source);
124
125        if let Some(returning) = &mut stmt.returning {
126            s.visit_returning_mut(returning);
127        }
128    }
129
130    fn visit_stmt_select_mut(&mut self, stmt: &mut stmt::Select) {
131        self.visit_source_mut(&mut stmt.source);
132
133        let mut s = self.scope(&stmt.source);
134
135        s.visit_filter_mut(&mut stmt.filter);
136        s.visit_returning_mut(&mut stmt.returning);
137    }
138
139    fn visit_stmt_update_mut(&mut self, stmt: &mut stmt::Update) {
140        self.visit_update_target_mut(&mut stmt.target);
141
142        let mut s = self.scope(&stmt.target);
143
144        s.visit_assignments_mut(&mut stmt.assignments);
145        s.visit_filter_mut(&mut stmt.filter);
146
147        if let Some(expr) = &mut stmt.condition.expr {
148            s.visit_expr_mut(expr);
149        }
150
151        if let Some(returning) = &mut stmt.returning {
152            s.visit_returning_mut(returning);
153        }
154    }
155}

162    fail: bool,

166/// Lift `expr IN (subquery)` into a foreign-key-based comparison when
167/// `expr` is a relation field reference and `subquery` targets the
168/// relation's target model.
169///
170/// Returns `None` when the LHS is not a relation field reference or when
171/// the lift cannot apply.  When the lift succeeds, the returned
172/// expression may itself contain unlowered references and the caller is
173/// expected to re-visit it through the lowering walk.
174pub(super) fn lift_in_subquery(
175    cx: &ExprContext,
176    expr: &stmt::Expr,
177    query: &stmt::Query,
178) -> Option<stmt::Expr> {
179    // The expression is a path expression referencing a relation.
180    let field = match expr {
181        // `Project(Ref(rel), [head, ...tail])` — the path traverses through a
182        // relation field (`rel`) before reaching the relation the subquery
183        // targets. Re-root the path at `rel`'s target model and rebuild as a
184        // nested IN-subquery on that model, then recurse to lift the outer
185        // `rel` hop.
186        stmt::Expr::Project(project_expr) => {
187            return lift_projection_in_subquery(cx, project_expr, query);
188        }
189        stmt::Expr::Reference(expr_reference @ stmt::ExprReference::Field { .. }) => {
190            cx.resolve_expr_reference(expr_reference).as_field_unwrap()
191        }
192        _ => {
193            return None;
194        }
195    };
196
197    // If the field is not a relation, abort. Direct relations lift through
198    // their paired foreign keys. A `via` has no single pair, so normalize its
199    // relation chain into a projected path and use the projection lift.
200    match &field.ty {
201        FieldTy::BelongsTo(belongs_to) => lift_belongs_to_in_subquery(cx, belongs_to, query),
202        FieldTy::Has(has) => lift_has_n_in_subquery(has.target, has.pair(&cx.schema().app), query),
203        FieldTy::Via(via) => lift_via_in_subquery(cx, via, query),
204        _ => None,
205    }
206}

208/// Lift an `IN` subquery whose left side names a `via` relation.
209///
210/// For `User.groups.any(Group.name == "Rust")`, where `groups` follows
211/// `User.memberships.group`, this builds:
212///
213/// ```text
214/// User.id IN (
215///     SELECT Membership.user_id FROM Membership
216///     WHERE Membership.group_id IN (
217///         SELECT Group.id FROM Group WHERE Group.name == "Rust"
218///     )
219/// )
220/// ```
221///
222/// The path is expanded into direct relation fields, converted to the same
223/// projected expression produced by an explicit relation chain, and then
224/// handled by the regular projection lift.
225fn lift_via_in_subquery(
226    cx: &ExprContext,
227    via: &app::Via,
228    query: &stmt::Query,
229) -> Option<stmt::Expr> {
230    if via.is_scalar() {
231        return None;
232    }
233
234    let fields = super::relation_path::flatten_via_path(cx.schema(), via)?;
235    let target = fields
236        .last()
237        .and_then(|field| cx.schema().app.field(*field).relation_target_id())?;
238
239    if target != via.target || target != query.body.as_select_unwrap().source.model_id_unwrap() {
240        return None;
241    }
242
243    let (base, projection) = fields.split_first()?;
244    let base = stmt::Expr::ref_self_field(*base);
245    let path = if projection.is_empty() {
246        base
247    } else {
248        let projection = projection
249            .iter()
250            .map(|field| field.index)
251            .collect::<Vec<_>>();
252        stmt::Expr::project(base, projection.as_slice())
253    };
254
255    lift_in_subquery(cx, &path, query)
256}

258/// Lifts an `IN`-subquery whose left side is a path through a relation field.
259///
260/// `.any()` on a relation chain — for example, `Release.project.topics.any(name
261/// == "rust")` — builds an `InSubquery` whose LHS projects through the
262/// chain:
263///
264/// ```text
265/// InSubquery {
266///     expr:  Project(Ref(Release.project), [Project.topics_idx]),
267///     query: SELECT FROM Topic WHERE name == "rust",
268/// }
269/// ```
270///
271/// Two paths handle this:
272///
273/// **Fused.** [`try_fuse_paired_relations`] recognizes the common
274/// `BelongsTo → Has` chain over a shared primary key and emits a single
275/// FK-on-FK `IN` that bypasses the intermediate model:
276///
277/// ```text
278/// Release.project_id IN (SELECT Topic.project_id FROM Topic WHERE name == "rust")
279/// ```
280///
281/// **General.** For chains the fast path doesn't match (multi-hop
282/// projections, non-paired relations), re-root the projection at the
283/// relation's target model and wrap the original subquery in a nested `IN`
284/// against that model, then recurse on the outer hop. For the same input,
285/// the fallback produces:
286///
287/// ```text
288/// Release.project IN (
289///     SELECT FROM Project
290///     WHERE Project.topics IN (SELECT FROM Topic WHERE name == "rust")
291/// )
292/// ```
293///
294/// The recursive call lifts the outer `Release.project` hop via the standard
295/// `BelongsTo` branch; the inner `Project.topics` hop is lifted on the next
296/// visitor pass when [`LiftInSubquery`]'s children walk reaches it.
297fn lift_projection_in_subquery(
298    cx: &ExprContext,
299    project_expr: &stmt::ExprProject,
300    query: &stmt::Query,
301) -> Option<stmt::Expr> {
302    let Expr::Reference(expr_ref) = &*project_expr.base else {
303        return None;
304    };
305    let ResolvedRef::Field(field) = cx.resolve_expr_reference(expr_ref) else {
306        return None;
307    };
308
309    let target_model_id = field.relation_target_id()?;
310
311    let (head_idx, tail) = project_expr.projection.as_slice().split_first()?;
312
313    if tail.is_empty()
314        && let Some(direct) =
315            try_fuse_paired_relations(cx, field, target_model_id, *head_idx, query)
316    {
317        return Some(direct);
318    }
319
320    let target_field = Expr::ref_self_field(FieldId {
321        model: target_model_id,
322        index: *head_idx,
323    });
324    let inner_lhs = if tail.is_empty() {
325        target_field
326    } else {
327        Expr::project(target_field, stmt::Projection::from(tail))
328    };
329
330    let new_subquery = stmt::Query::new_select(
331        stmt::Source::from(target_model_id),
332        Expr::in_subquery(inner_lhs, query.clone()),
333    );
334
335    lift_in_subquery(cx, &project_expr.base, &new_subquery)
336}

338/// Fuses a `BelongsTo → Has` chain into a single FK-on-FK `IN` when both
339/// relations meet at the same primary key.
340///
341/// # Example
342///
343/// Given the schema:
344///
345/// ```text
346/// Todo     { category_id: Category.id, ... }   // BelongsTo  Todo.category
347/// Category { todos: HasMany Todo, ... }        // Has, paired with Todo.category
348/// ```
349///
350/// The path `Todo.category.todos` lifts as follows:
351///
352/// ```text
353/// // Input
354/// InSubquery {
355///     expr:  Project(Ref(Todo.category), [Category.todos_idx]),
356///     query: SELECT FROM Todo WHERE title == "salad",
357/// }
358///
359/// // Output
360/// InSubquery {
361///     expr:  Ref(Todo.category_id),
362///     query: SELECT Todo.category_id FROM Todo WHERE title == "salad",
363/// }
364/// ```
365///
366/// The outer relation's FK source columns become the LHS; the inner
367/// relation's paired-BelongsTo FK source columns become the subquery's
368/// returning list. The user's original filter is preserved verbatim.
369///
370/// # Why the fusion is sound
371///
372/// Composing the two hops without fusion routes through the intermediate
373/// model:
374///
375/// ```text
376/// Todo.category_id IN (
377///     SELECT Category.id FROM Category
378///     WHERE Category.id IN (SELECT Todo.category_id FROM Todo WHERE title == "salad")
379/// )
380/// ```
381///
382/// Both FKs target `Category.id`, so every value the innermost subquery
383/// returns exists in `Category.id` under FK integrity (which the engine
384/// already assumes). The middle filter therefore admits every row the inner
385/// returns, and the middle `SELECT Category.id` simply re-emits them. The
386/// `Category` scan is a no-op and can be dropped.
387///
388/// Returns `None` when the chain doesn't match the pattern — outer isn't a
389/// `BelongsTo`, inner isn't a `Has`, or the FK columns don't line up.
390fn try_fuse_paired_relations(
391    cx: &ExprContext,
392    outer_field: &app::Field,
393    target_model_id: ModelId,
394    head_idx: usize,
395    query: &stmt::Query,
396) -> Option<stmt::Expr> {
397    let outer_belongs_to = match &outer_field.ty {
398        FieldTy::BelongsTo(rel) => rel,
399        _ => return None,
400    };
401
402    let target_model = cx.schema().app.model(target_model_id).as_root_unwrap();
403    let head_field = target_model.fields.get(head_idx)?;
404    let inner_has = match &head_field.ty {
405        FieldTy::Has(has) => has,
406        _ => return None,
407    };
408    let inner_pair = inner_has.pair(&cx.schema().app);
409
410    // Both FKs must reference the same PK columns in the same order.
411    if outer_belongs_to.foreign_key.fields.len() != inner_pair.foreign_key.fields.len() {
412        return None;
413    }
414    for (outer_fk, inner_fk) in outer_belongs_to
415        .foreign_key
416        .fields
417        .iter()
418        .zip(inner_pair.foreign_key.fields.iter())
419    {
420        if outer_fk.target != inner_fk.target {
421            return None;
422        }
423    }
424
425    lift_fk_in_subquery(
426        inner_has.target,
427        super::key_field_refs(
428            0,
429            outer_belongs_to
430                .foreign_key
431                .fields
432                .iter()
433                .map(|fk| fk.source),
434        ),
435        super::key_field_refs(0, inner_pair.foreign_key.fields.iter().map(|fk| fk.source)),
436        query,
437    )
438}

440/// Rewrites a comparison that walks a relation field into a foreign-key
441/// subquery.
442///
443/// `Profile::filter(Profile::fields().user().name().eq("alice"))` starts as
444/// the filter `profile.user.name = 'alice'`, where the left side projects
445/// through the `user` relation. This moves the comparison into a subquery on
446/// the target model and defers to [`lift_in_subquery`]:
447///
448/// ```text
449/// Profile.user_id IN (SELECT User.id FROM User WHERE User.name = 'alice')
450/// ```
451///
452/// Returns `None` when `project_side` does not walk a relation field.
453pub(super) fn try_lift_relation_path_comparison(
454    cx: &ExprContext,
455    op: stmt::BinaryOp,
456    project_side: &stmt::Expr,
457    other_side: &stmt::Expr,
458) -> Option<stmt::Expr> {
459    lift_relation_path_predicate(cx, project_side, |target_lhs| {
460        Expr::binary_op(target_lhs, op, other_side.clone())
461    })
462}

464/// Rewrites a `LIKE`/`ILIKE` that walks a relation field into a foreign-key
465/// subquery — [`try_lift_relation_path_comparison`] for pattern matches.
466///
467/// `Profile::filter(Profile::fields().user().name().like("al%"))` rewrites to:
468///
469/// ```text
470/// Profile.user_id IN (SELECT User.id FROM User WHERE User.name LIKE 'al%')
471/// ```
472///
473/// `LIKE`/`ILIKE` are `Expr::Like`, not `Expr::BinaryOp`, so they never reach
474/// [`try_lift_relation_path_comparison`] and need this entry point. Without
475/// it the `user.name` path stays a projection through the relation, which the
476/// rest of lowering cannot turn into a column and so panics.
477///
478/// Returns `None` when the pattern's subject does not walk a relation field.
479pub(super) fn try_lift_relation_path_like(
480    cx: &ExprContext,
481    like: &stmt::ExprLike,
482) -> Option<stmt::Expr> {
483    lift_relation_path_predicate(cx, &like.expr, |target_lhs| {
484        stmt::ExprLike {
485            expr: Box::new(target_lhs),
486            pattern: like.pattern.clone(),
487            escape: like.escape,
488            case_insensitive: like.case_insensitive,
489        }
490        .into()
491    })
492}

494/// Shared core of the relation-path lifts.
495///
496/// `project_side` is the side of the predicate that walks a relation — the
497/// `profile.user.name` in `profile.user.name = 'alice'`. It is a projection
498/// whose first step names the `user` relation field on `Profile` and whose
499/// remaining steps index into `User`. This re-roots the path at the target
500/// model (so it reads `User.name`), builds a `SELECT` over `User` whose filter
501/// `make_filter` produces from the re-rooted path, and defers to
502/// [`lift_in_subquery`] to turn the relation reference into the foreign-key
503/// `IN` form.
504///
505/// `make_filter` is the only difference between callers: a binary op for
506/// comparisons, a `LIKE` for pattern matches.
507///
508/// Returns `None` when `project_side` does not walk a relation field.
509fn lift_relation_path_predicate(
510    cx: &ExprContext,
511    project_side: &stmt::Expr,
512    make_filter: impl FnOnce(stmt::Expr) -> stmt::Expr,
513) -> Option<stmt::Expr> {
514    let Expr::Project(project_expr) = project_side else {
515        return None;
516    };
517    let Expr::Reference(expr_ref) = &*project_expr.base else {
518        return None;
519    };
520    let ResolvedRef::Field(field) = cx.resolve_expr_reference(expr_ref) else {
521        return None;
522    };
523
524    let target_model_id = field.relation_target_id()?;
525
526    // The first step names the relation field on the source model; the
527    // rest index into the target model. Drop the first step and re-root the
528    // remainder at the target model.
529    let (head_idx, tail) = project_expr.projection.as_slice().split_first()?;
530    let target_field = Expr::ref_self_field(FieldId {
531        model: target_model_id,
532        index: *head_idx,
533    });
534    let target_lhs = if tail.is_empty() {
535        target_field
536    } else {
537        Expr::project(target_field, stmt::Projection::from(tail))
538    };
539
540    let subquery =
541        stmt::Query::new_select(stmt::Source::from(target_model_id), make_filter(target_lhs));
542
543    lift_in_subquery(cx, &project_expr.base, &subquery)
544}

546/// Build a foreign-key `IN` subquery for one direct relation edge.
547///
548/// `lhs` references key fields on the current model. `returning` references
549/// the matching key fields on `target`, which is also the subquery source.
550fn lift_fk_in_subquery(
551    target: ModelId,
552    lhs: stmt::Expr,
553    returning: stmt::Expr,
554    query: &stmt::Query,
555) -> Option<stmt::Expr> {
556    if target != query.body.as_select_unwrap().source.model_id_unwrap() {
557        return None;
558    }
559
560    let mut subquery = query.clone();
561    subquery.body.as_select_mut_unwrap().returning = stmt::Returning::Project(returning);
562
563    Some(stmt::Expr::in_subquery(lhs, subquery))
564}

566/// BelongsTo branch: try to lift the subquery's filter into direct FK
567/// comparisons.  When the filter references only FK-mapped fields, return
568/// the AND of per-FK equalities.  Otherwise, fall back to an IN subquery
569/// on the foreign-key column(s) — a tuple-form IN for composite FKs.
570///
571/// Returns `None` when the subquery does not target the BelongsTo's
572/// target model.
573fn lift_belongs_to_in_subquery(
574    cx: &ExprContext,
575    belongs_to: &BelongsTo,
576    query: &stmt::Query,
577) -> Option<stmt::Expr> {
578    if belongs_to.target != query.body.as_select_unwrap().source.model_id_unwrap() {
579        return None;
580    }
581
582    let select = query.body.as_select_unwrap();
583
584    let mut lift = LiftBelongsTo {
585        cx: cx.scope(&select.source),
586        belongs_to,
587        fk_field_matches: vec![false; belongs_to.foreign_key.fields.len()],
588        operands: vec![],
589        fail: false,
590    };
591
592    lift.visit_filter(&select.filter);
593
594    // Fall back to the IN-subquery form whenever we couldn't account for
595    // every FK column with a direct equality from the filter. This covers
596    // both `fail=true` (a binary op referenced something that isn't on the
597    // FK) and the case where the filter contained no liftable binary ops at
598    // all — e.g. when it's a nested `InSubquery` that the LiftBelongsTo
599    // visitor deliberately skips (see `visit_expr_in_subquery`).
600    let all_fks_matched = lift.fk_field_matches.iter().all(|m| *m);
601
602    if lift.fail || !all_fks_matched {
603        lift_fk_in_subquery(
604            belongs_to.target,
605            super::key_field_refs(0, belongs_to.foreign_key.fields.iter().map(|fk| fk.source)),
606            super::key_field_refs(0, belongs_to.foreign_key.fields.iter().map(|fk| fk.target)),
607            query,
608        )
609    } else {
610        Some(if lift.operands.len() == 1 {
611            lift.operands.into_iter().next().unwrap()
612        } else {
613            stmt::ExprAnd {
614                operands: lift.operands,
615            }
616            .into()
617        })
618    }
619}

621/// HasOne/HasMany branch: rewrite to a foreign-key IN subquery against
622/// the related table. Single-column FKs produce a scalar IN; composite
623/// FKs produce a tuple-form IN that the SQL serializer renders as
624/// `(a, b) IN (SELECT a, b FROM ...)`.
625fn lift_has_n_in_subquery(
626    target: ModelId,
627    pair: &BelongsTo,
628    query: &stmt::Query,
629) -> Option<stmt::Expr> {
630    lift_fk_in_subquery(
631        target,
632        super::key_field_refs(0, pair.foreign_key.fields.iter().map(|fk| fk.target)),
633        super::key_field_refs(0, pair.foreign_key.fields.iter().map(|fk| fk.source)),
634        query,
635    )
636}
637
638impl Visit for LiftBelongsTo<'_> {
639    fn visit_expr_in_subquery(&mut self, _i: &stmt::ExprInSubquery) {
640        // Stop the walk at a nested IN-subquery boundary. Field references
641        // inside resolve in the nested query's own scope, but the visitor's
642        // `cx` is scoped to the current (outer) subquery's source — recursing
643        // would misresolve inner refs as outer-model fields and produce
644        // bogus FK matches. The main LiftInSubquery walker handles the
645        // nested subquery on its own pass, with the correct scope.
646    }
647
648    fn visit_expr_binary_op(&mut self, i: &stmt::ExprBinaryOp) {
649        match (&*i.lhs, &*i.rhs) {
650            (stmt::Expr::Reference(expr_reference), other)
651            | (other, stmt::Expr::Reference(expr_reference)) => {
652                assert!(i.op.is_eq() || i.op.is_ne());
653
654                if i.op.is_eq() || i.op.is_ne() {
655                    let field = self
656                        .cx
657                        .resolve_expr_reference(expr_reference)
658                        .as_field_unwrap();
659
660                    self.lift_fk_constraint(field.id, i.op, other);
661                } else {
662                    self.fail = true;
663                }
664            }
665            // Constraints we can't lift to a direct FK comparison (e.g. a
666            // projection through an embedded field).  Bail to the IN-subquery
667            // form so the filter is preserved verbatim; without this, the
668            // empty `operands` list silently produced an empty AND (= true)
669            // and the subquery returned every row.
670            _ => {
671                self.fail = true;
672            }
673        }
674    }
675}
676
677impl LiftBelongsTo<'_> {
678    fn lift_fk_constraint(&mut self, field: FieldId, op: stmt::BinaryOp, expr: &stmt::Expr) {
679        for (i, fk_field) in self.belongs_to.foreign_key.fields.iter().enumerate() {
680            if fk_field.target == field {
681                if self.fk_field_matches[i] {
682                    todo!("not handled");
683                }
684
685                self.operands.push(stmt::Expr::binary_op(
686                    stmt::Expr::ref_self_field(fk_field.source),
687                    op,
688                    expr.clone(),
689                ));
690                self.fk_field_matches[i] = true;
691
692                return;
693            }
694        }
695
696        self.fail = true;
697    }
698}

14/// High-level Intermediate Representation of a query.
15///
16/// [`HirStatement`] is a collection of related statements with tracked
17/// dependencies, produced by the lowering phase. It captures the structure of
18/// a query after model-to-table transformation but before operation graph
19/// construction.
20///
21/// The HIR tracks which statements depend on results from others, enabling the
22/// planner to determine execution order and identify opportunities for batching.
23/// The dependency graph may contain cycles when preloading associations (e.g.,
24/// users → todos → users), which the planning phase must break when building
25/// the final DAG.
26#[derive(Debug)]
27pub(super) struct HirStatement {
28    /// Storage for all statement metadata, indexed by [`StmtId`].
29    store: IndexVec<StmtId, StatementInfo>,
30}
31
32/// Metadata for a single statement within the HIR.
33///
34/// Each [`StatementInfo`] represents one database operation that will execute
35/// as part of the query. Not every sub-expression in the original query becomes
36/// a [`StatementInfo`]; only those that execute as separate database operations
37/// (e.g., the root query and any `include()` subqueries).
38///
39/// It tracks the statement's dependencies on other statements, arguments passed
40/// from parent or child statements, and references to the MIR nodes created
41/// during planning.
42#[derive(Debug)]
43pub(super) struct StatementInfo {
44    /// The lowered statement to execute.
45    ///
46    /// Initially `None`, populated during lowering after all transformations
47    /// are complete. Contains the table-level statement ready for planning.
48    pub(super) stmt: Option<Box<stmt::Statement>>,
49
50    /// Statement IDs that must execute before this statement.
51    ///
52    /// Dependencies ensure execution order for consistency, even when this
53    /// statement does not consume the dependency's result. For example, an
54    /// `UPDATE` may depend on a prior `INSERT` to maintain referential
55    /// integrity.
56    pub(super) deps: HashSet<StmtId>,
57
58    /// Arguments that flow into this statement from other statements.
59    ///
60    /// Each [`Arg`] describes data passed from a sub-statement ([`Arg::Sub`]) or
61    /// referenced from a parent statement ([`Arg::Ref`]). During planning, these
62    /// arguments become edges in the operation graph.
63    pub(super) args: Vec<Arg>,
64
65    /// Column references from child statements that point back to this one.
66    ///
67    /// When a child statement references columns from this statement (via
68    /// [`Arg::Ref`]), those columns must be included in this statement's
69    /// batch-load query. The key is the child's [`StmtId`], and the value
70    /// tracks which columns are referenced.
71    pub(super) back_refs: HashMap<StmtId, BackRef>,
72
73    /// MIR node that executes this statement's database query.
74    ///
75    /// Set during planning when the statement is converted to an operation.
76    /// Used to wire up dependencies between operations.
77    pub(super) load_data_statement: Cell<Option<mir::NodeId>>,
78
79    /// Items selected by the `exec_statement` operation.
80    ///
81    /// Populated during planning to track which items are fetched from the
82    /// database. Used to resolve column references in child statements.
83    pub(super) load_data_select_items: OnceCell<SelectItems>,
84
85    /// MIR node representing this statement's final output.
86    ///
87    /// May differ from `exec_statement` when post-processing is required
88    /// (e.g., filtering, projection, or nested merge). Dependent statements
89    /// reference this node to establish execution order.
90    pub(super) output: Cell<Option<mir::NodeId>>,
91
92    /// When true, the statement is independent. An independent statement does
93    /// not depend on any anestors itself nor do any of its sub-dependencies.
94    pub(super) independent: bool,
95}

101impl StatementInfo {
102    /// Creates a new [`StatementInfo`] with the given dependencies.
103    ///
104    /// All other fields are initialized to empty or `None` and populated
105    /// later during lowering and planning.
106    pub(super) fn new(deps: HashSet<StmtId>) -> StatementInfo {
107        StatementInfo {
108            stmt: None,
109            deps,
110            args: vec![],
111            back_refs: HashMap::new(),
112            load_data_statement: Cell::new(None),
113            load_data_select_items: OnceCell::new(),
114            output: Cell::new(None),
115            independent: true,
116        }
117    }
118
119    pub(super) fn stmt(&self) -> &stmt::Statement {
120        self.stmt.as_deref().unwrap()
121    }
122}

124/// Tracks columns referenced by a child statement.
125///
126/// When a child statement (e.g., an `include()` subquery) references columns
127/// from its parent, the parent must include those columns in its result set.
128/// [`BackRef`] records which columns are needed and the MIR node that projects
129/// them for the child's batch-load operation.
130#[derive(Debug, Default)]
131pub(super) struct BackRef {
132    /// Column expressions from this statement that are referenced by a child statement.
133    ///
134    /// When a child statement references columns from this statement (via
135    /// [`Arg::Ref`]), those columns must be included in this statement's
136    /// batch-load query. This set tracks which columns need to be loaded so they
137    /// can be used during nested merge.
138    pub(super) exprs: IndexSet<stmt::ExprReference>,
139
140    /// Node ID of the projection operation that extracts these back-ref columns.
141    ///
142    /// After executing this statement, a projection node is created to extract just
143    /// the columns needed by child statements. This projection's output is used as
144    /// input to the child statement's batch-load operation.
145    pub(super) node_id: Cell<Option<mir::NodeId>>,
146}

148/// An argument that flows between statements in the HIR.
149///
150/// Arguments represent data dependencies between statements. They describe how
151/// results from one statement are used by another, enabling the planner to wire
152/// up the operation graph correctly.
153#[derive(Debug)]
154pub(super) enum Arg {
155    /// Data from a sub-statement that feeds into this statement.
156    Sub {
157        /// The statement ID that provides the data for this argument.
158        stmt_id: StmtId,
159
160        /// True when the sub-statement is used in the returning clause, false when used in filters.
161        ///
162        /// Determines how the sub-statement is handled during planning:
163        /// - `true`: Data is merged with parent rows via [`NestedMerge`](crate::engine::mir::NestedMerge)
164        /// - `false`: Data is used as input to filter expressions
165        returning: bool,
166
167        /// Index in the operation's inputs list. Set during planning.
168        input: Cell<Option<usize>>,
169
170        /// Depending on the statement type, this is used in different ways. For
171        /// Query types, it is the index of the TableRef that will provide the
172        /// data for this ref. For Insert statements, it is the offset at which
173        /// the data should be fetched.
174        batch_load_index: Cell<Option<usize>>,
175    },
176
177    /// A reference to a parent statement's columns.
178    Ref {
179        /// The expression reference relative to the target statement.
180        target_expr_ref: stmt::ExprReference,
181
182        /// The parent statement that provides the data for this reference.
183        stmt_id: StmtId,
184
185        /// Number of nesting levels between this statement and the referenced parent.
186        ///
187        /// A value of 1 means the immediate parent, 2 means the grandparent, etc.
188        nesting: usize,
189
190        /// The MIR node input when the ref is used during the data loading phase
191        data_load_input: Cell<Option<usize>>,
192
193        /// The MIR node input when the ref is used during the returing phase
194        returning_input: Cell<Option<usize>>,
195
196        /// Depending on the statement type, this is used in different ways. For
197        /// Query types, it is the index of the TableRef that will provide the
198        /// data for this ref. For Insert statements, it is the offset at which
199        /// the data should be fetched.
200        batch_load_index: Cell<Option<usize>>,
201    },
202}

204impl HirStatement {
205    /// Creates an empty [`HirStatement`] with no statements.
206    pub(super) fn new() -> HirStatement {
207        HirStatement {
208            store: IndexVec::new(),
209        }
210    }
211
212    /// Inserts a [`StatementInfo`] and returns its assigned [`StmtId`].
213    pub(super) fn insert(&mut self, info: StatementInfo) -> StmtId {
214        self.store.push(info)
215    }
216
217    /// Creates and inserts a new [`StatementInfo`] with the given dependencies.
218    pub(super) fn new_statement_info(&mut self, deps: HashSet<StmtId>) -> StmtId {
219        self.insert(StatementInfo::new(deps))
220    }
221
222    /// Returns the [`StmtId`] of the root statement.
223    ///
224    /// The root statement is always the first one inserted (index 0) and
225    /// represents the top-level query that produces the final result.
226    pub(super) fn root_id(&self) -> StmtId {
227        StmtId::from(0)
228    }
229
230    /// Returns a reference to the root [`StatementInfo`].
231    pub(super) fn root(&self) -> &StatementInfo {
232        let root_id = self.root_id();
233        &self.store[root_id]
234    }
235}

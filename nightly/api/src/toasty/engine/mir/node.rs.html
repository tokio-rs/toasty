10/// A single node in the MIR operation graph.
11///
12/// Each [`Node`] represents one operation to execute. It contains the operation
13/// itself, its dependencies on other nodes, and metadata used during execution
14/// planning (variable assignment, reference counting, traversal state).
15#[derive(Debug)]
16pub(crate) struct Node {
17    /// The operation this node performs.
18    pub(crate) op: Operation,
19
20    /// Nodes that must execute before this one.
21    ///
22    /// This is a superset of the node's data inputs; it may include additional
23    /// ordering dependencies (e.g., an `UPDATE` depending on a prior `INSERT`).
24    pub(crate) deps: IndexSet<NodeId>,
25
26    /// Variable slot where this node's output is stored during execution.
27    ///
28    /// Set during execution planning when converting MIR to actions.
29    pub(crate) var: Cell<Option<exec::VarId>>,
30
31    /// Number of downstream nodes that consume this node's output.
32    ///
33    /// Used for reference counting; the output is freed after the last use.
34    pub(crate) num_uses: Cell<usize>,
35
36    /// Whether this node has been visited during topological sort.
37    pub(crate) visited: Cell<bool>,
38}
39
40impl Node {
41    pub(crate) fn ty(&self) -> &stmt::Type {
42        match &self.op {
43            Operation::Const(m) => &m.ty,
44            Operation::DeleteByKey(m) => &m.ty,
45            Operation::Eval(m) => &m.eval.ret,
46            Operation::ExecStatement(m) => &m.ty,
47            Operation::Filter(m) => &m.ty,
48            Operation::FindPkByIndex(m) => &m.ty,
49            Operation::GetByKey(m) => &m.ty,
50            Operation::Guard(m) => &m.ty,
51            Operation::QueryPk(m) => &m.ty,
52            Operation::Scan(m) => &m.ty,
53            Operation::Project(m) => &m.ty,
54            Operation::UpdateByKey(m) => &m.ty,
55            Operation::NestedMerge(_m) => todo!(),
56            Operation::ReadModifyWrite(m) => &m.ty,
57        }
58    }
59
60    pub(crate) fn to_exec(
61        &self,
62        logical_plan: &LogicalPlan,
63        var_table: &mut exec::VarDecls,
64    ) -> exec::Action {
65        match &self.op {
66            Operation::Const(op) => op.to_exec(self, var_table).into(),
67            Operation::DeleteByKey(op) => op.to_exec(logical_plan, self, var_table).into(),
68            Operation::Eval(op) => op.to_exec(logical_plan, self, var_table).into(),
69            Operation::ExecStatement(op) => op.to_exec(logical_plan, self, var_table).into(),
70            Operation::Filter(op) => op.to_exec(logical_plan, self, var_table).into(),
71            Operation::FindPkByIndex(op) => op.to_exec(logical_plan, self, var_table).into(),
72            Operation::GetByKey(op) => op.to_exec(logical_plan, self, var_table).into(),
73            Operation::Guard(op) => op.to_exec(logical_plan, self, var_table).into(),
74            Operation::NestedMerge(op) => op.to_exec(logical_plan, self, var_table).into(),
75            Operation::Project(op) => op.to_exec(logical_plan, self, var_table).into(),
76            Operation::ReadModifyWrite(op) => op.to_exec(logical_plan, self, var_table).into(),
77            Operation::QueryPk(op) => op.to_exec(logical_plan, self, var_table).into(),
78            Operation::Scan(op) => op.to_exec(logical_plan, self, var_table).into(),
79            Operation::UpdateByKey(op) => op.to_exec(logical_plan, self, var_table).into(),
80        }
81    }
82}

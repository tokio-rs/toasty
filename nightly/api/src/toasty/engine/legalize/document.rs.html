1//! Legalization rule: rewrite `#[document]` path reads into the shape
2//! drivers consume.
3//!
4//! The engine views a document column at the app level: the mapping's raising
5//! cast (`Cast(column, Type::Model)`) turns the driver's named `Value::Object`
6//! into the embed's positional `Value::Record`, and a path into the document
7//! is a positional projection through that cast —
8//! `Project(Cast(column, Model), steps)` — which the in-memory interpreter
9//! evaluates directly. Drivers and the database schema see only the
10//! structural view: the column is typed `Type::Object`, its value is a named
11//! `Value::Object`, and a path is a resolved [`stmt::FuncJsonExtract`] name
12//! path. The model identity never crosses the driver boundary.
13//!
14//! Value conversion between the two views is the mapping's job: the schema
15//! builder plants a lowering cast (`Model` → `Object`, carrying the source
16//! type — see `stmt::ExprCast::from`) in `model_to_table` and the raising
17//! cast in `table_to_model`, and the cast machinery (fold/simplify constant
18//! folding, `mir::Project` evaluation) converts values wherever the casts
19//! land. What cannot be expressed as a cast is the *path read*: turning a
20//! positional projection into a JSON key path is a per-backend expression
21//! rewrite (the leaf's comparison form depends on driver capabilities). That
22//! makes it a legalization rule rather than a simplifier rewrite (the
23//! simplifier runs identically for every backend) or a serializer concern
24//! (the statement must already be representable when the driver receives it).
25
26use toasty_core::{
27    driver::Capability,
28    schema::{Schema, app},
29    stmt::{self, VisitMut},
30};

32/// Rewrite every projection into a `#[document]` column into the
33/// [`FuncJsonExtract`](stmt::FuncJsonExtract) node the SQL serializer
34/// renders, and strip any residual raising cast (meaningless driver-side).
35pub(super) fn statement(schema: &Schema, capability: &Capability, stmt: &mut stmt::Statement) {
36    LegalizeDocumentPaths { schema, capability }.visit_mut(stmt);
37}
38
39/// Rewrite a driver-bound expression — a key-value operation's filter or
40/// condition — into its driver-consumable shape: document paths become
41/// [`stmt::FuncJsonExtract`] name paths, and text-compared document leaves
42/// get text operands, exactly as [`statement`] does for full statements.
43pub(super) fn table_expr(schema: &Schema, capability: &Capability, expr: &mut stmt::Expr) {
44    LegalizeDocumentPaths { schema, capability }.visit_expr_mut(expr);
45}

47/// Traversal backing [`statement`]. Needs no scope tracking: the rewrite
48/// matches the document-path shape structurally and takes the embed identity
49/// from the raising cast's type — it never resolves a column reference
50/// against the enclosing query's source.
51struct LegalizeDocumentPaths<'a> {
52    schema: &'a Schema,
53    // Read only by `leaf_compares_as_text`, whose non-trivial arms are gated
54    // on the features below.
55    #[cfg_attr(
56        not(any(feature = "jiff", feature = "rust_decimal", feature = "bigdecimal")),
57        allow(dead_code)
58    )]
59    capability: &'a Capability,
60}
61
62impl LegalizeDocumentPaths<'_> {
63    /// Rewrite a projection rooted at a `#[document]` column into the
64    /// [`FuncJsonExtract`](stmt::FuncJsonExtract) the SQL serializer renders.
65    ///
66    /// The lowered form of a document path is a positional projection through
67    /// the column's raising cast — `Project(Cast(column, Model), steps)`.
68    /// Driver-side there is no positional form: the projection becomes a JSON
69    /// key path rooted at the bare column reference.
70    fn lower_project(&self, expr: &mut stmt::Expr) {
71        let stmt::Expr::Project(project) = expr else {
72            return;
73        };
74        // Only a projection through a document raising cast rooted at a
75        // column reference is a document path; anything else is left for
76        // later phases.
77        let stmt::Expr::Cast(cast) = &mut *project.base else {
78            return;
79        };
80        let stmt::Type::Model(embed_id) = cast.ty else {
81            return;
82        };
83        if !matches!(&*cast.expr, stmt::Expr::Reference(_)) {
84            return;
85        }
86        let Some((path, ty)) =
87            build_json_path(self.schema, embed_id, project.projection.as_slice())
88        else {
89            return;
90        };
91        let base = Box::new(cast.expr.take());
92        *expr = stmt::Expr::from(stmt::FuncJsonExtract { base, path, ty });
93    }
94
95    /// Whether a document leaf of type `ty` compares as plain text on this
96    /// backend. PostgreSQL and MySQL cast the JSON extraction back to the
97    /// leaf's native SQL type, but a backend without that native type (SQLite
98    /// has no temporal or decimal column types) compares the extracted text
99    /// directly — so the bound operand must be the exact text the JSON codec
100    /// stores.
101    fn leaf_compares_as_text(&self, ty: &stmt::Type) -> bool {
102        match ty {
103            #[cfg(feature = "jiff")]
104            stmt::Type::Timestamp => !self.capability.native_timestamp,
105            #[cfg(feature = "jiff")]
106            stmt::Type::Date => !self.capability.native_date,
107            #[cfg(feature = "jiff")]
108            stmt::Type::Time => !self.capability.native_time,
109            #[cfg(feature = "jiff")]
110            stmt::Type::DateTime => !self.capability.native_datetime,
111            #[cfg(feature = "rust_decimal")]
112            stmt::Type::Decimal => !self.capability.native_decimal,
113            #[cfg(feature = "bigdecimal")]
114            stmt::Type::BigDecimal => !self.capability.native_decimal,
115            _ => false,
116        }
117    }
118
119    /// If `extract_side` is a JSON extraction whose leaf compares as text on
120    /// this backend, rewrite the constant `operand` to the leaf's document
121    /// text form (the exact text the JSON codec stores — see
122    /// [`document_text`]) and retype the extraction as a text read.
123    fn textify_comparison(&self, extract_side: &mut stmt::Expr, operand: &mut stmt::Expr) {
124        let stmt::Expr::Func(stmt::ExprFunc::JsonExtract(func)) = extract_side else {
125            return;
126        };
127        if !self.leaf_compares_as_text(&func.ty) {
128            return;
129        }
130        let stmt::Expr::Value(value) = operand else {
131            return;
132        };
133        let Some(text) = document_text(value) else {
134            return;
135        };
136        *value = stmt::Value::String(text);
137        func.ty = stmt::Type::String;
138    }
139}
140
141impl VisitMut for LegalizeDocumentPaths<'_> {
142    fn visit_expr_mut(&mut self, expr: &mut stmt::Expr) {
143        // A document path must be lowered before descending into its base:
144        // the base is the raising cast that names the embed, which the
145        // residual cast strip below would otherwise remove first (children
146        // are visited before their parent).
147        if let stmt::Expr::Project(_) = expr {
148            self.lower_project(expr);
149        }
150
151        stmt::visit_mut::visit_expr_mut(self, expr);
152
153        // Children are visited first, so by the time a comparison node is
154        // reached its document-path side is already a `FuncJsonExtract`.
155        match expr {
156            // A raising cast left outside a document path (a whole-document
157            // reference) is meaningless driver-side: the column already
158            // carries the named object. Strip it to the bare reference.
159            stmt::Expr::Cast(expr_cast) if expr_cast.ty.contains_model() => {
160                *expr = expr_cast.expr.take();
161            }
162            stmt::Expr::BinaryOp(binary) => {
163                self.textify_comparison(&mut binary.lhs, &mut binary.rhs);
164                self.textify_comparison(&mut binary.rhs, &mut binary.lhs);
165            }
166            stmt::Expr::InList(in_list) => {
167                let stmt::Expr::Func(stmt::ExprFunc::JsonExtract(func)) = &mut *in_list.expr else {
168                    return;
169                };
170                if !self.leaf_compares_as_text(&func.ty) {
171                    return;
172                }
173                match &mut *in_list.list {
174                    stmt::Expr::List(list) => {
175                        for item in &mut list.items {
176                            if let stmt::Expr::Value(value) = item
177                                && let Some(text) = document_text(value)
178                            {
179                                *value = stmt::Value::String(text);
180                            }
181                        }
182                    }
183                    stmt::Expr::Value(stmt::Value::List(items)) => {
184                        for value in items.iter_mut() {
185                            if let Some(text) = document_text(value) {
186                                *value = stmt::Value::String(text);
187                            }
188                        }
189                    }
190                    _ => return,
191                }
192                func.ty = stmt::Type::String;
193            }
194            _ => {}
195        }
196    }
197}

199/// Resolves a projection through (possibly nested) document types via the
200/// schema's shared [`project_fields`](app::Schema::project_fields) walk,
201/// collecting the JSON key path and the leaf field's type. Returns `None` if
202/// the projection does not resolve to a document path.
203fn build_json_path(
204    schema: &Schema,
205    embed_id: app::ModelId,
206    projection: &[usize],
207) -> Option<(Vec<String>, stmt::Type)> {
208    let mut path = Vec::with_capacity(projection.len());
209    let mut leaf_ty = None;
210
211    for field in schema.app.project_fields(embed_id, projection) {
212        path.push(field.name.app.as_deref()?.to_owned());
213        leaf_ty = Some(field.expr_ty().clone());
214    }
215
216    // `project_fields` yields fewer fields than asked when the projection does
217    // not resolve to a document path (an out-of-range step, or a descent past
218    // a non-document leaf).
219    (!path.is_empty() && path.len() == projection.len())
220        .then(|| (path, leaf_ty.expect("a non-empty path has a leaf type")))
221}

223/// The text form `value` takes inside a stored JSON document
224/// ([`stmt::Value::document_storage_text`]), for comparison operands bound
225/// against a plain-text extraction — exactly what the codec's `collect_str`
226/// writes. `None` for values with no document text form (including `Null`,
227/// which comparisons reach via `IsNull` instead).
228fn document_text(value: &stmt::Value) -> Option<String> {
229    #[cfg(any(feature = "jiff", feature = "rust_decimal", feature = "bigdecimal"))]
230    {
231        value.document_storage_text().map(|text| text.to_string())
232    }
233    #[cfg(not(any(feature = "jiff", feature = "rust_decimal", feature = "bigdecimal")))]
234    {
235        let _ = value;
236        None
237    }

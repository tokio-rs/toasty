14/// How to interpret a statement's output rows.
15///
16/// A conditional write (the SQL `#[version]` / OCC path compiled as a single
17/// CTE statement) prefixes its result with two probe columns: the number of
18/// rows matching the filter and, of those, the number satisfying the condition.
19/// The write applied only when the two agree; a mismatch is a condition
20/// failure, and zero matched rows means the record no longer exists.
21#[derive(Debug, Clone, Copy, PartialEq, Eq)]
22pub(crate) enum ConditionalOutput {
23    /// Not a conditional write. Output is passed through unchanged.
24    None,
25
26    /// Conditional write with no `RETURNING`. The two probe columns are the
27    /// only output; the action reports the matched-row count.
28    Count,
29
30    /// Conditional write with a `RETURNING`. The two probe columns are followed
31    /// by the changed rows' columns, which become the action's output.
32    Returning,
33}

45/// Information about a MySQL INSERT with RETURNING that needs special handling.
46///
47/// MySQL doesn't support RETURNING clauses, but we can work around this for
48/// auto-increment columns by using LAST_INSERT_ID().
49#[derive(Debug)]
50struct MySQLInsertReturning {
51    /// Number of rows being inserted
52    num_rows: u64,
53
54    /// The original returning expression that was removed from the statement
55    returning_expr: stmt::Expr,
56
57    /// The type of the auto-increment column
58    auto_column_type: stmt::Type,
59}
60
61/// Information about a MySQL UPDATE with RETURNING that needs special handling.
62///
63/// MySQL doesn't support `RETURNING` on `UPDATE`. The workaround is to strip
64/// the returning, run the UPDATE, then run a follow-up `SELECT` over the same
65/// table and filter to fetch the post-update column values. The two
66/// statements are not atomic relative to concurrent writers — see #881 for
67/// the broader design discussion.
68#[derive(Debug)]
69pub(super) struct MySQLUpdateReturning {
70    /// The `SELECT` statement that returns the post-update values. Carries
71    /// the same filter as the original `UPDATE` plus the projected
72    /// returning expression.
73    select_stmt: stmt::Statement,
74}

102impl Exec<'_> {
103    pub(super) async fn action_exec_statement(&mut self, action: &ExecStatement) -> Result<()> {
104        let mut stmt = action.stmt.clone();
105
106        // Collect input values and substitute into the statement
107        if !action.input.is_empty() {
108            let mut input_values = Vec::new();
109            for var_id in &action.input {
110                let response = self.vars.load(*var_id).await?;
111                let values = response.values.collect_as_value().await?;
112                input_values.push(values);
113            }
114            stmt.substitute(&input_values);
115
116            self.engine.simplify_stmt(&mut stmt);
117        }
118
119        debug_assert!(
120            stmt.returning()
121                .and_then(|returning| returning.as_project())
122                .map(|expr| expr.is_record())
123                .unwrap_or(true),
124            "stmt={stmt:#?}"
125        );
126
127        // MySQL does not support returning clauses with insert statements,
128        // which adds a wrinkle when we want to get the IDs for autoincrement
129        // IDs.
130        let mysql_insert_returning = self.process_stmt_insert_with_returning_on_mysql(&mut stmt);
131
132        // MySQL does not support `RETURNING` on `UPDATE`. Strip the returning
133        // and capture an equivalent `SELECT` to run after the UPDATE.
134        let mysql_update_returning = self.process_stmt_update_with_returning_on_mysql(&mut stmt);
135
136        // Short circuit if we can statically determine there are no results
137        if let stmt::Statement::Query(query) = &stmt
138            && let stmt::ExprSet::Values(values) = &query.body
139            && values.is_empty()
140        {
141            assert_eq!(action.conditional, ConditionalOutput::None);
142
143            let rows = if action.output.ty.is_some() {
144                Rows::Stream(stmt::ValueStream::default())
145            } else {
146                Rows::Count(0)
147            };
148
149            self.vars.store(
150                action.output.output.var,
151                action.output.output.num_uses,
152                ExecResponse::from_rows(rows),
153            );
154
155            return Ok(());
156        }
157
158        // Legalize the statement for the target backend and extract bind
159        // parameters (SQL drivers only; key-value drivers read values
160        // directly from the statement).
161        let params = self.engine.prepare_for_driver(&mut stmt);
162
163        let ret = match action.conditional {
164            // A conditional write prefixes its result with two `I64` probe
165            // counts; the `Returning` variant follows them with the changed
166            // rows' columns.
167            ConditionalOutput::Count => Some(vec![stmt::Type::I64, stmt::Type::I64]),
168            ConditionalOutput::Returning => {
169                let mut tys = vec![stmt::Type::I64, stmt::Type::I64];
170                tys.extend(
171                    action
172                        .output
173                        .ty
174                        .clone()
175                        .expect("conditional write with RETURNING has output columns"),
176                );
177                Some(tys)
178            }
179            ConditionalOutput::None if mysql_insert_returning.is_some() => {
180                // For MySQL INSERT with RETURNING, we don't send RETURNING to the database
181                // (it doesn't support it). The driver will fetch auto-increment IDs using LAST_INSERT_ID().
182                None
183            }
184            ConditionalOutput::None if mysql_update_returning.is_some() => {
185                // The UPDATE has had its RETURNING stripped; the driver runs
186                // a plain UPDATE that returns no rows. The follow-up SELECT
187                // below produces the returning values.
188                None
189            }
190            ConditionalOutput::None => action.output.ty.clone(),
191        };
192
193        let op = operation::QuerySql {
194            stmt,
195            params,
196            ret,
197            last_insert_id_hack: mysql_insert_returning.as_ref().map(|info| info.num_rows),
198        };
199
200        let mut res = self.connection.exec(&self.engine.schema, op.into()).await?;
201
202        match action.conditional {
203            ConditionalOutput::None => {
204                if let Some(mysql_info) = mysql_insert_returning {
205                    res.values = mysql_info.reconstruct_returning(res.values).await?;
206                } else if let Some(mysql_update) = mysql_update_returning {
207                    res = self
208                        .run_mysql_update_returning_select(mysql_update, action.output.ty.clone())
209                        .await?;
210                }
211            }
212            ConditionalOutput::Count | ConditionalOutput::Returning => {
213                let rows = collect_conditional_probe(res.values).await?;
214                let (matched, conditioned) = conditional_probe_counts(&rows[0])?;
215
216                // A conditional write targets a row the caller holds an
217                // instance of: zero matched rows means it has since been
218                // deleted.
219                if matched == 0 {
220                    return Err(toasty_core::Error::record_not_found(
221                        "conditional write matched no rows",
222                    ));
223                }
224                if matched != conditioned {
225                    return Err(toasty_core::Error::condition_failed(
226                        "write condition did not match",
227                    ));
228                }
229
230                res.values = match action.conditional {
231                    ConditionalOutput::Count => Rows::Count(matched as u64),
232                    _ => {
233                        // The probe locked the matched rows, so the write
234                        // applied to exactly those rows and every result row is
235                        // a real changed row — strip the two leading probe
236                        // columns.
237                        let changed = rows
238                            .into_iter()
239                            .map(|row| {
240                                let stmt::Value::Record(record) = row else {
241                                    return Err(toasty_core::Error::invalid_result(
242                                        "conditional write expected Record",
243                                    ));
244                                };
245                                Ok(stmt::Value::record_from_vec(
246                                    record.fields.into_iter().skip(2).collect(),
247                                ))
248                            })
249                            .collect::<Result<Vec<_>>>()?;
250
251                        Rows::value_stream(changed)
252                    }
253                };
254            }
255        }
256
257        // Apply pagination if configured
258        if let Some(pagination) = &action.pagination {
259            assert!(res.next_cursor.is_none() && res.prev_cursor.is_none());
260            res.values.buffer().await?;
261            self.apply_sql_pagination(&mut res, pagination)?;
262        }
263
264        self.vars
265            .store(action.output.output.var, action.output.output.num_uses, res);
266
267        Ok(())
268    }
269
270    /// Apply SQL pagination by extracting cursor from last row.
271    /// If we got a full page (page_size rows), extract cursor for potential next page.
272    /// The client will naturally discover there's no more data when the next request returns empty.
273    ///
274    /// The response values must already be buffered (via `Rows::buffer()`).
275    fn apply_sql_pagination(
276        &mut self,
277        res: &mut ExecResponse,
278        pagination: &PaginationConfig,
279    ) -> Result<()> {
280        let Some(extract_cursor) = &pagination.extract_cursor else {
281            return Ok(());
282        };
283
284        let Rows::Value(stmt::Value::List(ref row_vec)) = res.values else {
285            return Ok(());
286        };
287
288        let page_size = pagination.page_size as usize;
289
290        // Extract cursors for potential next/prev pages
291        res.next_cursor = if row_vec.len() == page_size {
292            let cursor_row = &row_vec[page_size - 1];
293            Some(extract_cursor.eval(&self.engine.schema, std::slice::from_ref(cursor_row))?)
294        } else {
295            // Got fewer than page_size rows, no more data
296            None
297        };
298
299        // Extract prev cursor from first row only when the driver supports backward pagination
300        res.prev_cursor = if !row_vec.is_empty() && self.engine.capability().backward_pagination {
301            let cursor_row = &row_vec[0];
302            Some(extract_cursor.eval(&self.engine.schema, std::slice::from_ref(cursor_row))?)
303        } else {
304            None
305        };
306
307        Ok(())
308    }
309}

311impl Exec<'_> {
312    /// Detects an UPDATE with a non-empty `RETURNING` on a MySQL backend
313    /// and rewrites the statement for the workaround path:
314    ///
315    /// - The returning clause is stripped from the UPDATE so the SQL
316    ///   serializer doesn't reject it.
317    /// - An equivalent `SELECT` over the same table + filter is captured,
318    ///   carrying the original returning expression as its projection.
319    ///
320    /// Returns `None` when the backend supports `RETURNING` natively (PG,
321    /// SQLite) or when the statement is not an UPDATE with a returning
322    /// project. The two-statement path is not atomic relative to concurrent
323    /// writers — see #881.
324    pub(super) fn process_stmt_update_with_returning_on_mysql(
325        &self,
326        stmt: &mut stmt::Statement,
327    ) -> Option<MySQLUpdateReturning> {
328        if self.engine.capability().returning_from_mutation || !self.engine.capability().sql {
329            return None;
330        }
331
332        let stmt::Statement::Update(update) = stmt else {
333            return None;
334        };
335
336        let table_id = match &update.target {
337            stmt::UpdateTarget::Table(table_id) => *table_id,
338            _ => return None,
339        };
340
341        let returning = update.returning.take()?;
342
343        let select = stmt::Select {
344            returning,
345            source: stmt::Source::table(table_id),
346            filter: update.filter.clone(),
347            distinct: false,
348        };
349        let select_stmt =
350            stmt::Statement::Query(stmt::Query::new(stmt::ExprSet::Select(Box::new(select))));
351
352        Some(MySQLUpdateReturning { select_stmt })
353    }
354
355    /// Runs the follow-up `SELECT` for a MySQL UPDATE with stripped
356    /// `RETURNING`. The driver receives a plain query whose result rows
357    /// take the place of the original RETURNING output.
358    pub(super) async fn run_mysql_update_returning_select(
359        &mut self,
360        mysql_update: MySQLUpdateReturning,
361        ret_ty: Option<Vec<stmt::Type>>,
362    ) -> Result<toasty_core::driver::ExecResponse> {
363        let mut select_stmt = mysql_update.select_stmt;
364        let select_params = self.engine.prepare_for_driver(&mut select_stmt);
365
366        let op = operation::QuerySql {
367            stmt: select_stmt,
368            params: select_params,
369            ret: ret_ty,
370            last_insert_id_hack: None,
371        };
372
373        self.connection.exec(&self.engine.schema, op.into()).await
374    }
375
376    /// Processes INSERT statements with RETURNING on MySQL, which doesn't support RETURNING.
377    ///
378    /// Returns information needed to reconstruct the RETURNING results using LAST_INSERT_ID()
379    /// if this is a MySQL INSERT with RETURNING. Returns None otherwise.
380    ///
381    /// # Panics
382    ///
383    /// Panics if the RETURNING clause includes non-auto-increment columns, as MySQL doesn't
384    /// support RETURNING and we can only work around it for auto-increment columns.
385    fn process_stmt_insert_with_returning_on_mysql(
386        &self,
387        stmt: &mut stmt::Statement,
388    ) -> Option<MySQLInsertReturning> {
389        if !self.engine.capability().sql || self.engine.capability().returning_from_mutation {
390            return None;
391        }
392
393        let stmt::Statement::Insert(insert) = stmt else {
394            return None;
395        };
396
397        let returning = insert.returning.take()?;
398
399        // Verify that all columns in the RETURNING clause are auto-increment columns.
400        // This is required because MySQL doesn't support RETURNING, but we can work around
401        // this limitation for auto-increment columns by using LAST_INSERT_ID().
402        let cx = self.engine.expr_cx_for(&*insert);
403
404        let mut ref_count = 0;
405        let mut auto_column_type = None;
406        stmt::visit::for_each_expr(&returning, |expr| {
407            if let stmt::Expr::Reference(expr_ref) = expr {
408                let column = cx.resolve_expr_reference(expr_ref).as_column_unwrap();
409
410                assert!(
411                    column.auto_increment,
412                    "MySQL does not support RETURNING clause for non-auto-increment columns. \
413                     Column '{}' in table '{}' is not auto-increment. \
414                     Only auto-increment columns can be returned from INSERT statements on MySQL.",
415                    column.name, self.engine.schema.db.tables[column.id.table.0].name
416                );
417
418                auto_column_type = Some(column.ty.clone());
419                ref_count += 1;
420            }
421        });
422
423        assert_eq!(
424            ref_count, 1,
425            "MySQL INSERT with RETURNING must have exactly one auto-increment column reference, found {ref_count}"
426        );
427
428        let auto_column_type = auto_column_type.expect("auto_column_type should be set");
429
430        // Extract the expression from the RETURNING clause and replace ExprReference with ExprArg
431        let mut returning_expr = match returning {
432            stmt::Returning::Project(expr) => expr,
433            _ => panic!(
434                "MySQL INSERT with RETURNING must have an Expr, got: {:#?}",
435                returning
436            ),
437        };
438
439        // Replace the ExprReference with ExprArg(position: 0) so we can pass the ID as a positional argument
440        stmt::visit_mut::for_each_expr_mut(&mut returning_expr, |expr| {
441            if matches!(expr, stmt::Expr::Reference(_)) {
442                *expr = stmt::Expr::Arg(stmt::ExprArg {
443                    position: 0,
444                    nesting: 0,
445                });
446            }
447        });
448
449        // Count the number of rows being inserted
450        let num_rows = match &insert.source.body {
451            stmt::ExprSet::Values(values) => values.rows.len() as u64,
452            _ => {
453                panic!(
454                    "MySQL INSERT with RETURNING only supports VALUES, got: {:#?}",
455                    insert.source.body
456                );
457            }
458        };
459
460        Some(MySQLInsertReturning {
461            num_rows,
462            returning_expr,
463            auto_column_type,
464        })
465    }
466}

468/// Collects a conditional write's result rows. The probe (a `COUNT` aggregate)
469/// always yields at least one row, so an empty result is a driver bug.
470async fn collect_conditional_probe(rows: Rows) -> Result<Vec<stmt::Value>> {
471    let Rows::Stream(rows) = rows else {
472        return Err(toasty_core::Error::invalid_result(format!(
473            "conditional write expected Stream, got {rows:?}"
474        )));
475    };
476
477    let rows = rows.collect().await?;
478    if rows.is_empty() {
479        return Err(toasty_core::Error::invalid_result(
480            "conditional write probe returned no rows",
481        ));
482    }
483
484    Ok(rows)
485}
486
487/// Reads the two leading probe counts (`matched`, `conditioned`) from a
488/// conditional write's result row.
489fn conditional_probe_counts(row: &stmt::Value) -> Result<(i64, i64)> {
490    let stmt::Value::Record(record) = row else {
491        return Err(toasty_core::Error::invalid_result(format!(
492            "conditional write expected Record, got {row:?}"
493        )));
494    };
495
496    match (record.fields.first(), record.fields.get(1)) {
497        (Some(stmt::Value::I64(matched)), Some(stmt::Value::I64(conditioned))) => {
498            Ok((*matched, *conditioned))
499        }
500        _ => Err(toasty_core::Error::invalid_result(format!(
501            "conditional write probe columns are not I64; row={row:?}"
502        ))),
503    }
504}

512impl MySQLInsertReturning {
513    /// Reconstructs RETURNING results from the ID rows returned by the driver.
514    ///
515    /// MySQL doesn't support RETURNING, but we fetch auto-increment IDs using LAST_INSERT_ID().
516    /// This method takes the ID rows returned by the driver and evaluates the original RETURNING
517    /// expression for each ID to produce the expected results.
518    async fn reconstruct_returning(self, rows: Rows) -> Result<Rows> {
519        // The driver executed SELECT LAST_INSERT_ID() and returned rows with IDs.
520        let Rows::Stream(id_rows) = rows else {
521            return Err(toasty_core::Error::invalid_result(format!(
522                "MySQL INSERT RETURNING expected Stream, got {:?}",
523                rows
524            )));
525        };
526
527        let id_values = id_rows.collect().await?;
528        assert_eq!(
529            id_values.len(),
530            self.num_rows as usize,
531            "Expected {} ID rows from driver, got {}",
532            self.num_rows,
533            id_values.len()
534        );
535
536        // Reconstruct the RETURNING results by evaluating the original returning expression
537        // for each ID row returned by the driver
538        let mut returning_rows = Vec::with_capacity(self.num_rows as usize);
539
540        for id_value_raw in id_values {
541            // The driver returns a record with one field containing the ID.
542            // Extract the ID value from the record wrapper.
543            let stmt::Value::Record(record) = id_value_raw else {
544                return Err(toasty_core::Error::invalid_result(format!(
545                    "MySQL INSERT RETURNING expected Record from driver, got {:?}",
546                    id_value_raw
547                )));
548            };
549
550            assert_eq!(
551                record.fields.len(),
552                1,
553                "Expected record with one field from driver"
554            );
555
556            // Cast the ID to the correct type for the auto-increment column.
557            // An auto-increment ID is a scalar, so the cast is schema-free.
558            let id_value = self.auto_column_type.cast(&(), record.fields[0].clone())?;
559            let input = vec![id_value];
560
561            // Evaluate the returning expression with the auto-increment ID
562            let row_value = self.returning_expr.eval(&input)?;
563
564            returning_rows.push(row_value);
565        }
566
567        Ok(Rows::Stream(stmt::ValueStream::from_iter(
568            returning_rows.into_iter().map(Ok),
569        )))
570    }

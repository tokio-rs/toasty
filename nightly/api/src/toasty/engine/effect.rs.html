1//! Statement effect classification.
2//!
3//! [`classify`] walks a [`Statement`] AST and returns whether the
4//! statement can be retried transparently on `connection_lost`.  The
5//! pool consults this before attempting a retry; downstream consumers
6//! (e.g. a runtime-checked read-only API surface) read the same
7//! classification.
8//!
9//! See [`docs/dev/design/retry-safe-recovery.md`] for the full
10//! contract.
11//!
12//! [`Statement`]: toasty_core::stmt::Statement
13//! [`docs/dev/design/retry-safe-recovery.md`]: ../../../docs/dev/design/retry-safe-recovery.md
14
15use toasty_core::stmt::{Delete, Insert, Statement, Update, Visit};
16
17/// Whether a statement mutates database state.
18///
19/// A statement is [`Effect::ReadOnly`] if it is a [`Statement::Query`]
20/// and contains no `Insert`, `Update`, or `Delete` anywhere in its
21/// tree. Otherwise it is [`Effect::Mutating`].
22///
23/// CTE-with-mutation queries (e.g.
24/// `WITH ins AS (INSERT ... RETURNING *) SELECT * FROM ins`) parse as
25/// `Statement::Query` values whose `WITH` clauses contain an
26/// `ExprSet::Insert`, `ExprSet::Update`, or `ExprSet::Delete` and are
27/// correctly classified as `Mutating`.
28#[derive(Debug, Clone, Copy, PartialEq, Eq)]
29pub(crate) enum Effect {
30    /// The statement reads but does not mutate state.  Safe to retry
31    /// after `connection_lost`.
32    ReadOnly,
33
34    /// The statement mutates state, either directly (top-level
35    /// `Insert` / `Update` / `Delete`) or via an embedded sub-statement.
36    /// Not safe to retry without further analysis.
37    Mutating,
38}
39
40/// Classify a statement's effect on database state.
41///
42/// O(n) in the size of the statement tree; no schema access.
43pub(crate) fn classify(stmt: &Statement) -> Effect {
44    let mut walker = Walker { mutating: false };
45    walker.visit_stmt(stmt);
46    if walker.mutating {
47        Effect::Mutating
48    } else {
49        Effect::ReadOnly
50    }
51}
52
53struct Walker {
54    mutating: bool,
55}
56
57impl Visit for Walker {
58    fn visit_stmt_delete(&mut self, _: &Delete) {
59        self.mutating = true;
60    }
61
62    fn visit_stmt_insert(&mut self, _: &Insert) {
63        self.mutating = true;
64    }
65
66    fn visit_stmt_update(&mut self, _: &Update) {
67        self.mutating = true;
68    }
69}

1use toasty_core::stmt;
2
3use crate::engine::simplify::Simplify;
4
5impl Simplify<'_> {
6    pub(super) fn simplify_stmt_query_when_empty(&mut self, stmt: &mut stmt::Query) {
7        if stmt.with.is_some() {
8            // Just skip if there are any CTEs for now
9            return;
10        }
11
12        if self.stmt_query_is_empty(stmt) {
13            stmt.body = stmt::ExprSet::Values(Default::default());
14            stmt.order_by = None;
15            stmt.limit = None;
16        }
17    }
18
19    pub(super) fn stmt_query_is_empty(&self, query: &stmt::Query) -> bool {
20        match &query.body {
21            stmt::ExprSet::Values(values) => values.is_empty(),
22            stmt::ExprSet::Select(select) => self.stmt_select_is_empty(select),
23            _ => false,
24        }
25    }
26
27    pub(super) fn stmt_select_is_empty(&self, select: &stmt::Select) -> bool {
28        // Only check table sources (lowered queries)
29        let stmt::Source::Table(source_table) = &select.source else {
30            return false;
31        };
32
33        if select.filter.is_false() {
34            return true;
35        }
36
37        // Check if the source guarantees empty results
38        self.source_table_is_empty(source_table)
39    }
40
41    fn source_table_is_empty(&self, source_table: &stmt::SourceTable) -> bool {
42        if source_table.from.is_empty() {
43            todo!(
44                "this case is not handled yet at this level; \
45                   Do we want to handle this or transform the statement \
46                   to something different first?"
47            );
48        }
49
50        // Get the primary relation from the first table
51        let stmt::TableFactor::Table(primary_table_id) = &source_table.from[0].relation;
52
53        // Check if primary source is empty VALUES
54        if self.table_ref_is_empty_values(&source_table.tables[primary_table_id.0]) {
55            // Primary source is empty - result will be empty unless there are
56            // LEFT JOINs that could still produce results
57            return !source_table
58                .from
59                .iter()
60                .flat_map(|twj| &twj.joins)
61                .any(|join| matches!(join.constraint, stmt::JoinOp::Left(_)));
62        }
63
64        // Check if any non-optional join has empty VALUES
65        // Currently only LEFT joins exist, so any other join would be INNER
66        // For now, this is conservative - we'd add INNER/CROSS join detection here
67        false
68    }
69
70    fn table_ref_is_empty_values(&self, table_ref: &stmt::TableRef) -> bool {
71        match table_ref {
72            stmt::TableRef::Derived(derived) => {
73                // Check if the derived table is empty VALUES. Inner queries
74                // would have already been simplified, so we only need to check
75                // for empty values here.
76                matches!(&derived.subquery.body,
77                    stmt::ExprSet::Values(values) if values.is_empty()
78                )
79            }
80            _ => false,
81        }
82    }
83}

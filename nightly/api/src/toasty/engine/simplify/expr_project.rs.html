1use toasty_core::stmt::{self, Project};
2
3use super::Simplify;
4
5impl Simplify<'_> {
6    pub(super) fn simplify_expr_project(
7        &mut self,
8        expr: &mut stmt::ExprProject,
9    ) -> Option<stmt::Expr> {
10        // Constant evaluation: if the base is an Expr::Value, we can evaluate
11        // the projection at compile time.
12        //
13        // Examples:
14        //   - `project(Value::I64(42), [])` → `Value::I64(42)`
15        //   - `project(Value::Record([1, 2, 3]), [0])` → `Value::I64(1)`
16        //
17        // Note: This only handles Expr::Value, not Expr::Record or Expr::List.
18        // Those variants represent expressions that will be evaluated later,
19        // not constant values that can be folded now.
20        if let stmt::Expr::Value(value) = &*expr.base {
21            // Use the value's project method to follow the projection path
22            if let Some(result) = value.project(&expr.projection) {
23                return Some(result);
24            }
25        }
26
27        // Handle projections through records (embedded fields lower to records of columns).
28        // After lowering, embedded field references become records where each field is a column.
29        // Uses `entry` to support arbitrary-depth projections (e.g., [1, 1] for nested embedded).
30        // Examples:
31        //   project([street_col, city_col, zip_col], [1]) → city_col
32        //   project([name_col, record([street_col, city_col])], [1, 1]) → city_col
33        if let stmt::Expr::Record(_) = &*expr.base
34            && let Some(entry) = expr.base.entry(&expr.projection)
35        {
36            return Some(entry.to_expr());
37        }
38
39        // Project into Match: distribute the projection into each arm's expression.
40        // Example: project(Match(d, [1 => Record([d, a]), 2 => Record([d, n])]), [0])
41        //        → Match(d, [1 => project(Record([d, a]), [0]), 2 => project(Record([d, n]), [0])])
42        //        → Match(d, [1 => d, 2 => d])   (after recursive simplification)
43        if let stmt::Expr::Match(match_expr) = &mut *expr.base {
44            for arm in &mut match_expr.arms {
45                arm.expr = stmt::Expr::project(arm.expr.take(), expr.projection.clone());
46            }
47            *match_expr.else_expr =
48                stmt::Expr::project(match_expr.else_expr.take(), expr.projection.clone());
49            return Some(expr.base.take());
50        }
51
52        None
53    }
54}

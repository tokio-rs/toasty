1//! Bind a fully-resolved statement's inline values into typed parameters.
2//!
3//! Three phases:
4//! 1. **Extract** ([`extract`]): replace scalar `Value` nodes with `Arg(n)`
5//!    placeholders, initializing each param's type from the value itself.
6//! 2. **Synthesize** (bottom-up): compute each node's inferred type from its
7//!    children (column refs get their storage type from the schema, records get
8//!    a tuple of field types, etc.).
9//! 3. **Check** (top-down): push refined types into `Arg(n)` nodes, upgrading a
10//!    param when context is more precise (e.g. `Enum` over `Text`).
11//!
12//! Synthesize and check ([`infer`]) run together in a single recursive walk.
13//! Types carry **provenance** (`Column` vs `Inferred`) so schema-authoritative
14//! column types win over value-inferred guesses when merging.
15//!
16//! `#[document]` values are named into their `Value::Object` form before this
17//! runs: the mapping's lowering casts convert them during statement
18//! lowering/simplification, and document *paths* are resolved by legalization
19//! ([`super::legalize`]) — `Engine::prepare_for_driver` runs both in order.
20

22    driver::{Capability, operation::TypedValue},
23    schema::db,
24    stmt,
25};

33/// Bind a statement's inline values: replace scalar `Value` nodes with
34/// `Expr::Arg(n)` placeholders and infer a precise `db::Type` for each. The
35/// returned `Vec<TypedValue>` is indexed by the `n` in each placeholder.
36///
37/// The statement must already be legalized ([`super::legalize`]): document
38/// values named, document paths resolved. Runs via
39/// `Engine::prepare_for_driver`, as the final engine step before a driver
40/// serializes the statement.
41pub(crate) fn run(
42    stmt: &mut stmt::Statement,
43    db_schema: &db::Schema,
44    capability: &Capability,
45) -> Vec<TypedValue> {
46    // Phase 1: mechanical extraction — replace values with Arg(n).
47    let mut params: Vec<Param> = Vec::new();
48    extract::extract_values(stmt, &mut params, capability);
49
50    // Phases 2+3: bidirectional type inference — refine param types.
51    infer::refine_param_types(stmt, db_schema, &mut params);
52
53    // Materialize the final TypedValues. `finalize_ty` panics if any param is
54    // still unresolved — synthesize/check is expected to type every param.
55    params
56        .into_iter()
57        .map(|p| {
58            let Param { value, ty } = p;
59            TypedValue {
60                ty: finalize_ty(&value, ty),
61                value,
62            }
63        })
64        .collect()
65}
66
67/// A bind parameter being inferred. Once inference completes, the `Ty` is
68/// converted to a concrete `db::Type` for the `TypedValue`.
69struct Param {
70    value: stmt::Value,
71    ty: Ty,
72}
73
74/// Resolve a `Ty` to a concrete `db::Type`. Panics on `Unknown` / `Record` —
75/// every param should be fully inferred by the synthesize/check pass; if a
76/// statement reaches here with an unresolved param, that's a bug worth
77/// surfacing so we can evaluate the specific case.
78fn finalize_ty(value: &stmt::Value, ty: Ty) -> db::Type {
79    match ty {
80        Ty::Column(t) | Ty::Inferred(t) => t,
81        Ty::List(elem) => db::Type::list(finalize_ty(value, *elem)),
82        Ty::Unknown => panic!("bind left {value:?} with unresolved type"),
83        Ty::Record(_) => panic!(
84            "bind left {value:?} typed as a record; only scalars and lists are extracted as params"
85        ),
86    }
87}

93/// The inferred database-level type of an expression node.
94///
95/// Each scalar type carries **provenance**: `Column` means the type came from
96/// the schema (authoritative), `Inferred` means it was guessed from the value.
97/// Column types always win when merging.
98#[derive(Debug, Clone)]
99enum Ty {
100    /// Type from a column reference or schema (authoritative).
101    Column(db::Type),
102    /// Type inferred from a value (initial guess — may be less specific).
103    Inferred(db::Type),
104    /// A tuple of types (one per field).
105    Record(Vec<Ty>),
106    /// A homogeneous list where all elements share a type.
107    List(Box<Ty>),
108    /// Type could not be determined.
109    Unknown,
110}
111
112impl Ty {
113    /// Extract the `db::Type`, regardless of provenance.
114    fn db_type(&self) -> Option<&db::Type> {
115        match self {
116            Ty::Column(ty) | Ty::Inferred(ty) => Some(ty),
117            _ => None,
118        }
119    }
120
121    /// Returns true if this type comes from the schema (authoritative).
122    #[cfg(test)]
123    fn is_column(&self) -> bool {
124        matches!(self, Ty::Column(_))
125    }
126}

30/// Infer the [`Scope`] type from a scope expression and return its fields
31/// path.
32///
33/// The `create!` macro uses this in the scoped form (`in expr { ... }`) to
34/// obtain the field struct for nested builders. Because the macro has no
35/// type information, it cannot call `S::new_path_root()` directly — this function
36/// lets Rust infer `S` from the scope argument.
37pub fn scope_fields<S: Scope>(_scope: &S) -> S::Path<S::Item> {
38    S::new_path_root()
39}

41/// Convert a value into an untyped [`core::stmt::Expr`] via the typed
42/// [`IntoExpr<T>`] trait.
43///
44/// Generated code (`#[derive(Model)]`, `#[derive(Embed)]`) splices this in
45/// instead of inlining the `let expr: Expr<T> = value.into_expr(); expr.into()`
46/// pattern at every field site. The explicit `T` type parameter
47/// disambiguates which `IntoExpr` impl to use.
48pub fn into_untyped_expr<T, V: IntoExpr<T>>(value: V) -> core::stmt::Expr {
49    let expr: stmt::Expr<T> = value.into_expr();
50    expr.into()
51}

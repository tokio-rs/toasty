1//! Raw SQL execution helpers for SQL backends.
2//!
3//! Use [`statement`] for SQL that returns an affected-row count and [`query`]
4//! for SQL that returns rows. Both builders execute through
5//! [`Executor`](crate::Executor), so they work with [`Db`](crate::Db),
6//! [`Connection`](crate::Connection), and [`Transaction`](crate::Transaction).
7//!
8//! Raw SQL is backend SQL. Toasty does not rewrite placeholders, quote
9//! identifiers, or hydrate models from raw query results. Use the placeholder
10//! syntax of the active driver. The syntax is exposed at runtime through
11//! [`Capability::sql_placeholder`](crate::Capability::sql_placeholder):
12//!
13//! | Backend | [`SqlPlaceholder`](crate::SqlPlaceholder) |
14//! |---|---|
15//! | SQLite | `NumberedQuestionMark` (`?1`, `?2`, ...) |
16//! | Turso | `NumberedQuestionMark` (`?1`, `?2`, ...) |
17//! | PostgreSQL | `DollarNumber` (`$1`, `$2`, ...) |
18//! | MySQL | `QuestionMark` (`?`, `?`, ...) |
19//!
20//! # Statements
21//!
22//! ```ignore
23//! let updated = toasty::sql::statement(
24//!     "UPDATE users SET name = ?1 WHERE id = ?2",
25//! )
26//! .bind("Alice")
27//! .bind(1_i64)
28//! .exec(&mut db)
29//! .await?;
30//!
31//! assert_eq!(updated, 1);
32//! ```
33//!
34//! # Queries
35//!
36//! Queries return dynamic [`Value`] rows. Each row is a [`Value::Record`] with
37//! fields in selected-column order.
38//!
39//! ```ignore
40//! let rows = toasty::sql::query(
41//!     "SELECT id, name FROM users WHERE active = ?1",
42//! )
43//! .bind(true)
44//! .exec(&mut db)
45//! .await?;
46//! ```
47//!
48//! The driver infers result value types from database metadata by default. Use
49//! [`Query::column_types`] when a result column is ambiguous, such as a SQLite
50//! boolean stored as an integer or a UUID stored as bytes.

63/// A raw SQL statement that returns an affected-row count.
64///
65/// Create one with [`statement`].
66pub struct Statement {
67    sql: String,
68    params: Vec<Param>,
69}

71/// A raw SQL query that returns dynamic rows.
72///
73/// Create one with [`query`].
74pub struct Query {
75    sql: String,
76    params: Vec<Param>,
77    ret: RawSqlRet,
78}
79
80#[derive(Debug, Clone)]
81enum Param {
82    Infer(Value),
83    Typed(TypedValue),
84}
85
86/// Create a raw SQL statement.
87///
88/// Statements are executed for their affected-row count. Use
89/// [`query`] for SQL that returns rows.
90///
91/// The SQL string must use the placeholder syntax reported by
92/// [`Capability::sql_placeholder`](crate::Capability::sql_placeholder).
93///
94/// ```ignore
95/// let count = toasty::sql::statement(
96///     "DELETE FROM users WHERE archived = ?1",
97/// )
98/// .bind(true)
99/// .exec(&mut db)
100/// .await?;
101/// ```
102pub fn statement(sql: impl Into<String>) -> Statement {
103    Statement {
104        sql: sql.into(),
105        params: vec![],
106    }
107}
108
109/// Create a raw SQL query.
110///
111/// Query rows are returned as dynamic [`Value`] records.
112///
113/// ```ignore
114/// let rows = toasty::sql::query(
115///     "SELECT id, name FROM users WHERE active = ?1",
116/// )
117/// .bind(true)
118/// .exec(&mut db)
119/// .await?;
120/// ```
121pub fn query(sql: impl Into<String>) -> Query {
122    Query {
123        sql: sql.into(),
124        params: vec![],
125        ret: RawSqlRet::Infer,
126    }
127}

129impl Statement {
130    /// Bind a value using the executor driver's default database type for
131    /// that value.
132    ///
133    /// Values are bound in the order this method is called. Use
134    /// [`bind_typed`](Self::bind_typed) for `NULL`, empty lists, or values
135    /// whose database type cannot be inferred from the value alone.
136    pub fn bind(mut self, value: impl Into<Value>) -> Self {
137        self.params.push(Param::Infer(value.into()));
138        self
139    }
140
141    /// Bind a value with an explicit database type.
142    ///
143    /// This is useful for `NULL`, empty lists, or backend-specific column
144    /// types.
145    ///
146    /// ```ignore
147    /// use toasty::schema::db;
148    ///
149    /// toasty::sql::statement(
150    ///     "UPDATE users SET archived_at = ?1 WHERE id = ?2",
151    /// )
152    /// .bind_typed(toasty::stmt::Value::Null, db::Type::Timestamp(6))
153    /// .bind(1_i64)
154    /// .exec(&mut db)
155    /// .await?;
156    /// ```
157    pub fn bind_typed(mut self, value: impl Into<Value>, ty: db::Type) -> Self {
158        self.params.push(Param::Typed(TypedValue {
159            value: value.into(),
160            ty,
161        }));
162        self
163    }
164
165    /// Execute the statement and return the affected-row count.
166    ///
167    /// The executor may be a [`Db`](crate::Db), [`Connection`](crate::Connection),
168    /// or [`Transaction`](crate::Transaction).
169    pub async fn exec(self, executor: &mut dyn Executor) -> Result<u64> {
170        let raw = into_raw_sql(
171            self.sql,
172            self.params,
173            RawSqlRet::None,
174            executor.capability(),
175        )?;
176        let response = executor.exec_raw_sql(raw).await?;
177
178        Ok(response.values.into_count())
179    }
180}

182impl Query {
183    /// Bind a value using the executor driver's default database type for
184    /// that value.
185    ///
186    /// Values are bound in the order this method is called. Use
187    /// [`bind_typed`](Self::bind_typed) for `NULL`, empty lists, or values
188    /// whose database type cannot be inferred from the value alone.
189    pub fn bind(mut self, value: impl Into<Value>) -> Self {
190        self.params.push(Param::Infer(value.into()));
191        self
192    }
193
194    /// Bind a value with an explicit database type.
195    ///
196    /// This is useful for `NULL`, empty lists, or backend-specific column
197    /// types.
198    pub fn bind_typed(mut self, value: impl Into<Value>, ty: db::Type) -> Self {
199        self.params.push(Param::Typed(TypedValue {
200            value: value.into(),
201            ty,
202        }));
203        self
204    }
205
206    /// Provide result column type hints for ambiguous backend values.
207    ///
208    /// Hints affect result decoding only. They do not change the SQL sent to
209    /// the database. Use this when backend metadata is not enough to choose the
210    /// desired Toasty value type.
211    ///
212    /// ```ignore
213    /// use toasty::stmt;
214    ///
215    /// let rows = toasty::sql::query(
216    ///     "SELECT id, enabled FROM users WHERE id = ?1",
217    /// )
218    /// .bind(1_i64)
219    /// .column_types([stmt::Type::I64, stmt::Type::Bool])
220    /// .exec(&mut db)
221    /// .await?;
222    /// ```
223    pub fn column_types(mut self, types: impl IntoIterator<Item = stmt::Type>) -> Self {
224        self.ret = RawSqlRet::Types(types.into_iter().collect());
225        self
226    }
227
228    /// Execute the query and return dynamic rows.
229    ///
230    /// Each row is a [`Value::Record`] with fields in selected-column order.
231    /// The executor may be a [`Db`](crate::Db), [`Connection`](crate::Connection),
232    /// or [`Transaction`](crate::Transaction).
233    pub async fn exec(self, executor: &mut dyn Executor) -> Result<Vec<Value>> {
234        let raw = into_raw_sql(self.sql, self.params, self.ret, executor.capability())?;
235        let response = executor.exec_raw_sql(raw).await?;
236
237        match response.values.collect_as_value().await? {
238            Value::List(items) => Ok(items),
239            value => Err(toasty_core::Error::invalid_result(format!(
240                "raw SQL query expected a list of rows, got {value:?}"
241            ))),
242        }
243    }
244}
245
246fn into_raw_sql(
247    sql: String,
248    params: Vec<Param>,
249    ret: RawSqlRet,
250    capability: &Capability,
251) -> Result<RawSql> {
252    if !capability.sql {
253        return Err(Error::unsupported_feature(format!(
254            "{} does not support raw SQL",
255            capability.driver_name
256        )));
257    }
258
259    Ok(RawSql {
260        sql,
261        params: params
262            .into_iter()
263            .map(|param| param.into_typed(capability))
264            .collect::<Result<_>>()?,
265        ret,
266    })
267}
268
269impl Param {
270    fn into_typed(self, capability: &Capability) -> Result<TypedValue> {
271        match self {
272            Param::Infer(value) => infer_typed_value(value, capability),
273            Param::Typed(value) => Ok(value),
274        }
275    }
276}
277
278fn infer_typed_value(value: Value, capability: &Capability) -> Result<TypedValue> {
279    let ty = value
280        .infer_db_ty(&capability.storage_types)
281        .map_err(|err| {
282            err.context(Error::invalid_statement(
283                "cannot infer raw SQL bind type; use bind_typed",
284            ))
285        })?;
286
287    Ok(TypedValue { value, ty })
288}

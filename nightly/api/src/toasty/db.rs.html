30/// Shared state between all `Db` clones.
31pub(crate) struct Shared {
32    pub(crate) engine: Engine,
33    pub(crate) pool: Pool,
34}

36/// A database handle backed by a connection pool.
37///
38/// Each operation acquires a connection from the pool, executes, and returns
39/// the connection. Use [`Db::connection`] to obtain a dedicated
40/// [`Connection`] when you need multiple statements to share the same
41/// physical connection (e.g. temporary tables or session-level state).
42///
43/// Cloning a `Db` is cheap — it shares the underlying pool.
44pub struct Db {
45    shared: Arc<Shared>,
46}
47
48impl Clone for Db {
49    fn clone(&self) -> Self {
50        Db {
51            shared: self.shared.clone(),
52        }
53    }
54}
55
56impl Db {
57    /// Create a new [`Builder`] for configuring and opening a database.
58    ///
59    /// # Examples
60    ///
61    /// ```
62    /// # tokio::runtime::Runtime::new().unwrap().block_on(async {
63    /// # #[derive(Debug, toasty::Model)]
64    /// # struct User {
65    /// #     #[key]
66    /// #     id: i64,
67    /// #     name: String,
68    /// # }
69    /// let driver = toasty_driver_sqlite::Sqlite::in_memory();
70    /// let db = toasty::Db::builder()
71    ///     .models(toasty::models!(User))
72    ///     .build(driver)
73    ///     .await
74    ///     .unwrap();
75    /// # });
76    /// ```
77    pub fn builder() -> Builder {
78        Builder::default()
79    }
80
81    /// Acquire a dedicated connection from the pool.
82    ///
83    /// The returned [`Connection`] implements [`Executor`] and pins all
84    /// operations to the same physical connection. This is useful when
85    /// multiple statements must share connection-level state such as
86    /// temporary tables or session variables.
87    ///
88    /// When the `Connection` is dropped it is returned to the pool for reuse.
89    pub async fn connection(&self) -> Result<Connection> {
90        self.shared.pool.get(self.shared.clone()).await
91    }
92
93    pub(crate) async fn exec_stmt(
94        &self,
95        stmt: stmt::Statement,
96        in_transaction: bool,
97    ) -> Result<ExecResponse> {
98        let conn = self.connection().await?;
99        conn.exec_stmt(stmt, in_transaction).await
100    }
101
102    /// Creates tables and indices defined in the schema on the database.
103    pub async fn push_schema(&self) -> Result<()> {
104        let conn = self.connection().await?;
105        conn.push_schema().await
106    }
107
108    /// Drops the entire database and recreates an empty one without applying migrations.
109    pub async fn reset_db(&self) -> Result<()> {
110        self.shared.pool.driver().reset_db().await
111    }
112
113    /// Returns a reference to the underlying database driver.
114    pub fn driver(&self) -> &dyn Driver {
115        self.shared.pool.driver()
116    }
117
118    /// Returns the compiled schema used by this database handle.
119    pub fn schema(&self) -> &Arc<Schema> {
120        &self.shared.engine.schema
121    }
122
123    /// Returns the capability flags reported by the driver.
124    ///
125    /// The query engine uses these to decide which operation types to generate
126    /// (e.g., SQL vs. key-value).
127    pub fn capability(&self) -> &Capability {
128        self.shared.engine.capability()
129    }
130
131    /// Begin a transaction, acquiring a connection from the pool.
132    ///
133    /// Takes `&mut self` so the `Db` handle is exclusively borrowed while the
134    /// transaction is open. This prevents accidentally running a statement
135    /// against the pool — which would execute on a separate connection and
136    /// bypass the transaction — when you meant to run it against `&mut tx`.
137    ///
138    /// If you need a second handle while the transaction is open, clone the
139    /// `Db` before calling this method. Clones share the same pool.
140    pub async fn transaction(&mut self) -> Result<Transaction<'_>> {
141        <Self as Executor>::transaction(self).await
142    }
143
144    /// Returns a [`TransactionBuilder`] that will acquire a connection from
145    /// the pool when [`begin`](TransactionBuilder::begin) is called.
146    ///
147    /// Like [`transaction`](Self::transaction), this takes `&mut self` so the
148    /// `Db` handle stays locked for the lifetime of the transaction. Clone
149    /// the `Db` beforehand if you need a separate handle.
150    pub fn transaction_builder(&mut self) -> TransactionBuilder<'_> {
151        TransactionBuilder::new(tx::TxSource::Db(self))
152    }
153
154    /// Returns a reference to the connection pool backing this handle.
155    #[doc(hidden)]
156    pub fn pool(&self) -> &Pool {
157        &self.shared.pool
158    }
159}

161impl std::fmt::Debug for Db {
162    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
163        f.debug_struct("Db")
164            .field("engine", &self.shared.engine)
165            .finish()
166    }
167}

169#[async_trait]
170impl Executor for Db {
171    async fn transaction(&mut self) -> Result<Transaction<'_>> {
172        let conn = self.connection().await?;
173        Transaction::begin(ConnRef::owned(conn)).await
174    }
175
176    async fn exec_untyped(&mut self, stmt: stmt::Statement) -> Result<ExecResponse> {
177        self.exec_stmt(stmt, false).await
178    }
179
180    fn schema(&mut self) -> &Arc<Schema> {
181        Db::schema(self)
182    }
183}

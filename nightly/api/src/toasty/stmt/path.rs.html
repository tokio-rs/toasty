9/// A typed path from a root model `T` to a field of type `U`.
10///
11/// `Path` represents a traversal through a model's fields and relations. The
12/// type parameter `T` is the root model the path starts from, and `U` is the
13/// type of the value at the end of the path.
14///
15/// Paths are the primary way to reference model fields in queries. Generated
16/// code provides accessor methods (e.g., `User::fields().name()`) that return
17/// paths, which you then use with comparison methods to build filter
18/// expressions:
19///
20/// ```
21/// # #[derive(Debug, toasty::Model)]
22/// # struct User {
23/// #     #[key]
24/// #     id: i64,
25/// #     name: String,
26/// # }
27/// // Path<User, String> — the "name" field on User
28/// let path = User::fields().name();
29///
30/// // Expr<bool> — a filter expression
31/// let filter = path.eq("Alice");
32/// ```
33///
34/// Paths can also be used to construct order-by clauses via
35/// [`asc`](Path::asc) and [`desc`](Path::desc), and can be chained to
36/// navigate through relations with [`chain`](Path::chain).
37pub struct Path<T, U> {
38    pub(super) untyped: stmt::Path,
39    _p: PhantomData<(T, U)>,
40}

95    /// Append `other` to this path, producing a new path from `T` to `V`.
96    ///
97    /// The origin of `other` is left unconstrained because list field structs
98    /// store `Path<Origin, List<M>>` while chaining segments rooted at `M`
99    /// (not `List<M>`). The untyped path concatenation is always correct; the
100    /// generic parameters serve only as compile-time markers.
101    ///
102    /// # Examples
103    ///
104    /// ```
105    /// # #[derive(Debug, toasty::Model)]
106    /// # struct User {
107    /// #     #[key]
108    /// #     id: i64,
109    /// #     name: String,
110    /// # }
111    /// use toasty::stmt::Path;
112    /// use toasty::schema::Model;
113    ///
114    /// let user_path = User::path_root();
115    /// let name_path = User::path_field::<String>(1);
116    /// let _chained: Path<User, String> = user_path.chain(name_path);
117    /// ```
118    pub fn chain<X, V>(mut self, other: impl Into<Path<X, V>>) -> Path<T, V> {
119        let other = other.into();
120        self.untyped.chain(&other.untyped);
121
122        Path {
123            untyped: self.untyped,
124            _p: PhantomData,
125        }
126    }

128    /// Build a filter `Expr<bool>` from this path, automatically wrapping
129    /// the body with an `is_variant(parent, variant_id)` AND-gate when the
130    /// path is variant-rooted.
131    ///
132    /// All boolean-producing methods on `Path` (`eq`, `ne`, `gt`, `is_none`,
133    /// `starts_with`, `any`, …) funnel through this so that filter-context
134    /// uses of a variant-rooted path implicitly require the variant to
135    /// match. Path-yielding contexts (`include`, `order_by`, `chain`)
136    /// bypass this helper and keep the bare path.
137    fn build_filter<F>(self, build_body: F) -> Expr<bool>
138    where
139        F: FnOnce(stmt::Expr) -> stmt::Expr,
140    {
141        let gate = match &self.untyped.root {
142            stmt::PathRoot::Variant { parent, variant_id } => {
143                let parent_stmt = parent.as_ref().clone().into_stmt();
144                Some(stmt::Expr::is_variant(parent_stmt, *variant_id))
145            }
146            _ => None,
147        };
148        let body = build_body(self.untyped.into_stmt());
149        let untyped = match gate {
150            Some(g) => stmt::Expr::and(g, body),
151            None => body,
152        };
153        Expr {
154            untyped,
155            _p: PhantomData,
156        }
157    }

391impl<T, U> Path<T, List<U>> {
392    /// Build an `IN subquery` expression that tests whether **any** associated
393    /// record satisfies `filter`.
394    ///
395    /// The path must point to a `HasMany` (or similar collection) field on the
396    /// parent model. The returned expression can be used as a filter on the
397    /// parent query.
398    ///
399    /// # Examples
400    ///
401    /// ```
402    /// # #[derive(Debug, toasty::Model)]
403    /// # struct User {
404    /// #     #[key]
405    /// #     id: i64,
406    /// #     name: String,
407    /// # }
408    /// # #[derive(Debug, toasty::Model)]
409    /// # struct Todo {
410    /// #     #[key]
411    /// #     id: i64,
412    /// #     title: String,
413    /// # }
414    /// use toasty::stmt::List;
415    /// use toasty::schema::Model;
416    ///
417    /// // Find users that have at least one todo with "urgent" in the title
418    /// let todos_path = User::path_field::<List<Todo>>(2);
419    /// let filter = todos_path.any(Todo::fields().title().eq("urgent"));
420    /// ```
421    pub fn any(self, filter: Expr<bool>) -> Expr<bool>
422    where
423        U: crate::schema::Model,
424    {
425        // Build a query on the child model filtered by `filter`
426        let child_query = super::Query::<List<U>>::all().filter(filter);
427        self.build_filter(move |path| stmt::Expr::in_subquery(path, child_query.untyped))
428    }
429
430    /// Build a `NOT IN subquery` expression that tests whether **all** associated
431    /// records satisfy `filter`.
432    ///
433    /// The path must point to a `HasMany` (or similar collection) field on the
434    /// parent model. Returns `true` when every associated record matches
435    /// `filter`, including the vacuous case where the parent has no associated
436    /// records (matching Rust's `[].iter().all()` semantics).
437    ///
438    /// # Examples
439    ///
440    /// ```
441    /// # #[derive(Debug, toasty::Model)]
442    /// # struct User {
443    /// #     #[key]
444    /// #     id: i64,
445    /// #     name: String,
446    /// # }
447    /// # #[derive(Debug, toasty::Model)]
448    /// # struct Todo {
449    /// #     #[key]
450    /// #     id: i64,
451    /// #     complete: bool,
452    /// # }
453    /// use toasty::stmt::List;
454    /// use toasty::schema::Model;
455    ///
456    /// // Find users whose todos are all complete
457    /// let todos_path = User::path_field::<List<Todo>>(2);
458    /// let filter = todos_path.all(Todo::fields().complete().eq(true));
459    /// ```
460    pub fn all(self, filter: Expr<bool>) -> Expr<bool>
461    where
462        U: crate::schema::Model,
463    {
464        // parent NOT IN (SELECT child_fk FROM child WHERE NOT filter)
465        let child_query = super::Query::<List<U>>::all().filter(filter.not());
466        self.build_filter(move |path| {
467            stmt::Expr::not(stmt::Expr::in_subquery(path, child_query.untyped))
468        })
469    }
470}

472/// Container-style predicates on a `Vec<scalar>` model field. The path target
473/// is the [`List<U>`] marker (matching `Field::Path<Origin>` for `Vec<U>`),
474/// and the element type `U` is constrained to a path-target scalar so the
475/// same `IntoExpr` infrastructure that powers `eq`/`in_list` covers the
476/// right-hand side.
477impl<T, U> Path<T, List<U>>
478where
479    U: crate::schema::Scalar,
480{
481    /// Test whether the array contains `value`.
482    ///
483    /// Mirrors [`Vec::contains`]. Lowers to `value = ANY(col)` on PostgreSQL.
484    ///
485    /// # Examples
486    ///
487    /// ```ignore
488    /// let filter = User::fields().tags().contains("admin");
489    /// ```
490    pub fn contains(self, value: impl IntoExpr<U>) -> Expr<bool> {
491        let value = value.into_expr().untyped;
492        self.build_filter(move |path| stmt::Expr::any_op(value, stmt::BinaryOp::Eq, path))
493    }
494
495    /// Test whether the array contains every element of `values`.
496    ///
497    /// Mirrors [`HashSet::is_superset`](std::collections::HashSet::is_superset).
498    /// Lowers to `col @> values` (PostgreSQL `@>` operator).
499    pub fn is_superset(self, values: impl IntoExpr<List<U>>) -> Expr<bool> {
500        let values = values.into_expr().untyped;
501        self.build_filter(move |path| stmt::Expr::array_is_superset(path, values))
502    }
503
504    /// Test whether the array shares at least one element with `values`.
505    ///
506    /// Negation of [`HashSet::is_disjoint`](std::collections::HashSet::is_disjoint).
507    /// Lowers to `col && values` (PostgreSQL `&&` operator).
508    pub fn intersects(self, values: impl IntoExpr<List<U>>) -> Expr<bool> {
509        let values = values.into_expr().untyped;
510        self.build_filter(move |path| stmt::Expr::array_intersects(path, values))
511    }
512
513    /// Returns the array length.
514    ///
515    /// Mirrors [`Vec::len`]. Lowers to `cardinality(col)` on PostgreSQL.
516    pub fn len(self) -> Expr<i64> {
517        Expr::from_untyped(stmt::Expr::array_length(self.untyped.into_stmt()))
518    }
519
520    /// Returns `true` if the array is empty.
521    ///
522    /// Equivalent to `.len().eq(0)`. Mirrors [`Vec::is_empty`].
523    pub fn is_empty(self) -> Expr<bool> {
524        let untyped = stmt::Expr::eq(
525            stmt::Expr::array_length(self.untyped.into_stmt()),
526            stmt::Expr::Value(stmt::Value::I64(0)),
527        );
528        Expr::from_untyped(untyped)
529    }
530}

574    /// Test whether this string field starts with `prefix`.
575    ///
576    /// Available on any string-valued field, including `String`,
577    /// `Option<String>`, and other wrappers whose `Field::Inner` is `String`.
578    /// For DynamoDB, this maps to `begins_with` in a `KeyConditionExpression`
579    /// (sort key) or `FilterExpression` (non-key attribute).
580    ///
581    /// # Examples
582    ///
583    /// ```
584    /// # #[derive(Debug, toasty::Model)]
585    /// # struct User {
586    /// #     #[key]
587    /// #     id: i64,
588    /// #     name: String,
589    /// #     nickname: Option<String>,
590    /// # }
591    /// let filter = User::fields().name().starts_with("Al".to_string());
592    /// let filter = User::fields().nickname().starts_with("Al".to_string());
593    /// ```
594    pub fn starts_with(self, prefix: impl IntoExpr<String>) -> Expr<bool> {
595        let prefix = prefix.into_expr().untyped;
596        self.build_filter(move |path| stmt::Expr::starts_with(path, prefix))
597    }
598
599    /// Test whether this string field matches a SQL `LIKE` pattern.
600    ///
601    /// Available on any string-valued field, including `String`,
602    /// `Option<String>`, and other wrappers whose `Field::Inner` is `String`.
603    /// The caller is responsible for including any `%` or `_` wildcard
604    /// characters in `pattern`. Not supported by the DynamoDB driver — use
605    /// [`starts_with`](Self::starts_with) instead.
606    ///
607    /// `.like()` is a pass-through to the database's own `LIKE`, whose case
608    /// sensitivity differs by backend: case-sensitive on PostgreSQL,
609    /// case-insensitive for ASCII on SQLite, and collation-dependent on MySQL.
610    /// For a case-insensitive match on PostgreSQL, use [`ilike`](Self::ilike).
611    ///
612    /// # Examples
613    ///
614    /// ```
615    /// # #[derive(Debug, toasty::Model)]
616    /// # struct User {
617    /// #     #[key]
618    /// #     id: i64,
619    /// #     name: String,
620    /// #     nickname: Option<String>,
621    /// # }
622    /// let filter = User::fields().name().like("Al%".to_string());
623    /// let filter = User::fields().nickname().like("Al%".to_string());
624    /// ```
625    pub fn like(self, pattern: impl IntoExpr<String>) -> Expr<bool> {
626        let pattern = pattern.into_expr().untyped;
627        self.build_filter(move |path| stmt::Expr::like(path, pattern))
628    }
629
630    /// Case-insensitive variant of [`like`](Self::like), mapping to
631    /// PostgreSQL's `ILIKE` operator.
632    ///
633    /// PostgreSQL is the only supported backend with a native `ILIKE`, so
634    /// `.ilike()` works only there. Toasty does not emulate it: on MySQL,
635    /// SQLite, and DynamoDB the query is rejected with an unsupported-feature
636    /// error. PostgreSQL's `LIKE` is case-sensitive and so provides `ILIKE` as
637    /// its case-insensitive companion; the other backends fold ASCII case in
638    /// `LIKE` (SQLite) or set it through the column collation (MySQL), so there
639    /// is no operator with matching semantics for `.ilike()` to pass through to.
640    ///
641    /// # Examples
642    ///
643    /// ```
644    /// # #[derive(Debug, toasty::Model)]
645    /// # struct User {
646    /// #     #[key]
647    /// #     id: i64,
648    /// #     name: String,
649    /// # }
650    /// // Matches "Alice", "ALICIA", and "alfred".
651    /// let filter = User::fields().name().ilike("al%".to_string());
652    /// ```
653    pub fn ilike(self, pattern: impl IntoExpr<String>) -> Expr<bool> {
654        Expr {
655            untyped: stmt::Expr::ilike(self.untyped.into_stmt(), pattern.into_expr().untyped),
656            _p: PhantomData,
657        }
658    }
659}

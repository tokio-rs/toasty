1//! Tracing spans for statement execution.
2//!
3//! Statements execute on a dedicated per-connection worker task, so a span
4//! entered by the caller does not naturally cover the driver's work. The
5//! spans created here are constructed on the caller's task — parenting them
6//! to the caller's current span (e.g. a per-request span) — then carried
7//! across the channel to the worker task and entered there, so every event
8//! the engine and driver emit lands inside the caller's span tree.
9
10use toasty_core::{
11    Schema,
12    schema::app::ModelId,
13    stmt::{self, ExprSet, InsertTarget, UpdateTarget},
14};
15
16/// Creates the `toasty::query` span for one statement execution.
17///
18/// When the span is disabled by the subscriber's filter, returns the
19/// caller's current span instead so request context still reaches the
20/// worker task's events.
21pub(crate) fn query_span(schema: &Schema, stmt: &stmt::Statement) -> tracing::Span {
22    let span = tracing::debug_span!(
23        target: "toasty::query",
24        "query",
25        stmt.kind = stmt.name(),
26        model = tracing::field::Empty,
27    );
28    if span.is_disabled() {
29        return tracing::Span::current();
30    }
31    if let Some(model_id) = stmt_model_id(stmt) {
32        let name = schema.app.model(model_id).name().upper_camel_case();
33        span.record("model", name.as_str());
34    }
35    span
36}
37
38/// Creates the `toasty::query` span for a user-authored SQL execution.
39/// Falls back to the caller's current span like [`query_span`].
40pub(crate) fn raw_sql_span() -> tracing::Span {
41    let span = tracing::debug_span!(target: "toasty::query", "query", stmt.kind = "raw_sql");
42    if span.is_disabled() {
43        return tracing::Span::current();
44    }
45    span
46}

48/// Resolves the model a statement targets, when it targets one directly.
49/// Statements arriving here are model-level; lowered (table-level) shapes
50/// return `None`.
51fn stmt_model_id(stmt: &stmt::Statement) -> Option<ModelId> {
52    match stmt {
53        stmt::Statement::Query(query) => query_model_id(query),
54        stmt::Statement::Insert(insert) => match &insert.target {
55            InsertTarget::Model(model_id) => Some(*model_id),
56            InsertTarget::Scope(query) => query_model_id(query),
57            InsertTarget::Table(_) => None,
58        },
59        stmt::Statement::Update(update) => match &update.target {
60            UpdateTarget::Model(model_id) => Some(*model_id),
61            UpdateTarget::Query(query) => query_model_id(query),
62            UpdateTarget::Table(_) => None,
63        },
64        stmt::Statement::Delete(delete) => delete.from.as_model().map(|source| source.id),
65    }
66}
67
68fn query_model_id(query: &stmt::Query) -> Option<ModelId> {
69    match &query.body {
70        ExprSet::Select(select) => select.source.as_model().map(|source| source.id),
71        ExprSet::Update(update) => match &update.target {
72            UpdateTarget::Model(model_id) => Some(*model_id),
73            UpdateTarget::Query(query) => query_model_id(query),
74            UpdateTarget::Table(_) => None,
75        },
76        _ => None,
77    }
78}

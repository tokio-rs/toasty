1//! cms-article-fields: a CMS publishing flow on a single Article — draft it, count views,
2//! curate tags, lazily load the body, then publish. Along the way it answers "what can I
3//! attach to one field?": custom column names, create/update defaults, auto timestamps, a
4//! queryable `Vec<scalar>`, a deferred column, and opaque JSON.
5//!
6//! Run it cold (`cargo run -p example-cms-article-fields`). In-memory SQLite by default;
7//! set `TOASTY_CONNECTION_URL` for another backend.
8
9// A plain serde type. `Json<SeoMeta>` stores it as one opaque JSON column — handy for a
10// value with no native column representation, at the cost of not being queryable.
11#[derive(Debug, serde::Serialize, serde::Deserialize)]
12struct SeoMeta {
13    description: String,
14    canonical_url: Option<String>,
15}
16
17// A fixed set of lifecycle states is better modeled as an embedded enum than a `String`: typos
18// become compile errors and `match`es stay exhaustive. With no `#[column(variant)]`, each
19// variant is stored in one column as its name in snake_case ("draft", "published", "archived");
20// override the stored label with `#[column(variant = "d")]` or an integer if you need to.
21#[derive(Debug, PartialEq, toasty::Embed)]
22enum ArticleStatus {
23    Draft,
24    Published,
25    Archived,
26}
27
28#[derive(Debug, toasty::Model)]
29// Custom table name. By default Toasty snake_cases and pluralizes the struct name, so `Article`
30// would map to `articles`; we override it to namespace the table for a CMS schema.
31#[table = "cms_articles"]
32struct Article {
33    #[key]
34    #[auto]
35    id: uuid::Uuid,
36
37    // The Rust field stays `title`; the column is `headline`.
38    #[column("headline")]
39    title: String,
40
41    // A second auto field that is NOT the key. `uuid(v4)` is a random id (vs the default
42    // time-ordered v7), which is what you want for an unguessable share token.
43    #[auto(uuid(v4))]
44    share_token: uuid::Uuid,
45
46    // `Option<T>` is a nullable column. Left unset on create, it stores NULL / reads None.
47    subtitle: Option<String>,
48
49    // `#[default(expr)]` runs at insert time and applies ONLY on create. The expression is any
50    // Rust value — a literal, or an enum variant.
51    #[default(0)]
52    view_count: i64,
53    #[default(ArticleStatus::Draft)]
54    status: ArticleStatus,
55
56    // `#[default]` sets this once, on create. `#[update]` re-evaluates on every write, so
57    // `updated_at` tracks the last change while `created_at` is fixed.
58    #[default(jiff::Timestamp::now())]
59    created_at: jiff::Timestamp,
60    #[update(jiff::Timestamp::now())]
61    updated_at: jiff::Timestamp,
62
63    // A queryable scalar collection (one column). Unlike `Json<T>`, you can filter and
64    // incrementally mutate it.
65    tags: Vec<String>,
66
67    // A large column omitted from the default SELECT; load it on demand with `.include()`.
68    body: toasty::Deferred<String>,
69
70    seo: toasty::Json<SeoMeta>,
71}
72
73#[tokio::main]
74async fn main() -> toasty::Result<()> {
75    let url =
76        std::env::var("TOASTY_CONNECTION_URL").unwrap_or_else(|_| "sqlite::memory:".to_string());
77    let mut db = toasty::Db::builder()
78        .models(toasty::models!(crate::*))
79        .connect(&url)
80        .await?;
81    db.push_schema().await?;
82
83    // Set only what we want; everything else comes from #[default]/#[auto]/#[update].
84    let mut article = toasty::create!(Article {
85        title: "Hello, Toasty",
86        body: "a long body we usually don't read in list views".to_string(),
87        tags: ["rust", "orm"], // array literal — no vec! needed
88        seo: toasty::Json(SeoMeta {
89            description: "an intro to Toasty".into(),
90            canonical_url: None,
91        }),
92    })
93    .exec(&mut db)
94    .await?;
95
96    assert_eq!(article.view_count, 0); // #[default(0)] fired on create
97    assert_eq!(article.status, ArticleStatus::Draft); // #[default(..)] value
98    assert!(article.subtitle.is_none()); // omitted Option -> None / SQL NULL
99    println!(
100        "drafted {:?}: status={:?}, views={}, share_token={}",
101        article.title, article.status, article.view_count, article.share_token
102    );
103
104    // Register a page view. A relative numeric update folds read+write into ONE atomic
105    // statement, so concurrent viewers can't clobber each other (a Rust-side
106    // read-modify-write would race).
107    toasty::update!(article { view_count.increment() })
108        .exec(&mut db)
109        .await?;
110    println!("after a page view: view_count={}", article.view_count);
111
112    // Incrementally mutate the tag collection. push/extend/clear/set are portable;
113    // pop/remove/remove_at are PostgreSQL-only (see postgres-directory).
114    toasty::update!(article { tags.push("published") })
115        .exec(&mut db)
116        .await?;
117    println!("tags: {:?}", article.tags);
118
119    // A deferred column is unloaded after a normal query — calling `.get()` on it panics.
120    let listed = Article::filter_by_id(article.id).get(&mut db).await?;
121    assert!(listed.body.is_unloaded());
122    // `.include()` adds it to the SAME query (no extra round-trip); then `.get()` is sync.
123    let full = Article::filter_by_id(article.id)
124        .include(Article::fields().body())
125        .get(&mut db)
126        .await?;
127    println!("loaded the body on demand: {} bytes", full.body.get().len());
128
129    // Fill in the previously-null subtitle, then publish. Setting `status` explicitly
130    // overrides its default; `updated_at` advances while `created_at` stays put.
131    toasty::update!(article {
132        subtitle: "a gentle introduction"
133    })
134    .exec(&mut db)
135    .await?;
136    toasty::update!(article {
137        status: ArticleStatus::Published
138    })
139    .exec(&mut db)
140    .await?;
141    println!(
142        "published: status={:?}, subtitle={:?}, created==updated? {}",
143        article.status,
144        article.subtitle,
145        article.created_at == article.updated_at
146    );
147
148    Ok(())
149}

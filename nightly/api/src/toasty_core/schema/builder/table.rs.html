34/// Computes a model's mapping, creating table columns and mapping expressions
35/// in a single recursive pass over the model's fields.
36///
37/// Holds state that persists across the entire mapping process: the shared
38/// mutable accumulators (columns, lowering expressions, bit counter) plus
39/// references to the table and schema. The recursive field-mapping logic lives
40/// on [`MapField`], which borrows `BuildMapping` and carries per-level context.
41struct BuildMapping<'a> {
42    app: &'a app::Schema,
43    db: &'a driver::Capability,
44    table: &'a mut Table,
45    mapping: &'a mut mapping::Model,
46    /// Model-name prefix used when multiple models share one table, separated
47    /// from the rest of the column name with `__`. None for single-model tables.
48    schema_prefix: Option<String>,
49    /// Optional prefix for database object names (enum types).
50    name_prefix: Option<String>,
51    next_bit: usize,
52    lowering_columns: Vec<ColumnId>,
53    model_to_table: Vec<stmt::Expr>,
54    table_to_model: Vec<stmt::Expr>,
55}
56
57/// Per-level state for the recursive `map_field*` methods.
58///
59/// Analogous to `LowerStatement` in `lower.rs`: `MapField` holds context that
60/// may change between recursive calls, while [`BuildMapping`] holds the shared
61/// mutable accumulators (columns, lowering expressions, bit counter) that
62/// persist across the entire mapping process.
63struct MapField<'a, 'b> {
64    /// State shared across the entire mapping process.
65    build: &'a mut BuildMapping<'b>,
66
67    /// Accumulated embed-prefix components (without schema_prefix), pushed on
68    /// entry to each nested field and popped on exit.
69    ///
70    /// Final column names join these with `_`, append the field name, then
71    /// prepend the schema prefix (if any) with `__`. Keeping components
72    /// separate ensures schema_prefix is applied exactly once.
73    prefix: Vec<String>,
74
75    /// When true, columns are created nullable regardless of the field's own
76    /// nullability. Set while processing fields whose columns are only populated
77    /// conditionally: enum variant fields (only the active variant's columns are
78    /// populated) and nullable embedded-struct fields (all columns are NULL when
79    /// the embed is `None`).
80    force_nullable: bool,
81
82    /// Base expression for the current nesting level.
83    ///
84    /// `None` at the top level. `Expr::Project(source_ref, proj)` at any nested
85    /// level, where `source_ref` is the top-level source field reference and
86    /// `proj` is the projection from that source down to the current container
87    /// (not including the final `field_index` step). `field_expr` and
88    /// `sub_projection` extend it by `field_index` to reach a specific field.
89    field_base: Option<stmt::Expr>,
90
91    /// A template expression with `Expr::arg(0)` as a placeholder for the raw
92    /// field expression. `field_expr` substitutes the raw field expression into
93    /// this template before returning. The identity value is `Expr::arg(0)`
94    /// itself, which substitutes to the raw expression unchanged.
95    ///
96    /// Used by variant-specific `MapField` instances to automatically wrap
97    /// field expressions in the discriminant match guard.
98    field_expr_base: stmt::Expr,
99
100    /// True when an outer single-field embed flagged its column as
101    /// auto-increment. Single-field newtype embeds flatten to one column, so
102    /// the outer field's `#[auto]` (Increment) must apply to that flattened
103    /// column. ORed into the column's flag at creation time so an outer or
104    /// inner declaration both take effect.
105    inherited_auto_increment: bool,
106}

108impl BuildSchema<'_> {
109    pub(super) fn build_table_stub_for_model(&mut self, model: &ModelRoot) -> TableId {
110        if let Some(table_name) = model.table_name.as_ref() {
111            let table_name = self.prefix_table_name(table_name);
112
113            if !self.table_lookup.contains_key(&table_name) {
114                let id = self.register_table(&table_name);
115                self.tables.push(Table::new(id, table_name.clone()));
116            }
117
118            *self.table_lookup.get(&table_name).unwrap()
119        } else {
120            let name = self.table_name_from_model(&model.name);
121            let id = self.register_table(&name);
122
123            self.tables.push(Table::new(id, name));
124            id
125        }
126    }
127
128    pub(super) fn build_tables_from_models(
129        &mut self,
130        app: &app::Schema,
131        db: &driver::Capability,
132    ) -> Result<()> {
133        for table in &mut self.tables {
134            let models = app
135                .models()
136                .filter(|model| model.is_root())
137                .filter(|model| self.mapping.model(model.id()).table == table.id)
138                .collect::<Vec<_>>();
139
140            assert!(
141                models.len() == 1,
142                "TODO: handle mapping many models to one table"
143            );
144
145            BuildTableFromModels {
146                app,
147                db,
148                table,
149                mapping: &mut self.mapping,
150                prefix_table_names: models.len() > 1,
151                name_prefix: self.builder.table_name_prefix.clone(),
152            }
153            .build(models[0])?;
154        }
155
156        Ok(())
157    }
158
159    pub(super) fn register_table(&mut self, name: impl AsRef<str>) -> TableId {
160        assert!(!self.table_lookup.contains_key(name.as_ref()));
161        let id = TableId(self.table_lookup.len());
162        self.table_lookup.insert(name.as_ref().to_string(), id);
163        id
164    }
165
166    fn table_name_from_model(&self, model_name: &Name) -> String {
167        let base = pluralizer::pluralize(&model_name.snake_case(), 2, false);
168        self.prefix_table_name(&base)
169    }
170
171    fn prefix_table_name(&self, name: &str) -> String {
172        if let Some(prefix) = &self.builder.table_name_prefix {
173            format!("{prefix}{name}")
174        } else {
175            name.to_string()
176        }
177    }

180impl BuildTableFromModels<'_> {
181    fn build(&mut self, model: &Model) -> Result<()> {
182        self.map_model_fields(model)?;
183        self.update_index_names();
184        Ok(())
185    }
186
187    fn map_model_fields(&mut self, model: &Model) -> Result<()> {
188        let root = model.as_root_unwrap();
189        let schema_prefix = if self.prefix_table_names {
190            Some(model.name().snake_case())
191        } else {
192            None
193        };
194
195        BuildMapping {
196            app: self.app,
197            db: self.db,
198            table: self.table,
199            mapping: self.mapping.model_mut(model),
200            schema_prefix,
201            name_prefix: self.name_prefix.clone(),
202            next_bit: 0,
203            lowering_columns: vec![],
204            model_to_table: vec![],
205            table_to_model: vec![],
206        }
207        .build_mapping(root)?;
208
209        let model_fields = &self.mapping.model(model.id()).fields;
210        let mut indices = Vec::new();
211        self.collect_indices(&root.fields, model_fields, &root.indices, &mut indices)?;
212
213        for index in indices {
214            if index.primary_key {
215                self.table.primary_key.columns = index.columns.iter().map(|c| c.column).collect();
216            }
217
218            self.table.indices.push(index);
219        }
220
221        Ok(())
222    }
223
224    /// Collects DB-level indices from app-level index definitions, then recurses
225    /// into embedded struct fields to collect their indices as well.
226    fn collect_indices(
227        &self,
228        fields: &[app::Field],
229        field_mappings: &[mapping::Field],
230        indices: &[app::Index],
231        out: &mut Vec<db::Index>,
232    ) -> Result<()> {
233        for app_index in indices {
234            let mut index = db::Index {
235                id: IndexId {
236                    table: self.table.id,
237                    index: out.len(),
238                },
239                name: app_index.name.clone().unwrap_or_default(),
240                on: self.table.id,
241                columns: vec![],
242                unique: app_index.unique,
243                primary_key: app_index.primary_key,
244            };
245
246            for index_field in &app_index.fields {
247                let mapping = &field_mappings[index_field.field.index];
248
249                // Resolve the mapped column for this indexed field. Primitive
250                // fields map directly. Newtype embedded structs (a single
251                // unnamed field) are transparent wrappers around a primitive, so
252                // we unwrap one level.
253                //
254                // Multi-field or named-field embedded structs are not yet
255                // supported in indices because the column ordering within the
256                // index matters and there is no syntax to specify it. That will
257                // likely require an explicit field-order annotation on the
258                // index.
259                let column = self.resolve_indexed_column(mapping);
260
261                index.columns.push(db::IndexColumn {
262                    column,
263                    op: index_field.op,
264                    scope: index_field.scope,
265                });
266            }
267
268            out.push(index);
269        }
270
271        for (field_index, field) in fields.iter().enumerate() {
272            let app::FieldTy::Embedded(embedded) = &field.ty else {
273                continue;
274            };
275
276            let target = lookup_embedded_model(self.app, embedded.target, field)?;
277
278            match target {
279                app::Model::EmbeddedStruct(embedded_struct) => {
280                    let field_mapping = field_mappings[field_index]
281                        .as_struct()
282                        .expect("embedded struct field should have struct mapping");
283
284                    self.collect_indices(
285                        &embedded_struct.fields,
286                        &field_mapping.fields,
287                        &embedded_struct.indices,
288                        out,
289                    )?;
290                }
291                app::Model::EmbeddedEnum(embedded_enum) => {
292                    if embedded_enum.indices.is_empty() {
293                        continue;
294                    }
295
296                    let field_mapping = field_mappings[field_index]
297                        .as_enum()
298                        .expect("embedded enum field should have enum mapping");
299
300                    // Build a flat mapping from global field index to the
301                    // field's mapping within its variant. Enum fields use
302                    // global indices; each field belongs to a specific variant
303                    // and has a local offset within that variant.
304                    let mut flat_mappings: Vec<mapping::Field> =
305                        vec![
306                            mapping::Field::Relation(mapping::FieldRelation {
307                                field_mask: crate::stmt::PathFieldSet::new(),
308                            });
309                            embedded_enum.fields.len()
310                        ];
311
312                    for (variant_idx, variant_mapping) in field_mapping.variants.iter().enumerate()
313                    {
314                        let mut local_idx = 0;
315                        for (global_idx, f) in embedded_enum.fields.iter().enumerate() {
316                            let app::VariantId { index: vi, .. } =
317                                f.variant.expect("enum field must have variant");
318                            if vi != variant_idx {
319                                continue;
320                            }
321                            flat_mappings[global_idx] = variant_mapping.fields[local_idx].clone();
322                            local_idx += 1;
323                        }
324                    }
325
326                    self.collect_indices(
327                        &embedded_enum.fields,
328                        &flat_mappings,
329                        &embedded_enum.indices,
330                        out,
331                    )?;
332                }
333                _ => continue,
334            }
335        }
336
337        Ok(())
338    }
339
340    /// Walks newtype embed layers down to the underlying primitive column.
341    ///
342    /// Indexed fields may be either a primitive directly or a chain of
343    /// single-unnamed-field embeds wrapping a primitive. Multi-field or
344    /// named-field embeds are not supported as index targets — the index
345    /// would have to choose a column ordering and there is no syntax for
346    /// that yet.
347    fn resolve_indexed_column(&self, mut mapping: &mapping::Field) -> ColumnId {
348        loop {
349            match mapping {
350                mapping::Field::Primitive(p) => return p.column,
351                mapping::Field::Enum(p) => {
352                    // Only unit (data-less) enums can be indexed: the
353                    // discriminant column alone represents the value.
354                    // Data-carrying enums span multiple columns and have no
355                    // single index column. The derive macro rejects this at
356                    // compile time via `IndexableField`; this assert is the
357                    // backstop for schemas built without the macro.
358                    assert!(
359                        p.variants.iter().all(|v| v.fields.is_empty()),
360                        "only unit (data-less) embedded enums can be indexed; \
361                         data-carrying enum variants span multiple columns"
362                    );
363                    return p.discriminant.column;
364                }
365                mapping::Field::Struct(s) => {
366                    let embedded_struct = self.app.model(s.id).as_embedded_struct_unwrap();
367                    assert!(
368                        embedded_struct.fields.len() == 1
369                            && embedded_struct.fields[0].name.app.is_none(),
370                        "only newtype embedded structs (single unnamed \
371                         field) can be indexed; multi-field or named-field \
372                         embedded structs require explicit index field \
373                         ordering"
374                    );
375                    mapping = &s.fields[0];
376                }
377                _ => panic!(
378                    "only primitive and newtype embedded structs can be \
379                     indexed"
380                ),
381            }
382        }
383    }
384
385    fn update_index_names(&mut self) {
386        for index in &mut self.table.indices {
387            // Preserve user-provided names from `#[index(name = "...", ...)]`.
388            if !index.name.is_empty() {
389                continue;
390            }
391
392            let mut name = format!("index_{}_by", self.table.name);
393
394            for (i, index_column) in index.columns.iter().enumerate() {
395                let column = &self.table.columns[index_column.column.index];
396
397                if i > 0 {
398                    name.push_str("_and");
399                }
400
401                name.push('_');
402                name.push_str(&column.name);
403            }
404
405            index.name = if let Some(limit) = self.db.max_identifier_length {
406                truncate_identifier(name, limit)
407            } else {
408                name
409            };
410        }
411    }
412}

414impl BuildMapping<'_> {
415    fn build_mapping(mut self, model: &ModelRoot) -> Result<()> {
416        let mut fields = MapField::new(&mut self).map_fields(&model.fields)?;
417
418        assert!(!self.model_to_table.is_empty());
419        assert_eq!(self.model_to_table.len(), self.lowering_columns.len());
420
421        self.build_table_to_model(model, &fields)?;
422
423        // Compute the default `RETURNING` expression for the model and for
424        // each nested embedded type. Mutates `fields` to populate the
425        // per-embed `default_returning` along the way.
426        let default_returning = self.build_default_returning_root(model, &mut fields)?;
427
428        self.mapping.fields = fields;
429        self.mapping.columns = self.lowering_columns;
430        self.mapping.model_to_table = stmt::ExprRecord::from_vec(self.model_to_table);
431        self.mapping.table_to_model =
432            TableToModel::new(stmt::ExprRecord::from_vec(self.table_to_model));
433        self.mapping.default_returning = default_returning;
434
435        Ok(())
436    }
437
438    /// Builds the model's default `RETURNING` expression — the same shape as
439    /// `table_to_model` but with every deferred field, at this level or
440    /// inside a nested embedded type, pre-masked to `Null`. Also writes each
441    /// embed's own default expression into the corresponding mapping node so
442    /// lowering can splice it in when an `.include()` activates a deferred
443    /// embed.
444    fn build_default_returning_root(
445        &self,
446        model: &ModelRoot,
447        fields: &mut [mapping::Field],
448    ) -> Result<stmt::Expr> {
449        let exprs: Vec<stmt::Expr> = model
450            .fields
451            .iter()
452            .zip(fields.iter_mut())
453            .map(|(field, mapping)| self.build_default_returning_field(field, mapping))
454            .collect::<Result<_>>()?;
455        Ok(stmt::Expr::record(exprs))
456    }
457
458    /// Builds the default returning expression for a single field and, if
459    /// the field is an embedded type, populates the embed's own
460    /// `default_returning` cache.
461    fn build_default_returning_field(
462        &self,
463        field: &app::Field,
464        mapping: &mut mapping::Field,
465    ) -> Result<stmt::Expr> {
466        // Deferred fields are `Null` in the default expression. Still
467        // recurse through deferred embeds so the nested `default_returning`
468        // is populated — `process_includes` reads it during a `.include()`
469        // splice.
470        if field.deferred {
471            if matches!(&field.ty, app::FieldTy::Embedded(_)) {
472                self.populate_embed_default_returning(field, mapping)?;
473            }
474            return Ok(stmt::Expr::null());
475        }
476
477        match &field.ty {
478            app::FieldTy::Primitive(primitive) => {
479                let column_id = mapping.as_primitive().unwrap().column;
480                Ok(self.map_table_column_to_model(column_id, primitive))
481            }
482            app::FieldTy::Embedded(_) => self.populate_embed_default_returning(field, mapping),
483            app::FieldTy::BelongsTo(_) | app::FieldTy::Has(_) | app::FieldTy::Via(_) => {
484                Ok(stmt::Value::Null.into())
485            }
486        }
487    }
488
489    /// Resolves the embedded target, recurses to compute its default
490    /// expression, stores it on the mapping node, and returns a clone for
491    /// the caller to splice in for the parent field.
492    fn populate_embed_default_returning(
493        &self,
494        field: &app::Field,
495        mapping: &mut mapping::Field,
496    ) -> Result<stmt::Expr> {
497        let app::FieldTy::Embedded(embedded) = &field.ty else {
498            unreachable!("populate_embed_default_returning called on non-embed");
499        };
500        let target = lookup_embedded_model(self.app, embedded.target, field)?;
501
502        match (target, mapping) {
503            (app::Model::EmbeddedStruct(embed_model), mapping::Field::Struct(s)) => {
504                let record = {
505                    let exprs: Vec<stmt::Expr> = embed_model
506                        .fields
507                        .iter()
508                        .zip(s.fields.iter_mut())
509                        .map(|(f, m)| self.build_default_returning_field(f, m))
510                        .collect::<Result<_>>()?;
511                    stmt::Expr::record(exprs)
512                };
513                // A nullable embedded struct guards its record on the head
514                // column, matching `build_table_to_model_field_struct`.
515                let expr = match s.presence {
516                    None => record,
517                    Some(presence) => self.wrap_presence_match(presence.index, record),
518                };
519                s.default_returning = expr.clone();
520                Ok(expr)
521            }
522            (app::Model::EmbeddedEnum(embed_model), mapping::Field::Enum(e)) => {
523                let expr = self.build_default_returning_enum(embed_model, e, field.nullable)?;
524                e.default_returning = expr.clone();
525                Ok(expr)
526            }
527            _ => unreachable!("invalid schema: embedded field maps to root model"),
528        }
529    }
530
531    /// Builds the enum's default `Match` expression, recursing into each
532    /// variant's fields so a deferred sub-field nested inside a variant's
533    /// embed-struct is pre-masked. The shape mirrors
534    /// [`Self::build_table_to_model_field_enum`] but routes each variant
535    /// field through [`Self::build_default_returning_field`] instead of
536    /// the raw column emitter.
537    ///
538    /// Deferred variant fields route through the same default-returning
539    /// masking as fields inside embedded structs.
540    fn build_default_returning_enum(
541        &self,
542        model: &app::EmbeddedEnum,
543        mapping: &mut mapping::FieldEnum,
544        nullable: bool,
545    ) -> Result<stmt::Expr> {
546        let disc_col_ref = stmt::Expr::column(stmt::ExprColumn {
547            nesting: 0,
548            table: 0,
549            column: mapping.discriminant.column.index,
550        });
551
552        if !model.has_data_variants() {
553            return Ok(disc_col_ref);
554        }
555
556        let mut arms = Vec::new();
557        for (variant_index, (variant, variant_mapping)) in model
558            .variants
559            .iter()
560            .zip(mapping.variants.iter_mut())
561            .enumerate()
562        {
563            let variant_fields: Vec<&app::Field> = model.variant_fields(variant_index).collect();
564            let arm_expr = if variant_fields.is_empty() {
565                disc_col_ref.clone()
566            } else {
567                let mut record_elems = vec![disc_col_ref.clone()];
568                for (local_idx, field) in variant_fields.iter().enumerate() {
569                    let mapping_field = &mut variant_mapping.fields[local_idx];
570                    record_elems.push(self.build_default_returning_field(field, mapping_field)?);
571                }
572                stmt::Expr::record(record_elems)
573            };
574            arms.push(stmt::MatchArm {
575                pattern: variant.discriminant.clone(),
576                expr: arm_expr,
577            });
578        }
579
580        let else_expr = self.enum_decode_else(model, nullable, &disc_col_ref);
581        Ok(stmt::Expr::match_expr(disc_col_ref, arms, else_expr))
582    }

584    fn next_bit(&mut self) -> usize {
585        let bit = self.next_bit;
586        self.next_bit += 1;
587        bit
588    }

590    fn build_table_to_model(
591        &mut self,
592        model: &ModelRoot,
593        mapping: &[mapping::Field],
594    ) -> Result<()> {
595        for (index, field) in model.fields.iter().enumerate() {
596            let expr = self.build_table_to_model_field(field, &mapping[index])?;
597            self.table_to_model.push(expr);
598        }
599        Ok(())
600    }

602    /// Builds the `table_to_model` expression for an embedded enum field.
603    ///
604    /// For unit-only enums the discriminant column reference suffices.
605    /// For mixed/data-carrying enums a `Match` expression dispatches on the
606    /// discriminant: unit arms return the discriminant directly, data arms
607    /// return `Record([disc, field1, ...])` matching the shape expected by
608    /// `Primitive::load`.
609    ///
610    /// When `nullable` (an `Option<EmbeddedEnum>`), the discriminant column is
611    /// `NULL` for `None`: a unit-only enum's column reference already reads back
612    /// as `None`, and a data enum's `Match` else branch yields `null` so a
613    /// `NULL` (unmatched) discriminant decodes to `None`.
614    fn build_table_to_model_field_enum(
615        &self,
616        model: &app::EmbeddedEnum,
617        mapping: &mapping::FieldEnum,
618        nullable: bool,
619    ) -> Result<stmt::Expr> {
620        let disc_col_ref = stmt::Expr::column(stmt::ExprColumn {
621            nesting: 0,
622            table: 0,
623            column: mapping.discriminant.column.index,
624        });
625
626        if !model.has_data_variants() {
627            return Ok(disc_col_ref);
628        }
629
630        let mut arms = Vec::new();
631
632        for (variant_index, (variant, mapping)) in
633            model.variants.iter().zip(&mapping.variants).enumerate()
634        {
635            let variant_fields: Vec<_> = model.variant_fields(variant_index).collect();
636            let arm_expr = if variant_fields.is_empty() {
637                disc_col_ref.clone()
638            } else {
639                let mut record_elems = vec![disc_col_ref.clone()];
640
641                for (local_idx, field) in variant_fields.iter().enumerate() {
642                    let expr =
643                        self.build_table_to_model_field(field, &mapping.fields[local_idx])?;
644                    record_elems.push(expr);
645                }
646                stmt::Expr::record(record_elems)
647            };
648            arms.push(stmt::MatchArm {
649                pattern: variant.discriminant.clone(),
650                expr: arm_expr,
651            });
652        }
653
654        let else_expr = self.enum_decode_else(model, nullable, &disc_col_ref);
655        Ok(stmt::Expr::match_expr(disc_col_ref, arms, else_expr))
656    }
657
658    /// The `else` branch shared by the enum decode and default-returning
659    /// `Match` expressions.
660    ///
661    /// When `nullable`, a `NULL` (unmatched) discriminant is the `None` case, so
662    /// the else is `null`. Otherwise it mirrors the data-arm `Record` shape but
663    /// fills field positions with `Expr::Error`, making projections distribute
664    /// uniformly: projecting `[0]` extracts the discriminant (pruning the
665    /// errors) while a deeper projection yields `Expr::Error` (unreachable at
666    /// runtime).
667    fn enum_decode_else(
668        &self,
669        model: &app::EmbeddedEnum,
670        nullable: bool,
671        disc_col_ref: &stmt::Expr,
672    ) -> stmt::Expr {
673        if nullable {
674            return stmt::Expr::null();
675        }
676        let max_fields = (0..model.variants.len())
677            .map(|i| model.variant_fields(i).count())
678            .max()
679            .unwrap_or(0);
680        if max_fields == 0 {
681            stmt::Expr::error("unexpected enum discriminant")
682        } else {
683            let mut elems = vec![disc_col_ref.clone()];
684            for _ in 0..max_fields {
685                elems.push(stmt::Expr::error("unexpected enum discriminant"));
686            }
687            stmt::Expr::record(elems)
688        }
689    }
690
691    /// Encodes `expr` for `column_id`, appends the result to `model_to_table`,
692    /// records the column in `lowering_columns`, and returns the lowering index.
693    fn push_lowering(
694        &mut self,
695        column_id: ColumnId,
696        ty: &stmt::Type,
697        expr: impl Into<stmt::Expr>,
698    ) -> usize {
699        let lowering_expr = self.encode_column(column_id, ty, expr);
700        let lowering_index = self.model_to_table.len();
701        self.lowering_columns.push(column_id);
702        self.model_to_table.push(lowering_expr);
703        lowering_index
704    }
705
706    fn encode_column(
707        &self,
708        column_id: ColumnId,
709        ty: &stmt::Type,
710        expr: impl Into<stmt::Expr>,
711    ) -> stmt::Expr {
712        let expr = expr.into();
713        let column = self.table.column(column_id);
714
715        assert_ne!(stmt::Type::Null, *ty);
716
717        match &column.ty {
718            column_ty if column_ty == ty => expr,
719            // If the types do not match, attempt casting as a fallback.
720            _ => stmt::Expr::cast(expr, &column.ty),
721        }
722    }

724    /// Maps table columns to model field expressions during query lowering.
725    ///
726    /// Called during query planning to replace model field references with the
727    /// appropriate table column expressions. Handles type conversions between
728    /// table storage and model types.
729    fn map_table_column_to_model(
730        &self,
731        column_id: ColumnId,
732        primitive: &app::FieldPrimitive,
733    ) -> stmt::Expr {
734        let column = self.table.column(column_id);
735
736        // NOTE: nesting and table are stubs here (though often the actual values).
737        // The engine must substitute these with the actual TableRef index in the query's TableSource.
738        let expr_column = stmt::Expr::column(stmt::ExprColumn {
739            nesting: 0,
740            table: 0,
741            column: column_id.index,
742        });
743
744        match &column.ty {
745            c_ty if *c_ty == primitive.ty => expr_column,
746            // If the types do not match, attempt casting as a fallback.
747            _ => stmt::Expr::cast(expr_column, &primitive.ty),
748        }
749    }
750
751    fn build_table_to_model_field_struct(
752        &self,
753        model: &app::EmbeddedStruct,
754        mapping: &mapping::FieldStruct,
755    ) -> Result<stmt::Expr> {
756        let exprs: Vec<stmt::Expr> = model
757            .fields
758            .iter()
759            .enumerate()
760            .map(|(index, field)| self.build_table_to_model_field(field, &mapping.fields[index]))
761            .collect::<Result<_>>()?;
762        let record = stmt::Expr::record(exprs);
763
764        // A nullable embedded struct decodes through a `Match` on its head
765        // column (see `wrap_presence_match`).
766        match mapping.presence {
767            None => Ok(record),
768            Some(presence) => Ok(self.wrap_presence_match(presence.index, record)),
769        }
770    }
771
772    /// Wraps `record` in the presence `Match` that decodes a nullable embedded
773    /// struct: when the head column is non-`NULL` the record is reconstructed,
774    /// otherwise the value is `null`, which `impl Load for Option<T>` reads
775    /// back as `None`. The engine evaluates this `Match` against the returned
776    /// row before `Load` runs.
777    ///
778    /// The head column is either a dedicated `bool` presence column (only ever
779    /// `true`/`NULL`) or, for a single-column embed, the embed's own flattened
780    /// leaf column — in both cases its null-ness is the option's none-ness, so
781    /// guarding on `is_null(head) == false` (present) is the uniform check. We
782    /// match `is_null(head)` against `false` rather than wrapping in `not(..)`
783    /// because the table→model evaluator handles `IsNull`/`Match`/`Project`
784    /// but not `Not`.
785    fn wrap_presence_match(&self, presence_column: usize, record: stmt::Expr) -> stmt::Expr {
786        stmt::Expr::match_expr(
787            stmt::Expr::is_null(stmt::Expr::column(stmt::ExprColumn {
788                nesting: 0,
789                table: 0,
790                column: presence_column,
791            })),
792            vec![stmt::MatchArm {
793                pattern: stmt::Value::Bool(false),
794                expr: record,
795            }],
796            stmt::Expr::null(),
797        )
798    }
799
800    fn build_table_to_model_field(
801        &self,
802        field: &app::Field,
803        mapping: &mapping::Field,
804    ) -> Result<stmt::Expr> {
805        match &field.ty {
806            app::FieldTy::Primitive(primitive) => {
807                let column_id = mapping.as_primitive().unwrap().column;
808                Ok(self.map_table_column_to_model(column_id, primitive))
809            }
810            app::FieldTy::Embedded(embedded) => {
811                let target = lookup_embedded_model(self.app, embedded.target, field)?;
812
813                match target {
814                    app::Model::EmbeddedEnum(embedded) => {
815                        let mapping = mapping
816                            .as_enum()
817                            .expect("embedded enum field should have enum mapping");
818                        self.build_table_to_model_field_enum(embedded, mapping, field.nullable)
819                    }
820                    app::Model::EmbeddedStruct(embedded) => {
821                        let mapping = mapping
822                            .as_struct()
823                            .expect("embedded struct field should have struct mapping");
824                        self.build_table_to_model_field_struct(embedded, mapping)
825                    }
826                    _ => unreachable!("invalid schema"),
827                }
828            }
829            app::FieldTy::BelongsTo(_) | app::FieldTy::Has(_) | app::FieldTy::Via(_) => {
830                Ok(stmt::Value::Null.into())
831            }
832        }
833    }
834}

836impl<'a, 'b> MapField<'a, 'b> {
837    fn new(build: &'a mut BuildMapping<'b>) -> Self {
838        MapField {
839            build,
840            prefix: vec![],
841            force_nullable: false,
842            field_base: None,
843            field_expr_base: stmt::Expr::arg(0),
844            inherited_auto_increment: false,
845        }

848    fn map_fields(&mut self, fields: &[app::Field]) -> Result<Vec<mapping::Field>> {
849        fields
850            .iter()
851            .enumerate()
852            .map(|(index, field)| self.map_field(index, field))
853            .collect()
854    }
855
856    fn map_field(&mut self, index: usize, field: &app::Field) -> Result<mapping::Field> {
857        match &field.ty {
858            app::FieldTy::Primitive(primitive) => {
859                Ok(self.map_field_primitive(index, field, primitive))
860            }
861            app::FieldTy::Embedded(embedded) => {
862                let target = lookup_embedded_model(self.build.app, embedded.target, field)?;
863
864                match target {
865                    app::Model::EmbeddedEnum(embedded_enum) => {
866                        self.map_field_enum(index, field, embedded_enum)
867                    }
868                    app::Model::EmbeddedStruct(embedded_struct) => {
869                        self.map_field_struct(index, field, embedded.target, embedded_struct)
870                    }
871                    _ => unreachable!(),
872                }
873            }
874            app::FieldTy::BelongsTo(_) | app::FieldTy::Has(_) | app::FieldTy::Via(_) => {
875                assert!(!self.force_nullable);
876                let bit = self.build.next_bit();
877                Ok(mapping::Field::Relation(mapping::FieldRelation {
878                    field_mask: stmt::PathFieldSet::from_iter([bit]),
879                }))
880            }
881        }
882    }

884    /// Creates the column and builds the mapping for a primitive field in one step.
885    fn map_field_primitive(
886        &mut self,
887        field_index: usize,
888        field: &app::Field,
889        primitive: &app::FieldPrimitive,
890    ) -> mapping::Field {
891        let column_id = self.create_column(field, primitive);
892        let expr = self.field_expr(field, field_index);
893        let lowering_index = self.build.push_lowering(column_id, &primitive.ty, expr);
894        let bit = self.build.next_bit();
895        let sub_projection = self.sub_projection(field_index);
896        let column_expr = self.build.map_table_column_to_model(column_id, primitive);
897
898        mapping::Field::Primitive(mapping::FieldPrimitive {
899            column: column_id,
900            lowering: lowering_index,
901            field_mask: stmt::PathFieldSet::from_iter([bit]),
902            sub_projection,
903            column_expr,
904        })
905    }
906
907    /// Creates the discriminant and variant-field columns, then builds the
908    /// enum mapping — all in a single pass.
909    fn map_field_enum(
910        &mut self,
911        field_index: usize,
912        field: &app::Field,
913        embedded_enum: &app::EmbeddedEnum,
914    ) -> Result<mapping::Field> {
915        // Create the discriminant column. It inherits nullability from the enum field.
916        let column_id = self.create_column(field, &embedded_enum.discriminant);
917        let field_expr = self.field_expr(field, field_index);
918
919        // For data-carrying enums the model value is Record([I64(disc), ...]),
920        // so project [0] to extract the discriminant; for unit-only enums the
921        // value IS the I64 discriminant directly.
922        let disc_expr = if embedded_enum.has_data_variants() {
923            stmt::Expr::project(field_expr.clone(), stmt::Projection::single(0))
924        } else {
925            field_expr.clone()
926        };
927
928        let lowering_index =
929            self.build
930                .push_lowering(column_id, &embedded_enum.discriminant.ty, disc_expr);
931
932        let bit = self.build.next_bit();
933        let sub_projection = self.sub_projection(field_index);
934
935        let disc_proj = stmt::Expr::project(field_expr.clone(), stmt::Projection::single(0));
936
937        let variants = embedded_enum
938            .variants
939            .iter()
940            .enumerate()
941            .map(|(variant_index, variant)| {
942                let mut mapper =
943                    self.for_variant(field, field_index, disc_proj.clone(), &variant.discriminant);
944
945                let fields: Vec<mapping::Field> = embedded_enum
946                    .variant_fields(variant_index)
947                    .enumerate()
948                    .map(|(index, field)| {
949                        // Variant fields are stored at positions 1.. in the Record
950                        // (position 0 is the discriminant), so adjust the index.
951                        mapper.map_field(index + 1, field)
952                    })
953                    .collect::<Result<_>>()?;
954
955                Ok(mapping::EnumVariant {
956                    discriminant: variant.discriminant.clone(),
957                    fields,
958                })
959            })
960            .collect::<Result<_>>()?;
961
962        let field_mask = stmt::PathFieldSet::from_iter([bit]);
963
964        let disc_column_expr = self
965            .build
966            .map_table_column_to_model(column_id, &embedded_enum.discriminant);
967
968        Ok(mapping::Field::Enum(mapping::FieldEnum {
969            discriminant: mapping::FieldPrimitive {
970                column: column_id,
971                lowering: lowering_index,
972                field_mask: field_mask.clone(),
973                sub_projection: stmt::Projection::identity(),
974                column_expr: disc_column_expr,
975            },
976            variants,
977            field_mask,
978            sub_projection,
979            default_returning: stmt::Expr::null(),
980        }))
981    }
982
983    fn map_field_struct(
984        &mut self,
985        field_index: usize,
986        field: &app::Field,
987        model_id: ModelId,
988        embedded_struct: &app::EmbeddedStruct,
989    ) -> Result<mapping::Field> {
990        let sub_projection = self.sub_projection(field_index);
991
992        // A nullable embedded struct (`Option<Embed>`) is decoded/encoded
993        // through a *head* column whose null-ness is the option's none-ness
994        // (`NULL` = `None`), keeping `None` represented as `NULL` like
995        // `Option<scalar>` and an embedded enum's (nullable) discriminant. The
996        // flattened leaf columns are forced nullable and presence-guarded, so a
997        // `None` value writes all-`NULL` rather than panicking when projecting
998        // `Null`. There are two ways to obtain the head column:
999        //
1000        //  * Multi-column embed (`Option<Address>` over `Address { street, city
1001        //    }`): a *dedicated* nullable `bool` presence column (`NULL`/`true`)
1002        //    named after the field. `None` → `(NULL, NULL, NULL)`; `Some(..)` →
1003        //    `(true, "..", "..")`. The presence column distinguishes `None`
1004        //    from a `Some` whose own fields are all `NULL`.
1005        //
1006        //  * Newtype embed (`Option<Email(String)>` over `struct Email(String)`):
1007        //    the one flattened leaf is unnamed and collapses to the field name,
1008        //    so a dedicated presence column would *collide* with it. Reuse that
1009        //    leaf as the head — a non-`Option` inner is non-`NULL` exactly when
1010        //    the embed is present, so its null-ness already encodes none-ness,
1011        //    like `Option<scalar>`. (A *named* single field flattens to
1012        //    `{field}_{name}`, doesn't collide, and keeps the disambiguating
1013        //    presence column above.)
1014        //
1015        //    A newtype wrapping an *optional* (`Option<MaybeBody>` over
1016        //    `MaybeBody(Option<String>)`) is rejected: the single column can't
1017        //    tell `None` from `Some(MaybeBody(None))`, and a dedicated presence
1018        //    column would collide with the leaf, so there is no sound mapping.
1019        let reuse_leaf = if field.nullable {
1020            match self.newtype_leaf_nullable(embedded_struct) {
1021                // Newtype over a non-optional scalar → reuse the leaf as head.
1022                Some(false) => true,
1023                // Newtype over an optional → no unambiguous single-column mapping.
1024                Some(true) => {
1025                    let parent = self
1026                        .build
1027                        .app
1028                        .get_model(field.id.model)
1029                        .map(|m| m.name().upper_camel_case())
1030                        .unwrap_or_else(|| "?".to_string());
1031                    let embed = embedded_struct.name.upper_camel_case();
1032                    return Err(Error::invalid_schema(format!(
1033                        "field `{parent}::{}` is `Option<{embed}>`, but `{embed}` is \
1034                         a newtype wrapping an optional value; it can't be flattened \
1035                         to a single column because that column can't distinguish \
1036                         `None` from `Some({embed}(None))`. Use a named struct field \
1037                         instead of a tuple newtype, or remove a layer of `Option`.",
1038                        field.name,
1039                    )));
1040                }
1041                // Not a newtype → dedicated presence column below.
1042                None => false,
1043            }
1044        } else {
1045            false
1046        };
1047
1048        // The dedicated presence column is created *before* the leaves so it
1049        // sorts first in the table. Only the `(column, lowering)` pair is
1050        // needed: `lowering` records the column's encode expression in
1051        // `columns`, and the column id becomes the struct's presence head.
1052        let dedicated_presence = if field.nullable && !reuse_leaf {
1053            let presence_primitive = app::FieldPrimitive {
1054                ty: stmt::Type::Bool,
1055                storage_ty: None,
1056                serialize: None,
1057            };
1058            // Nullable because `field.nullable` is set.
1059            let column = self.create_column(field, &presence_primitive);
1060            // Encode `true` for `Some`, `NULL` for `None` — the same presence
1061            // guard the leaf columns use, applied to the sentinel `true`.
1062            let presence_expr = self.presence_guard(field, field_index, stmt::Expr::from(true));
1063            let lowering = self
1064                .build
1065                .push_lowering(column, &stmt::Type::Bool, presence_expr);
1066            Some((column, lowering))
1067        } else {
1068            None
1069        };
1070
1071        // For a single-field newtype the outer field's `#[auto]` flattens
1072        // down to the one inner column. Multi-field embeds have no clear
1073        // target column, so the inherited flag stops at the boundary —
1074        // `Foo(Bar(u64))` with `#[auto]` on the outer flows through both
1075        // newtypes, but `Foo { a, b: Bar(u64) }` would not propagate it
1076        // from any outer past the `Foo` layer.
1077        let single_field = embedded_struct.fields.len() == 1;
1078        let mut child = if field.nullable {
1079            self.for_nullable_struct(field, field_index)
1080        } else {
1081            self.for_struct(field, field_index)
1082        };
1083        if single_field {
1084            child.inherited_auto_increment |= field.is_auto_increment();
1085        } else {
1086            child.inherited_auto_increment = false;
1087        }
1088        let nested_fields = child.map_fields(&embedded_struct.fields)?;
1089
1090        let mut columns: indexmap::IndexMap<ColumnId, usize> =
1091            nested_fields.iter().flat_map(|f| f.columns()).collect();
1092
1093        // The head column: the dedicated presence column (added to `columns`
1094        // here so its encode lowering is tracked), or the reused single leaf
1095        // for a single-column embed (already in `columns` as a real leaf).
1096        let presence = if let Some((column, lowering)) = dedicated_presence {
1097            columns.insert(column, lowering);
1098            Some(column)
1099        } else if reuse_leaf {
1100            Some(
1101                single_leaf_primitive(&nested_fields[0])
1102                    .expect("single-column embed flattens to one primitive leaf")
1103                    .column,
1104            )
1105        } else {
1106            None
1107        };
1108
1109        let field_mask = nested_fields
1110            .iter()
1111            .fold(stmt::PathFieldSet::new(), |acc, f| acc | f.field_mask());
1112
1113        Ok(mapping::Field::Struct(mapping::FieldStruct {
1114            id: model_id,
1115            fields: nested_fields,
1116            columns,
1117            field_mask,
1118            sub_projection,
1119            default_returning: stmt::Expr::null(),
1120            presence,
1121        }))
1122    }
1123
1124    /// If `embed` is a newtype chain — a single *unnamed* field (recursively,
1125    /// through nested single-field newtype wrappers) bottoming out in a
1126    /// primitive — returns `Some(leaf_nullable)`, the bottom field's own
1127    /// nullability. Returns `None` when `embed` is not a newtype.
1128    ///
1129    /// A newtype flattens to one column that takes the parent field's own name,
1130    /// so when the parent is nullable it reuses that single leaf as the head
1131    /// column (a dedicated presence column would collide). That reuse is sound
1132    /// only when the leaf is non-`NULL` whenever the embed is present —
1133    /// `Some(false)`. A `Some(true)` leaf (the newtype wraps an `Option`) has no
1134    /// unambiguous single-column mapping and is rejected by the caller. A
1135    /// *named* single field (`Foo { value }`) flattens to `{field}_value`,
1136    /// doesn't collide, so it returns `None` here and keeps a dedicated presence
1137    /// column (preserving the `None`-vs-`Some(all-NULL)` disambiguation).
1138    fn newtype_leaf_nullable(&self, embed: &app::EmbeddedStruct) -> Option<bool> {
1139        let [field] = &embed.fields[..] else {
1140            return None;
1141        };
1142        if field.name.storage_name().is_some() {
1143            return None;
1144        }
1145        match &field.ty {
1146            app::FieldTy::Primitive(_) => Some(field.nullable),
1147            app::FieldTy::Embedded(embedded) => {
1148                match lookup_embedded_model(self.build.app, embedded.target, field) {
1149                    Ok(app::Model::EmbeddedStruct(inner)) => self.newtype_leaf_nullable(inner),
1150                    _ => None,
1151                }
1152            }
1153            _ => None,
1154        }
1155    }

1157    /// Builds the final database column name for `field` at the current nesting level.
1158    ///
1159    /// Joins `self.prefix` components with `_`, appends the field name, then
1160    /// prepends `schema_prefix` (if any) with `__`. Because `schema_prefix` is
1161    /// applied here — never stored in `self.prefix` — it is always applied
1162    /// exactly once regardless of nesting depth.
1163    fn column_name(&self, field: &app::Field) -> String {
1164        let embed = match field.name.storage_name() {
1165            Some(field_name) => {
1166                if self.prefix.is_empty() {
1167                    field_name.to_owned()
1168                } else {
1169                    format!("{}_{field_name}", self.prefix.join("_"))
1170                }
1171            }
1172            None => {
1173                // Unnamed field (newtype wrapper) — use just the prefix
1174                // components so the column name is identical to the parent
1175                // field's name.
1176                assert!(
1177                    !self.prefix.is_empty(),
1178                    "unnamed field with empty prefix; a newtype field must be \
1179                     nested inside a parent field"
1180                );
1181                self.prefix.join("_")
1182            }
1183        };
1184        match self.build.schema_prefix.as_deref() {
1185            None => embed,
1186            Some(sp) => format!("{sp}__{embed}"),
1187        }
1188    }

1190    /// Creates a column for `field` using `primitive` for the storage type.
1191    ///
1192    /// Derives the column name from `self.column_name(field)`, nullability from
1193    /// `field.nullable || self.force_nullable`, and auto-increment from
1194    /// `field.is_auto_increment()`.
1195    fn create_column(&mut self, field: &app::Field, primitive: &app::FieldPrimitive) -> ColumnId {
1196        let nullable = field.nullable || self.force_nullable;
1197        let is_auto_increment = (field.is_auto_increment() || self.inherited_auto_increment)
1198            && self.build.db.auto_increment;
1199
1200        let mut storage_ty = db::Type::from_app_column(
1201            &primitive.ty,
1202            primitive.storage_ty.as_ref(),
1203            self.build.db,
1204            is_auto_increment,
1205        )
1206        .expect("unsupported storage type");
1207
1208        // Prefix native enum type names so they don't collide across test runs
1209        // or multi-tenant deployments.
1210        if let db::Type::Enum(ref mut type_enum) = storage_ty
1211            && let (Some(prefix), Some(name)) = (&self.build.name_prefix, &mut type_enum.name)
1212        {
1213            *name = format!("{prefix}{name}");
1214        }
1215
1216        let id = ColumnId {
1217            table: self.build.table.id,
1218            index: self.build.table.columns.len(),
1219        };
1220
1221        self.build.table.columns.push(db::Column {
1222            id,
1223            name: self.column_name(field),
1224            ty: storage_ty.bridge_type(&primitive.ty),
1225            storage_ty,
1226            nullable,
1227            primary_key: false,
1228            auto_increment: is_auto_increment,
1229            versionable: field.is_versionable(),
1230        });
1231
1232        id
1233    }
1234
1235    /// Extends `parent_base` by one projection step to produce the `field_base`
1236    /// for a child `MapField` entering `field` at `field_index` within the parent.
1237    ///
1238    /// If `parent_base` is the top-level sentinel (`Expr::arg(0)`), the child
1239    /// starts a fresh projection rooted at `field.id`. Otherwise the existing
1240    /// `ExprProject` is extended by `field_index`.
1241    fn extend_field_base(&self, field: &app::Field, field_index: usize) -> stmt::Expr {
1242        match &self.field_base {
1243            None => stmt::Expr::ref_self_field(field.id),
1244            Some(stmt::Expr::Project(ep)) => {
1245                let mut proj = ep.projection.clone();
1246                proj.push(field_index);
1247                stmt::Expr::project(*ep.base.clone(), proj)
1248            }
1249            Some(expr) => stmt::Expr::project(expr.clone(), [field_index]),
1250        }
1251    }
1252
1253    /// Returns the sub-projection from the root source field to a field at
1254    /// `field_index` within the current nesting level.
1255    ///
1256    /// If `field_base` is an `ExprProject`, the sub-projection is its
1257    /// projection extended by `field_index`. At the top level (`field_base`
1258    /// is `Expr::arg(0)`) the field is its own root, so identity is returned.
1259    fn sub_projection(&self, field_index: usize) -> stmt::Projection {
1260        match &self.field_base {
1261            None => stmt::Projection::identity(),
1262            Some(stmt::Expr::Project(ep)) => {
1263                let mut proj = ep.projection.clone();
1264                proj.push(field_index);
1265                proj
1266            }
1267            Some(_) => [field_index].into(),
1268        }
1269    }
1270
1271    /// Builds the lowering expression for a field at the current nesting level.
1272    ///
1273    /// At the top level (`field_base` is `Expr::arg(0)`) each field references
1274    /// itself directly. Inside an embedded struct/variant the expression extends
1275    /// `field_base` by `field_index`. The raw expression is then substituted
1276    /// into `field_expr_base` (which may wrap it in a match guard).
1277    fn field_expr(&self, field: &app::Field, field_index: usize) -> stmt::Expr {
1278        let raw = match self.field_base.clone() {
1279            None => stmt::Expr::ref_self_field(field.id),
1280            Some(stmt::Expr::Project(mut expr_project)) => {
1281                expr_project.projection.push(field_index);
1282                expr_project.into()
1283            }
1284            Some(expr) => stmt::Expr::project(expr, [field_index]),
1285        };
1286
1287        let mut result = self.field_expr_base.clone();
1288        result.substitute(&[raw]);
1289        result
1290    }
1291
1292    /// Creates a child `MapField` for recursing into an embedded field.
1293    ///
1294    /// The child inherits the current prefix extended by `name` and inherits
1295    /// `force_nullable`, `field_base`, and `field_expr_base` unchanged. Used
1296    /// when entering struct/variant fields so that sub-field columns are named
1297    /// `{..prefix..}_{name}_{sub_field}`.
1298    fn with_prefix(&mut self, name: Option<&str>) -> MapField<'_, 'b> {
1299        let mut prefix = self.prefix.clone();
1300        if let Some(name) = name {
1301            prefix.push(name.to_owned());
1302        }
1303        MapField {
1304            build: self.build,
1305            prefix,
1306            force_nullable: self.force_nullable,
1307            field_base: self.field_base.clone(),
1308            field_expr_base: self.field_expr_base.clone(),
1309            inherited_auto_increment: self.inherited_auto_increment,
1310        }
1311    }
1312
1313    /// Creates a child `MapField` for an embedded type whose columns are only
1314    /// conditionally populated: an enum variant (only the active variant's
1315    /// columns are written) or a nullable struct embed (every column is `NULL`
1316    /// when the embed is `None`).
1317    ///
1318    /// Sets `field_base` so that `field_expr` on the child projects from
1319    /// `field`, forces the flattened columns nullable, and installs `guard` as
1320    /// the `field_expr_base` so that every `field_expr` call is automatically
1321    /// wrapped in it. `guard` is a `Match` template with `Expr::arg(0)` as the
1322    /// placeholder for the raw field expression — a discriminant check (see
1323    /// [`Self::for_variant`]) or a presence guard (see
1324    /// [`Self::for_nullable_struct`]) — whose `else` branch encodes the
1325    /// inactive/`None` case to `null` instead of projecting into `Null`.
1326    fn for_guarded_embed(
1327        &mut self,
1328        field: &app::Field,
1329        field_index: usize,
1330        guard: stmt::Expr,
1331    ) -> MapField<'_, 'b> {
1332        let field_base = self.extend_field_base(field, field_index);
1333        let mut child = self.with_prefix(field.name.storage_name());
1334        child.force_nullable = true;
1335        child.field_base = Some(field_base);
1336        child.field_expr_base.substitute(&[guard]);
1337        child
1338    }
1339
1340    /// Creates a variant-specific child `MapField`, guarding each `field_expr`
1341    /// in the discriminant check `match_expr(disc_proj, [arm(discriminant,
1342    /// Expr::arg(0))], null())`. See [`Self::for_guarded_embed`].
1343    fn for_variant(
1344        &mut self,
1345        field: &app::Field,
1346        field_index: usize,
1347        disc_proj: stmt::Expr,
1348        discriminant: &stmt::Value,
1349    ) -> MapField<'_, 'b> {
1350        let guard = stmt::Expr::match_expr(
1351            disc_proj,
1352            vec![stmt::MatchArm {
1353                pattern: discriminant.clone(),
1354                expr: stmt::Expr::arg(0),
1355            }],
1356            stmt::Expr::null(),
1357        );
1358        self.for_guarded_embed(field, field_index, guard)
1359    }
1360
1361    /// Creates a child `MapField` for recursing into an embedded struct field.
1362    ///
1363    /// Updates `field_base` to reflect the new nesting level: if entering the
1364    /// first embedded level, sets the source to this field with an identity
1365    /// projection; at deeper levels, extends the existing projection by
1366    /// `field_index`.
1367    fn for_struct(&mut self, field: &app::Field, field_index: usize) -> MapField<'_, 'b> {
1368        let field_base = self.extend_field_base(field, field_index);
1369        // Unnamed (newtype) inner fields keep the parent's prefix so the
1370        // flattened column name comes from the outermost named field.
1371        let mut child = self.with_prefix(field.name.storage_name());
1372        child.field_base = Some(field_base);
1373        child
1374    }
1375
1376    /// Builds the presence guard that encodes a nullable struct embed
1377    /// (`Option<Embed>`):
1378    ///
1379    /// `Match(is_not_null(field), [Bool(true) => present], else => null())`.
1380    ///
1381    /// `present` is what's written when the embed is `Some`: the sentinel
1382    /// `Bool(true)` for the presence column itself, or `Expr::arg(0)` (the raw
1383    /// leaf expression) for the guard wrapped around each flattened leaf column.
1384    /// A `None` value (the field is `Null`, so `is_not_null` is `false`) takes
1385    /// the `else` branch and encodes `null`.
1386    fn presence_guard(
1387        &self,
1388        field: &app::Field,
1389        field_index: usize,
1390        present: stmt::Expr,
1391    ) -> stmt::Expr {
1392        stmt::Expr::match_expr(
1393            stmt::Expr::is_not_null(self.field_expr(field, field_index)),
1394            vec![stmt::MatchArm {
1395                pattern: stmt::Value::Bool(true),
1396                expr: present,
1397            }],
1398            stmt::Expr::null(),
1399        )
1400    }
1401
1402    /// Creates a child `MapField` for recursing into a nullable embedded struct
1403    /// (`Option<Embed>`), guarding each `field_expr` in a [`Self::presence_guard`]
1404    /// over `Expr::arg(0)` so a `None` value encodes each leaf to `null` (the
1405    /// guard's `else` branch). See [`Self::for_guarded_embed`].
1406    fn for_nullable_struct(&mut self, field: &app::Field, field_index: usize) -> MapField<'_, 'b> {
1407        let guard = self.presence_guard(field, field_index, stmt::Expr::arg(0));
1408        self.for_guarded_embed(field, field_index, guard)
1409    }
1410}

1412/// Returns the single primitive leaf of a newtype-chain embed, recursing
1413/// through nested single-field newtype wrappers. Paired with
1414/// [`MapField::newtype_leaf_nullable`]: when that returns `Some(_)`, the mapped
1415/// field has exactly one nested field reachable here as a primitive.
1416fn single_leaf_primitive(field: &mapping::Field) -> Option<&mapping::FieldPrimitive> {
1417    match field {
1418        mapping::Field::Primitive(primitive) => Some(primitive),
1419        mapping::Field::Struct(s) if s.fields.len() == 1 => single_leaf_primitive(&s.fields[0]),
1420        _ => None,
1421    }
1422}

1424/// Truncate `name` to `limit` bytes, appending a 5-character stable hash
1425/// suffix (`_XXXX`) so the result is unique and deterministic across builds.
1426///
1427/// The hash is FNV-1a over the *full* pre-truncation name so two names that
1428/// share a long prefix produce different suffixes. `DefaultHasher` is
1429/// intentionally avoided — it is unstable across Rust versions and would
1430/// churn migration snapshots.
1431fn truncate_identifier(name: String, limit: usize) -> String {
1432    if name.len() <= limit {
1433        return name;
1434    }
1435
1436    // FNV-1a 32-bit hash of the full name for a stable 4-hex-digit suffix.
1437    const FNV_OFFSET: u32 = 2_166_136_261;
1438    const FNV_PRIME: u32 = 16_777_619;
1439    let hash = name
1440        .bytes()
1441        .fold(FNV_OFFSET, |acc, b| acc.wrapping_mul(FNV_PRIME) ^ b as u32);
1442    let suffix = format!("_{:04x}", hash & 0xFFFF);
1443
1444    // Truncate at a char boundary so we never split a multi-byte character.
1445    // The condition uses the char's END position so we never exceed `budget`.
1446    let budget = limit.saturating_sub(suffix.len());
1447    let truncated = &name[..name
1448        .char_indices()
1449        .take_while(|(i, c)| i + c.len_utf8() <= budget)
1450        .last()
1451        .map(|(i, c)| i + c.len_utf8())
1452        .unwrap_or(0)];
1453
1454    format!("{truncated}{suffix}")
1455}

1457/// Look up an embedded model by ID, returning a descriptive error if not found.
1458fn lookup_embedded_model<'a>(
1459    app: &'a app::Schema,
1460    target: ModelId,
1461    field: &app::Field,
1462) -> Result<&'a Model> {
1463    app.get_model(target).ok_or_else(|| {
1464        let parent_name = app
1465            .get_model(field.id.model)
1466            .map(|m| m.name().upper_camel_case())
1467            .unwrap_or_else(|| "?".to_string());
1468
1469        Error::invalid_schema(format!(
1470            "field `{parent_name}::{}` references an embedded type that is not registered \
1471             in the schema; did you forget to include the embedded type in \
1472             `toasty::models!(..)`?",
1473            field.name,
1474        ))
1475    })

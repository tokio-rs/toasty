34/// Computes a model's mapping, creating table columns and mapping expressions
35/// in a single recursive pass over the model's fields.
36///
37/// Holds state that persists across the entire mapping process: the shared
38/// mutable accumulators (columns, lowering expressions, bit counter) plus
39/// references to the table and schema. The recursive field-mapping logic lives
40/// on [`MapField`], which borrows `BuildMapping` and carries per-level context.
41struct BuildMapping<'a> {
42    app: &'a app::Schema,
43    db: &'a driver::Capability,
44    table: &'a mut Table,
45    mapping: &'a mut mapping::Model,
46    /// Model-name prefix used when multiple models share one table, separated
47    /// from the rest of the column name with `__`. None for single-model tables.
48    schema_prefix: Option<String>,
49    /// Optional prefix for database object names (enum types).
50    name_prefix: Option<String>,
51    next_bit: usize,
52    lowering_columns: Vec<ColumnId>,
53    model_to_table: Vec<stmt::Expr>,
54    table_to_model: Vec<stmt::Expr>,
55    /// Column-sharing registry, active only while mapping the variant fields of
56    /// an embedded enum. `None` outside an enum's variant mapping. Saved and
57    /// restored around each enum so `#[shared]` identifiers are scoped to a
58    /// single enum's variants (and never leak across nested enums); the
59    /// physical column `names` a nested enum creates are merged back into the
60    /// outer registry on restore, since they live in the parent's namespace.
61    enum_columns: Option<EnumColumns>,
62}
63
64/// Per-enum column bookkeeping, active while mapping an embedded enum's
65/// variant fields.
66#[derive(Default)]
67struct EnumColumns {
68    /// Maps a shared logical-field identifier (`#[shared(<ident>)]`) to the
69    /// column an earlier variant already created for it, so later variants
70    /// declaring the same identifier share that column instead of producing a
71    /// duplicate.
72    shared: std::collections::HashMap<Name, SharedColumn>,
73
74    /// Every flattened column name created so far for this enum's variant
75    /// fields. Sharing is always explicit: two fields resolving to the same
76    /// column name without a common `#[shared]` identifier is an invalid
77    /// schema, not an implicit merge.
78    names: std::collections::HashSet<String>,
79}

81/// A column created by an earlier enum variant that later variants may reuse.
82#[derive(Clone)]
83struct SharedColumn {
84    column: ColumnId,
85    /// Index into `model_to_table` of the column's encode expression. Later
86    /// variants merge their discriminant-guarded arm into this `Match`.
87    lowering: usize,
88    /// App type of the field that first created the column. Reused fields must
89    /// match it exactly so the column needs no per-variant casting.
90    ty: stmt::Type,
91    /// Discriminant of the variant that created the column. Reuse is only sound
92    /// across *different* variants; a field in the same variant reaching the same
93    /// column is an invalid mapping (both would be active for one discriminant).
94    variant_discriminant: Option<stmt::Value>,
95}

97/// Per-level state for the recursive `map_field*` methods.
98///
99/// Analogous to `LowerStatement` in `lower.rs`: `MapField` holds context that
100/// may change between recursive calls, while [`BuildMapping`] holds the shared
101/// mutable accumulators (columns, lowering expressions, bit counter) that
102/// persist across the entire mapping process.
103struct MapField<'a, 'b> {
104    /// State shared across the entire mapping process.
105    build: &'a mut BuildMapping<'b>,
106
107    /// Accumulated embed-prefix components (without schema_prefix), pushed on
108    /// entry to each nested field and popped on exit.
109    ///
110    /// Final column names join these with `_`, append the field name, then
111    /// prepend the schema prefix (if any) with `__`. Keeping components
112    /// separate ensures schema_prefix is applied exactly once.
113    prefix: Vec<String>,
114
115    /// When true, columns are created nullable regardless of the field's own
116    /// nullability. Set while processing fields whose columns are only populated
117    /// conditionally: enum variant fields (only the active variant's columns are
118    /// populated) and nullable embedded-struct fields (all columns are NULL when
119    /// the embed is `None`).
120    force_nullable: bool,
121
122    /// Base expression for the current nesting level.
123    ///
124    /// `None` at the top level. `Expr::Project(source_ref, proj)` at any nested
125    /// level, where `source_ref` is the top-level source field reference and
126    /// `proj` is the projection from that source down to the current container
127    /// (not including the final `field_index` step). `field_expr` and
128    /// `sub_projection` extend it by `field_index` to reach a specific field.
129    field_base: Option<stmt::Expr>,
130
131    /// A template expression with `Expr::arg(0)` as a placeholder for the raw
132    /// field expression. `field_expr` substitutes the raw field expression into
133    /// this template before returning. The identity value is `Expr::arg(0)`
134    /// itself, which substitutes to the raw expression unchanged.
135    ///
136    /// Used by variant-specific `MapField` instances to automatically wrap
137    /// field expressions in the discriminant match guard.
138    field_expr_base: stmt::Expr,
139
140    /// True when an outer single-field embed flagged its column as
141    /// auto-increment. Single-field newtype embeds flatten to one column, so
142    /// the outer field's `#[auto]` (Increment) must apply to that flattened
143    /// column. ORed into the column's flag at creation time so an outer or
144    /// inner declaration both take effect.
145    inherited_auto_increment: bool,
146
147    /// Discriminant of the enum variant currently being mapped, set by
148    /// [`Self::for_variant`] and inherited by nested embeds within the variant.
149    /// `None` outside an enum variant. Used to distinguish a sound cross-variant
150    /// column share from an invalid same-variant collision.
151    variant_discriminant: Option<stmt::Value>,
152}

155    pub(super) fn build_table_stub_for_model(&mut self, model: &ModelRoot) -> TableId {
156        let table_name = self.prefix_table_name(&model.table_name);
157
158        if !self.table_lookup.contains_key(&table_name) {
159            let id = self.register_table(&table_name);
160            self.tables.push(Table::new(id, table_name.clone()));
161        }
162
163        *self.table_lookup.get(&table_name).unwrap()
164    }
165
166    pub(super) fn build_tables_from_models(
167        &mut self,
168        app: &app::Schema,
169        db: &driver::Capability,
170    ) -> Result<()> {
171        for table in &mut self.tables {
172            let models = app
173                .models()
174                .filter(|model| model.is_root())
175                .filter(|model| self.mapping.model(model.id()).table == table.id)
176                .collect::<Vec<_>>();
177
178            assert!(
179                models.len() == 1,
180                "TODO: handle mapping many models to one table"
181            );
182
183            BuildTableFromModels {
184                app,
185                db,
186                table,
187                mapping: &mut self.mapping,
188                prefix_table_names: models.len() > 1,
189                name_prefix: self.builder.table_name_prefix.clone(),
190            }
191            .build(models[0])?;
192        }
193
194        Ok(())
195    }
196
197    pub(super) fn register_table(&mut self, name: impl AsRef<str>) -> TableId {
198        assert!(!self.table_lookup.contains_key(name.as_ref()));
199        let id = TableId(self.table_lookup.len());
200        self.table_lookup.insert(name.as_ref().to_string(), id);
201        id
202    }

204    fn prefix_table_name(&self, name: &str) -> String {
205        if let Some(prefix) = &self.builder.table_name_prefix {
206            format!("{prefix}{name}")
207        } else {
208            name.to_string()
209        }
210    }

213impl BuildTableFromModels<'_> {
214    fn build(&mut self, model: &Model) -> Result<()> {
215        self.map_model_fields(model)?;
216        self.update_index_names();
217        Ok(())
218    }

220    fn map_model_fields(&mut self, model: &Model) -> Result<()> {
221        let root = model.as_root_unwrap();
222        let schema_prefix = if self.prefix_table_names {
223            Some(model.name().snake_case())
224        } else {
225            None
226        };
227
228        BuildMapping {
229            app: self.app,
230            db: self.db,
231            table: self.table,
232            mapping: self.mapping.model_mut(model),
233            schema_prefix,
234            name_prefix: self.name_prefix.clone(),
235            next_bit: 0,
236            lowering_columns: vec![],
237            model_to_table: vec![],
238            table_to_model: vec![],
239            enum_columns: None,
240        }
241        .build_mapping(root)?;
242
243        let model_fields = &self.mapping.model(model.id()).fields;
244        let mut indices = Vec::new();
245        self.collect_indices(&root.fields, model_fields, &root.indices, &mut indices)?;
246
247        for index in indices {
248            if index.primary_key {
249                self.table.primary_key.columns = index.columns.iter().map(|c| c.column).collect();
250            }
251
252            self.table.indices.push(index);
253        }
254
255        Ok(())
256    }
257
258    /// Collects DB-level indices from app-level index definitions, then recurses
259    /// into embedded struct fields to collect their indices as well.
260    fn collect_indices(
261        &self,
262        fields: &[app::Field],
263        field_mappings: &[mapping::Field],
264        indices: &[app::Index],
265        out: &mut Vec<db::Index>,
266    ) -> Result<()> {
267        for app_index in indices {
268            let mut index = db::Index {
269                id: IndexId {
270                    table: self.table.id,
271                    index: out.len(),
272                },
273                name: app_index.name.clone().unwrap_or_default(),
274                on: self.table.id,
275                columns: vec![],
276                unique: app_index.unique,
277                primary_key: app_index.primary_key,
278            };
279
280            for index_field in &app_index.fields {
281                let mapping = &field_mappings[index_field.field.index];
282
283                // Resolve the mapped column for this indexed field. Primitive
284                // fields map directly. Newtype embedded structs (a single
285                // unnamed field) are transparent wrappers around a primitive, so
286                // we unwrap one level.
287                //
288                // Multi-field or named-field embedded structs are not yet
289                // supported in indices because the column ordering within the
290                // index matters and there is no syntax to specify it. That will
291                // likely require an explicit field-order annotation on the
292                // index.
293                let column = self.resolve_indexed_column(mapping);
294
295                index.columns.push(db::IndexColumn {
296                    column,
297                    op: index_field.op,
298                    scope: index_field.scope,
299                });
300            }
301
302            out.push(index);
303        }
304
305        for (field_index, field) in fields.iter().enumerate() {
306            let app::FieldTy::Embedded(embedded) = &field.ty else {
307                continue;
308            };
309
310            let target = lookup_embedded_model(self.app, embedded.target, field)?;
311
312            match target {
313                app::Model::EmbeddedStruct(embedded_struct) => {
314                    let field_mapping = field_mappings[field_index]
315                        .as_struct()
316                        .expect("embedded struct field should have struct mapping");
317
318                    self.collect_indices(
319                        &embedded_struct.fields,
320                        &field_mapping.fields,
321                        &embedded_struct.indices,
322                        out,
323                    )?;
324                }
325                app::Model::EmbeddedEnum(embedded_enum) => {
326                    if embedded_enum.indices.is_empty() {
327                        continue;
328                    }
329
330                    let field_mapping = field_mappings[field_index]
331                        .as_enum()
332                        .expect("embedded enum field should have enum mapping");
333
334                    // Build a flat mapping from global field index to the
335                    // field's mapping within its variant. Enum fields use
336                    // global indices; each field belongs to a specific variant
337                    // and has a local offset within that variant.
338                    let mut flat_mappings: Vec<mapping::Field> =
339                        vec![
340                            mapping::Field::Relation(mapping::FieldRelation {
341                                field_mask: crate::stmt::PathFieldSet::new(),
342                            });
343                            embedded_enum.fields.len()
344                        ];
345
346                    for (variant_idx, variant_mapping) in field_mapping.variants.iter().enumerate()
347                    {
348                        let mut local_idx = 0;
349                        for (global_idx, f) in embedded_enum.fields.iter().enumerate() {
350                            let app::VariantId { index: vi, .. } =
351                                f.variant.expect("enum field must have variant");
352                            if vi != variant_idx {
353                                continue;
354                            }
355                            flat_mappings[global_idx] = variant_mapping.fields[local_idx].clone();
356                            local_idx += 1;
357                        }
358                    }
359
360                    self.collect_indices(
361                        &embedded_enum.fields,
362                        &flat_mappings,
363                        &embedded_enum.indices,
364                        out,
365                    )?;
366                }
367                _ => continue,
368            }
369        }
370
371        Ok(())
372    }

374    /// Walks newtype embed layers down to the underlying primitive column.
375    ///
376    /// Indexed fields may be either a primitive directly or a chain of
377    /// single-unnamed-field embeds wrapping a primitive. Multi-field or
378    /// named-field embeds are not supported as index targets — the index
379    /// would have to choose a column ordering and there is no syntax for
380    /// that yet.
381    fn resolve_indexed_column(&self, mut mapping: &mapping::Field) -> ColumnId {
382        loop {
383            match mapping {
384                mapping::Field::Primitive(p) => return p.column,
385                mapping::Field::Enum(p) => {
386                    // Only unit (data-less) enums can be indexed: the
387                    // discriminant column alone represents the value.
388                    // Data-carrying enums span multiple columns and have no
389                    // single index column. The derive macro rejects this at
390                    // compile time via `IndexableField`; this assert is the
391                    // backstop for schemas built without the macro.
392                    assert!(
393                        p.variants.iter().all(|v| v.fields.is_empty()),
394                        "only unit (data-less) embedded enums can be indexed; \
395                         data-carrying enum variants span multiple columns"
396                    );
397                    return p.discriminant.column;
398                }
399                mapping::Field::Struct(s) => {
400                    let embedded_struct = self.app.model(s.id).as_embedded_struct_unwrap();
401                    assert!(
402                        embedded_struct.fields.len() == 1
403                            && embedded_struct.fields[0].name.app.is_none(),
404                        "only newtype embedded structs (single unnamed \
405                         field) can be indexed; multi-field or named-field \
406                         embedded structs require explicit index field \
407                         ordering"
408                    );
409                    mapping = &s.fields[0];
410                }
411                _ => panic!(
412                    "only primitive and newtype embedded structs can be \
413                     indexed"
414                ),
415            }
416        }
417    }

419    fn update_index_names(&mut self) {
420        for index in &mut self.table.indices {
421            // Preserve user-provided names from `#[index(name = "...", ...)]`.
422            if !index.name.is_empty() {
423                continue;
424            }
425
426            let mut name = format!("index_{}_by", self.table.name);
427
428            for (i, index_column) in index.columns.iter().enumerate() {
429                let column = &self.table.columns[index_column.column.index];
430
431                if i > 0 {
432                    name.push_str("_and");
433                }
434
435                name.push('_');
436                name.push_str(&column.name);
437            }
438
439            index.name = if let Some(limit) = self.db.max_identifier_length {
440                truncate_identifier(name, limit)
441            } else {
442                name
443            };
444        }
445    }

448impl BuildMapping<'_> {
449    fn build_mapping(mut self, model: &ModelRoot) -> Result<()> {
450        let mut fields = MapField::new(&mut self).map_fields(&model.fields)?;
451
452        assert!(!self.model_to_table.is_empty());
453        assert_eq!(self.model_to_table.len(), self.lowering_columns.len());
454
455        self.build_table_to_model(model, &fields)?;
456
457        // Compute the default `RETURNING` expression for the model and for
458        // each nested embedded type. Mutates `fields` to populate the
459        // per-embed `default_returning` along the way.
460        let default_returning = self.build_default_returning_root(model, &mut fields)?;
461
462        self.mapping.fields = fields;
463        self.mapping.columns = self.lowering_columns;
464        self.mapping.model_to_table = stmt::ExprRecord::from_vec(self.model_to_table);
465        self.mapping.table_to_model =
466            TableToModel::new(stmt::ExprRecord::from_vec(self.table_to_model));
467        self.mapping.default_returning = default_returning;
468
469        Ok(())
470    }
471
472    /// Builds the model's default `RETURNING` expression — the same shape as
473    /// `table_to_model` but with every deferred field, at this level or
474    /// inside a nested embedded type, pre-masked to `Null`. Also writes each
475    /// embed's own default expression into the corresponding mapping node so
476    /// lowering can splice it in when an `.include()` activates a deferred
477    /// embed.
478    fn build_default_returning_root(
479        &self,
480        model: &ModelRoot,
481        fields: &mut [mapping::Field],
482    ) -> Result<stmt::Expr> {
483        let exprs: Vec<stmt::Expr> = model
484            .fields
485            .iter()
486            .zip(fields.iter_mut())
487            .map(|(field, mapping)| self.build_default_returning_field(field, mapping))
488            .collect::<Result<_>>()?;
489        Ok(stmt::Expr::record(exprs))
490    }
491
492    /// Builds the default returning expression for a single field and, if
493    /// the field is an embedded type, populates the embed's own
494    /// `default_returning` cache.
495    fn build_default_returning_field(
496        &self,
497        field: &app::Field,
498        mapping: &mut mapping::Field,
499    ) -> Result<stmt::Expr> {
500        // Deferred fields are `Null` in the default expression. Still
501        // recurse through deferred embeds so the nested `default_returning`
502        // is populated — `process_includes` reads it during a `.include()`
503        // splice.
504        if field.deferred {
505            if matches!(&field.ty, app::FieldTy::Embedded(_)) {
506                self.populate_embed_default_returning(field, mapping)?;
507            }
508            return Ok(stmt::Expr::null());
509        }
510
511        match &field.ty {
512            app::FieldTy::Primitive(primitive) => {
513                let column_id = mapping.as_primitive().unwrap().column;
514                Ok(self.map_table_column_to_model(column_id, primitive))
515            }
516            app::FieldTy::Embedded(_) => self.populate_embed_default_returning(field, mapping),
517            app::FieldTy::BelongsTo(_) | app::FieldTy::Has(_) | app::FieldTy::Via(_) => {
518                Ok(stmt::Value::Null.into())
519            }
520        }
521    }
522
523    /// Resolves the embedded target, recurses to compute its default
524    /// expression, stores it on the mapping node, and returns a clone for
525    /// the caller to splice in for the parent field.
526    fn populate_embed_default_returning(
527        &self,
528        field: &app::Field,
529        mapping: &mut mapping::Field,
530    ) -> Result<stmt::Expr> {
531        let app::FieldTy::Embedded(embedded) = &field.ty else {
532            unreachable!("populate_embed_default_returning called on non-embed");
533        };
534        let target = lookup_embedded_model(self.app, embedded.target, field)?;
535
536        match (target, mapping) {
537            (app::Model::EmbeddedStruct(embed_model), mapping::Field::Struct(s)) => {
538                let record = {
539                    let exprs: Vec<stmt::Expr> = embed_model
540                        .fields
541                        .iter()
542                        .zip(s.fields.iter_mut())
543                        .map(|(f, m)| self.build_default_returning_field(f, m))
544                        .collect::<Result<_>>()?;
545                    stmt::Expr::record(exprs)
546                };
547                // A nullable embedded struct guards its record on the head
548                // column, matching `build_table_to_model_field_struct`.
549                let expr = match s.presence {
550                    None => record,
551                    Some(presence) => self.wrap_presence_match(presence.index, record),
552                };
553                s.default_returning = expr.clone();
554                Ok(expr)
555            }
556            (app::Model::EmbeddedEnum(embed_model), mapping::Field::Enum(e)) => {
557                let expr = self.build_default_returning_enum(embed_model, e, field.nullable)?;
558                e.default_returning = expr.clone();
559                Ok(expr)
560            }
561            _ => unreachable!("invalid schema: embedded field maps to root model"),
562        }
563    }
564
565    /// Builds the enum's default `Match` expression, recursing into each
566    /// variant's fields so a deferred sub-field nested inside a variant's
567    /// embed-struct is pre-masked. The shape mirrors
568    /// [`Self::build_table_to_model_field_enum`] but routes each variant
569    /// field through [`Self::build_default_returning_field`] instead of
570    /// the raw column emitter.
571    ///
572    /// Deferred variant fields route through the same default-returning
573    /// masking as fields inside embedded structs.
574    fn build_default_returning_enum(
575        &self,
576        model: &app::EmbeddedEnum,
577        mapping: &mut mapping::FieldEnum,
578        nullable: bool,
579    ) -> Result<stmt::Expr> {
580        let disc_col_ref =
581            self.map_table_column_to_model(mapping.discriminant.column, &model.discriminant);
582
583        if !model.has_data_variants() {
584            return Ok(disc_col_ref);
585        }
586
587        let mut arms = Vec::new();
588        for (variant_index, (variant, variant_mapping)) in model
589            .variants
590            .iter()
591            .zip(mapping.variants.iter_mut())
592            .enumerate()
593        {
594            let variant_fields: Vec<&app::Field> = model.variant_fields(variant_index).collect();
595            let arm_expr = if variant_fields.is_empty() {
596                disc_col_ref.clone()
597            } else {
598                let mut record_elems = vec![disc_col_ref.clone()];
599                for (local_idx, field) in variant_fields.iter().enumerate() {
600                    let mapping_field = &mut variant_mapping.fields[local_idx];
601                    record_elems.push(self.build_default_returning_field(field, mapping_field)?);
602                }
603                stmt::Expr::record(record_elems)
604            };
605            arms.push(stmt::MatchArm {
606                pattern: variant.discriminant.clone(),
607                expr: arm_expr,
608            });
609        }
610
611        let else_expr = self.enum_decode_else(model, nullable, &disc_col_ref);
612        Ok(stmt::Expr::match_expr(disc_col_ref, arms, else_expr))
613    }

621    fn build_table_to_model(
622        &mut self,
623        model: &ModelRoot,
624        mapping: &[mapping::Field],

626        for (index, field) in model.fields.iter().enumerate() {

633    /// Builds the `table_to_model` expression for an embedded enum field.
634    ///
635    /// For unit-only enums the discriminant column reference suffices.
636    /// For mixed/data-carrying enums a `Match` expression dispatches on the
637    /// discriminant: unit arms return the discriminant directly, data arms
638    /// return `Record([disc, field1, ...])` matching the shape expected by
639    /// `Primitive::load`.
640    ///
641    /// When `nullable` (an `Option<EmbeddedEnum>`), the discriminant column is
642    /// `NULL` for `None`: a unit-only enum's column reference already reads back
643    /// as `None`, and a data enum's `Match` else branch yields `null` so a
644    /// `NULL` (unmatched) discriminant decodes to `None`.
645    fn build_table_to_model_field_enum(
646        &self,
647        model: &app::EmbeddedEnum,
648        mapping: &mapping::FieldEnum,
649        nullable: bool,
650    ) -> Result<stmt::Expr> {
651        let disc_col_ref =
652            self.map_table_column_to_model(mapping.discriminant.column, &model.discriminant);
653
654        if !model.has_data_variants() {
655            return Ok(disc_col_ref);
656        }
657
658        let mut arms = Vec::new();
659
660        for (variant_index, (variant, mapping)) in
661            model.variants.iter().zip(&mapping.variants).enumerate()
662        {
663            let variant_fields: Vec<_> = model.variant_fields(variant_index).collect();
664            let arm_expr = if variant_fields.is_empty() {
665                disc_col_ref.clone()
666            } else {
667                let mut record_elems = vec![disc_col_ref.clone()];
668
669                for (local_idx, field) in variant_fields.iter().enumerate() {
670                    let expr =
671                        self.build_table_to_model_field(field, &mapping.fields[local_idx])?;
672                    record_elems.push(expr);
673                }
674                stmt::Expr::record(record_elems)
675            };
676            arms.push(stmt::MatchArm {
677                pattern: variant.discriminant.clone(),
678                expr: arm_expr,
679            });
680        }
681
682        let else_expr = self.enum_decode_else(model, nullable, &disc_col_ref);
683        Ok(stmt::Expr::match_expr(disc_col_ref, arms, else_expr))
684    }
685
686    /// The `else` branch shared by the enum decode and default-returning
687    /// `Match` expressions.
688    ///
689    /// When `nullable`, a `NULL` (unmatched) discriminant is the `None` case, so
690    /// the else is `null`. Otherwise it mirrors the data-arm `Record` shape but
691    /// fills field positions with `Expr::Error`, making projections distribute
692    /// uniformly: projecting `[0]` extracts the discriminant (pruning the
693    /// errors) while a deeper projection yields `Expr::Error` (unreachable at
694    /// runtime).
695    fn enum_decode_else(
696        &self,
697        model: &app::EmbeddedEnum,
698        nullable: bool,
699        disc_col_ref: &stmt::Expr,
700    ) -> stmt::Expr {
701        if nullable {
702            return stmt::Expr::null();
703        }
704        let max_fields = (0..model.variants.len())
705            .map(|i| model.variant_fields(i).count())
706            .max()
707            .unwrap_or(0);
708        if max_fields == 0 {
709            stmt::Expr::error("unexpected enum discriminant")
710        } else {
711            let mut elems = vec![disc_col_ref.clone()];
712            for _ in 0..max_fields {
713                elems.push(stmt::Expr::error("unexpected enum discriminant"));
714            }
715            stmt::Expr::record(elems)
716        }
717    }

719    /// Encodes `expr` for `column_id`, appends the result to `model_to_table`,
720    /// records the column in `lowering_columns`, and returns the lowering index.
721    fn push_lowering(
722        &mut self,
723        column_id: ColumnId,
724        ty: &stmt::Type,
725        expr: impl Into<stmt::Expr>,
726    ) -> usize {
727        let lowering_expr = self.encode_column(column_id, ty, expr);
728        let lowering_index = self.model_to_table.len();
729        self.lowering_columns.push(column_id);
730        self.model_to_table.push(lowering_expr);
731        lowering_index
732    }

734    /// Merges a later variant field's encode into the shared column's existing
735    /// encode at `lowering`, so one column encodes every contributing variant.
736    ///
737    /// Both encodes are produced by [`MapField::for_variant`], so each is a
738    /// discriminant-guarded `Match` (optionally wrapped in a `Cast`) whose arm
739    /// fires only for that variant. Appending `new_encode`'s arm to the existing
740    /// `Match` yields `match disc { d1 => v1, d2 => v2, .. } else null` — exactly
741    /// the per-variant dispatch a shared column needs.
742    fn merge_shared_encode(&mut self, lowering: usize, new_encode: stmt::Expr) -> Result<()> {
743        let new_arms = into_match_arms(new_encode).ok_or_else(|| {
744            Error::invalid_schema(
745                "shared enum column encode is not a discriminant match expression",
746            )
747        })?;
748        let existing = match_arms_mut(&mut self.model_to_table[lowering]).ok_or_else(|| {
749            Error::invalid_schema(
750                "shared enum column encode is not a discriminant match expression",
751            )
752        })?;
753        existing.extend(new_arms);
754        Ok(())
755    }

757    fn encode_column(
758        &self,
759        column_id: ColumnId,
760        ty: &stmt::Type,
761        expr: impl Into<stmt::Expr>,
762    ) -> stmt::Expr {
763        let expr = expr.into();
764        let column = self.table.column(column_id);
765
766        assert_ne!(stmt::Type::Null, *ty);
767
768        match &column.ty {
769            column_ty if column_ty == ty => expr,
770            // A `#[document]` column lowers through a schema-directed cast.
771            // The structural target (`Object`) does not name the embedded
772            // model and a positional record is not self-describing, so the
773            // cast carries the model-level source type (see
774            // `stmt::ExprCast::from`).
775            _ if column.is_document() => stmt::Expr::cast_from(expr, ty, &column.ty),
776            // If the types do not match, attempt casting as a fallback.
777            _ => stmt::Expr::cast(expr, &column.ty),
778        }
779    }

781    /// Maps table columns to model field expressions during query lowering.
782    ///
783    /// Called during query planning to replace model field references with the
784    /// appropriate table column expressions. Handles type conversions between
785    /// table storage and model types.
786    fn map_table_column_to_model(
787        &self,
788        column_id: ColumnId,
789        primitive: &app::FieldPrimitive,
790    ) -> stmt::Expr {
791        let column = self.table.column(column_id);
792
793        // NOTE: nesting and table are stubs here (though often the actual values).
794        // The engine must substitute these with the actual TableRef index in the query's TableSource.
795        let expr_column = stmt::Expr::column(stmt::ExprColumn {
796            nesting: 0,
797            table: 0,
798            column: column_id.index,
799        });
800
801        match &column.ty {
802            c_ty if *c_ty == primitive.ty => expr_column,
803            // If the types do not match, attempt casting as a fallback. A
804            // `#[document]` column takes this path too: the raising cast
805            // (`Object` → `Model`) turns the driver's named wire object into
806            // the embed's positional record when the engine evaluates it.
807            _ => stmt::Expr::cast(expr_column, &primitive.ty),
808        }
809    }

811    fn build_table_to_model_field_struct(
812        &self,
813        model: &app::EmbeddedStruct,
814        mapping: &mapping::FieldStruct,
815    ) -> Result<stmt::Expr> {
816        let exprs: Vec<stmt::Expr> = model
817            .fields
818            .iter()
819            .enumerate()
820            .map(|(index, field)| self.build_table_to_model_field(field, &mapping.fields[index]))
821            .collect::<Result<_>>()?;
822        let record = stmt::Expr::record(exprs);
823
824        // A nullable embedded struct decodes through a `Match` on its head
825        // column (see `wrap_presence_match`).
826        match mapping.presence {
827            None => Ok(record),
828            Some(presence) => Ok(self.wrap_presence_match(presence.index, record)),
829        }
830    }
831
832    /// Wraps `record` in the presence `Match` that decodes a nullable embedded
833    /// struct: when the head column is non-`NULL` the record is reconstructed,
834    /// otherwise the value is `null`, which `impl Load for Option<T>` reads
835    /// back as `None`. The engine evaluates this `Match` against the returned
836    /// row before `Load` runs.
837    ///
838    /// The head column is either a dedicated `bool` presence column (only ever
839    /// `true`/`NULL`) or, for a single-column embed, the embed's own flattened
840    /// leaf column — in both cases its null-ness is the option's none-ness, so
841    /// guarding on `is_null(head) == false` (present) is the uniform check. We
842    /// match `is_null(head)` against `false` rather than wrapping in `not(..)`
843    /// because the table→model evaluator handles `IsNull`/`Match`/`Project`
844    /// but not `Not`.
845    fn wrap_presence_match(&self, presence_column: usize, record: stmt::Expr) -> stmt::Expr {
846        stmt::Expr::match_expr(
847            stmt::Expr::is_null(stmt::Expr::column(stmt::ExprColumn {
848                nesting: 0,
849                table: 0,
850                column: presence_column,
851            })),
852            vec![stmt::MatchArm {
853                pattern: stmt::Value::Bool(false),
854                expr: record,
855            }],
856            stmt::Expr::null(),
857        )
858    }
859
860    fn build_table_to_model_field(
861        &self,
862        field: &app::Field,
863        mapping: &mapping::Field,
864    ) -> Result<stmt::Expr> {
865        match &field.ty {
866            app::FieldTy::Primitive(primitive) => {
867                let column_id = mapping.as_primitive().unwrap().column;
868                Ok(self.map_table_column_to_model(column_id, primitive))
869            }
870            app::FieldTy::Embedded(embedded) => {
871                let target = lookup_embedded_model(self.app, embedded.target, field)?;
872
873                match target {
874                    app::Model::EmbeddedEnum(embedded) => {
875                        let mapping = mapping
876                            .as_enum()
877                            .expect("embedded enum field should have enum mapping");
878                        self.build_table_to_model_field_enum(embedded, mapping, field.nullable)
879                    }
880                    app::Model::EmbeddedStruct(embedded) => {
881                        let mapping = mapping
882                            .as_struct()
883                            .expect("embedded struct field should have struct mapping");
884                        self.build_table_to_model_field_struct(embedded, mapping)
885                    }
886                    _ => unreachable!("invalid schema"),
887                }
888            }
889            app::FieldTy::BelongsTo(_) | app::FieldTy::Has(_) | app::FieldTy::Via(_) => {
890                Ok(stmt::Value::Null.into())
891            }
892        }
893    }

917    fn map_field(&mut self, index: usize, field: &app::Field) -> Result<mapping::Field> {
918        // A shared logical field maps to exactly one column, so only
919        // single-column (primitive) fields can participate in sharing.
920        if field.shared.is_some() && !matches!(&field.ty, app::FieldTy::Primitive(_)) {
921            return Err(Error::invalid_schema(format!(
922                "field `{}` declares a shared field but is not a primitive; \
923                 only single-column fields can share a column",
924                field.name,
925            )));
926        }
927
928        match &field.ty {
929            app::FieldTy::Primitive(primitive) => self.map_field_primitive(index, field, primitive),
930            app::FieldTy::Embedded(embedded) => {
931                let target = lookup_embedded_model(self.build.app, embedded.target, field)?;
932
933                match target {
934                    app::Model::EmbeddedEnum(embedded_enum) => self.map_field_enum(
935                        index,
936                        field,
937                        embedded_enum,
938                        embedded.storage_ty.as_ref(),
939                    ),
940                    app::Model::EmbeddedStruct(embedded_struct) => {
941                        if embedded.storage_ty.is_some() {
942                            return Err(Error::invalid_schema(format!(
943                                "field `{}` sets a database type on an embedded struct; \
944                                 type overrides are supported only on embedded enum fields",
945                                field.name,
946                            )));
947                        }
948                        self.map_field_struct(index, field, embedded.target, embedded_struct)
949                    }
950                    _ => unreachable!(),
951                }
952            }
953            app::FieldTy::BelongsTo(_) | app::FieldTy::Has(_) | app::FieldTy::Via(_) => {
954                assert!(!self.force_nullable);
955                let bit = self.build.next_bit();
956                Ok(mapping::Field::Relation(mapping::FieldRelation {
957                    field_mask: stmt::PathFieldSet::from_iter([bit]),
958                }))
959            }
960        }
961    }

963    /// Creates the column and builds the mapping for a primitive field in one step.
964    ///
965    /// When column sharing is active (i.e. mapping enum variant fields) and an
966    /// earlier variant already created the column for this field's `#[shared]`
967    /// identifier, this reuses that column — merging the field's
968    /// discriminant-guarded encode into the shared column's `Match` — instead
969    /// of creating a duplicate. Sharing is always explicit: without a common
970    /// `#[shared]` identifier, two variant fields resolving to the same column
971    /// name are a duplicate-column error, never an implicit merge.
972    fn map_field_primitive(
973        &mut self,
974        field_index: usize,
975        field: &app::Field,
976        primitive: &app::FieldPrimitive,
977    ) -> Result<mapping::Field> {
978        let expr = self.field_expr(field, field_index);
979        let column_name = self.column_name(field);
980
981        let shared = field.shared.as_ref().and_then(|ident| {
982            self.build
983                .enum_columns
984                .as_ref()
985                .and_then(|columns| columns.shared.get(ident).cloned())
986        });
987
988        let (column_id, lowering_index) = if let Some(shared) = shared {
989            let ident = field.shared.as_ref().unwrap().snake_case();
990
991            // Reuse is only sound across *different* variants — the merged encode
992            // dispatches on the discriminant, and at most one variant is active
993            // per row. Two fields in the *same* variant reaching one column would
994            // append a second arm under the same discriminant, so the encode
995            // would silently keep only the first and both fields would decode from
996            // the one column. Reject that invalid mapping. The `Embed` derive
997            // catches it at compile time; this is the backstop for schemas built
998            // without the macro.
999            if shared.variant_discriminant.is_some()
1000                && shared.variant_discriminant == self.variant_discriminant
1001            {
1002                return Err(Error::invalid_schema(format!(
1003                    "two fields in the same enum variant declare `#[shared({ident})]`; \
1004                     a column can be shared only across different variants",
1005                )));
1006            }
1007
1008            // A shared column needs no per-variant casting, so the contributing
1009            // fields must have identical types. The `Embed` derive rejects a
1010            // mismatch at compile time via `SameColumnType`; this is the
1011            // backstop for schemas built without the macro. Width or kind
1012            // mismatches (e.g. `i32` vs `i64`, or `String` vs `i64`) are
1013            // rejected here rather than silently coerced.
1014            if shared.ty != primitive.ty {
1015                return Err(Error::invalid_schema(format!(
1016                    "enum variant fields sharing `{ident}` have incompatible types \
1017                     ({:?} and {:?}); fields sharing a column must have the same type",
1018                    shared.ty, primitive.ty,
1019                )));
1020            }
1021
1022            // Every sharing field must resolve to the same column name; a
1023            // `#[column("...")]` override on one variant that disagrees with
1024            // another's is an error, not two columns. The `Embed` derive
1025            // rejects this at compile time; this is the backstop.
1026            let existing = &self.build.table.column(shared.column).name;
1027            if *existing != column_name {
1028                return Err(Error::invalid_schema(format!(
1029                    "enum variant fields sharing `{ident}` resolve to different \
1030                     column names (`{existing}` and `{column_name}`); fields sharing \
1031                     a column must agree on the column name",
1032                )));
1033            }
1034
1035            self.build.merge_shared_encode(shared.lowering, expr)?;
1036            (shared.column, shared.lowering)
1037        } else {
1038            if let Some(columns) = &self.build.enum_columns
1039                && columns.names.contains(&column_name)
1040            {
1041                return Err(Error::invalid_schema(format!(
1042                    "two enum variant fields map to the column `{column_name}` \
1043                     without declaring a shared field; annotate both with \
1044                     `#[shared(<ident>)]` to share the column",
1045                )));
1046            }
1047
1048            let column_id = self.create_column(field, primitive);
1049            let lowering_index = self.build.push_lowering(column_id, &primitive.ty, expr);
1050
1051            if let Some(columns) = self.build.enum_columns.as_mut() {
1052                columns.names.insert(column_name);
1053                if let Some(ident) = &field.shared {
1054                    columns.shared.insert(
1055                        ident.clone(),
1056                        SharedColumn {
1057                            column: column_id,
1058                            lowering: lowering_index,
1059                            ty: primitive.ty.clone(),
1060                            variant_discriminant: self.variant_discriminant.clone(),
1061                        },
1062                    );
1063                }
1064            }
1065
1066            (column_id, lowering_index)
1067        };
1068
1069        let bit = self.build.next_bit();
1070        let sub_projection = self.sub_projection(field_index);
1071        let column_expr = self.build.map_table_column_to_model(column_id, primitive);
1072
1073        Ok(mapping::Field::Primitive(mapping::FieldPrimitive {
1074            column: column_id,
1075            lowering: lowering_index,
1076            field_mask: stmt::PathFieldSet::from_iter([bit]),
1077            sub_projection,
1078            column_expr,
1079        }))
1080    }

1082    /// Creates the discriminant and variant-field columns, then builds the
1083    /// enum mapping — all in a single pass.
1084    fn map_field_enum(
1085        &mut self,
1086        field_index: usize,
1087        field: &app::Field,
1088        embedded_enum: &app::EmbeddedEnum,
1089        storage_ty: Option<&db::Type>,
1090    ) -> Result<mapping::Field> {
1091        let mut discriminant = embedded_enum.discriminant.clone();
1092        if let Some(storage_ty) = storage_ty {
1093            let Some(max) = integer_storage_max(storage_ty) else {
1094                return Err(Error::invalid_schema(format!(
1095                    "field `{}` sets a database type incompatible with its \
1096                     embedded enum discriminant",
1097                    field.name,
1098                )));
1099            };
1100            if embedded_enum.discriminant.ty != stmt::Type::I64 {
1101                return Err(Error::invalid_schema(format!(
1102                    "field `{}` sets an integer database type on a \
1103                     non-integer embedded enum",
1104                    field.name,
1105                )));
1106            }
1107
1108            for variant in &embedded_enum.variants {
1109                let stmt::Value::I64(value) = variant.discriminant else {
1110                    unreachable!("integer enum has a non-integer discriminant")
1111                };
1112                if value < 0 || value > max {
1113                    return Err(Error::invalid_schema(format!(
1114                        "variant `{}` has discriminant {value}, which does not fit \
1115                         the database type requested for field `{}`",
1116                        variant.name, field.name,
1117                    )));
1118                }
1119            }
1120            discriminant.storage_ty = Some(storage_ty.clone());
1121        }
1122
1123        // Create the discriminant column. It inherits nullability from the enum field.
1124        let column_id = self.create_column(field, &discriminant);
1125        let field_expr = self.field_expr(field, field_index);
1126
1127        // A nested enum's discriminant column lives in the parent enum's
1128        // column namespace; record it so a later variant field of the parent
1129        // resolving to the same name is caught as a duplicate.
1130        if let Some(columns) = self.build.enum_columns.as_mut() {
1131            let column_name = self.build.table.column(column_id).name.clone();
1132            if !columns.names.insert(column_name.clone()) {
1133                return Err(Error::invalid_schema(format!(
1134                    "two enum variant fields map to the column `{column_name}` \
1135                     without declaring a shared field; annotate both with \
1136                     `#[shared(<ident>)]` to share the column",
1137                )));
1138            }
1139        }
1140
1141        // For data-carrying enums the model value is Record([I64(disc), ...]),
1142        // so project [0] to extract the discriminant; for unit-only enums the
1143        // value IS the I64 discriminant directly.
1144        let disc_expr = if embedded_enum.has_data_variants() {
1145            stmt::Expr::project(field_expr.clone(), stmt::Projection::single(0))
1146        } else {
1147            field_expr.clone()
1148        };
1149
1150        let lowering_index = self
1151            .build
1152            .push_lowering(column_id, &discriminant.ty, disc_expr);
1153
1154        let bit = self.build.next_bit();
1155        let sub_projection = self.sub_projection(field_index);
1156
1157        let disc_proj = stmt::Expr::project(field_expr.clone(), stmt::Projection::single(0));
1158
1159        // Activate a fresh column-sharing registry for this enum's variant
1160        // fields. A nested enum installs (and restores) its own, so sharing
1161        // never crosses enum boundaries — but the physical `names` are seeded
1162        // from the outer registry, since a nested enum's columns live in the
1163        // parent's namespace and must collide against columns the outer enum
1164        // already created. Restored after the variants are mapped.
1165        let inner = EnumColumns {
1166            shared: Default::default(),
1167            names: self
1168                .build
1169                .enum_columns
1170                .as_ref()
1171                .map(|outer| outer.names.clone())
1172                .unwrap_or_default(),
1173        };
1174        let mut saved_shared = self.build.enum_columns.replace(inner);
1175
1176        let variants: Result<Vec<mapping::EnumVariant>> = embedded_enum
1177            .variants
1178            .iter()
1179            .enumerate()
1180            .map(|(variant_index, variant)| {
1181                let mut mapper =
1182                    self.for_variant(field, field_index, disc_proj.clone(), &variant.discriminant);
1183
1184                let fields: Vec<mapping::Field> = embedded_enum
1185                    .variant_fields(variant_index)
1186                    .enumerate()
1187                    .map(|(index, field)| {
1188                        // Variant fields are stored at positions 1.. in the Record
1189                        // (position 0 is the discriminant), so adjust the index.
1190                        mapper.map_field(index + 1, field)
1191                    })
1192                    .collect::<Result<_>>()?;
1193
1194                Ok(mapping::EnumVariant {
1195                    discriminant: variant.discriminant.clone(),
1196                    fields,
1197                })
1198            })
1199            .collect();
1200
1201        // Restore the outer registry. Shared-identifier scoping stays local to
1202        // this enum, but the physical column names it created live in the
1203        // parent's namespace — merge them back so a later field of the outer
1204        // enum colliding with a nested enum's column is still caught.
1205        if let (Some(outer), Some(inner)) = (saved_shared.as_mut(), self.build.enum_columns.take())
1206        {
1207            outer.names.extend(inner.names);
1208        }
1209        self.build.enum_columns = saved_shared;
1210        let variants = variants?;
1211
1212        let field_mask = stmt::PathFieldSet::from_iter([bit]);
1213
1214        let disc_column_expr = self
1215            .build
1216            .map_table_column_to_model(column_id, &discriminant);
1217
1218        Ok(mapping::Field::Enum(mapping::FieldEnum {
1219            discriminant: mapping::FieldPrimitive {
1220                column: column_id,
1221                lowering: lowering_index,
1222                field_mask: field_mask.clone(),
1223                sub_projection: stmt::Projection::identity(),
1224                column_expr: disc_column_expr,
1225            },
1226            variants,
1227            field_mask,
1228            sub_projection,
1229            default_returning: stmt::Expr::null(),
1230        }))
1231    }

1233    fn map_field_struct(
1234        &mut self,
1235        field_index: usize,
1236        field: &app::Field,
1237        model_id: ModelId,
1238        embedded_struct: &app::EmbeddedStruct,
1239    ) -> Result<mapping::Field> {
1240        let sub_projection = self.sub_projection(field_index);
1241
1242        // A nullable embedded struct (`Option<Embed>`) is decoded/encoded
1243        // through a *head* column whose null-ness is the option's none-ness
1244        // (`NULL` = `None`), keeping `None` represented as `NULL` like
1245        // `Option<scalar>` and an embedded enum's (nullable) discriminant. The
1246        // flattened leaf columns are forced nullable and presence-guarded, so a
1247        // `None` value writes all-`NULL` rather than panicking when projecting
1248        // `Null`. There are two ways to obtain the head column:
1249        //
1250        //  * Multi-column embed (`Option<Address>` over `Address { street, city
1251        //    }`): a *dedicated* nullable `bool` presence column (`NULL`/`true`)
1252        //    named after the field. `None` → `(NULL, NULL, NULL)`; `Some(..)` →
1253        //    `(true, "..", "..")`. The presence column distinguishes `None`
1254        //    from a `Some` whose own fields are all `NULL`.
1255        //
1256        //  * Newtype embed (`Option<Email(String)>` over `struct Email(String)`):
1257        //    the one flattened leaf is unnamed and collapses to the field name,
1258        //    so a dedicated presence column would *collide* with it. Reuse that
1259        //    leaf as the head — a non-`Option` inner is non-`NULL` exactly when
1260        //    the embed is present, so its null-ness already encodes none-ness,
1261        //    like `Option<scalar>`. (A *named* single field flattens to
1262        //    `{field}_{name}`, doesn't collide, and keeps the disambiguating
1263        //    presence column above.)
1264        //
1265        //    A newtype wrapping an *optional* (`Option<MaybeBody>` over
1266        //    `MaybeBody(Option<String>)`) is rejected: the single column can't
1267        //    tell `None` from `Some(MaybeBody(None))`, and a dedicated presence
1268        //    column would collide with the leaf, so there is no sound mapping.
1269        let reuse_leaf = if field.nullable {
1270            match self.newtype_leaf_nullable(embedded_struct) {
1271                // Newtype over a non-optional scalar → reuse the leaf as head.
1272                Some(false) => true,
1273                // Newtype over an optional → no unambiguous single-column mapping.
1274                Some(true) => {
1275                    let parent = self
1276                        .build
1277                        .app
1278                        .get_model(field.id.model)
1279                        .map(|m| m.name().upper_camel_case())
1280                        .unwrap_or_else(|| "?".to_string());
1281                    let embed = embedded_struct.name.upper_camel_case();
1282                    return Err(Error::invalid_schema(format!(
1283                        "field `{parent}::{}` is `Option<{embed}>`, but `{embed}` is \
1284                         a newtype wrapping an optional value; it can't be flattened \
1285                         to a single column because that column can't distinguish \
1286                         `None` from `Some({embed}(None))`. Use a named struct field \
1287                         instead of a tuple newtype, or remove a layer of `Option`.",
1288                        field.name,
1289                    )));
1290                }
1291                // Not a newtype → dedicated presence column below.
1292                None => false,
1293            }
1294        } else {
1295            false
1296        };
1297
1298        // The dedicated presence column is created *before* the leaves so it
1299        // sorts first in the table. Only the `(column, lowering)` pair is
1300        // needed: `lowering` records the column's encode expression in
1301        // `columns`, and the column id becomes the struct's presence head.
1302        let dedicated_presence = if field.nullable && !reuse_leaf {
1303            let presence_primitive = app::FieldPrimitive {
1304                ty: stmt::Type::Bool,
1305                storage_ty: None,
1306                serialize: None,
1307            };
1308            // Nullable because `field.nullable` is set.
1309            let column = self.create_column(field, &presence_primitive);
1310            // Encode `true` for `Some`, `NULL` for `None` — the same presence
1311            // guard the leaf columns use, applied to the sentinel `true`.
1312            let presence_expr = self.presence_guard(field, field_index, stmt::Expr::from(true));
1313            let lowering = self
1314                .build
1315                .push_lowering(column, &stmt::Type::Bool, presence_expr);
1316            Some((column, lowering))
1317        } else {
1318            None
1319        };
1320
1321        // For a single-field newtype the outer field's `#[auto]` flattens
1322        // down to the one inner column. Multi-field embeds have no clear
1323        // target column, so the inherited flag stops at the boundary —
1324        // `Foo(Bar(u64))` with `#[auto]` on the outer flows through both
1325        // newtypes, but `Foo { a, b: Bar(u64) }` would not propagate it
1326        // from any outer past the `Foo` layer.
1327        let single_field = embedded_struct.fields.len() == 1;
1328        let mut child = if field.nullable {
1329            self.for_nullable_struct(field, field_index)
1330        } else {
1331            self.for_struct(field, field_index)
1332        };
1333        if single_field {
1334            child.inherited_auto_increment |= field.is_auto_increment();
1335        } else {
1336            child.inherited_auto_increment = false;
1337        }
1338        let nested_fields = child.map_fields(&embedded_struct.fields)?;
1339
1340        let mut columns: indexmap::IndexMap<ColumnId, usize> =
1341            nested_fields.iter().flat_map(|f| f.columns()).collect();
1342
1343        // The head column: the dedicated presence column (added to `columns`
1344        // here so its encode lowering is tracked), or the reused single leaf
1345        // for a single-column embed (already in `columns` as a real leaf).
1346        let presence = if let Some((column, lowering)) = dedicated_presence {
1347            columns.insert(column, lowering);
1348            Some(column)
1349        } else if reuse_leaf {
1350            Some(
1351                single_leaf_primitive(&nested_fields[0])
1352                    .expect("single-column embed flattens to one primitive leaf")
1353                    .column,
1354            )
1355        } else {
1356            None
1357        };
1358
1359        let field_mask = nested_fields
1360            .iter()
1361            .fold(stmt::PathFieldSet::new(), |acc, f| acc | f.field_mask());
1362
1363        Ok(mapping::Field::Struct(mapping::FieldStruct {
1364            id: model_id,
1365            fields: nested_fields,
1366            columns,
1367            field_mask,
1368            sub_projection,
1369            default_returning: stmt::Expr::null(),
1370            presence,
1371        }))
1372    }

1374    /// If `embed` is a newtype chain — a single *unnamed* field (recursively,
1375    /// through nested single-field newtype wrappers) bottoming out in a
1376    /// primitive — returns `Some(leaf_nullable)`, the bottom field's own
1377    /// nullability. Returns `None` when `embed` is not a newtype.
1378    ///
1379    /// A newtype flattens to one column that takes the parent field's own name,
1380    /// so when the parent is nullable it reuses that single leaf as the head
1381    /// column (a dedicated presence column would collide). That reuse is sound
1382    /// only when the leaf is non-`NULL` whenever the embed is present —
1383    /// `Some(false)`. A `Some(true)` leaf (the newtype wraps an `Option`) has no
1384    /// unambiguous single-column mapping and is rejected by the caller. A
1385    /// *named* single field (`Foo { value }`) flattens to `{field}_value`,
1386    /// doesn't collide, so it returns `None` here and keeps a dedicated presence
1387    /// column (preserving the `None`-vs-`Some(all-NULL)` disambiguation).
1388    fn newtype_leaf_nullable(&self, embed: &app::EmbeddedStruct) -> Option<bool> {
1389        let [field] = &embed.fields[..] else {
1390            return None;
1391        };
1392        if field.name.storage_name().is_some() {
1393            return None;
1394        }
1395        match &field.ty {
1396            app::FieldTy::Primitive(_) => Some(field.nullable),
1397            app::FieldTy::Embedded(embedded) => {
1398                match lookup_embedded_model(self.build.app, embedded.target, field) {
1399                    Ok(app::Model::EmbeddedStruct(inner)) => self.newtype_leaf_nullable(inner),
1400                    _ => None,
1401                }
1402            }
1403            _ => None,
1404        }
1405    }

1407    /// Builds the final database column name for `field` at the current nesting level.
1408    ///
1409    /// Joins `self.prefix` components with `_`, appends the field name, then
1410    /// prepends `schema_prefix` (if any) with `__`. Because `schema_prefix` is
1411    /// applied here — never stored in `self.prefix` — it is always applied
1412    /// exactly once regardless of nesting depth.
1413    ///
1414    /// The field-name component prefers an explicit storage override, then the
1415    /// shared logical-field identifier (so every variant of a `#[shared]`
1416    /// group derives the same column name regardless of its Rust field name),
1417    /// then the app name.
1418    fn column_name(&self, field: &app::Field) -> String {
1419        let name = field
1420            .name
1421            .storage
1422            .clone()
1423            .or_else(|| field.shared.as_ref().map(Name::snake_case))
1424            .or_else(|| field.name.app.clone());
1425        let embed = match name {
1426            Some(field_name) => {
1427                if self.prefix.is_empty() {
1428                    field_name
1429                } else {
1430                    format!("{}_{field_name}", self.prefix.join("_"))
1431                }
1432            }
1433            None => {
1434                // Unnamed field (newtype wrapper) — use just the prefix
1435                // components so the column name is identical to the parent
1436                // field's name.
1437                assert!(
1438                    !self.prefix.is_empty(),
1439                    "unnamed field with empty prefix; a newtype field must be \
1440                     nested inside a parent field"
1441                );
1442                self.prefix.join("_")
1443            }
1444        };
1445        match self.build.schema_prefix.as_deref() {
1446            None => embed,
1447            Some(sp) => format!("{sp}__{embed}"),
1448        }
1449    }

1451    /// Creates a column for `field` using `primitive` for the storage type.
1452    ///
1453    /// Derives the column name from `self.column_name(field)`, nullability from
1454    /// `field.nullable || self.force_nullable`, and auto-increment from
1455    /// `field.is_auto_increment()`.
1456    fn create_column(&mut self, field: &app::Field, primitive: &app::FieldPrimitive) -> ColumnId {
1457        let nullable = field.nullable || self.force_nullable;
1458        let is_auto_increment = (field.is_auto_increment() || self.inherited_auto_increment)
1459            && self.build.db.auto_increment;
1460
1461        let mut storage_ty = db::Type::from_app_column(
1462            &primitive.ty,
1463            primitive.storage_ty.as_ref(),
1464            self.build.db,
1465            is_auto_increment,
1466        )
1467        .expect("unsupported storage type");
1468
1469        // Prefix native enum type names so they don't collide across test runs
1470        // or multi-tenant deployments.
1471        if let Some(type_enum) = storage_ty.named_enum_mut()
1472            && let (Some(prefix), Some(name)) = (&self.build.name_prefix, &mut type_enum.name)
1473        {
1474            *name = format!("{prefix}{name}");
1475        }
1476
1477        let id = ColumnId {
1478            table: self.build.table.id,
1479            index: self.build.table.columns.len(),
1480        };
1481
1482        self.build.table.columns.push(db::Column {
1483            id,
1484            name: self.column_name(field),
1485            ty: storage_ty.bridge_type(&primitive.ty),
1486            storage_ty,
1487            nullable,
1488            primary_key: false,
1489            auto_increment: is_auto_increment,
1490            versionable: field.is_versionable(),
1491        });
1492
1493        id
1494    }

1496    /// Extends `parent_base` by one projection step to produce the `field_base`
1497    /// for a child `MapField` entering `field` at `field_index` within the parent.
1498    ///
1499    /// If `parent_base` is the top-level sentinel (`Expr::arg(0)`), the child
1500    /// starts a fresh projection rooted at `field.id`. Otherwise the existing
1501    /// `ExprProject` is extended by `field_index`.
1502    fn extend_field_base(&self, field: &app::Field, field_index: usize) -> stmt::Expr {
1503        match &self.field_base {
1504            None => stmt::Expr::ref_self_field(field.id),
1505            Some(stmt::Expr::Project(ep)) => {
1506                let mut proj = ep.projection.clone();
1507                proj.push(field_index);
1508                stmt::Expr::project(*ep.base.clone(), proj)
1509            }
1510            Some(expr) => stmt::Expr::project(expr.clone(), [field_index]),
1511        }
1512    }
1513
1514    /// Returns the sub-projection from the root source field to a field at
1515    /// `field_index` within the current nesting level.
1516    ///
1517    /// If `field_base` is an `ExprProject`, the sub-projection is its
1518    /// projection extended by `field_index`. At the top level (`field_base`
1519    /// is `Expr::arg(0)`) the field is its own root, so identity is returned.
1520    fn sub_projection(&self, field_index: usize) -> stmt::Projection {
1521        match &self.field_base {
1522            None => stmt::Projection::identity(),
1523            Some(stmt::Expr::Project(ep)) => {
1524                let mut proj = ep.projection.clone();
1525                proj.push(field_index);
1526                proj
1527            }
1528            Some(_) => [field_index].into(),
1529        }
1530    }
1531
1532    /// Builds the lowering expression for a field at the current nesting level.
1533    ///
1534    /// At the top level (`field_base` is `Expr::arg(0)`) each field references
1535    /// itself directly. Inside an embedded struct/variant the expression extends
1536    /// `field_base` by `field_index`. The raw expression is then substituted
1537    /// into `field_expr_base` (which may wrap it in a match guard).
1538    fn field_expr(&self, field: &app::Field, field_index: usize) -> stmt::Expr {
1539        let raw = match self.field_base.clone() {
1540            None => stmt::Expr::ref_self_field(field.id),
1541            Some(stmt::Expr::Project(mut expr_project)) => {
1542                expr_project.projection.push(field_index);
1543                expr_project.into()
1544            }
1545            Some(expr) => stmt::Expr::project(expr, [field_index]),
1546        };
1547
1548        let mut result = self.field_expr_base.clone();
1549        result.substitute(&[raw]);
1550        result
1551    }

1553    /// Creates a child `MapField` for recursing into an embedded field.
1554    ///
1555    /// The child inherits the current prefix extended by `name` and inherits
1556    /// `force_nullable`, `field_base`, and `field_expr_base` unchanged. Used
1557    /// when entering struct/variant fields so that sub-field columns are named
1558    /// `{..prefix..}_{name}_{sub_field}`.
1559    fn with_prefix(&mut self, name: Option<&str>) -> MapField<'_, 'b> {
1560        let mut prefix = self.prefix.clone();
1561        if let Some(name) = name {
1562            prefix.push(name.to_owned());
1563        }
1564        MapField {
1565            build: self.build,
1566            prefix,
1567            force_nullable: self.force_nullable,
1568            field_base: self.field_base.clone(),
1569            field_expr_base: self.field_expr_base.clone(),
1570            inherited_auto_increment: self.inherited_auto_increment,
1571            // A nested embed stays within its enclosing variant, so it inherits
1572            // the variant discriminant; `for_variant` overrides it per variant.
1573            variant_discriminant: self.variant_discriminant.clone(),
1574        }
1575    }

1577    /// Creates a child `MapField` for an embedded type whose columns are only
1578    /// conditionally populated: an enum variant (only the active variant's
1579    /// columns are written) or a nullable struct embed (every column is `NULL`
1580    /// when the embed is `None`).
1581    ///
1582    /// Sets `field_base` so that `field_expr` on the child projects from
1583    /// `field`, forces the flattened columns nullable, and installs `guard` as
1584    /// the `field_expr_base` so that every `field_expr` call is automatically
1585    /// wrapped in it. `guard` is a `Match` template with `Expr::arg(0)` as the
1586    /// placeholder for the raw field expression — a discriminant check (see
1587    /// [`Self::for_variant`]) or a presence guard (see
1588    /// [`Self::for_nullable_struct`]) — whose `else` branch encodes the
1589    /// inactive/`None` case to `null` instead of projecting into `Null`.
1590    fn for_guarded_embed(
1591        &mut self,
1592        field: &app::Field,
1593        field_index: usize,
1594        guard: stmt::Expr,
1595    ) -> MapField<'_, 'b> {
1596        let field_base = self.extend_field_base(field, field_index);
1597        let mut child = self.with_prefix(field.name.storage_name());
1598        child.force_nullable = true;
1599        child.field_base = Some(field_base);
1600        child.field_expr_base.substitute(&[guard]);
1601        child
1602    }

1604    /// Creates a variant-specific child `MapField`, guarding each `field_expr`
1605    /// in the discriminant check `match_expr(disc_proj, [arm(discriminant,
1606    /// Expr::arg(0))], null())`. See [`Self::for_guarded_embed`].
1607    fn for_variant(
1608        &mut self,
1609        field: &app::Field,
1610        field_index: usize,
1611        disc_proj: stmt::Expr,
1612        discriminant: &stmt::Value,
1613    ) -> MapField<'_, 'b> {
1614        let guard = stmt::Expr::match_expr(
1615            disc_proj,
1616            vec![stmt::MatchArm {
1617                pattern: discriminant.clone(),
1618                expr: stmt::Expr::arg(0),
1619            }],
1620            stmt::Expr::null(),
1621        );
1622        let mut child = self.for_guarded_embed(field, field_index, guard);
1623        child.variant_discriminant = Some(discriminant.clone());
1624        child
1625    }

1627    /// Creates a child `MapField` for recursing into an embedded struct field.
1628    ///
1629    /// Updates `field_base` to reflect the new nesting level: if entering the
1630    /// first embedded level, sets the source to this field with an identity
1631    /// projection; at deeper levels, extends the existing projection by
1632    /// `field_index`.
1633    fn for_struct(&mut self, field: &app::Field, field_index: usize) -> MapField<'_, 'b> {
1634        let field_base = self.extend_field_base(field, field_index);
1635        // Unnamed (newtype) inner fields keep the parent's prefix so the
1636        // flattened column name comes from the outermost named field.
1637        let mut child = self.with_prefix(field.name.storage_name());
1638        child.field_base = Some(field_base);
1639        child
1640    }

1642    /// Builds the presence guard that encodes a nullable struct embed
1643    /// (`Option<Embed>`):
1644    ///
1645    /// `Match(is_not_null(field), [Bool(true) => present], else => null())`.
1646    ///
1647    /// `present` is what's written when the embed is `Some`: the sentinel
1648    /// `Bool(true)` for the presence column itself, or `Expr::arg(0)` (the raw
1649    /// leaf expression) for the guard wrapped around each flattened leaf column.
1650    /// A `None` value (the field is `Null`, so `is_not_null` is `false`) takes
1651    /// the `else` branch and encodes `null`.
1652    fn presence_guard(
1653        &self,
1654        field: &app::Field,
1655        field_index: usize,
1656        present: stmt::Expr,
1657    ) -> stmt::Expr {
1658        stmt::Expr::match_expr(
1659            stmt::Expr::is_not_null(self.field_expr(field, field_index)),
1660            vec![stmt::MatchArm {
1661                pattern: stmt::Value::Bool(true),
1662                expr: present,
1663            }],
1664            stmt::Expr::null(),
1665        )
1666    }

1668    /// Creates a child `MapField` for recursing into a nullable embedded struct
1669    /// (`Option<Embed>`), guarding each `field_expr` in a [`Self::presence_guard`]
1670    /// over `Expr::arg(0)` so a `None` value encodes each leaf to `null` (the
1671    /// guard's `else` branch). See [`Self::for_guarded_embed`].
1672    fn for_nullable_struct(&mut self, field: &app::Field, field_index: usize) -> MapField<'_, 'b> {
1673        let guard = self.presence_guard(field, field_index, stmt::Expr::arg(0));
1674        self.for_guarded_embed(field, field_index, guard)
1675    }

1701/// Maximum non-negative discriminant representable by an integer storage
1702/// type. Enum discriminants use `i64` in the app schema, so unsigned 8-byte
1703/// storage has the same effective maximum as signed 8-byte storage.
1704fn integer_storage_max(ty: &db::Type) -> Option<i64> {
1705    match ty {
1706        db::Type::Integer(size @ 1..=7) => Some((1_i64 << (u32::from(*size) * 8 - 1)) - 1),
1707        db::Type::Integer(8) | db::Type::UnsignedInteger(8) => Some(i64::MAX),
1708        db::Type::UnsignedInteger(size @ 1..=7) => Some((1_i64 << (u32::from(*size) * 8)) - 1),
1709        _ => None,
1710    }
1711}

1725/// Truncate `name` to `limit` bytes, appending a 5-character stable hash
1726/// suffix (`_XXXX`) so the result is unique and deterministic across builds.
1727///
1728/// The hash is FNV-1a over the *full* pre-truncation name so two names that
1729/// share a long prefix produce different suffixes. `DefaultHasher` is
1730/// intentionally avoided — it is unstable across Rust versions and would
1731/// churn migration snapshots.
1732fn truncate_identifier(name: String, limit: usize) -> String {
1733    if name.len() <= limit {
1734        return name;
1735    }
1736
1737    // FNV-1a 32-bit hash of the full name for a stable 4-hex-digit suffix.
1738    const FNV_OFFSET: u32 = 2_166_136_261;
1739    const FNV_PRIME: u32 = 16_777_619;
1740    let hash = name
1741        .bytes()
1742        .fold(FNV_OFFSET, |acc, b| acc.wrapping_mul(FNV_PRIME) ^ b as u32);
1743    let suffix = format!("_{:04x}", hash & 0xFFFF);
1744
1745    // Truncate at a char boundary so we never split a multi-byte character.
1746    // The condition uses the char's END position so we never exceed `budget`.
1747    let budget = limit.saturating_sub(suffix.len());
1748    let truncated = &name[..name
1749        .char_indices()
1750        .take_while(|(i, c)| i + c.len_utf8() <= budget)
1751        .last()
1752        .map(|(i, c)| i + c.len_utf8())
1753        .unwrap_or(0)];
1754
1755    format!("{truncated}{suffix}")
1756}

1758/// Look up an embedded model by ID, returning a descriptive error if not found.
1759fn lookup_embedded_model<'a>(
1760    app: &'a app::Schema,
1761    target: ModelId,
1762    field: &app::Field,
1763) -> Result<&'a Model> {
1764    app.get_model(target).ok_or_else(|| {
1765        let parent_name = app
1766            .get_model(field.id.model)
1767            .map(|m| m.name().upper_camel_case())
1768            .unwrap_or_else(|| "?".to_string());
1769
1770        Error::invalid_schema(format!(
1771            "field `{parent_name}::{}` references an embedded type that is not registered \
1772             in the schema; did you forget to include the embedded type in \
1773             `toasty::models!(..)`?",
1774            field.name,
1775        ))
1776    })

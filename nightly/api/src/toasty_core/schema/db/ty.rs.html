4/// Database-level storage types representing how values are stored in the target database.
5///
6/// `db::Type` represents the **external** types used by specific database systems
7/// (PostgreSQL, MySQL, SQLite, DynamoDB, etc.) to store column values. These are the actual
8/// storage types that appear in `CREATE TABLE` statements and database schemas.
9///
10/// # Type System Hierarchy
11///
12/// Toasty has two distinct type systems:
13///
14/// 1. **[`stmt::Type`](crate::stmt::Type)**: Application and query engine types (internal to Toasty)
15///    - Represents Rust types: `I8`, `I16`, `String`, etc.
16///    - Types of [`stmt::Value`] and [`stmt::Expr`]
17///    - Used throughout Toasty's query processing at both application and engine levels
18///
19/// 2. **`db::Type`** (this type): Database storage types (external)
20///    - Represents database column types: `Integer(n)`, `Text`, `VarChar(n)`, etc.
21///    - External representation specific to the target database
22///    - Specified in the schema, used by drivers
23///
24/// # Mapping from Application to Database Types
25///
26/// The mapping from [`stmt::Type`] to `db::Type` happens at the driver boundary:
27///
28/// ```text
29/// stmt::Type::String  →  db::Type::Text         (default for most databases)
30///                     →  db::Type::VarChar(255)  (if specified in schema)
31///
32/// stmt::Type::I64     →  db::Type::Integer(8)   (8-byte integer)
33/// stmt::Type::I32     →  db::Type::Integer(4)   (4-byte integer)
34/// stmt::Type::Bool    →  db::Type::Boolean
35/// ```
36///
37/// See [`Type::from_app`] for the complete mapping logic.
38///
39/// # Schema Usage
40///
41/// Each column in the database schema ([`Column`](crate::schema::db::Column)) stores both:
42/// - `column.ty: stmt::Type` - How Toasty views the column internally (application/engine type)
43/// - `column.storage_ty: Option<db::Type>` - How the database stores it externally (storage type)
44///
45/// When `storage_ty` is `None`, the driver uses default mappings from `stmt::Type`.
46/// When specified, it allows fine-grained control over storage (e.g., `VARCHAR(50)` vs `TEXT`).
47///
48/// # Database-Specific Behavior
49///
50/// Different databases support different storage types. The driver's capability
51/// structure ([`driver::Capability`]) describes what types are available:
52///
53/// - **PostgreSQL**: Supports `Text`, `VarChar`, `Integer`, `Boolean`, etc.
54/// - **SQLite**: Uses type affinity; most types map to `TEXT`, `INTEGER`, `REAL`, or `BLOB`
55/// - **DynamoDB**: Uses NoSQL types like `S` (string), `N` (number), `BOOL`, etc.
56///
57/// # See Also
58///
59/// - [`stmt::Type`](crate::stmt::Type) - Application and query engine type system with detailed flow documentation
60/// - [`Type::from_app`] - Mapping logic from statement types to database types
61/// - [`Column`](crate::schema::db::Column) - Schema representation with both type systems
62#[derive(Debug, Clone, PartialEq)]

89    Numeric(Option<(u32, u32)>),

132impl Type {
133    /// Construct a list storage type, collapsing a list of documents into a
134    /// single document.
135    ///
136    /// A `#[document]` collection (`Vec<embed>`) is stored as one JSON blob,
137    /// never a native array, so `db::Type` must never nest a [`Type::Document`]
138    /// inside a [`Type::List`]. Routing list construction through here makes
139    /// that invariant hold by construction: the list *is* the document.
140    pub fn list(elem: Type) -> Type {
141        match elem {
142            Type::Document { binary } => Type::Document { binary },
143            elem => Type::List(Box::new(elem)),
144        }
145    }

147    /// The named enum this storage type carries: a scalar [`Type::Enum`] or the
148    /// element of an enum array (`List(Enum)`).
149    pub fn named_enum(&self) -> Option<&TypeEnum> {
150        match self {
151            Type::Enum(type_enum) => Some(type_enum),
152            Type::List(elem) => match elem.as_ref() {
153                Type::Enum(type_enum) => Some(type_enum),
154                _ => None,
155            },
156            _ => None,
157        }
158    }

160    /// Mutable counterpart to [`named_enum`](Self::named_enum).
161    pub fn named_enum_mut(&mut self) -> Option<&mut TypeEnum> {
162        match self {
163            Type::Enum(type_enum) => Some(type_enum),
164            Type::List(elem) => match elem.as_mut() {
165                Type::Enum(type_enum) => Some(type_enum),
166                _ => None,
167            },
168            _ => None,
169        }

172    /// Maps an application-level type to a database-level storage type.
173    pub fn from_app(
174        ty: &stmt::Type,
175        hint: Option<&Type>,
176        db: &driver::StorageTypes,
177    ) -> Result<Type> {
178        match hint {
179            Some(ty) => Ok(ty.clone()),
180            None => match ty {
181                stmt::Type::Bool => Ok(Type::Boolean),
182                stmt::Type::I8 => Ok(Type::Integer(1)),
183                stmt::Type::I16 => Ok(Type::Integer(2)),
184                stmt::Type::I32 => Ok(Type::Integer(4)),
185                stmt::Type::I64 => Ok(Type::Integer(8)),
186                // Map unsigned types to UnsignedInteger with appropriate byte width
187                stmt::Type::U8 => Ok(Type::UnsignedInteger(1)),
188                stmt::Type::U16 => Ok(Type::UnsignedInteger(2)),
189                stmt::Type::U32 => Ok(Type::UnsignedInteger(4)),
190                stmt::Type::U64 => Ok(Type::UnsignedInteger(8)),
191                stmt::Type::F32 => Ok(Type::Float(4)),
192                stmt::Type::F64 => Ok(Type::Float(8)),
193                stmt::Type::String => Ok(db.default_string_type.clone()),
194                stmt::Type::Uuid => Ok(db.default_uuid_type.clone()),
195                stmt::Type::Bytes => Ok(db.default_bytes_type.clone()),
196                // Decimal type
197                #[cfg(feature = "rust_decimal")]
198                stmt::Type::Decimal => Ok(db.default_decimal_type.clone()),
199                // BigDecimal type
200                #[cfg(feature = "bigdecimal")]
201                stmt::Type::BigDecimal => Ok(db.default_bigdecimal_type.clone()),
202                // Date/time types from jiff
203                #[cfg(feature = "jiff")]
204                stmt::Type::Timestamp => Ok(db.default_timestamp_type.clone()),
205                #[cfg(feature = "jiff")]
206                stmt::Type::Zoned => Ok(db.default_zoned_type.clone()),
207                #[cfg(feature = "jiff")]
208                stmt::Type::Date => Ok(db.default_date_type.clone()),
209                #[cfg(feature = "jiff")]
210                stmt::Type::Time => Ok(db.default_time_type.clone()),
211                #[cfg(feature = "jiff")]
212                stmt::Type::DateTime => Ok(db.default_datetime_type.clone()),
213                // An embedded model column is stored as a single document
214                // (`jsonb` on PG, `JSON` elsewhere) — never a native array.
215                // `Type::list` collapses a list-of-documents back to one
216                // document, so a `#[document]` collection (`List(Model)`) lands
217                // here as a plain `Document`.
218                stmt::Type::Model(_) => Ok(Type::Document { binary: true }),
219                stmt::Type::List(elem) => Ok(Type::list(Self::from_app(elem, None, db)?)),
220                _ => Err(crate::Error::unsupported_feature(format!(
221                    "type {:?} is not supported by this database",
222                    ty
223                ))),
224            },
225        }
226    }

228    pub(crate) fn from_app_column(
229        ty: &stmt::Type,
230        hint: Option<&Type>,
231        db: &driver::Capability,
232        auto_increment: bool,
233    ) -> Result<Type> {
234        let mut storage_ty = Self::from_app(ty, hint, &db.storage_types)?;
235
236        if auto_increment && let Some(max) = db.max_auto_increment_integer_width {
237            match &mut storage_ty {
238                Type::Integer(size) | Type::UnsignedInteger(size) if *size > max => {
239                    *size = max;
240                }
241                _ => {}
242            }
243        }
244
245        Ok(storage_ty)
246    }

248    /// Determines the [`stmt::Type`] closest to this [`db::Type`] that should be used
249    /// as an intermediate conversion step to lessen the work done by each individual driver.
250    pub fn bridge_type(&self, ty: &stmt::Type) -> stmt::Type {
251        match (self, ty) {
252            (Self::Blob | Self::Binary(_), stmt::Type::Uuid) => stmt::Type::Bytes,
253            (Self::Text | Self::VarChar(_), _) => stmt::Type::String,
254            // Enum values are always strings at the application level
255            (Self::Enum(_), _) => stmt::Type::String,
256            // Let engine handle UTC conversion
257            #[cfg(feature = "jiff")]
258            (Self::Timestamp(_) | Self::DateTime(_), stmt::Type::Zoned) => stmt::Type::Timestamp,
259            // Bool key/index fields stored as 1-byte integer (e.g. DynamoDB N("1"/"0")).
260            // The engine casts Bool <-> I8 transparently via encode_column /
261            // map_table_column_to_model; the driver handles them as plain numbers.
262            (Self::Integer(1), stmt::Type::Bool) => stmt::Type::I8,
263            // A `#[document]` column stores a structural document: the column
264            // is typed by `Type::Object` (mirroring `Value::Object`), not by
265            // the embedded model. The model identity stays an app/engine
266            // concept, carried by `mapping::Mapping::document_columns`. A
267            // document collection (`List(Model)`, whose storage collapses to
268            // one document) keeps its list shape with `Object` elements.
269            (Self::Document { .. }, stmt::Type::Model(_)) => stmt::Type::Object,
270            (Self::Document { .. }, stmt::Type::List(elem))
271                if matches!(**elem, stmt::Type::Model(_)) =>
272            {
273                stmt::Type::List(Box::new(stmt::Type::Object))
274            }
275            _ => ty.clone(),
276        }
277    }

279    pub(crate) fn verify(&self, db: &driver::Capability) -> Result<()> {
280        match *self {
281            Type::VarChar(size) => match db.storage_types.varchar {
282                Some(max) if size > max => Err(crate::Error::unsupported_feature(format!(
283                    "VARCHAR({}) exceeds database maximum of {}",
284                    size, max
285                ))),
286                None => Err(crate::Error::unsupported_feature(
287                    "VARCHAR type is not supported by this database",
288                )),
289                _ => Ok(()),
290            },
291            _ => Ok(()),
292        }
293    }

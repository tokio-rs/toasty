1//! Mutation methods (`insert`, `remove`) on a multi-step relation traversal
2//! are not supported. The check is at exec time — the type system allows the
3//! call but the executor returns an error.
4//!
5//! Before unifying `Many`, `One`, and `OptionOne` into a single per-model
6//! `Query<T>`, calling `.insert()` on a `via` relation was a compile error
7//! because `Many<Via>` had no such method. The unified design moves that
8//! rejection to runtime so the surface area stays small.
9
10use crate::prelude::*;
11
12#[driver_test(
13    id(ID),
14    requires(sql),
15    scenario(crate::scenarios::user_org_project_todo)
16)]
17pub async fn insert_on_multi_step_traversal_errors(test: &mut Test) -> Result<()> {
18    let mut db = setup(test).await;
19
20    let user = toasty::create!(User { name: "Alice" })
21        .exec(&mut db)
22        .await?;
23    let org = toasty::create!(Organization {
24        name: "Acme",
25        user: &user
26    })
27    .exec(&mut db)
28    .await?;
29    let project = toasty::create!(Project {
30        name: "p",
31        organization: &org
32    })
33    .exec(&mut db)
34    .await?;
35    let _todo = toasty::create!(Todo {
36        title: "t",
37        project: &project
38    })
39    .exec(&mut db)
40    .await?;
41
42    // `user.organizations()` is a single-step relation; chaining `.projects()`
43    // produces a query whose underlying association is two steps. `.insert`
44    // requires a single-step scope, so the executor rejects it.
45    let chained = user.organizations().projects();
46    let err = assert_err!(chained.insert(&mut db, &project).await);
47    assert!(err.to_string().to_lowercase().contains("multi-step"));
48
49    let chained = user.organizations().projects();
50    let err = assert_err!(chained.remove(&mut db, &project).await);
51    assert!(err.to_string().to_lowercase().contains("multi-step"));
52
53    Ok(())
54}

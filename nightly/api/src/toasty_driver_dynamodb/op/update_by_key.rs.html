10/// An [`stmt::Input`] that resolves column references into a record produced
11/// by `item_to_record`. After lowering, filter/condition expressions reference
12/// columns via `ExprReference::Column { column: i }` where `i` is the column's
13/// position in `table.columns`. `item_to_record` builds the record in that same
14/// order, so indexing by `col.column` gives the right field.
15struct RecordInput<'a>(&'a stmt::ValueRecord);
16
17impl stmt::Input for RecordInput<'_> {
18    fn resolve_ref(
19        &mut self,
20        expr_reference: &stmt::ExprReference,
21        projection: &stmt::Projection,
22    ) -> Option<stmt::Expr> {
23        match expr_reference {
24            stmt::ExprReference::Column(col) => {
25                Some(self.0.fields[col.column].entry(projection).to_expr())
26            }
27            _ => None,
28        }
29    }
30}

32/// Returns `true` when the DynamoDB `ConditionalCheckFailedException` was
33/// caused by the *filter* expression failing (→ return count 0), or `false`
34/// when it was caused by the *condition* expression failing (→ return an
35/// error).
36///
37/// Strategy: DynamoDB returns the item's pre-update state when
38/// `ReturnValuesOnConditionCheckFailure::AllOld` is set.  We evaluate the
39/// filter in-memory against that snapshot:
40///
41/// - No old item → the record didn't exist; the filter trivially didn't
42///   match → count 0.
43/// - Old item exists, filter evaluates to `false` → count 0.
44/// - Old item exists, filter evaluates to `true` (or there is no filter) →
45///   the condition must have been the failing part → error.
46fn filter_failed(
47    old_item: Option<&HashMap<String, AttributeValue>>,
48    table: &db::Table,
49    filter: Option<&stmt::Expr>,
50) -> bool {
51    let Some(filter) = filter else {
52        return false;
53    };
54
55    let Some(item) = old_item else {
56        return true;
57    };
58
59    let record = item_to_record(item, table.columns.iter()).unwrap();
60    !filter.eval_bool(RecordInput(&record)).unwrap_or(false)
61}
62
63/// Interprets a `ConditionalCheckFailedException` from `update_item`: if the
64/// filter was the failing predicate return an empty response; otherwise surface
65/// a condition error.
66fn on_update_item_condition_failed(
67    item: Option<&HashMap<String, AttributeValue>>,
68    message: Option<&str>,
69    table: &db::Table,
70    filter: Option<&stmt::Expr>,
71    returning: bool,
72) -> Result<ExecResponse> {
73    if filter_failed(item, table, filter) {
74        if returning {
75            Ok(ExecResponse::empty_value_stream())
76        } else {
77            Ok(ExecResponse::count(0))
78        }
79    } else {
80        Err(toasty_core::Error::condition_failed(
81            message
82                .unwrap_or("DynamoDB conditional check failed")
83                .to_string(),
84        ))
85    }
86}
87
88/// Interprets a `TransactionCanceledException` from `transact_write_items`:
89/// if every `ConditionalCheckFailed` reason was caused by the filter return an
90/// empty response; if any was caused by the condition expression surface a
91/// condition error.
92fn on_transaction_cancelled(
93    reasons: &[CancellationReason],
94    message: Option<&str>,
95    table: &db::Table,
96    filter: Option<&stmt::Expr>,
97    returning: bool,
98) -> Result<ExecResponse> {
99    let any_condition_failed = reasons
100        .iter()
101        .filter(|r| r.code() == Some("ConditionalCheckFailed"))
102        .any(|r| !filter_failed(r.item(), table, filter));
103
104    if any_condition_failed {
105        Err(toasty_core::Error::condition_failed(
106            message
107                .unwrap_or("DynamoDB conditional check failed")
108                .to_string(),
109        ))
110    } else if returning {
111        Ok(ExecResponse::empty_value_stream())
112    } else {
113        Ok(ExecResponse::count(0))
114    }
115}

117impl Connection {
118    pub(crate) async fn exec_update_by_key(
119        &mut self,
120        schema: &db::Schema,
121        op: operation::UpdateByKey,
122    ) -> Result<ExecResponse> {
123        let table = schema.table(op.table);
124        let cx = ExprContext::new_with_target(schema, table);
125
126        let mut expr_attrs = ExprAttrs::default();
127
128        let unique_indices = table
129            .indices
130            .iter()
131            .filter(|index| {
132                if !index.primary_key && index.unique {
133                    index.columns.iter().any(|index_column| {
134                        let column = index_column.table_column(schema);
135                        op.assignments
136                            .keys()
137                            .any(|projection| *projection == column.id.index)
138                    })
139                } else {
140                    false
141                }
142            })
143            .collect::<Vec<_>>();
144
145        let filter_expression = match (&op.filter, &op.condition) {
146            (Some(filter), None) => Some(ddb_expression(&cx, &mut expr_attrs, false, filter)),
147            (None, Some(condition)) => Some(ddb_expression(&cx, &mut expr_attrs, false, condition)),
148            (Some(filter), Some(condition)) => {
149                let f = ddb_expression(&cx, &mut expr_attrs, false, filter);
150                let c = ddb_expression(&cx, &mut expr_attrs, false, condition);
151                Some(format!("({f}) AND ({c})"))
152            }
153            _ => None,
154        };
155
156        let mut update_expression_set = String::new();
157        let mut update_expression_remove = String::new();
158
159        // Bound value per assigned column index. Used below to seed the
160        // returning-row placeholders; only the transact path (which can't
161        // fetch `UPDATED_NEW`) actually surfaces these seeds.
162        let mut bound_values: HashMap<usize, &stmt::Value> = HashMap::new();
163
164        for (projection, assignment) in op.assignments.iter() {
165            enum AssignKind {
166                Set,
167                Append,
168                Add,
169                Subtract,
170            }
171
172            let (expr, kind) = match assignment {
173                stmt::Assignment::Set(expr) => (expr, AssignKind::Set),
174                stmt::Assignment::Append(expr) => (expr, AssignKind::Append),
175                stmt::Assignment::Add(expr) => (expr, AssignKind::Add),
176                stmt::Assignment::Subtract(expr) => (expr, AssignKind::Subtract),
177                stmt::Assignment::Remove(_)
178                | stmt::Assignment::Pop
179                | stmt::Assignment::RemoveAt(_) => {
180                    // Collection mutations are gated by `vec_remove` /
181                    // `vec_pop` / `vec_remove_at` capability flags, all
182                    // currently `false` on DynamoDB. The lowering rejects
183                    // them before reaching the driver; if one slips through
184                    // it's a bug worth surfacing.
185                    unreachable!(
186                        "collection mutation reached DynamoDB driver — capability flag is off; assignment={assignment:#?}",
187                    )
188                }
189                _ => todo!(
190                    "only SET / APPEND / ADD / SUBTRACT supported in DynamoDB; got {assignment:#?}"
191                ),
192            };
193            let value = match expr {
194                stmt::Expr::Value(value) => value,
195                _ => todo!("op = {:#?}", op),
196            };
197
198            let column_ref = table.resolve(projection);
199            bound_values.insert(column_ref.id.index, value);
200
201            let column = expr_attrs.column(column_ref).to_string();
202
203            match kind {
204                AssignKind::Append => {
205                    // `stmt::push` / `stmt::extend` on a `Vec<scalar>` field
206                    // map to DynamoDB's `list_append(path, :v)`, which
207                    // atomically concatenates the given List `L` onto the
208                    // existing list attribute (creating the attribute if
209                    // absent).
210                    let value = expr_attrs.value(value);
211
212                    if !update_expression_set.is_empty() {
213                        write!(update_expression_set, ", ").unwrap();
214                    }
215
216                    write!(
217                        update_expression_set,
218                        "{column} = list_append(if_not_exists({column}, :__toasty_empty_list), {value})"
219                    )
220                    .unwrap();
221                    expr_attrs.attr_values.insert(
222                        ":__toasty_empty_list".to_string(),
223                        aws_sdk_dynamodb::types::AttributeValue::L(Vec::new()),
224                    );
225                }
226                AssignKind::Set if value.is_null() => {
227                    if !update_expression_remove.is_empty() {
228                        write!(update_expression_remove, ", ").unwrap();
229                    }
230
231                    write!(update_expression_remove, "{column}").unwrap();
232                }
233                AssignKind::Set => {
234                    let value = expr_attrs.value(value);
235
236                    if !update_expression_set.is_empty() {
237                        write!(update_expression_set, ", ").unwrap();
238                    }
239
240                    write!(update_expression_set, "{column} = {value}").unwrap();
241                }
242                // `stmt::add` / `stmt::subtract` / `stmt::increment` /
243                // `stmt::decrement` map to DynamoDB's `SET col = col + :v` /
244                // `SET col = col - :v` form, which atomically combines the
245                // bound value with the current attribute value.
246                AssignKind::Add | AssignKind::Subtract => {
247                    let op = if matches!(kind, AssignKind::Add) {
248                        "+"
249                    } else {
250                        "-"
251                    };
252                    let value = expr_attrs.value(value);
253
254                    if !update_expression_set.is_empty() {
255                        write!(update_expression_set, ", ").unwrap();
256                    }
257
258                    write!(update_expression_set, "{column} = {column} {op} {value}").unwrap();
259                }
260            }
261        }
262
263        let mut update_expression = String::new();
264
265        if !update_expression_set.is_empty() {
266            write!(update_expression, "SET {update_expression_set}").unwrap();
267        }
268
269        if !update_expression_remove.is_empty() {
270            write!(update_expression, " REMOVE {update_expression_remove}").unwrap();
271        }
272
273        // Build the returning row from the explicit column list the engine
274        // requested — exactly these columns, in this order. The engine inlines
275        // `Set` values at plan time and injects the engine-managed `#[version]`
276        // bump outside the returning projection, so neither appears in
277        // `op.returning`; the driver no longer has to infer the row shape from
278        // the assignments. Each column is seeded with a placeholder, then
279        // refreshed below from the `UPDATED_NEW` response with its post-update
280        // value.
281        let mut ret = vec![];
282        let mut refresh_after_update: Vec<(usize, &db::Column)> = vec![];
283
284        if let Some(columns) = &op.returning {
285            for column_id in columns {
286                let column = schema.column(*column_id);
287                let placeholder = bound_values
288                    .get(&column_id.index)
289                    .map(|value| (*value).clone())
290                    .unwrap_or(stmt::Value::Null);
291                refresh_after_update.push((ret.len(), column));
292                ret.push(placeholder);
293            }
294        }
295
296        // The seeded placeholders are not the post-update column values.
297        // Request `UPDATED_NEW` so the response carries the actual new
298        // attribute values and we can replace the placeholders below.
299        let needs_updated_new = !refresh_after_update.is_empty();
300
301        match &unique_indices[..] {
302            [] => {
303                // The engine shreds multi-key updates into one op per key, so a
304                // non-unique-index update always carries exactly one key.
305                let [key] = &op.keys[..] else {
306                    panic!("expected exactly 1 key, got {}", op.keys.len());
307                };
308
309                let res = self
310                    .client
311                    .update_item()
312                    .table_name(&table.name)
313                    .set_key(Some(ddb_key(table, key)))
314                    .set_update_expression(Some(update_expression))
315                    .set_expression_attribute_names(Some(expr_attrs.attr_names))
316                    .set_expression_attribute_values(if !expr_attrs.attr_values.is_empty() {
317                        Some(expr_attrs.attr_values)
318                    } else {
319                        None
320                    })
321                    .set_condition_expression(filter_expression)
322                    .return_values_on_condition_check_failure(
323                        ReturnValuesOnConditionCheckFailure::AllOld,
324                    )
325                    .set_return_values(needs_updated_new.then_some(ReturnValue::UpdatedNew))
326                    .send()
327                    .await;
328
329                let output = match res {
330                    Ok(output) => output,
331                    Err(SdkError::ServiceError(e)) => {
332                        if let UpdateItemError::ConditionalCheckFailedException(cce) = e.err() {
333                            return on_update_item_condition_failed(
334                                cce.item(),
335                                cce.message.as_deref(),
336                                table,
337                                op.filter.as_ref(),
338                                op.returning.is_some(),
339                            );
340                        }
341                        return Err(toasty_core::Error::driver_operation_failed(
342                            SdkError::ServiceError(e),
343                        ));
344                    }
345                    Err(other) => {
346                        return Err(toasty_core::Error::driver_operation_failed(other));
347                    }
348                };
349
350                if needs_updated_new && let Some(attrs) = output.attributes() {
351                    for (idx, column) in &refresh_after_update {
352                        if let Some(attr) = attrs.get(&column.name) {
353                            ret[*idx] = Value::from_ddb(&column.ty, attr).into_inner();
354                        }
355                    }
356                }
357            }
358            [index] => {
359                // Updating a unique-indexed column requires synchronizing the separate
360                // DynamoDB index table. Because DynamoDB has no native unique-constraint
361                // support, Toasty maintains a dedicated table per unique index and keeps
362                // it in sync manually.
363                //
364                // The sequence is:
365                //   1. Read the current unique column value(s) with get_item.
366                //   2. Compare against the incoming assignment to decide whether the
367                //      unique value actually changed.
368                //   3a. Unchanged → plain update_item; no index surgery needed.
369                //   3b. Changed (or first-time set) → transact_write_items that
370                //       atomically: updates the base table, deletes the old index entry
371                //       (if any), and inserts the new index entry with an
372                //       attribute_not_exists guard to enforce uniqueness.
373                //
374                // Concurrency contract:
375                //   The get_item in step 1 is NOT atomic with the write in step 3. A
376                //   concurrent writer could mutate the unique column between those two
377                //   operations.
378                //
379                //   Changed branch (3b): the base-table Update inside the transaction
380                //   carries `<unique_col> = :old_value` as its condition, so a
381                //   concurrent mutation is detected atomically at commit time and the
382                //   transaction is cancelled.
383                //
384                //   Unchanged branch (3a): the only concurrent-safety guarantee comes
385                //   from op.condition (e.g. a #[version] check). If the model has no
386                //   version field, a narrow ABA race is possible where a concurrent
387                //   writer changes the unique column away and back between the read and
388                //   the update_item, leaving the index in an inconsistent state. Models
389                //   with mutable unique fields should use #[version] to close this gap.
390                assert!(op.keys.len() == 1, "TODO: handle multiple keys");
391                let key = &op.keys[0];
392
393                let mut transact_items = vec![];
394
395                let attributes_to_get = index
396                    .columns
397                    .iter()
398                    .map(|index_column| index_column.table_column(schema).name.clone())
399                    .collect();
400
401                // Records that have had their unique values set initially
402                // (previously were null).
403                let mut set_unique_attrs = HashMap::new();
404                // Records that have had their unique attribute updated from a
405                // previous value.
406                let mut updated_unique_attrs = HashMap::new();
407
408                // Read the current unique column value(s) to determine whether index
409                // surgery is needed. Version is not fetched here; it is verified
410                // atomically at write time via op.condition.
411                let res = self
412                    .client
413                    .get_item()
414                    .table_name(&table.name)
415                    .set_key(Some(ddb_key(table, key)))
416                    .set_attributes_to_get(Some(attributes_to_get))
417                    .send()
418                    .await
419                    .map_err(toasty_core::Error::driver_operation_failed)?;
420
421                let Some(mut curr_unique_values) = res.item else {
422                    return Err(toasty_core::Error::record_not_found(format!(
423                        "table={} key={:?}",
424                        table.name, key
425                    )));
426                };
427
428                // Resolve each unique-column assignment to a concrete post-update
429                // value. `Set(v)` resolves to `v`; `Add(x)` / `Subtract(x)` resolve
430                // to `prev ± x` using the GET result above. The main-table update
431                // expression keeps the atomic `SET col = col ± :v` form, so
432                // atomicity is preserved by the transaction's `<col> = <prev>`
433                // condition below — if the column changed concurrently, the
434                // transaction is cancelled before the precomputed index entry is
435                // written.
436                let mut resolved_unique_values = HashMap::new();
437                for index_column in &index.columns {
438                    let column = index_column.table_column(schema);
439                    for (projection, assignment) in op.assignments.iter() {
440                        if *projection != column.id.index {
441                            continue;
442                        }
443                        let resolved = match assignment {
444                            stmt::Assignment::Set(stmt::Expr::Value(v)) => v.clone(),
445                            stmt::Assignment::Add(stmt::Expr::Value(delta))
446                            | stmt::Assignment::Subtract(stmt::Expr::Value(delta)) => {
447                                let prev = curr_unique_values.get(&column.name).ok_or_else(
448                                    || {
449                                        toasty_core::Error::invalid_statement(format!(
450                                            "arithmetic update on unique column {} requires an existing value",
451                                            column.name,
452                                        ))
453                                    },
454                                )?;
455                                let prev_value = Value::from_ddb(&column.ty, prev).into_inner();
456                                let result = match assignment {
457                                    stmt::Assignment::Add(_) => prev_value.checked_add(delta),
458                                    _ => prev_value.checked_sub(delta),
459                                };
460                                result.ok_or_else(|| {
461                                    toasty_core::Error::invalid_statement(format!(
462                                        "arithmetic overflow on unique column {}",
463                                        column.name,
464                                    ))
465                                })?
466                            }
467                            other => {
468                                return Err(toasty_core::Error::invalid_statement(format!(
469                                    "unsupported assignment on unique column {}: {other:#?}",
470                                    column.name,
471                                )));
472                            }
473                        };
474                        resolved_unique_values.insert(column.id, resolved);
475                    }
476                }
477
478                for index_column in &index.columns {
479                    let column = index_column.table_column(schema);
480
481                    for (projection, _) in op.assignments.iter() {
482                        if *projection != column.id.index {
483                            continue;
484                        }
485                        let new_value = &resolved_unique_values[&column.id];
486                        if let Some(prev) = curr_unique_values.remove(&column.name) {
487                            if Value::from_ddb(&column.ty, &prev).into_inner() != *new_value {
488                                updated_unique_attrs.insert(column.id, prev);
489                            }
490                        } else {
491                            set_unique_attrs.insert(column.id, ());
492                        }
493                    }
494                }
495
496                if updated_unique_attrs.is_empty() && set_unique_attrs.is_empty() {
497                    // The unique column appears in the assignment but its value is
498                    // unchanged — no index table surgery needed; just update the
499                    // main table directly.
500                    let res = self
501                        .client
502                        .update_item()
503                        .table_name(&table.name)
504                        .set_key(Some(ddb_key(table, key)))
505                        .set_update_expression(Some(update_expression))
506                        .set_expression_attribute_names(Some(expr_attrs.attr_names))
507                        .set_expression_attribute_values(if !expr_attrs.attr_values.is_empty() {
508                            Some(expr_attrs.attr_values)
509                        } else {
510                            None
511                        })
512                        .set_condition_expression(filter_expression)
513                        .return_values_on_condition_check_failure(
514                            ReturnValuesOnConditionCheckFailure::AllOld,
515                        )
516                        .send()
517                        .await;
518
519                    if let Err(SdkError::ServiceError(e)) = res {
520                        if let UpdateItemError::ConditionalCheckFailedException(cce) = e.err() {
521                            return on_update_item_condition_failed(
522                                cce.item(),
523                                cce.message.as_deref(),
524                                table,
525                                op.filter.as_ref(),
526                                op.returning.is_some(),
527                            );
528                        }
529                        return Err(toasty_core::Error::driver_operation_failed(
530                            SdkError::ServiceError(e),
531                        ));
532                    }
533                } else {
534                    assert!(
535                        updated_unique_attrs.len() + set_unique_attrs.len() == 1,
536                        "TODO: support more than one unique attr"
537                    );
538
539                    let mut condition_expression = String::new();
540
541                    for column_id in set_unique_attrs.keys() {
542                        let column = expr_attrs.column(schema.column(*column_id)).to_string();
543                        condition_expression = format!("attribute_not_exists({column})");
544                    }
545
546                    for (column_id, prev) in &updated_unique_attrs {
547                        let column = expr_attrs.column(schema.column(*column_id)).to_string();
548                        let value = expr_attrs.ddb_value(prev.clone());
549
550                        condition_expression = format!("{column} = {value}");
551                    }
552
553                    if let Some(filter_expression) = filter_expression {
554                        condition_expression.push_str(" AND ");
555                        condition_expression.push_str(&filter_expression);
556                    }
557
558                    transact_items.push(
559                        TransactWriteItem::builder()
560                            .update(
561                                Update::builder()
562                                    .table_name(&table.name)
563                                    .set_key(Some(ddb_key(table, key)))
564                                    .condition_expression(condition_expression)
565                                    .set_update_expression(Some(update_expression))
566                                    .set_expression_attribute_names(Some(expr_attrs.attr_names))
567                                    .set_expression_attribute_values(Some(expr_attrs.attr_values))
568                                    .build()
569                                    .unwrap(),
570                            )
571                            .build(),
572                    );
573
574                    for (column_id, prev) in &updated_unique_attrs {
575                        let name = &schema.column(*column_id).name;
576                        transact_items.push(
577                            TransactWriteItem::builder()
578                                .delete(
579                                    Delete::builder()
580                                        .table_name(&index.name)
581                                        .key(name.clone(), prev.clone())
582                                        .build()
583                                        .unwrap(),
584                                )
585                                .build(),
586                        );
587                    }
588
589                    for column_id in updated_unique_attrs.keys().chain(set_unique_attrs.keys()) {
590                        let name = &schema.column(*column_id).name;
591
592                        let mut index_insert_items = HashMap::new();
593
594                        for index_column in &index.columns {
595                            let column = index_column.table_column(schema);
596                            let value = &resolved_unique_values[column_id];
597                            if !value.is_null() {
598                                index_insert_items.insert(
599                                    column.name.clone(),
600                                    Value::from(value.clone()).to_ddb(),
601                                );
602                            }
603                        }
604
605                        // This will be empty if **unsetting** a unique attribute.
606                        if index_insert_items.is_empty() {
607                            continue;
608                        }
609
610                        for (index, column) in table.primary_key_columns().enumerate() {
611                            let key_field = match key {
612                                stmt::Value::Record(record) => &record[index],
613                                value => value,
614                            };
615                            index_insert_items.insert(
616                                column.name.clone(),
617                                Value::from(key_field.clone()).to_ddb(),
618                            );
619                        }
620
621                        let mut expression_names = HashMap::new();
622                        let expr_name = format!("#{name}");
623                        let condition_expression = format!("attribute_not_exists({expr_name})");
624                        expression_names.insert(expr_name, name.clone());
625
626                        transact_items.push(
627                            TransactWriteItem::builder()
628                                .put(
629                                    Put::builder()
630                                        .table_name(&index.name)
631                                        .set_item(Some(index_insert_items))
632                                        .condition_expression(condition_expression)
633                                        .set_expression_attribute_names(Some(expression_names))
634                                        .build()
635                                        .unwrap(),
636                                )
637                                .build(),
638                        );
639                    }
640
641                    let res = self
642                        .client
643                        .transact_write_items()
644                        .set_transact_items(Some(transact_items))
645                        .send()
646                        .await;
647
648                    if let Err(SdkError::ServiceError(e)) = res {
649                        if let TransactWriteItemsError::TransactionCanceledException(tce) = e.err()
650                        {
651                            return on_transaction_cancelled(
652                                tce.cancellation_reasons(),
653                                tce.message(),
654                                table,
655                                op.filter.as_ref(),
656                                op.returning.is_some(),
657                            );
658                        }
659                        return Err(toasty_core::Error::driver_operation_failed(
660                            SdkError::ServiceError(e),
661                        ));
662                    }
663                }
664            }
665            _ => todo!(),
666        }
667
668        // If we get here, then returning should be None
669        Ok(if op.returning.is_some() {
670            let values = stmt::ValueStream::from_value(stmt::Value::record_from_vec(ret));
671            ExecResponse::value_stream(values)
672        } else {
673            ExecResponse::count(op.keys.len() as _)
674        })
675    }
676}

717    /// Build `status = "active"` as a column-reference filter expression.
718    fn status_eq_active() -> Expr {
719        Expr::BinaryOp(ExprBinaryOp {
720            lhs: Box::new(Expr::Reference(ExprReference::Column(ExprColumn {
721                nesting: 0,
722                table: 0,
723                column: 0, // column 0 in the table → "status"
724            }))),
725            op: BinaryOp::Eq,
726            rhs: Box::new(Expr::Value(stmt::Value::String("active".to_string()))),
727        })
728    }
729
730    fn item_with_status(status: &str) -> HashMap<String, AttributeValue> {
731        HashMap::from([("status".to_string(), AttributeValue::S(status.to_string()))])
732    }
733
734    // No filter at all: the condition expression failed → caller should surface an error,
735    // not return count 0.  filter_failed must return false.

739        assert!(!filter_failed(None, &table, None));
740    }

742    // Filter present but item is missing (record was deleted between read and check):
743    // treat as "filter didn't match" → count 0.
744    #[test]

747        let filter = status_eq_active();

749    }

751    // Item present and filter matches: the filter was NOT the failing part, so the
752    // condition expression must have failed → return false (surface an error).

759    }

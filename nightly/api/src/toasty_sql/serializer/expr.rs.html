7impl ToSql for &stmt::Expr {
8    fn to_sql(self, f: &mut super::Formatter<'_>) {
9        match self {
10            stmt::Expr::And(expr) => {
11                fmt!(f, Delimited(expr.operands.iter().map(AndOperand), " AND "));
12            }
13            stmt::Expr::Between(expr) => {
14                fmt!(f, expr.expr " BETWEEN " expr.low " AND " expr.high);
15            }
16            stmt::Expr::BinaryOp(expr) => {
17                assert!(!expr.lhs.is_value_null());
18                assert!(!expr.rhs.is_value_null());
19
20                fmt!(f, expr.lhs " " expr.op " " expr.rhs);
21            }
22            stmt::Expr::Exists(expr) => {
23                f.depth += 1;
24                fmt!(f, "EXISTS (" expr.subquery ")");
25                f.depth -= 1;
26            }
27            stmt::Expr::Func(stmt::ExprFunc::Count(func)) => match (&func.arg, &func.filter) {
28                (None, None) => fmt!(f, "COUNT(*)"),
29                // Mysql does not support filters, so translate it to an expression
30                (None, Some(expr)) if f.serializer.is_mysql() => {
31                    fmt!(f, "COUNT(CASE WHEN " expr " THEN 1 END)")
32                }
33                (None, Some(expr)) => fmt!(f, "COUNT(*) FILTER (WHERE " expr ")"),
34                _ => todo!("func={func:#?}"),
35            },
36            stmt::Expr::Func(stmt::ExprFunc::LastInsertId(_)) => {
37                fmt!(f, "LAST_INSERT_ID()")
38            }
39            stmt::Expr::Func(stmt::ExprFunc::JsonExtract(func)) => {
40                serialize_json_extract(f, func);
41            }
42            stmt::Expr::Project(project)
43                if let stmt::Expr::Incoming(incoming) = project.base.as_ref() =>
44            {
45                let stmt::ExprIncoming::Table(table) = incoming else {
46                    panic!("incoming projection was not lowered")
47                };
48                let [column] = project.projection.as_slice() else {
49                    panic!("lowered incoming projection must reference one column")
50                };
51                let column = ColumnId {
52                    table: *table,
53                    index: *column,
54                };
55                fmt!(f, "excluded." f.serializer.column_name(column));
56            }
57            stmt::Expr::Incoming(_) => panic!("incoming row must be projected"),
58            stmt::Expr::IsSuperset(e) => match f.serializer.flavor {
59                Flavor::Postgresql => fmt!(f, e.lhs.as_ref() " @> " e.rhs.as_ref()),
60                // The rhs Value::List is bound as one JSON string. MySQL's
61                // `JSON_CONTAINS(target, candidate)` matches when every
62                // element of `candidate` appears in `target`.
63                Flavor::Mysql => {
64                    fmt!(f, "JSON_CONTAINS(" e.lhs.as_ref() ", " e.rhs.as_ref() ")")
65                }
66                // SQLite has no direct superset operator; emulate via
67                // `NOT EXISTS (rhs element with no match in lhs)`.
68                Flavor::Sqlite => fmt!(
69                    f,
70                    "NOT EXISTS (SELECT 1 FROM json_each(" e.rhs.as_ref()
71                    ") AS r WHERE r.value NOT IN (SELECT l.value FROM json_each("
72                    e.lhs.as_ref() ") AS l))"
73                ),
74            },
75            stmt::Expr::Intersects(e) => match f.serializer.flavor {
76                Flavor::Postgresql => fmt!(f, e.lhs.as_ref() " && " e.rhs.as_ref()),
77                Flavor::Mysql => {
78                    fmt!(f, "JSON_OVERLAPS(" e.lhs.as_ref() ", " e.rhs.as_ref() ")")
79                }
80                Flavor::Sqlite => fmt!(
81                    f,
82                    "EXISTS (SELECT 1 FROM json_each(" e.rhs.as_ref()
83                    ") AS r WHERE r.value IN (SELECT l.value FROM json_each("
84                    e.lhs.as_ref() ") AS l))"
85                ),
86            },
87            stmt::Expr::Length(e) => match f.serializer.flavor {
88                Flavor::Postgresql => fmt!(f, "cardinality(" e.expr.as_ref() ")"),
89                Flavor::Mysql => fmt!(f, "JSON_LENGTH(" e.expr.as_ref() ")"),
90                Flavor::Sqlite => fmt!(f, "json_array_length(" e.expr.as_ref() ")"),
91            },
92            stmt::Expr::Ident(name) => {
93                fmt!(f, Ident(name));
94            }
95            stmt::Expr::InList(expr) => {
96                fmt!(f, expr.expr " IN " expr.list);
97            }
98            stmt::Expr::AnyOp(expr) => match f.serializer.flavor {
99                // `value = ANY(col)` — PostgreSQL's array membership operator.
100                // Drives `Path::contains` for native-array columns and the
101                // IN-list rewrite.
102                Flavor::Postgresql => {
103                    fmt!(f, expr.lhs " " expr.op " ANY(" expr.rhs ")");
104                }
105                // MySQL's `value MEMBER OF (json_array)` (8.0.17+). Only the
106                // equality form makes sense; `Path::contains` is the only
107                // current emitter and the lowering pass never produces
108                // ANY on MySQL since `predicate_match_any` is false.
109                Flavor::Mysql if matches!(expr.op, stmt::BinaryOp::Eq) => {
110                    fmt!(f, expr.lhs " MEMBER OF (" expr.rhs ")");
111                }
112                Flavor::Mysql => unreachable!("AnyOp with non-Eq operator on MySQL: {expr:?}"),
113                // SQLite renders `value = ANY(col)` (i.e. `Path::contains`)
114                // as `value IN (SELECT value FROM json_each(col))`.
115                Flavor::Sqlite if matches!(expr.op, stmt::BinaryOp::Eq) => {
116                    fmt!(
117                        f,
118                        expr.lhs " IN (SELECT value FROM json_each(" expr.rhs "))"
119                    );
120                }
121                Flavor::Sqlite => unreachable!("AnyOp with non-Eq operator on SQLite: {expr:?}"),
122            },
123            stmt::Expr::AllOp(expr) => {
124                fmt!(f, expr.lhs " " expr.op " ALL(" expr.rhs ")");
125            }
126            stmt::Expr::InSubquery(expr) => {
127                fmt!(f, expr.expr " IN (" expr.query ")");
128            }
129            stmt::Expr::IsNull(expr) => {
130                fmt!(f, expr.expr " IS NULL");
131            }
132            stmt::Expr::Like(expr) => {
133                let op =
134                    if expr.case_insensitive && matches!(f.serializer.flavor, Flavor::Postgresql) {
135                        " ILIKE "
136                    } else {
137                        " LIKE "
138                    };
139                fmt!(f, expr.expr op expr.pattern);
140                if let Some(escape) = expr.escape {
141                    let escape = if f.serializer.is_mysql() && escape == '\\' {
142                        stmt::Value::String("\\\\".to_string())
143                    } else {
144                        stmt::Value::String(escape.to_string())
145                    };
146                    let escape = &escape;
147                    fmt!(f, " ESCAPE " escape);
148                }
149            }
150            stmt::Expr::StartsWith(expr) => {
151                match f.serializer.flavor {
152                    // PostgreSQL's `^@` prefix-match operator; prefix is bound
153                    // as a plain string parameter.
154                    Flavor::Postgresql => {
155                        fmt!(f, expr.expr " ^@ " expr.prefix);
156                    }
157                    // SQLite GLOB is case-sensitive.  extract_params has already
158                    // escaped GLOB metacharacters and appended `*` to the prefix
159                    // parameter, so we only need to emit the right operator.
160                    Flavor::Sqlite => {
161                        fmt!(f, expr.expr " GLOB " expr.prefix);
162                    }
163                    // MySQL LIKE is case-insensitive by default; casting the
164                    // column side to BINARY forces a case-sensitive byte
165                    // comparison.  extract_params has escaped `%`/`_`/`!` and
166                    // appended `%` to the prefix parameter.
167                    Flavor::Mysql => {
168                        fmt!(f, "BINARY " expr.expr " LIKE " expr.prefix " ESCAPE '!'");
169                    }
170                }
171            }
172            stmt::Expr::Not(expr) => {
173                fmt!(f, "NOT (" expr.expr ")");
174            }
175            stmt::Expr::Or(expr) => {
176                fmt!(f, Delimited(&expr.operands, " OR "));
177            }
178            stmt::Expr::Record(expr) => {
179                let fields = Comma(expr.fields.iter());
180                fmt!(f, "(" fields ")");
181            }
182            stmt::Expr::Reference(expr_reference @ stmt::ExprReference::Column(expr_column)) => {
183                if f.alias {
184                    let depth = f.depth - expr_column.nesting;
185
186                    match f.cx.resolve_expr_reference(expr_reference) {
187                        ResolvedRef::Column(column) => {
188                            let name = Ident(&column.name);
189                            fmt!(f, "tbl_" depth "_" expr_column.table "." name)
190                        }
191                        ResolvedRef::Cte { .. } | ResolvedRef::Derived(_) => {
192                            fmt!(f, "tbl_" depth "_" expr_column.table "." ColumnAlias(expr_column.column))
193                        }
194                        ResolvedRef::Model(model) => {
195                            panic!("Model references cannot be serialized to SQL; model={model:?}")
196                        }
197                        ResolvedRef::Field(field) => {
198                            panic!("Field references cannot be serialized to SQL; field={field:?}")
199                        }
200                    }
201                } else {
202                    let column =
203                        f.cx.resolve_expr_reference(expr_reference)
204                            .as_column_unwrap();
205                    if matches!(f.serializer.flavor, Flavor::Postgresql)
206                        && expr_column.nesting == 0
207                        && f.assignment_table == Some(column.id.table)
208                    {
209                        fmt!(f, f.serializer.table_name(column.id.table) "." Ident(&column.name))
210                    } else {
211                        fmt!(f, Ident(&column.name))
212                    }
213                }
214            }
215            stmt::Expr::Stmt(expr) => {
216                let stmt = &*expr.stmt;
217                fmt!(f, "(" stmt ")");
218            }
219            stmt::Expr::List(expr) => {
220                let items = Comma(expr.items.iter());
221                fmt!(f, "(" items ")");
222            }
223            stmt::Expr::Value(expr) => expr.to_sql(f),
224            // Schema-fixed leaf rendered inline as a SQL literal.  Reuses
225            // the same `Value::to_sql` path the inline DDL serializer uses,
226            // which escapes `String` defensively.
227            stmt::Expr::Static(expr) => expr.to_sql(f),
228            stmt::Expr::Arg(arg) => {
229                // Pre-extracted bind parameter placeholder — render as a
230                // positional parameter. The arg position is 0-based; the
231                // placeholder is 1-based.
232                f.arg_positions.push(arg.position);
233                let placeholder = super::Placeholder(arg.position + 1);
234                fmt!(f, placeholder);
235            }
236            stmt::Expr::Default => match f.serializer.flavor {
237                Flavor::Postgresql | Flavor::Mysql => fmt!(f, "DEFAULT"),
238                // SQLite does not support the DEFAULT keyword but NULL acts similarly.
239                Flavor::Sqlite => fmt!(f, "NULL"),
240            },
241            _ => todo!("expr={:#?}", self),
242        }
243    }
244}

246/// A single operand of an `AND` chain.
247///
248/// `OR` binds looser than `AND` in SQL, so an `Or` operand must be
249/// parenthesized: `a AND (b OR c)` would otherwise serialize as
250/// `a AND b OR c`, which parses as `(a AND b) OR c` and silently changes the
251/// query's meaning. Operands of other kinds bind at least as tightly as `AND`
252/// (comparisons, `IS NULL`, `NOT (..)`, nested `AND`), so they need no parens.
253struct AndOperand<'a>(&'a stmt::Expr);
254
255impl ToSql for AndOperand<'_> {
256    fn to_sql(self, f: &mut super::Formatter<'_>) {
257        if matches!(self.0, stmt::Expr::Or(_)) {
258            fmt!(f, "(" self.0 ")");
259        } else {
260            fmt!(f, self.0);
261        }
262    }
263}

265/// Serializes a document path extraction per dialect: `json_extract(col,
266/// '$.a.b')` on SQLite, `CAST(JSON_UNQUOTE(JSON_EXTRACT(col, '$.a.b')) AS ...)`
267/// on MySQL, and `(col->'a'->>'b')::cast` on PostgreSQL — the latter two unwrap
268/// the leaf to text and cast it to match the bound parameter's type.
269fn serialize_json_extract(f: &mut super::Formatter<'_>, func: &stmt::FuncJsonExtract) {
270    match f.serializer.flavor {
271        Flavor::Sqlite => {
272            // SQLite's `json_extract` returns SQL-native scalars (unquoted text,
273            // integers, reals), so a path read compares directly against a bound
274            // parameter with no cast. The path is a single-quoted JSONPath like
275            // `$.a.b`.
276            fmt!(
277                f,
278                "json_extract(" func.base.as_ref() ", '$"
279                Delimited(func.path.iter().map(|key| (".", key.as_str())), "")
280                "')"
281            );
282        }
283        Flavor::Mysql => serialize_mysql_json_extract(f, func),
284        Flavor::Postgresql => {
285            // Descend with `->`, take the leaf as text with `->>`, then cast the
286            // text to the leaf type so it compares against a bound parameter.
287            let (leaf, parents) = func
288                .path
289                .split_last()
290                .expect("json extract path has at least one key");
291            fmt!(
292                f,
293                "(" func.base.as_ref()
294                Delimited(parents.iter().map(|key| ("->'", key.as_str(), "'")), "")
295                "->>'" leaf.as_str() "')"
296                pg_json_cast(&func.ty).map(|cast| ("::", cast))
297            );
298        }
299    }
300}

302/// Serializes a MySQL document path read. `JSON_EXTRACT` yields a *JSON-typed*
303/// value, which compares against a bound SQL parameter only by luck — a JSON
304/// string (e.g. an ISO timestamp) never equals a native `DATETIME`, and JSON
305/// string comparison is `utf8mb4_bin` (case-sensitive), unlike a `VARCHAR`
306/// column. So unwrap the leaf to text with `JSON_UNQUOTE` and `CAST` it to the
307/// leaf's SQL type (`CHAR` for strings, to recover a `VARCHAR`-matching
308/// collation — see [`mysql_json_cast`]), mirroring the `->>`-plus-cast
309/// PostgreSQL path. Booleans are the exception: the unquoted text
310/// `'true'`/`'false'` casts to `0`, so cast the *bare* JSON boolean to
311/// `UNSIGNED` instead (`true` -> 1, `false` -> 0), matching a bound bool param.
312fn serialize_mysql_json_extract(f: &mut super::Formatter<'_>, func: &stmt::FuncJsonExtract) {
313    if matches!(func.ty, stmt::Type::Bool) {
314        fmt!(f, "CAST(");
315        mysql_json_extract(f, func);
316        fmt!(f, " AS UNSIGNED)");
317    } else if let Some(cast) = mysql_json_cast(&func.ty) {
318        fmt!(f, "CAST(JSON_UNQUOTE(");
319        mysql_json_extract(f, func);
320        fmt!(f, ") AS " cast ")");
321    } else {
322        fmt!(f, "JSON_UNQUOTE(");
323        mysql_json_extract(f, func);
324        fmt!(f, ")");
325    }
326}

328/// Emits the bare `JSON_EXTRACT(col, '$.a.b')` every MySQL path read is built
329/// on, with the path as a single-quoted JSONPath argument.
330fn mysql_json_extract(f: &mut super::Formatter<'_>, func: &stmt::FuncJsonExtract) {
331    fmt!(
332        f,
333        "JSON_EXTRACT(" func.base.as_ref() ", '$"
334        Delimited(func.path.iter().map(|key| (".", key.as_str())), "")
335        "')"
336    );
337}

339/// The MySQL `CAST(... AS <type>)` target wrapped around a
340/// `JSON_UNQUOTE(JSON_EXTRACT(...))` text extraction so it compares against a
341/// bound parameter of the leaf type. Mirrors [`pg_json_cast`]; `None` leaves the
342/// extraction as bare unquoted text (only floats, which compare via numeric
343/// coercion). `Bool` is absent because it is cast separately — see
344/// [`serialize_mysql_json_extract`].
345///
346/// `String`/`Uuid` cast to `CHAR` not for the type but for the *collation*:
347/// `JSON_UNQUOTE` yields `utf8mb4_bin` (case-sensitive), while `CAST(... AS
348/// CHAR)` adopts the connection's default collation — the same one a bound
349/// literal and a `VARCHAR` column use — so a string filter on a document leaf
350/// matches the case sensitivity of a plain column. `AS CHAR` (no length) does
351/// not truncate, and inheriting the server default keeps it portable across
352/// server collation configs rather than hardcoding a collation name.
353///
354/// The temporal targets carry `(6)` precision: a bare `CAST(... AS DATETIME)`
355/// truncates to whole seconds, which would drop the microseconds the JSON codec
356/// writes and break an equality filter on a sub-second value.
357fn mysql_json_cast(ty: &stmt::Type) -> Option<&'static str> {
358    use crate::stmt::Type;
359    Some(match ty {
360        Type::String | Type::Uuid => "CHAR",
361        Type::I8 | Type::I16 | Type::I32 | Type::I64 => "SIGNED",
362        Type::U8 | Type::U16 | Type::U32 | Type::U64 => "UNSIGNED",
363        #[cfg(feature = "rust_decimal")]
364        Type::Decimal => "DECIMAL(65, 30)",
365        #[cfg(feature = "bigdecimal")]
366        Type::BigDecimal => "DECIMAL(65, 30)",
367        #[cfg(feature = "jiff")]
368        Type::Timestamp => "DATETIME(6)",
369        #[cfg(feature = "jiff")]
370        Type::Date => "DATE",
371        #[cfg(feature = "jiff")]
372        Type::Time => "TIME(6)",
373        #[cfg(feature = "jiff")]
374        Type::DateTime => "DATETIME(6)",
375        _ => return None,
376    })
377}

379/// The PostgreSQL cast applied to a `->>'` text extraction so it compares
380/// against a bound parameter of the leaf type. `String` (and any non-scalar
381/// leaf) needs no cast — `->>` already yields text.
382///
383/// Every scalar a `#[document]` leaf can hold must appear here: an unlisted
384/// scalar falls through to `None` and renders as an *uncast* text extraction,
385/// which PostgreSQL then refuses to compare against a typed parameter
386/// (`operator does not exist: text = ...`). The temporal casts pair with the
387/// microsecond-truncated text the JSON codec writes (see `toasty_sql::json`),
388/// so the extracted value parses cleanly into the SQL temporal type. `Zoned` is
389/// intentionally absent: it is rejected at schema-build because jiff renders it
390/// with an RFC 9557 `[IANA]` annotation that no PostgreSQL cast can parse.
391fn pg_json_cast(ty: &stmt::Type) -> Option<&'static str> {
392    use crate::stmt::Type;
393    Some(match ty {
394        Type::Bool => "boolean",
395        Type::I8 | Type::I16 | Type::I32 | Type::I64 => "bigint",
396        Type::U8 | Type::U16 | Type::U32 | Type::U64 => "bigint",
397        Type::F32 | Type::F64 => "double precision",
398        Type::Uuid => "uuid",
399        #[cfg(feature = "rust_decimal")]
400        Type::Decimal => "numeric",
401        #[cfg(feature = "bigdecimal")]
402        Type::BigDecimal => "numeric",
403        #[cfg(feature = "jiff")]
404        Type::Timestamp => "timestamptz",
405        #[cfg(feature = "jiff")]
406        Type::Date => "date",
407        #[cfg(feature = "jiff")]
408        Type::Time => "time",
409        #[cfg(feature = "jiff")]
410        Type::DateTime => "timestamp",
411        _ => return None,
412    })
413}

415impl ToSql for &stmt::BinaryOp {
416    fn to_sql(self, f: &mut super::Formatter<'_>) {
417        f.dst.push_str(match self {
418            stmt::BinaryOp::Eq => "=",
419            stmt::BinaryOp::Gt => ">",
420            stmt::BinaryOp::Ge => ">=",
421            stmt::BinaryOp::Lt => "<",
422            stmt::BinaryOp::Le => "<=",
423            stmt::BinaryOp::Ne => "<>",
424            stmt::BinaryOp::Add => "+",
425            stmt::BinaryOp::Sub => "-",
426        })
427    }

1use super::{Flavor, ToSql};
2
3use toasty_core::schema::db;
4
5impl ToSql for &db::Type {
6    fn to_sql(self, f: &mut super::Formatter<'_>) {
7        match self {
8            db::Type::Boolean => fmt!(f, "BOOLEAN"),
9            db::Type::Integer(1..=2) => fmt!(f, "SMALLINT"),
10            db::Type::Integer(3..=4) => fmt!(f, "INTEGER"),
11            db::Type::Integer(5..=8) => fmt!(f, "BIGINT"),
12            db::Type::Integer(_) => todo!(),
13            db::Type::UnsignedInteger(size) => {
14                match f.serializer.flavor {
15                    Flavor::Mysql => match size {
16                        1 => fmt!(f, "TINYINT UNSIGNED"),
17                        2 => fmt!(f, "SMALLINT UNSIGNED"),
18                        3..=4 => fmt!(f, "INT UNSIGNED"),
19                        5..=8 => fmt!(f, "BIGINT UNSIGNED"),
20                        _ => todo!("Unsupported unsigned integer size: {}", size),
21                    },
22                    Flavor::Postgresql => {
23                        match size {
24                            1 => fmt!(f, "SMALLINT"),   // u8 -> SMALLINT (i16)
25                            2 => fmt!(f, "INTEGER"),    // u16 -> INTEGER (i32)
26                            3..=4 => fmt!(f, "BIGINT"), // u32 -> BIGINT (i64)
27                            5..=8 => fmt!(f, "BIGINT"), // u64 -> BIGINT (i64) with capability limits
28                            _ => todo!("Unsupported unsigned integer size: {}", size),
29                        }
30                    }
31                    Flavor::Sqlite => {
32                        // SQLite uses INTEGER for all integer types
33                        fmt!(f, "INTEGER")
34                    }
35                }
36            }
37            db::Type::Float(size) => match f.serializer.flavor {
38                Flavor::Sqlite => fmt!(f, "REAL"),
39                Flavor::Postgresql => {
40                    if *size <= 4 {
41                        fmt!(f, "REAL")
42                    } else {
43                        fmt!(f, "DOUBLE PRECISION")
44                    }
45                }
46                Flavor::Mysql => {
47                    if *size <= 4 {
48                        fmt!(f, "FLOAT")
49                    } else {
50                        fmt!(f, "DOUBLE")
51                    }
52                }
53            },
54            db::Type::Text => fmt!(f, "TEXT"),
55            db::Type::VarChar(size) => fmt!(f, "VARCHAR(" size ")"),
56            db::Type::Uuid => {
57                fmt!(
58                    f,
59                    match f.serializer.flavor {
60                        Flavor::Postgresql => "UUID",
61                        _ => todo!("Unsupported type UUID"),
62                    }
63                );
64            }
65            db::Type::Numeric(None) => match f.serializer.flavor {
66                Flavor::Postgresql => fmt!(f, "NUMERIC"),
67                Flavor::Mysql => todo!(
68                    "MySQL does not support arbitrary-precision NUMERIC; precision and scale must be specified"
69                ),
70                Flavor::Sqlite => todo!("SQLite does not support NUMERIC type"),
71            },
72            db::Type::Numeric(Some((precision, scale))) => match f.serializer.flavor {
73                Flavor::Postgresql => fmt!(f, "NUMERIC(" precision ", " scale ")"),
74                Flavor::Mysql => fmt!(f, "DECIMAL(" precision ", " scale ")"),
75                Flavor::Sqlite => todo!("SQLite does not support NUMERIC type"),
76            },
77            db::Type::Binary(size) => match f.serializer.flavor {
78                Flavor::Mysql => fmt!(f, "BINARY(" size ")"),
79                _ => todo!("Unsupported fixed size binary type"),
80            },
81            db::Type::Blob => match f.serializer.flavor {
82                Flavor::Postgresql => fmt!(f, "BYTEA"),
83                Flavor::Mysql => fmt!(f, "BLOB"),
84                Flavor::Sqlite => fmt!(f, "BLOB"),
85            },
86            db::Type::Timestamp(precision) => match f.serializer.flavor {
87                Flavor::Postgresql => fmt!(f, "TIMESTAMPTZ(" precision ")"),
88                Flavor::Mysql => fmt!(f, "TIMESTAMP(" precision ")"),
89                Flavor::Sqlite => todo!("SQLite does not support Timestamp"),
90            },
91            db::Type::Date => match f.serializer.flavor {
92                Flavor::Postgresql | Flavor::Mysql => fmt!(f, "DATE"),
93                Flavor::Sqlite => todo!("SQLite does not support Date"),
94            },
95            db::Type::Time(precision) => match f.serializer.flavor {
96                Flavor::Postgresql => fmt!(f, "TIME(" precision ")"),
97                Flavor::Mysql => fmt!(f, "TIME(" precision ")"),
98                Flavor::Sqlite => todo!("SQLite does not support Time"),
99            },
100            db::Type::DateTime(precision) => match f.serializer.flavor {
101                Flavor::Postgresql => fmt!(f, "TIMESTAMP(" precision ")"),
102                Flavor::Mysql => fmt!(f, "DATETIME(" precision ")"),
103                Flavor::Sqlite => todo!("SQLite does not support DateTime"),
104            },
105            db::Type::Enum(type_enum) => match f.serializer.flavor {
106                // PostgreSQL: reference the named enum type created with CREATE TYPE.
107                Flavor::Postgresql => {
108                    let name = type_enum
109                        .name
110                        .as_deref()
111                        .expect("PostgreSQL enums require a type name");
112                    fmt!(f, name);
113                }
114                // MySQL: inline ENUM('label1', 'label2', ...) column type.
115                Flavor::Mysql => {
116                    use toasty_core::stmt::Value;
117
118                    f.dst.push_str("ENUM(");
119                    for (i, variant) in type_enum.variants.iter().enumerate() {
120                        if i > 0 {
121                            f.dst.push_str(", ");
122                        }
123                        Value::String(variant.name.clone()).to_sql(f);
124                    }
125                    f.dst.push(')');
126                }
127                // SQLite: TEXT column (CHECK constraint added in ColumnDef).
128                Flavor::Sqlite => fmt!(f, "TEXT"),
129            },
130            db::Type::List(elem) => match f.serializer.flavor {
131                Flavor::Postgresql => fmt!(f, elem.as_ref() "[]"),
132                // MySQL stores `Vec<scalar>` as a JSON document; SQLite uses
133                // TEXT (JSON1 functions operate on either, but TEXT is the
134                // idiomatic affinity). The element type is tracked by the
135                // engine — it doesn't surface in the column DDL.
136                Flavor::Mysql => fmt!(f, "JSON"),
137                Flavor::Sqlite => fmt!(f, "TEXT"),
138            },
139            db::Type::Document { binary } => match f.serializer.flavor {
140                // `binary` selects `jsonb` over `json` on PostgreSQL; the text
141                // encoding (`#[document(text)]`) is not yet wired up.
142                Flavor::Postgresql if *binary => fmt!(f, "JSONB"),
143                Flavor::Postgresql => fmt!(f, "JSON"),
144                Flavor::Mysql => fmt!(f, "JSON"),
145                Flavor::Sqlite => fmt!(f, "TEXT"),
146            },
147            db::Type::Custom(custom) => fmt!(f, custom.as_str()),
148        }
149    }
150}

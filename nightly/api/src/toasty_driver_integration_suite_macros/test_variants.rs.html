9/// Input to the `generate_driver_test_variants!` macro
10/// Expects: driver_expr, crate, module_name::test_name, #[driver_test(...)], attrs(...), capability(...)
11#[derive(Debug)]
12struct Input {
13    /// Crate
14    krate: syn::Path,
15
16    /// The full test path (module::test_name)
17    test_path: syn::Path,
18
19    // The driver expression
20    driver_expr: syn::Expr,
21
22    /// The driver_test attribute
23    driver_test_attr: syn::Attribute,
24
25    /// Extra attributes (e.g., #[should_panic], #[ignore])
26    extra_attrs: Vec<syn::Attribute>,
27
28    /// Parsed capability flags
29    capabilities: HashMap<syn::Ident, bool>,
30}
31
32impl syn::parse::Parse for Input {
33    fn parse(input: syn::parse::ParseStream) -> syn::Result<Self> {
34        let driver_expr: syn::Expr = input.parse()?;
35        input.parse::<Token![,]>()?;
36
37        let krate: syn::Path = input.parse()?;
38        input.parse::<Token![,]>()?;
39
40        let test_path: syn::Path = input.parse()?;
41        input.parse::<Token![,]>()?;
42
43        // Parse a single attribute using Attribute::parse_outer
44        let attrs = syn::Attribute::parse_outer(input)?;
45        let driver_test_attr = attrs
46            .into_iter()
47            .next()
48            .ok_or_else(|| input.error("expected driver_test attribute"))?;
49
50        input.parse::<Token![,]>()?;
51
52        // Parse attrs[...]
53        let attrs_ident: syn::Ident = input.parse()?;
54        if attrs_ident != "attrs" {
55            return Err(input.error("expected 'attrs'"));
56        }
57
58        let attrs_content;
59        syn::bracketed!(attrs_content in input);
60
61        // Parse attributes from the bracketed content
62        let extra_attrs = syn::Attribute::parse_outer(&attrs_content)?;
63
64        input.parse::<Token![,]>()?;
65
66        // Parse capability(...)
67        let capability_ident: syn::Ident = input.parse()?;
68        if capability_ident != "capability" {
69            return Err(input.error("expected 'capability'"));
70        }
71
72        let content;
73        syn::parenthesized!(content in input);
74
75        let mut capabilities = HashMap::new();
76        while !content.is_empty() {
77            let key: syn::Ident = content.parse()?;
78            content.parse::<Token![:]>()?;
79            let lit: syn::LitBool = content.parse()?;
80
81            capabilities.insert(key, lit.value);
82
83            // Parse trailing comma if present
84            if content.peek(Token![,]) {
85                content.parse::<Token![,]>()?;
86            }
87        }
88
89        Ok(Input {
90            driver_expr,
91            krate,
92            test_path,
93            driver_test_attr,
94            extra_attrs,
95            capabilities,
96        })
97    }
98}

100impl Input {
101    /// Convert capabilities from HashMap<Ident, bool> to HashMap<String, bool>
102    fn capabilities_as_strings(&self) -> HashMap<String, bool> {
103        self.capabilities
104            .iter()
105            .map(|(k, v)| (k.to_string(), *v))
106            .collect()
107    }
108
109    /// Get the base test name (last segment of the test path)
110    fn test_name(&self) -> &syn::Ident {
111        &self.test_path.segments.last().unwrap().ident
112    }
113
114    /// Create a capability lookup closure for use with Expansion::should_include
115    fn capability_checker(&self) -> impl Fn(&str) -> ThreeValuedBool + '_ {
116        let capabilities = self.capabilities_as_strings();
117        move |ident: &str| {
118            match capabilities.get(ident) {
119                Some(true) => ThreeValuedBool::True,
120                Some(false) => ThreeValuedBool::False,
121                None => ThreeValuedBool::True, // Default to true if not specified
122            }
123        }
124    }
125
126    /// Generate the test function name and path for a given expansion
127    fn generate_test_name_and_path(
128        &self,
129        expansion: &crate::parse::Expansion,
130    ) -> (syn::Ident, proc_macro2::TokenStream) {
131        let krate = &self.krate;
132        let test_path = &self.test_path;
133
134        if let Some(expansion_ident) = expansion.to_ident() {
135            // Has expansion: the function is in a module, call module::expansion_name
136            // Test function name is just the expansion name (e.g., "id_uuid")
137            let fn_path = quote! { #krate::tests::#test_path::#expansion_ident };
138            (expansion_ident, fn_path)
139        } else {
140            // No expansion: call the function directly with its original name
141            let base_test_name = self.test_name().clone();
142            let fn_path = quote! { #krate::tests::#test_path };
143            (base_test_name, fn_path)
144        }
145    }
146}

148pub fn expand(input: TokenStream) -> TokenStream {
149    let input = parse_macro_input!(input as Input);
150
151    let driver_expr = &input.driver_expr;
152    let krate = &input.krate;
153    let extra_attrs = &input.extra_attrs;
154
155    // Parse the driver_test attribute to get the expansions
156    let attr = DriverTestAttr::from_attribute(&input.driver_test_attr)
157        .expect("Failed to parse driver_test attribute");
158
159    // Generate expansions
160    let expansions = crate::parse::DriverTest::generate_expansions(&attr);
161
162    // Create capability checker closure
163    let capability_checker = input.capability_checker();
164
165    // Generate test functions for each expansion that passes the predicate
166    let test_functions: Vec<_> = expansions
167        .iter()
168        .filter_map(|expansion| {
169            // Evaluate the predicate for this expansion using should_include
170            if !expansion.should_include(&capability_checker) {
171                return None; // Skip this expansion
172            }
173
174            // Generate the test function name and path
175            let (test_fn_name, fn_path) = input.generate_test_name_and_path(expansion);
176
177            // Optionally emit set_serial(true) for serial tests
178            let serial_call = if attr.serial {
179                quote! { test.set_serial(true); }
180            } else {
181                quote! {}
182            };
183
184            // Generate the test function with extra attributes
185            Some(quote! {
186                #(#extra_attrs)*
187                #[test]
188                fn #test_fn_name() {
189                    let mut test = #krate::Test::new(
190                        ::std::sync::Arc::new(#driver_expr)
191                    );
192
193                    #serial_call
194                    test.run(async move |t| { #fn_path(t).await });
195                }
196            })
197        })
198        .collect();
199
200    // Check if we need to wrap test functions in a module
201    if expansions.needs_module_wrapper() {
202        // Wrap test functions in a module named after the test
203        let test_module_name = input.test_name();
204        quote! {
205            mod #test_module_name {
206                use super::*;
207                #(#test_functions)*
208            }
209        }
210        .into()
211    } else {
212        // Single test with no expansion - no module wrapper needed
213        quote! {
214            #(#test_functions)*
215        }
216        .into()
217    }
218}

8pub fn expand(attr: TokenStream, item: TokenStream) -> TokenStream {
9    let input = parse_macro_input!(item as ItemFn);
10    let attr = parse_macro_input!(attr as DriverTestAttr);
11
12    // Parse the driver test using shared logic
13    let driver_test = DriverTest::from_item_fn(input, attr);
14
15    let mod_name = &driver_test.name;
16    let vis = &driver_test.input.vis;
17
18    assert!(
19        !driver_test.expansions.is_empty(),
20        "driver_test={driver_test:#?}"
21    );
22
23    // Check if we need to wrap variants in a module
24    if driver_test.expansions.needs_module_wrapper() {
25        // Generate variants using expansion logic
26        // When inside a module, use the expansion name as the function name
27        let variant_fns: Vec<_> = driver_test
28            .expansions
29            .iter()
30            .map(|expansion| {
31                generate_variant(
32                    &driver_test.input,
33                    expansion,
34                    &driver_test.requires,
35                    driver_test.attr.scenario.as_ref(),
36                    true,
37                )
38            })
39            .collect();
40
41        quote! {
42            #vis mod #mod_name {
43                use super::*;
44
45                #(#variant_fns)*
46            }
47        }
48        .into()
49    } else {
50        // Single expansion with no name - return the function directly
51        let variant = generate_variant(
52            &driver_test.input,
53            &driver_test.expansions[0],
54            &driver_test.requires,
55            driver_test.attr.scenario.as_ref(),
56            false, // Don't use expansion name as function name
57        );
58        quote! {
59            #variant
60        }
61        .into()
62    }
63}

65/// Generate a test variant with ID rewritten to the target type
66fn generate_variant(
67    input: &ItemFn,
68    expansion: &Expansion,
69    requires: &Option<BoolExpr>,
70    scenario: Option<&syn::Path>,
71    use_expansion_name: bool,
72) -> ItemFn {
73    let mut variant = input.clone();
74
75    // Update function name based on whether we're inside a module
76    if use_expansion_name && let Some(expansion_ident) = expansion.to_ident() {
77        // Inside a module: use just the expansion name (e.g., "id_uuid")
78        variant.sig.ident = expansion_ident;
79    }
80    // Otherwise keep the original function name
81
82    // Don't add #[tokio::test] or #[test] attributes - the test registry in the consuming
83    // crate will add them. If we add test attributes here, the functions become test-only
84    // items that aren't accessible as regular library functions.
85
86    // Process driver_test_cfg attributes
87    process_driver_test_cfg_attrs(&mut variant, expansion);
88
89    // Rewrite driver_test_cfg! macro calls to boolean literals
90    rewrite_driver_test_cfg_macros(&mut variant, expansion);
91
92    // Rewrite ID types if expansion has an ID variant
93    if let (Some(id_ident), Some(id_variant)) = (&expansion.id_ident, &expansion.id_variant) {
94        let target_type = match id_variant {
95            crate::parse::KindVariant::IdU64 => syn::parse_quote!(u64),
96            crate::parse::KindVariant::IdUuid => syn::parse_quote!(uuid::Uuid),
97        };
98        let mut rewriter = IdRewriter::new(id_ident, target_type);
99        rewriter.visit_item_fn_mut(&mut variant);
100    }
101
102    // Inject scenario use-import if specified
103    if let Some(scenario_path) = scenario {
104        let use_stmt: syn::Stmt = if let Some(ref id_variant) = expansion.id_variant {
105            let variant_ident = syn::Ident::new(id_variant.name(), proc_macro2::Span::call_site());
106            syn::parse_quote! {
107                use #scenario_path::#variant_ident::*;
108            }
109        } else {
110            syn::parse_quote! {
111                use #scenario_path::*;
112            }
113        };
114        variant.block.stmts.insert(0, use_stmt);
115    }
116
117    // Add capability checks at the beginning of the function if there are requires
118    if let Some(requires_expr) = requires {
119        add_capability_checks_from_expr(&mut variant, requires_expr, expansion);
120    }
121
122    variant
123}

125/// Process driver_test_cfg attributes, either keeping or removing them based on the expansion
126fn process_driver_test_cfg_attrs(func: &mut ItemFn, expansion: &Expansion) {
127    // Process attributes in the function body (on items like struct definitions)
128    struct AttrProcessor<'a> {
129        expansion: &'a Expansion,
130    }
131
132    impl<'a> VisitMut for AttrProcessor<'a> {
133        fn visit_item_struct_mut(&mut self, node: &mut syn::ItemStruct) {
134            process_attrs(&mut node.attrs, self.expansion);
135            syn::visit_mut::visit_item_struct_mut(self, node);
136        }
137
138        fn visit_field_mut(&mut self, node: &mut syn::Field) {
139            process_attrs(&mut node.attrs, self.expansion);
140            syn::visit_mut::visit_field_mut(self, node);
141        }
142    }
143
144    let mut processor = AttrProcessor { expansion };
145    processor.visit_item_fn_mut(func);
146}

148/// Process attributes for a single item
149fn process_attrs(attrs: &mut Vec<syn::Attribute>, expansion: &Expansion) {
150    let mut new_attrs = Vec::new();
151
152    for attr in attrs.drain(..) {
153        if attr.path().is_ident("driver_test_cfg") {
154            // Parse driver_test_cfg(condition, attr)
155            // We expect: #[driver_test_cfg(condition, attr(...))]
156            if let syn::Meta::List(ref meta_list) = attr.meta {
157                let tokens = &meta_list.tokens;
158
159                // Try to parse manually: condition_ident, remaining_tokens
160                let token_string = tokens.to_string();
161                if let Some(comma_pos) = token_string.find(',') {
162                    let condition_ident = token_string[..comma_pos].trim();
163                    let remaining = token_string[comma_pos + 1..].trim();
164
165                    // Check if the condition is true for this expansion
166                    if expansion.is_ident_true(condition_ident) {
167                        // Parse the remaining part as an attribute (without the #[...] wrapper)
168                        // We need to add the #[...] wrapper to parse it correctly
169                        let attr_string = format!("#[{}]", remaining);
170                        if let Ok(parsed_attrs) =
171                            syn::parse::Parser::parse_str(syn::Attribute::parse_outer, &attr_string)
172                        {
173                            new_attrs.extend(parsed_attrs);
174                        }
175                    }
176                }
177            }
178        } else {
179            // Keep non-driver_test_cfg attributes
180            new_attrs.push(attr);
181        }
182    }
183
184    *attrs = new_attrs;
185}

187/// Add runtime capability checks for database capabilities in the requires expression
188fn add_capability_checks_from_expr(
189    func: &mut ItemFn,
190    requires_expr: &BoolExpr,
191    expansion: &Expansion,
192) {
193    use syn::{Ident, Stmt, parse_quote};
194
195    // Extract database capability identifiers from the expression
196    let db_capabilities = extract_db_capabilities(requires_expr, expansion);
197
198    if db_capabilities.is_empty() {
199        return;
200    }
201
202    // Get the test parameter name (first parameter of the function)
203    let test_param = func
204        .sig
205        .inputs
206        .first()
207        .and_then(|arg| {
208            if let syn::FnArg::Typed(pat_type) = arg {
209                if let syn::Pat::Ident(pat_ident) = &*pat_type.pat {
210                    Some(&pat_ident.ident)
211                } else {
212                    None
213                }
214            } else {
215                None
216            }
217        })
218        .expect("Test function must have at least one parameter");
219
220    // Generate capability check statements
221    let capability_checks: Vec<Stmt> = db_capabilities
222        .iter()
223        .map(|(cap_name, is_negated)| {
224            let cap_ident = Ident::new(cap_name, proc_macro2::Span::call_site());
225            if *is_negated {
226                // For negated capabilities, check that the capability is NOT present
227                parse_quote! {
228                    assert!(
229                        !#test_param.capability().#cap_ident,
230                        "Driver should not support capability: {}",
231                        stringify!(#cap_ident)
232                    );
233                }
234            } else {
235                // For regular capabilities, check that it IS present
236                parse_quote! {
237                    assert!(
238                        #test_param.capability().#cap_ident,
239                        "Driver does not support required capability: {}",
240                        stringify!(#cap_ident)
241                    );
242                }
243            }
244        })
245        .collect();
246
247    // Prepend the checks to the function body
248    let original_block = &func.block;
249    func.block = parse_quote! {
250        {
251            #(#capability_checks)*
252            #original_block
253        }
254    };
255}

257/// Extract database capability requirements from a boolean expression
258fn extract_db_capabilities(expr: &BoolExpr, expansion: &Expansion) -> Vec<(String, bool)> {
259    fn extract_recursive(
260        expr: &BoolExpr,
261        expansion: &Expansion,
262        negated: bool,
263        result: &mut Vec<(String, bool)>,
264    ) {
265        match expr {
266            BoolExpr::Ident(name) => {
267                // Check if this is a matrix identifier
268                if expansion.matrix_values.contains_key(name) {
269                    return; // Skip matrix identifiers
270                }
271
272                // Check if this is an ID variant identifier
273                if name == "id_u64" || name == "id_uuid" {
274                    return; // Skip ID variant identifiers
275                }
276
277                // This must be a database capability
278                result.push((name.clone(), negated));
279            }
280            BoolExpr::Or(exprs) | BoolExpr::And(exprs) => {
281                for e in exprs {
282                    extract_recursive(e, expansion, negated, result);
283                }
284            }
285            BoolExpr::Not(inner) => {
286                extract_recursive(inner, expansion, !negated, result);
287            }
288        }
289    }
290
291    let mut result = Vec::new();
292    extract_recursive(expr, expansion, false, &mut result);
293    result
294}

296/// Rewrite driver_test_cfg! macro calls to boolean literals based on the expansion
297fn rewrite_driver_test_cfg_macros(func: &mut ItemFn, expansion: &Expansion) {
298    struct MacroRewriter<'a> {
299        expansion: &'a Expansion,
300    }
301
302    impl<'a> VisitMut for MacroRewriter<'a> {
303        fn visit_expr_mut(&mut self, expr: &mut syn::Expr) {
304            // Check if this is a macro call to driver_test_cfg!
305            if let syn::Expr::Macro(expr_macro) = expr
306                && expr_macro.mac.path.is_ident("driver_test_cfg")
307            {
308                // Parse the boolean expression from the macro tokens
309                let tokens = expr_macro.mac.tokens.clone();
310                if let Ok(bool_expr) = syn::parse::Parser::parse2(
311                    |input: syn::parse::ParseStream| BoolExpr::parse(input),
312                    tokens,
313                ) {
314                    // Evaluate the expression for this expansion
315                    let result = self.expansion.evaluate_predicate(&bool_expr, &|_ident| {
316                        // Database capabilities are unknown at compile time
317                        // Return Unknown, which should not affect evaluation of
318                        // compile-time known values (matrix and ID variants)
319                        crate::parse::ThreeValuedBool::Unknown
320                    });
321
322                    // Convert to boolean literal
323                    let bool_value = match result {
324                        crate::parse::ThreeValuedBool::True => true,
325                        crate::parse::ThreeValuedBool::False => false,
326                        crate::parse::ThreeValuedBool::Unknown => {
327                            // Unknown means the expression references database capabilities,
328                            // which can only be checked at runtime, not compile time.
329                            // This is a compile error - driver_test_cfg! should only be used
330                            // for compile-time known values (ID variants, matrix dimensions).
331                            panic!(
332                                "driver_test_cfg! can only be used with compile-time known values \
333                                     (id_u64, id_uuid, matrix dimensions). Database capabilities must \
334                                     be checked at runtime using test.capability()"
335                            );
336                        }
337                    };
338
339                    // Replace the macro call with a boolean literal
340                    *expr = syn::parse_quote!(#bool_value);
341                    return;
342                }
343            }
344
345            // Continue visiting nested expressions
346            syn::visit_mut::visit_expr_mut(self, expr);
347        }
348    }
349
350    let mut rewriter = MacroRewriter { expansion };
351    rewriter.visit_item_fn_mut(func);

4/// Converts a database storage type to a PostgreSQL wire type.
5pub(crate) fn to_postgres_type(ty: &db::Type) -> &'static Type {
6    match ty {
7        db::Type::Boolean => &Type::BOOL,
8        db::Type::Float(4) => &Type::FLOAT4,
9        db::Type::Float(8) => &Type::FLOAT8,
10        db::Type::Integer(1) => &Type::INT2,
11        db::Type::Integer(2) => &Type::INT2,
12        db::Type::Integer(4) => &Type::INT4,
13        db::Type::Integer(8) => &Type::INT8,
14        db::Type::UnsignedInteger(1) => &Type::INT2,
15        db::Type::UnsignedInteger(2) => &Type::INT4,
16        db::Type::UnsignedInteger(4) => &Type::INT8,
17        db::Type::UnsignedInteger(8) => &Type::INT8,
18        db::Type::Text | db::Type::VarChar(_) => &Type::TEXT,
19        db::Type::Uuid => &Type::UUID,
20        db::Type::Numeric(_) => &Type::NUMERIC,
21        db::Type::Blob | db::Type::Binary(_) => &Type::BYTEA,
22        db::Type::Timestamp(_) => &Type::TIMESTAMPTZ,
23        db::Type::Date => &Type::DATE,
24        db::Type::Time(_) => &Type::TIME,
25        db::Type::DateTime(_) => &Type::TIMESTAMP,
26        // Enum types are handled separately via the cached OID map;
27        // fall back to TEXT if we reach here (shouldn't happen in practice).
28        db::Type::Enum(_) => &Type::TEXT,
29        db::Type::List(elem) => array_type_of(to_postgres_type(elem)),
30        _ => todo!("to_postgres_type; db_ty={ty:#?}"),
31    }
32}

34/// Returns the PostgreSQL array type whose element type is `elem`.
35pub(crate) fn array_type_of(elem: &Type) -> &'static Type {
36    match *elem {
37        Type::BOOL => &Type::BOOL_ARRAY,
38        Type::INT2 => &Type::INT2_ARRAY,
39        Type::INT4 => &Type::INT4_ARRAY,
40        Type::INT8 => &Type::INT8_ARRAY,
41        Type::FLOAT4 => &Type::FLOAT4_ARRAY,
42        Type::FLOAT8 => &Type::FLOAT8_ARRAY,
43        Type::TEXT => &Type::TEXT_ARRAY,
44        Type::VARCHAR => &Type::VARCHAR_ARRAY,
45        Type::BYTEA => &Type::BYTEA_ARRAY,
46        Type::UUID => &Type::UUID_ARRAY,
47        Type::NUMERIC => &Type::NUMERIC_ARRAY,
48        Type::TIMESTAMP => &Type::TIMESTAMP_ARRAY,
49        Type::TIMESTAMPTZ => &Type::TIMESTAMPTZ_ARRAY,
50        Type::DATE => &Type::DATE_ARRAY,
51        Type::TIME => &Type::TIME_ARRAY,
52        _ => todo!("no PostgreSQL array type for element type {elem:?}"),
53    }
54}

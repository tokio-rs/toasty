9/// Wrapper for reading string values from PostgreSQL enum columns.
10///
11/// The standard `String::FromSql::accepts()` rejects custom enum types.
12/// This wrapper accepts `Kind::Enum` types and reads the value as UTF-8 text.
13struct EnumString(String);
14
15impl<'a> postgres_types::FromSql<'a> for EnumString {
16    fn from_sql(
17        _ty: &Type,
18        raw: &'a [u8],
19    ) -> std::result::Result<Self, Box<dyn std::error::Error + Sync + Send>> {
20        Ok(EnumString(
21            std::str::from_utf8(raw)
22                .map_err(|e| Box::new(e) as Box<dyn std::error::Error + Sync + Send>)?
23                .to_string(),
24        ))
25    }
26
27    fn accepts(ty: &Type) -> bool {
28        matches!(ty.kind(), Kind::Enum(_))
29    }
30}

32/// Captures the raw wire bytes of a column without interpreting them.
33/// Used by the array decode path to hand the column body to
34/// [`array_from_sql`] directly, bypassing tokio-postgres's
35/// `Vec<Option<T>>` element decoder.
36struct RawBytes<'a>(&'a [u8]);
37
38impl<'a> postgres_types::FromSql<'a> for RawBytes<'a> {
39    fn from_sql(
40        _ty: &Type,
41        raw: &'a [u8],
42    ) -> std::result::Result<Self, Box<dyn std::error::Error + Sync + Send>> {
43        Ok(RawBytes(raw))
44    }
45
46    fn accepts(_ty: &Type) -> bool {
47        true
48    }
49}

66    /// Converts a PostgreSQL value within a row to a Toasty value.
67    pub(crate) fn from_sql(
68        index: usize,
69        row: &Row,
70        column: &Column,
71        expected_ty: &stmt::Type,
72    ) -> Self {
73        // Gets the value from the row as Option<T> and return stmt::Value::Null if the Option is
74        // None.
75        macro_rules! get_or_return_null {
76            ($ty:ty) => {{
77                match row.get::<usize, Option<$ty>>(index) {
78                    Some(inner) => inner,
79                    None => return Self(stmt::Value::Null),
80                }
81            }};
82        }
83
84        // NOTE: unfortunately, the inner representation of the PostgreSQL type enum is not
85        // accessible, so we must manually match each type like so.
86        let core_value = if column.type_() == &Type::TEXT || column.type_() == &Type::VARCHAR {
87            text_to_value(get_or_return_null!(String), expected_ty)
88        } else if column.type_() == &Type::BOOL {
89            stmt::Value::Bool(get_or_return_null!(bool))
90        } else if column.type_() == &Type::INT2 {
91            int2_to_value(get_or_return_null!(i16), expected_ty)
92        } else if column.type_() == &Type::INT4 {
93            int4_to_value(get_or_return_null!(i32), expected_ty)
94        } else if column.type_() == &Type::INT8 {
95            int8_to_value(get_or_return_null!(i64), expected_ty)
96        } else if column.type_() == &Type::UUID {
97            let v = get_or_return_null!(uuid::Uuid);
98            match expected_ty {
99                stmt::Type::Uuid => stmt::Value::Uuid(v),
100                stmt::Type::String => stmt::Value::String(v.to_string()),
101                _ => stmt::Value::Uuid(v),
102            }
103        } else if column.type_() == &Type::BYTEA {
104            let v = get_or_return_null!(Vec<u8>);
105            match expected_ty {
106                stmt::Type::Uuid => stmt::Value::Uuid(v.try_into().expect("invalid uuid bytes")),
107                stmt::Type::Bytes => stmt::Value::Bytes(v),
108                _ => todo!(
109                    "unsupported conversion from {:#?} to {expected_ty:?}",
110                    column.type_()
111                ),
112            }
113        } else if column.type_() == &Type::TIMESTAMPTZ {
114            #[cfg(feature = "jiff")]
115            {
116                stmt::Value::Timestamp(get_or_return_null!(jiff::Timestamp))
117            }
118            #[cfg(not(feature = "jiff"))]
119            {
120                panic!("TIMESTAMPTZ requires jiff feature to be enabled")
121            }
122        } else if column.type_() == &Type::TIMESTAMP {
123            #[cfg(feature = "jiff")]
124            {
125                stmt::Value::DateTime(get_or_return_null!(jiff::civil::DateTime))
126            }
127            #[cfg(not(feature = "jiff"))]
128            {
129                panic!("TIMESTAMP requires jiff feature to be enabled")
130            }
131        } else if column.type_() == &Type::DATE {
132            #[cfg(feature = "jiff")]
133            {
134                stmt::Value::Date(get_or_return_null!(jiff::civil::Date))
135            }
136            #[cfg(not(feature = "jiff"))]
137            {
138                panic!("DATE requires jiff feature to be enabled")
139            }
140        } else if column.type_() == &Type::TIME {
141            #[cfg(feature = "jiff")]
142            {
143                stmt::Value::Time(get_or_return_null!(jiff::civil::Time))
144            }
145            #[cfg(not(feature = "jiff"))]
146            {
147                panic!("TIME requires jiff feature to be enabled")
148            }
149        } else if column.type_() == &Type::FLOAT4 {
150            float4_to_value(get_or_return_null!(f32), expected_ty)
151        } else if column.type_() == &Type::FLOAT8 {
152            float8_to_value(get_or_return_null!(f64), expected_ty)
153        } else if column.type_() == &Type::NUMERIC {
154            #[cfg(feature = "rust_decimal")]
155            {
156                stmt::Value::Decimal(get_or_return_null!(rust_decimal::Decimal))
157            }
158            #[cfg(not(feature = "rust_decimal"))]
159            {
160                panic!("NUMERIC requires rust_decimal feature to be enabled")
161            }
162        } else if matches!(column.type_().kind(), Kind::Enum(_)) {
163            // Native database enum types (CREATE TYPE ... AS ENUM) are read as strings.
164            // We use EnumString instead of String because String::FromSql::accepts()
165            // rejects custom enum types.
166            match row.get::<usize, Option<EnumString>>(index) {
167                Some(EnumString(v)) => stmt::Value::String(v),
168                None => return Self(stmt::Value::Null),
169            }
170        } else if column.type_() == &Type::JSONB || column.type_() == &Type::JSON {
171            // `#[document]` columns. Decode the raw JSON wire bytes
172            // shape-directed to the named `Value::Object` wire form; the
173            // engine raises it to the embed's positional record.
174            let raw = match row.get::<usize, Option<RawBytes<'_>>>(index) {
175                Some(RawBytes(raw)) => raw,
176                None => return Self(stmt::Value::Null),
177            };
178            // `jsonb` wire format: a 1-byte version header, then UTF-8 JSON
179            // text. `json` is plain UTF-8 text.
180            let json_bytes = if column.type_() == &Type::JSONB {
181                &raw[1..]
182            } else {
183                raw
184            };
185            toasty_sql::json::from_slice(json_bytes, expected_ty)
186                .expect("invalid JSON in document column")
187        } else if let Kind::Array(_) = column.type_().kind() {
188            // Native array column (e.g. `text[]`, `int8[]`) — decoded
189            // directly into `Vec<stmt::Value>` via `array_from_sql`. The
190            // element type is taken from `expected_ty.as_list_unwrap()` so
191            // the per-element conversion mirrors a column of that scalar
192            // type.
193            let elem_ty = match expected_ty {
194                stmt::Type::List(elem) => elem.as_ref(),
195                other => panic!("array column expected stmt::Type::List, got {other:?}"),
196            };
197            let items = read_array_items(index, row, column, elem_ty);
198            match items {
199                Some(items) => stmt::Value::List(items),
200                None => return Self(stmt::Value::Null),
201            }
202        } else {
203            todo!(
204                "implement PostgreSQL to toasty conversion for `{:#?}`",
205                column.type_()
206            );
207        };
208
209        Value(core_value)
210    }

303// ============================================================================
304// Per-primitive conversions
305// ----------------------------------------------------------------------------
306// These functions translate a single decoded PostgreSQL primitive (the value
307// you get from `Row::get` or from an array element) into a `stmt::Value`,
308// respecting Toasty's expected element type. Sharing them between the column
309// path ([`Value::from_sql`]) and the array path ([`read_array_items`]) keeps
310// the two reading paths consistent: a `text[]` element and a `text` column
311// decode through the same logic.
312// ============================================================================
313
314fn text_to_value(v: String, expected_ty: &stmt::Type) -> stmt::Value {
315    match expected_ty {
316        stmt::Type::String => stmt::Value::String(v),
317        stmt::Type::Uuid => stmt::Value::Uuid(
318            v.parse()
319                .unwrap_or_else(|_| panic!("uuid could not be parsed from text")),
320        ),
321        _ => stmt::Value::String(v),
322    }
323}
324
325fn int2_to_value(v: i16, expected_ty: &stmt::Type) -> stmt::Value {
326    match expected_ty {
327        stmt::Type::I8 => stmt::Value::I8(v as i8),
328        stmt::Type::I16 => stmt::Value::I16(v),
329        stmt::Type::U8 => stmt::Value::U8(
330            u8::try_from(v).unwrap_or_else(|_| panic!("u8 value out of range: {v}")),
331        ),
332        stmt::Type::U16 => stmt::Value::U16(v as u16),
333        _ => panic!("unexpected type for INT2: {expected_ty:#?}"),
334    }
335}
336
337fn int4_to_value(v: i32, expected_ty: &stmt::Type) -> stmt::Value {
338    match expected_ty {
339        stmt::Type::I32 => stmt::Value::I32(v),
340        stmt::Type::U16 => stmt::Value::U16(
341            u16::try_from(v).unwrap_or_else(|_| panic!("u16 value out of range: {v}")),
342        ),
343        stmt::Type::U32 => stmt::Value::U32(v as u32),
344        _ => stmt::Value::I32(v),
345    }
346}
347
348fn int8_to_value(v: i64, expected_ty: &stmt::Type) -> stmt::Value {
349    match expected_ty {
350        stmt::Type::I64 => stmt::Value::I64(v),
351        stmt::Type::U32 => stmt::Value::U32(
352            u32::try_from(v).unwrap_or_else(|_| panic!("u32 value out of range: {v}")),
353        ),
354        stmt::Type::U64 => stmt::Value::U64(
355            u64::try_from(v).unwrap_or_else(|_| panic!("u64 value out of range: {v}")),
356        ),
357        _ => stmt::Value::I64(v),
358    }
359}

377/// Decode a PostgreSQL array column into a list of Toasty values. Returns
378/// `None` for SQL NULL. The column body is parsed with [`array_from_sql`]
379/// and each element is decoded straight into a `stmt::Value` via the
380/// per-primitive helpers above — the same ones used by [`Value::from_sql`]
381/// for scalar columns. No intermediate `Vec<Option<T>>` is allocated.
382fn read_array_items(
383    index: usize,
384    row: &Row,
385    column: &Column,
386    elem_ty: &stmt::Type,
387) -> Option<Vec<stmt::Value>> {
388    let elem_pg_ty = match column.type_().kind() {
389        Kind::Array(elem) => elem,
390        _ => panic!(
391            "read_array_items called on non-array column: {:?}",
392            column.type_()
393        ),
394    };
395
396    let RawBytes(raw) = row.get::<usize, Option<RawBytes<'_>>>(index)?;
397    let array = array_from_sql(raw).expect("invalid PostgreSQL array wire format");
398
399    let ndims = array
400        .dimensions()
401        .count()
402        .expect("invalid PostgreSQL array dimensions header");
403    if ndims > 1 {
404        panic!(
405            "multi-dimensional PostgreSQL arrays are not supported \
406             (got {ndims} dimensions). See https://github.com/tokio-rs/toasty/issues/870"
407        );
408    }
409
410    let mut values = array.values();
411    let (cap, _) = values.size_hint();
412    let mut out = Vec::with_capacity(cap);
413    while let Some(elem) = values
414        .next()
415        .expect("invalid PostgreSQL array element framing")
416    {
417        out.push(match elem {
418            None => stmt::Value::Null,
419            Some(bytes) => decode_array_element(elem_pg_ty, bytes, elem_ty),
420        });
421    }
422    Some(out)
423}

425/// Decode a single PostgreSQL array element from its raw wire bytes and
426/// project it into the Toasty value space using `elem_ty`. Reuses the
427/// scalar-column `_to_value` helpers so a `text[]` element and a `text`
428/// column decode through identical logic.
429fn decode_array_element(elem_pg_ty: &Type, bytes: &[u8], elem_ty: &stmt::Type) -> stmt::Value {
430    if elem_pg_ty == &Type::TEXT || elem_pg_ty == &Type::VARCHAR {
431        text_to_value(
432            String::from_sql(elem_pg_ty, bytes).expect("decode TEXT array element"),
433            elem_ty,
434        )
435    } else if elem_pg_ty == &Type::BOOL {
436        stmt::Value::Bool(bool::from_sql(elem_pg_ty, bytes).expect("decode BOOL array element"))
437    } else if elem_pg_ty == &Type::INT2 {
438        int2_to_value(
439            i16::from_sql(elem_pg_ty, bytes).expect("decode INT2 array element"),
440            elem_ty,
441        )
442    } else if elem_pg_ty == &Type::INT4 {
443        int4_to_value(
444            i32::from_sql(elem_pg_ty, bytes).expect("decode INT4 array element"),
445            elem_ty,
446        )
447    } else if elem_pg_ty == &Type::INT8 {
448        int8_to_value(
449            i64::from_sql(elem_pg_ty, bytes).expect("decode INT8 array element"),
450            elem_ty,
451        )
452    } else if elem_pg_ty == &Type::FLOAT4 {
453        float4_to_value(
454            f32::from_sql(elem_pg_ty, bytes).expect("decode FLOAT4 array element"),
455            elem_ty,
456        )
457    } else if elem_pg_ty == &Type::FLOAT8 {
458        float8_to_value(
459            f64::from_sql(elem_pg_ty, bytes).expect("decode FLOAT8 array element"),
460            elem_ty,
461        )
462    } else if elem_pg_ty == &Type::NUMERIC {
463        #[cfg(feature = "rust_decimal")]
464        {
465            stmt::Value::Decimal(
466                rust_decimal::Decimal::from_sql(elem_pg_ty, bytes)
467                    .expect("decode NUMERIC array element"),
468            )
469        }
470        #[cfg(not(feature = "rust_decimal"))]
471        {
472            panic!("NUMERIC array elements require rust_decimal feature to be enabled")
473        }
474    } else if elem_pg_ty == &Type::UUID {
475        stmt::Value::Uuid(
476            uuid::Uuid::from_sql(elem_pg_ty, bytes).expect("decode UUID array element"),
477        )
478    } else if matches!(elem_pg_ty.kind(), Kind::Enum(_)) {
479        // Enum labels are plain UTF-8; `EnumString` accepts `Kind::Enum` where
480        // `String` won't.
481        let EnumString(v) =
482            EnumString::from_sql(elem_pg_ty, bytes).expect("decode ENUM array element");
483        text_to_value(v, elem_ty)
484    } else {
485        todo!(
486            "implement PostgreSQL array decoding for element type `{:#?}`",
487            elem_pg_ty
488        )
489    }
490}

492impl ToSql for Value {
493    fn to_sql(
494        &self,
495        ty: &Type,
496        out: &mut BytesMut,
497    ) -> std::result::Result<IsNull, Box<dyn std::error::Error + Sync + Send>>
498    where
499        Self: Sized,
500    {
501        value_to_sql(&self.0, ty, out)
502    }
503
504    fn accepts(ty: &Type) -> bool {
505        matches!(
506            *ty,
507            Type::BOOL
508                | Type::INT2
509                | Type::INT4
510                | Type::INT8
511                | Type::TEXT
512                | Type::FLOAT4
513                | Type::FLOAT8
514                | Type::VARCHAR
515                | Type::BYTEA
516                | Type::UUID
517                | Type::NUMERIC
518                | Type::TIMESTAMP
519                | Type::TIMESTAMPTZ
520                | Type::DATE
521                | Type::TIME
522                | Type::JSONB
523                | Type::JSON
524        ) || matches!(ty.kind(), Kind::Enum(_) | Kind::Array(_))
525    }
526    to_sql_checked!();
527}

529/// Free-fn form of `Value::to_sql` so the array-element closure can call it
530/// with a `&CoreValue` borrowed from the input slice — no per-element clone
531/// or wrapper construction.
532fn value_to_sql(
533    value: &CoreValue,
534    ty: &Type,
535    out: &mut BytesMut,
536) -> std::result::Result<IsNull, Box<dyn std::error::Error + Sync + Send>> {
537    match (value, ty) {
538        (stmt::Value::Bool(value), _) => value.to_sql(ty, out),
539        (stmt::Value::I8(value), &Type::INT2) => (*value as i16).to_sql(ty, out),
540        (stmt::Value::I8(value), &Type::INT4) => (*value as i32).to_sql(ty, out),
541        (stmt::Value::I8(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
542        (stmt::Value::I16(value), &Type::INT2) => value.to_sql(ty, out),
543        (stmt::Value::I16(value), &Type::INT4) => (*value as i32).to_sql(ty, out),
544        (stmt::Value::I16(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
545        (stmt::Value::I32(value), &Type::INT4) => value.to_sql(ty, out),
546        (stmt::Value::I32(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
547        (stmt::Value::I64(value), &Type::INT4) => (*value as i32).to_sql(ty, out),
548        (stmt::Value::I64(value), &Type::INT8) => value.to_sql(ty, out),
549        (stmt::Value::U8(value), &Type::INT2) => (*value as i16).to_sql(ty, out),
550        (stmt::Value::U8(value), &Type::INT4) => (*value as i32).to_sql(ty, out),
551        (stmt::Value::U8(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
552        (stmt::Value::U16(value), &Type::INT4) => (*value as i32).to_sql(ty, out),
553        (stmt::Value::U16(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
554        (stmt::Value::U32(value), &Type::INT8) => (*value as i64).to_sql(ty, out),
555        (stmt::Value::U64(value), &Type::INT8) => {
556            if *value > i64::MAX as u64 {
557                return Err(Box::new(std::io::Error::new(
558                    std::io::ErrorKind::InvalidData,
559                    format!(
560                        "u64 value {} exceeds i64::MAX ({}), cannot store in PostgreSQL BIGINT",
561                        value,
562                        i64::MAX
563                    ),
564                )));
565            }
566            (*value as i64).to_sql(ty, out)
567        }
568        (stmt::Value::F32(value), &Type::FLOAT4) => value.to_sql(ty, out),
569        (stmt::Value::F32(value), &Type::FLOAT8) => (*value as f64).to_sql(ty, out),
570        (stmt::Value::F64(value), &Type::FLOAT4) => (*value as f32).to_sql(ty, out),
571        (stmt::Value::F64(value), &Type::FLOAT8) => value.to_sql(ty, out),
572        (stmt::Value::Null, _) => Ok(IsNull::Yes),
573        // PG enums are wire-encoded as plain UTF-8 text. `String::ToSql::accepts`
574        // rejects `Kind::Enum`, so write the bytes directly.
575        (stmt::Value::String(value), _) if matches!(ty.kind(), Kind::Enum(_)) => {
576            out.extend_from_slice(value.as_bytes());
577            Ok(IsNull::No)
578        }
579        (stmt::Value::String(value), _) => value.to_sql(ty, out),
580        (stmt::Value::Bytes(value), &Type::BYTEA) => value.to_sql(ty, out),
581        (stmt::Value::Uuid(value), &Type::UUID) => value.to_sql(ty, out),
582        #[cfg(feature = "rust_decimal")]
583        (stmt::Value::Decimal(value), _) => value.to_sql(ty, out),
584        #[cfg(feature = "jiff")]
585        (stmt::Value::Timestamp(value), _) => value.to_sql(ty, out),
586        #[cfg(feature = "jiff")]
587        (stmt::Value::Date(value), _) => value.to_sql(ty, out),
588        #[cfg(feature = "jiff")]
589        (stmt::Value::Time(value), _) => value.to_sql(ty, out),
590        #[cfg(feature = "jiff")]
591        (stmt::Value::DateTime(value), _) => value.to_sql(ty, out),
592        // `#[document]` columns: serialize the structural value (a
593        // `Value::List` of `Value::Object`s) to JSON and bind it. The engine
594        // has already converted positional records to named objects, so this
595        // is a plain structural encode.
596        (value, &Type::JSONB) => {
597            let bytes = toasty_sql::json::to_vec(value)
598                .map_err(|e| Box::new(e) as Box<dyn std::error::Error + Sync + Send>)?;
599            // `jsonb` wire format: a 1-byte version header, then UTF-8 text.
600            out.extend_from_slice(&[1]);
601            out.extend_from_slice(&bytes);
602            Ok(IsNull::No)
603        }
604        (value, &Type::JSON) => {
605            let bytes = toasty_sql::json::to_vec(value)
606                .map_err(|e| Box::new(e) as Box<dyn std::error::Error + Sync + Send>)?;
607            out.extend_from_slice(&bytes);
608            Ok(IsNull::No)
609        }
610        // List → bind as a PostgreSQL array via the streaming `array_to_sql`
611        // primitive: the closure runs per element and writes directly into
612        // `out`, so there's no intermediate `Vec<Option<T>>`. The element PG
613        // type (carried by the prepared statement, see `db::Type::List` →
614        // `to_postgres_type`) drives per-item conversion via this same fn.
615        (stmt::Value::List(items), _) => {
616            let Kind::Array(elem) = ty.kind() else {
617                return Err(format!("Value::List bound to non-array PG type {ty:?}").into());
618            };
619            let len = i32::try_from(items.len())
620                .map_err(|_| format!("array length {} exceeds i32::MAX", items.len()))?;
621            array_to_sql(
622                [ArrayDimension {
623                    len,
624                    lower_bound: 1,
625                }],
626                elem.oid(),
627                items.iter(),
628                |v, buf| match value_to_sql(v, elem, buf)? {
629                    IsNull::No => Ok(postgres_protocol::IsNull::No),
630                    IsNull::Yes => Ok(postgres_protocol::IsNull::Yes),
631                },
632                out,
633            )?;
634            Ok(IsNull::No)
635        }
636        (value, _) => todo!("unsupported Value for PostgreSQL type: {value:#?}, type: {ty:#?}"),
637    }
638}

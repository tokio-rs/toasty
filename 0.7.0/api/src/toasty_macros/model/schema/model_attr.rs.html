1use super::{ErrorSet, KeyAttr};
2
3#[derive(Debug, Default)]
4pub(crate) struct ModelAttr {
5    /// Primary key definition
6    pub(crate) key: Option<KeyAttr>,
7
8    /// Model-level secondary index definitions
9    pub(crate) indices: Vec<KeyAttr>,
10
11    /// Optional database table name to map the model to
12    pub(crate) table: Option<syn::LitStr>,
13}
14
15impl ModelAttr {
16    pub(super) fn populate_from_ast(
17        &mut self,
18        attrs: &Vec<syn::Attribute>,
19        names: &[syn::Ident],
20    ) -> syn::Result<()> {
21        let mut errs = ErrorSet::new();
22
23        for attr in attrs {
24            if attr.path().is_ident("key") {
25                if self.key.is_some() {
26                    errs.push(syn::Error::new_spanned(attr, "duplicate #[key] attribute"));
27                } else {
28                    match KeyAttr::from_ast(attr, names) {
29                        Ok(key_attr) => self.key = Some(key_attr),
30                        Err(e) => errs.push(e),
31                    }
32                }
33            } else if attr.path().is_ident("index") {
34                match KeyAttr::from_ast(attr, names) {
35                    Ok(index_attr) => self.indices.push(index_attr),
36                    Err(e) => errs.push(e),
37                }
38            } else if attr.path().is_ident("table") {
39                if self.table.is_some() {
40                    return Err(syn::Error::new_spanned(attr, "duplicate `table` attribute"));
41                }
42
43                let syn::Meta::NameValue(meta) = &attr.meta else {
44                    return Err(syn::Error::new_spanned(
45                        attr,
46                        "expected `table = \"table_name\"`",
47                    ));
48                };
49
50                let syn::Expr::Lit(lit) = &meta.value else {
51                    return Err(syn::Error::new_spanned(
52                        attr,
53                        "expected `table = \"table_name\"`",
54                    ));
55                };
56
57                let syn::Lit::Str(lit) = &lit.lit else {
58                    return Err(syn::Error::new_spanned(
59                        attr,
60                        "expected `table = \"table_name\"`",
61                    ));
62                };
63
64                self.table = Some(lit.clone());
65            }
66        }
67
68        if let Some(err) = errs.collect() {
69            return Err(err);
70        }
71
72        Ok(())
73    }
74}

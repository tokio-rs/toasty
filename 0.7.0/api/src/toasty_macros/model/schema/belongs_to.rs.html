3#[derive(Debug)]
4pub(crate) struct BelongsTo {
5    /// Target type
6    pub(crate) ty: syn::Type,
7
8    /// Foreign key
9    pub(crate) foreign_key: Vec<ForeignKeyField>,
10}
11
12impl BelongsTo {
13    pub(super) fn from_ast(
14        attr: &syn::Attribute,
15        ty: &syn::Type,
16        names: &[syn::Ident],
17    ) -> syn::Result<Self> {
18        let mut fk_sources: Option<Vec<syn::Ident>> = None;
19        let mut fk_targets: Option<Vec<syn::Ident>> = None;
20
21        attr.parse_nested_meta(|meta| {
22            if meta.path.is_ident("key") {
23                if fk_sources.is_some() {
24                    return Err(meta.error(
25                        "`key` specified more than once; use `key = [a, b]` for composite foreign keys",
26                    ));
27                }
28                fk_sources = Some(parse_idents(&meta)?);
29                Ok(())
30            } else if meta.path.is_ident("references") {
31                if fk_targets.is_some() {
32                    return Err(meta.error(
33                        "`references` specified more than once; use `references = [a, b]` for composite foreign keys",
34                    ));
35                }
36                fk_targets = Some(parse_idents(&meta)?);
37                Ok(())
38            } else {
39                Err(meta.error("expected `key` or `references`"))
40            }
41        })?;
42
43        let fk_sources = fk_sources
44            .ok_or_else(|| syn::Error::new_spanned(attr, "missing `key = ...` attribute"))?;
45        let fk_targets = fk_targets
46            .ok_or_else(|| syn::Error::new_spanned(attr, "missing `references = ...` attribute"))?;
47
48        if fk_sources.is_empty() || fk_targets.is_empty() {
49            return Err(syn::Error::new_spanned(
50                attr,
51                "`key` and `references` must each name at least one field",
52            ));
53        }
54
55        if fk_sources.len() != fk_targets.len() {
56            return Err(syn::Error::new_spanned(
57                attr,
58                format!(
59                    "`key` has {} field(s) but `references` has {} field(s); they must match",
60                    fk_sources.len(),
61                    fk_targets.len(),
62                ),
63            ));
64        }
65
66        let mut foreign_key = vec![];
67
68        for (source, target) in fk_sources.into_iter().zip(fk_targets) {
69            let source_idx = names
70                .iter()
71                .position(|name| name == &source)
72                .ok_or_else(|| {
73                    syn::Error::new_spanned(
74                        &source,
75                        format!("source field `{source}` not found in names"),
76                    )
77                })?;
78
79            foreign_key.push(ForeignKeyField {
80                source: source_idx,
81                target,
82            });
83        }
84
85        Ok(Self {
86            ty: ty.clone(),
87            foreign_key,
88        })
89    }
90}

92fn parse_idents(meta: &syn::meta::ParseNestedMeta<'_>) -> syn::Result<Vec<syn::Ident>> {
93    let value = meta.value()?;
94
95    if value.peek(syn::token::Bracket) {
96        let content;
97        syn::bracketed!(content in value);
98        let punctuated =
99            syn::punctuated::Punctuated::<syn::Ident, syn::Token![,]>::parse_terminated(&content)?;
100        Ok(punctuated.into_iter().collect())
101    } else {
102        Ok(vec![value.parse::<syn::Ident>()?])
103    }

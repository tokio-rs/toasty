1use crate::{Result, schema::Load};
2use toasty_core::stmt::{self, Value};
3
4/// Trait for types that can serve as the target of an update operation.
5///
6/// This trait is implemented by types that represent different update targets:
7/// - Generated query struct: builds the update from its inner query, producing
8///   `Update<List<Model>>` for multi-row updates
9/// - `&mut Model`: builds the update from the model's primary key, producing
10///   `Update<Model>` for single-row updates
11///
12/// The associated type `Returning` determines the statement return type.
13pub trait UpdateTarget {
14    /// The type parameter for the typed `Update<R>` statement.
15    type Returning: Load;
16
17    /// Build the update statement by combining this target's selection with the
18    /// provided assignments.
19    ///
20    /// For query-based targets, this takes the inner query (replacing it with a
21    /// default) and wraps it in an `Update<List<Model>>`.
22    /// For `&mut Model`, this builds an `Update<Model>` from the model's PK.
23    fn to_update_stmt(
24        &mut self,
25        assignments: stmt::Assignments,
26    ) -> crate::stmt::Update<Self::Returning>;
27
28    /// Apply the result of an update operation.
29    ///
30    /// For query-based updates, this discards the values.
31    /// For instance updates, this reloads the model from the value.
32    fn apply_result(self, value: Value) -> Result<()>;
33}

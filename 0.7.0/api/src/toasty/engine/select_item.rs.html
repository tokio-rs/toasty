5/// Represents an item that the engine can select from the database.
6///
7/// This generalizes `ExprReference` to support both column references and
8/// computed expressions like `COUNT(*)`. Using an enum that derives `Hash` and
9/// `Eq` allows the planner to continue deduplicating select items via
10/// `SelectItems`.
11#[derive(Debug, Clone, Copy, PartialEq, Hash, Eq)]
12pub(crate) enum SelectItem {
13    /// A reference to a column or field — the traditional case.
14    ExprReference(stmt::ExprReference),
15
16    /// The `COUNT(*)` aggregate. SQL-only.
17    CountStar,
18}
19
20impl SelectItem {
21    /// Returns the inner `ExprReference`, panicking if this is not one.
22    #[track_caller]
23    pub(crate) fn as_expr_reference_unwrap(&self) -> &stmt::ExprReference {
24        match self {
25            SelectItem::ExprReference(r) => r,
26            other => panic!("expected ExprReference, got {other:?}"),
27        }
28    }
29
30    /// Infer the type of this select item given the expression context.
31    pub(crate) fn infer_ty(&self, cx: &stmt::ExprContext<'_>) -> stmt::Type {
32        match self {
33            SelectItem::ExprReference(expr_reference) => cx.infer_expr_reference_ty(expr_reference),
34            SelectItem::CountStar => stmt::Type::U64,
35        }
36    }
37
38    /// Convert this select item into the corresponding expression.
39    pub(crate) fn to_expr(self) -> stmt::Expr {
40        match self {
41            SelectItem::ExprReference(expr_reference) => stmt::Expr::from(expr_reference),
42            SelectItem::CountStar => stmt::Expr::count_star(),
43        }
44    }
45}
46
47impl From<stmt::ExprReference> for SelectItem {
48    fn from(value: stmt::ExprReference) -> Self {
49        SelectItem::ExprReference(value)
50    }
51}
52
53/// A set of [`SelectItem`]s backed by an [`IndexSet`](indexmap::IndexSet) for
54/// deduplication and index-based lookup.
55#[derive(Debug, Default, Clone)]
56pub(crate) struct SelectItems(indexmap::IndexSet<SelectItem>);
57
58impl SelectItems {
59    pub(crate) fn new() -> Self {
60        Self(indexmap::IndexSet::new())
61    }
62
63    /// Find the index of an `ExprReference` item, returning `None` if not
64    /// present.
65    pub(crate) fn try_get_index_of_expr_reference(
66        &self,
67        expr_reference: impl Into<stmt::ExprReference>,
68    ) -> Option<usize> {
69        let item = SelectItem::ExprReference(expr_reference.into());
70        self.0.get_index_of(&item)
71    }
72
73    /// Find the index of an `ExprReference` item.
74    pub(crate) fn get_index_of_expr_reference(
75        &self,
76        expr_reference: impl Into<stmt::ExprReference>,
77    ) -> usize {
78        self.try_get_index_of_expr_reference(expr_reference)
79            .unwrap()
80    }
81
82    /// Find the index of the `CountStar` item.
83    pub(crate) fn get_index_of_count_star(&self) -> usize {
84        self.0.get_index_of(&SelectItem::CountStar).unwrap()
85    }
86
87    /// Returns `Type::List(Type::Record(field_tys))` where each `field_ty` is
88    /// inferred from the corresponding select item.
89    pub(crate) fn infer_record_list_ty(&self, cx: &stmt::ExprContext<'_>) -> stmt::Type {
90        let field_tys = self.0.iter().map(|item| item.infer_ty(cx)).collect();
91        stmt::Type::list(stmt::Type::Record(field_tys))
92    }
93
94    /// Extract only the `ExprReference` items from this set.
95    ///
96    /// Used by the NoSQL path where only column references are valid.
97    pub(crate) fn extract_expr_references(&self) -> indexmap::IndexSet<stmt::ExprReference> {
98        self.0
99            .iter()
100            .map(|item| *item.as_expr_reference_unwrap())
101            .collect()
102    }
103}
104
105impl<'a> IntoIterator for &'a SelectItems {
106    type Item = &'a SelectItem;
107    type IntoIter = indexmap::set::Iter<'a, SelectItem>;
108
109    fn into_iter(self) -> Self::IntoIter {
110        self.0.iter()
111    }
112}
113
114impl Deref for SelectItems {
115    type Target = indexmap::IndexSet<SelectItem>;
116
117    fn deref(&self) -> &Self::Target {
118        &self.0
119    }
120}
121
122impl DerefMut for SelectItems {
123    fn deref_mut(&mut self) -> &mut Self::Target {
124        &mut self.0
125    }

14#[derive(Debug)]
15struct HirPlanner<'a> {
16    engine: &'a Engine,
17
18    /// Root statement and all nested statements.
19    hir: &'a HirStatement,
20
21    /// Graph of operations needed to execute the statement
22    mir: mir::Store,
23}
24
25#[derive(Debug)]
26struct ExecPlanner<'a> {
27    logical_plan: &'a LogicalPlan,
28    var_decls: VarDecls,
29    actions: Vec<exec::Action>,
30    use_transactions: bool,
31}
32
33impl Engine {
34    pub(super) fn plan_hir_statement(&self, hir: HirStatement) -> Result<ExecPlan> {
35        // Build the logical plan
36        let logical_plan = HirPlanner {
37            engine: self,
38            hir: &hir,
39            mir: mir::Store::new(),
40        }
41        .build_logical_plan()?;
42
43        // Build the execution plan from the logical plan
44        Ok(self.plan_execution(logical_plan))
45    }
46
47    fn plan_execution(&self, logical_plan: mir::LogicalPlan) -> ExecPlan {
48        ExecPlanner {
49            logical_plan: &logical_plan,
50            var_decls: VarDecls::default(),
51            actions: vec![],
52            use_transactions: self.capability().sql,
53        }
54        .plan_execution()
55    }
56}
57
58impl HirPlanner<'_> {
59    fn build_logical_plan(mut self) -> Result<mir::LogicalPlan> {
60        let root_id = self.hir.root_id();
61        self.plan_statement(root_id)?;
62
63        let exit = self.hir.root().output.get().unwrap();
64        let exit_node = &self.mir.store[exit];
65
66        // Increment num uses for the exit node. This counts as the "engines"
67        // use of the variable to return to the use.
68        exit_node.num_uses.set(exit_node.num_uses.get() + 1);
69
70        Ok(mir::LogicalPlan::new(self.mir, exit))
71    }
72}

1use toasty_core::{
2    driver::Capability, schema::db::TableId, stmt, stmt::ExprContext, stmt::ValueSet,
3};
4
5use crate::engine::simplify;
6
7use super::Exec;

9impl Exec<'_> {
10    /// Split a composite filter into individual key predicates.
11    ///
12    /// Recognizes these forms and decomposes them:
13    /// - `ANY(MAP(Value::List([v1, v2, ...]), pred))` — substitutes each vi
14    ///   into pred
15    /// - `InList(expr, Value::List([v1, v2, ...]))` — produces `expr == vi`
16    ///   for each value
17    ///
18    /// For any other form (including a single equality), simplifies and returns
19    /// it as a single-element vec (or empty if unsatisfiable).
20    ///
21    /// Each returned predicate has been simplified and is guaranteed
22    /// satisfiable.
23    pub(super) fn split_filter(&self, filter: stmt::Expr, table: TableId) -> Vec<stmt::Expr> {
24        let db_table = self.engine.schema.db.table(table);
25        let cx = self.engine.expr_cx_for(db_table);
26
27        let capability = self.engine.capability;
28        match filter {
29            stmt::Expr::Any(any) => Self::split_filter_any_map(*any.expr, cx, capability),
30            stmt::Expr::InList(in_list) => {
31                Self::split_filter_in_list(*in_list.expr, *in_list.list, cx, capability)
32            }
33            mut other => {
34                simplify::simplify_expr(cx, self.engine.capability, &mut other);
35                if other.is_unsatisfiable() {
36                    vec![]
37                } else {
38                    vec![other]
39                }
40            }
41        }
42    }
43
44    /// `ANY(MAP(Value::List([v1, v2, ...]), pred))` — substitutes each value
45    /// into the predicate template.
46    ///
47    /// Duplicate values are collapsed: each kv-layer fan-out becomes one
48    /// driver call per partition key, and a downstream `HashIndex` build
49    /// over the merged rows requires unique keys.
50    fn split_filter_any_map(
51        map_expr: stmt::Expr,
52        cx: ExprContext<'_>,
53        capability: &Capability,
54    ) -> Vec<stmt::Expr> {
55        let stmt::Expr::Map(map) = map_expr else {
56            unreachable!()
57        };
58        let stmt::Expr::Value(stmt::Value::List(items)) = *map.base else {
59            unreachable!()
60        };
61
62        let mut seen = ValueSet::with_capacity(items.len());
63        items
64            .into_iter()
65            .filter(|item| !item.is_null())
66            .filter(|item| seen.insert(item.clone()))
67            .filter_map(|item| {
68                let mut pred = *map.map.clone();
69                // Unpack Record fields so arg(i) binds to field i.
70                match item {
71                    stmt::Value::Record(r) => pred.substitute(&r.fields[..]),
72                    item => pred.substitute([item]),
73                }
74                simplify::simplify_expr(cx, capability, &mut pred);
75                (!pred.is_unsatisfiable()).then_some(pred)
76            })
77            .collect()
78    }
79
80    /// `InList(expr, Value::List([v1, v2, ...]))` — produces `expr == vi` for
81    /// each value.
82    ///
83    /// Duplicate values are collapsed; see `split_filter_any_map`.
84    fn split_filter_in_list(
85        expr: stmt::Expr,
86        list: stmt::Expr,
87        cx: ExprContext<'_>,
88        capability: &Capability,
89    ) -> Vec<stmt::Expr> {
90        let stmt::Expr::Value(stmt::Value::List(values)) = list else {
91            unreachable!()
92        };
93
94        let mut seen = ValueSet::with_capacity(values.len());
95        values
96            .into_iter()
97            .filter(|v| !v.is_null())
98            .filter(|v| seen.insert(v.clone()))
99            .filter_map(|v| {
100                let mut pred = stmt::Expr::binary_op(expr.clone(), stmt::BinaryOp::Eq, v);
101                simplify::simplify_expr(cx, capability, &mut pred);
102                (!pred.is_unsatisfiable()).then_some(pred)
103            })
104            .collect()
105    }
106}

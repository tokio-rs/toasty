11/// Get a model by key
12#[derive(Debug)]
13pub(crate) struct GetByKey {
14    /// Where to get the keys to load
15    pub input: VarId,
16
17    /// Where to store the result
18    pub output: Output,
19
20    /// Table to query
21    pub table: TableId,
22
23    /// Columns to get
24    pub columns: Vec<ColumnId>,
25}
26
27impl Exec<'_> {
28    pub(super) async fn action_get_by_key(&mut self, action: &GetByKey) -> Result<()> {
29        let mut seen = ValueSet::new();
30        let keys: Vec<_> = self
31            .vars
32            .load(action.input)
33            .await?
34            .values
35            .collect_as_value()
36            .await?
37            .into_list_unwrap()
38            .into_iter()
39            .filter(|k| !k.is_null())
40            .filter(|k| seen.insert(k.clone()))
41            .collect();
42
43        let res = if keys.is_empty() {
44            Rows::value_stream(ValueStream::default())
45        } else {
46            let op = operation::GetByKey {
47                table: action.table,
48                select: action.columns.clone(),
49                keys,
50            };
51
52            let res = self.connection.exec(&self.engine.schema, op.into()).await?;
53            res.values
54        };
55
56        self.vars.store(
57            action.output.var,
58            action.output.num_uses,
59            ExecResponse::from_rows(res),
60        );
61        Ok(())
62    }
63}
64
65impl From<GetByKey> for Action {
66    fn from(src: GetByKey) -> Self {
67        Self::GetByKey(src)
68    }

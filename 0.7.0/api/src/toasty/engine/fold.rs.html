1//! Cheap, schema-free, idempotent rewrites.
2//!
3//! `fold` is the canonicalization pass shared between `lower`, `simplify`, and
4//! `exec_statement`. Each rule is local, O(n) per pass, and produces output
5//! that satisfies `fold(fold(x)) == fold(x)`. Rules pattern-match on local
6//! structure only; no schema lookups.
7//!
8//! See `docs/dev/design/lower-then-simplify.md` for the full taxonomy.
9

26use toasty_core::stmt::{self, Expr, Node, VisitMut};
27
28/// Folds an expression tree in place: canonicalizes structure, evaluates
29/// constants, propagates nulls, drops boolean units. Idempotent.
30pub(crate) fn fold_stmt<T: Node>(stmt: &mut T) {
31    Fold.visit_mut(stmt);
32}
33
34/// Visitor that applies fold rules bottom-up.
35struct Fold;
36
37impl VisitMut for Fold {
38    fn visit_expr_mut(&mut self, i: &mut Expr) {
39        // Recurse into children first so parent rules see canonical input.
40        stmt::visit_mut::visit_expr_mut(self, i);
41
42        // A single rule's output may itself be fold-eligible (e.g.
43        // `not(eq(x, y))` becomes `ne(x, y)`, which then constant-folds
44        // to a literal when both sides are values, or De Morgan creates a
45        // new `Or` of `Not`s that need their own pass). Recurse on the
46        // replacement so the fixed point is reached in one traversal.
47        if let Some(mut expr) = fold_one(i) {
48            self.visit_expr_mut(&mut expr);
49            *i = expr;
50        }
51    }
52}
53
54/// Apply one round of fold rules to `i`. Returns `Some` if a rule fired and
55/// produced a replacement expression; returns `None` if no rule applied or
56/// the rule mutated `i` in place.
57fn fold_one(i: &mut Expr) -> Option<Expr> {
58    match i {
59        Expr::And(expr) => expr_and::fold_expr_and(expr),
60        Expr::BinaryOp(expr) => {
61            expr_binary_op::fold_expr_binary_op(expr.op, &mut expr.lhs, &mut expr.rhs)
62        }
63        Expr::Cast(expr) => expr_cast::fold_expr_cast(expr),
64        Expr::InList(expr) => expr_in_list::fold_expr_in_list(expr),
65        Expr::Intersects(expr) => expr_intersects::fold_expr_intersects(expr),
66        Expr::IsNull(expr) => expr_is_null::fold_expr_is_null(expr),
67        Expr::IsSuperset(expr) => expr_is_superset::fold_expr_is_superset(expr),
68        Expr::List(expr) => expr_list::fold_expr_list(expr),
69        Expr::Match(expr) => expr_match::fold_expr_match(expr),
70        Expr::Not(expr) => expr_not::fold_expr_not(expr),
71        Expr::Or(expr) => expr_or::fold_expr_or(expr),
72        Expr::Record(expr) => expr_record::fold_expr_record(expr),
73        // No fold rules yet — leaves and not-yet-migrated variants pass
74        // through untouched. Listed explicitly so adding a new `Expr`
75        // variant forces a decision about whether it has cheap rewrites.
76        Expr::AllOp(_)
77        | Expr::Any(_)
78        | Expr::AnyOp(_)
79        | Expr::Arg(_)
80        | Expr::Default
81        | Expr::Error(_)
82        | Expr::Exists(_)
83        | Expr::Func(_)
84        | Expr::Ident(_)
85        | Expr::InSubquery(_)
86        | Expr::IsVariant(_)
87        | Expr::Length(_)
88        | Expr::Let(_)
89        | Expr::Like(_)
90        | Expr::Map(_)
91        | Expr::Project(_)
92        | Expr::Reference(_)
93        | Expr::StartsWith(_)
94        | Expr::Stmt(_)
95        | Expr::Value(_) => None,
96    }
97}

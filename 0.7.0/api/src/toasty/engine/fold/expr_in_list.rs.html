3/// Cheap canonicalization for `x IN (..)`: empty list → false, null
4/// propagation, literal dedup, single-item collapse.
5///
6/// Heavyweight rewrites (model-reference → primary-key-reference) live in
7/// `simplify/expr_in_list.rs` and run after this fold pass on canonical
8/// input.
9pub(super) fn fold_expr_in_list(expr: &mut stmt::ExprInList) -> Option<Expr> {
10    // `x in ()` → `false`
11    if expr.list.is_list_empty() {
12        return Some(Expr::Value(Value::Bool(false)));
13    }
14
15    // Null propagation, `null in (x, y, z)` → `null`
16    if expr.expr.is_value_null() {
17        return Some(Expr::null());
18    }
19
20    // Deduplicate literal value lists: `x in (1, 1, 2)` → `x in (1, 2)`.
21    // Lists carry literal `Value`s only (no symbolic expressions), so value
22    // equality is safe — there is no non-determinism to preserve.
23    if let Expr::Value(Value::List(values)) = &mut *expr.list {
24        let mut seen = ValueSet::new();
25        values.retain(|v| seen.insert(v.clone()));
26    }
27
28    // Rewrite single-item lists into equalities.
29    rewrite_expr_in_list_with_single_item(expr)
30}
31
32fn rewrite_expr_in_list_with_single_item(expr: &mut stmt::ExprInList) -> Option<Expr> {
33    let rhs = match &mut *expr.list {
34        Expr::Value(value) => {
35            let values = match value {
36                Value::List(value) => &value[..],
37                _ => todo!("{value:#?}"),
38            };
39
40            if values.len() != 1 {
41                return None;
42            }
43
44            Expr::Value(values[0].clone())
45        }
46        Expr::List(expr_list) => {
47            if expr_list.items.len() != 1 {
48                return None;
49            }
50
51            expr_list.items[0].take()
52        }
53        Expr::Record(_) => todo!("should not happen"),
54        _ => return None,
55    };
56
57    Some(Expr::eq(expr.expr.take(), rhs))
58}

6    /// Heavyweight OR rewrites. Cheap canonicalization (flatten, true
7    /// short-circuit, drop false, null propagation) runs in `fold::expr_or`
8    /// before this is reached.
9    pub(super) fn simplify_expr_or(&mut self, expr: &mut stmt::ExprOr) -> Option<stmt::Expr> {
10        // Idempotent law, `a or a` → `a`
11        // Note: O(n) lookups are acceptable here since operand lists are typically small.
12        // `is_equivalent_to` (not `PartialEq`) keeps this sound for non-deterministic
13        // operands like `LAST_INSERT_ID()` — two syntactically identical calls may
14        // return different values, so the second occurrence must survive.
15        let mut seen: Vec<stmt::Expr> = Vec::new();
16        expr.operands.retain(|operand| {
17            if seen.iter().any(|e| e.is_equivalent_to(operand)) {
18                false
19            } else {
20                seen.push(operand.clone());
21                true
22            }
23        });
24
25        // Absorption law, `x or (x and y)` → `x`
26        // If an operand is an AND that contains another operand of the OR, remove the AND.
27        let non_and_operands: Vec<_> = expr
28            .operands
29            .iter()
30            .filter(|op| !matches!(op, stmt::Expr::And(_)))
31            .cloned()
32            .collect();
33
34        expr.operands.retain(|operand| {
35            if let stmt::Expr::And(and_expr) = operand {
36                // Remove this AND if any of its operands appears as a direct operand of the OR
37                !and_expr
38                    .operands
39                    .iter()
40                    .any(|op| non_and_operands.iter().any(|e| e.is_equivalent_to(op)))
41            } else {
42                true
43            }
44        });
45
46        // Factoring, `(a and b) or (a and c)` → `a and (b or c)`
47        // Find common factors across all AND operands and factor them out.
48        if let Some(factored) = self.try_factor_or(expr) {
49            return Some(factored);
50        }
51
52        // Complement law, `a or not(a)` → `true` (only if `a` is non-nullable)
53        if self.try_complement_or(expr) {
54            return Some(true.into());
55        }
56
57        // The variant-tautology rewrite (`is_variant(x, 0) or is_variant(x, 1)`
58        // covering all variants → `true`) fires in the pre-lowering
59        // `LowerStatement::visit_expr_mut` `Expr::Or` arm, not here.
60
61        // OR-to-IN conversion, `a = 1 or a = 2 or a = 3` → `a in (1, 2, 3)`
62        if let Some(in_list) = self.try_or_to_in_list(expr) {
63            return Some(in_list);
64        }
65
66        if expr.operands.is_empty() {
67            Some(false.into())
68        } else if expr.operands.len() == 1 {
69            Some(expr.operands.remove(0))
70        } else {
71            None
72        }
73    }

75    /// Attempts to factor common terms from AND expressions within an OR.
76    /// `(a and b) or (a and c)` → `a and (b or c)`
77    /// `(a and b and c) or (a and b and d)` → `a and b and (c or d)`
78    fn try_factor_or(&self, expr: &mut stmt::ExprOr) -> Option<stmt::Expr> {
79        // Need at least 2 operands, all must be ANDs
80        if expr.operands.len() < 2 {
81            return None;
82        }
83
84        if !expr
85            .operands
86            .iter()
87            .all(|op| matches!(op, stmt::Expr::And(_)))
88        {
89            return None;
90        }
91
92        // Find all common factors by checking which operands from the first AND
93        // appear in all other ANDs
94        let first_and = match &expr.operands[0] {
95            stmt::Expr::And(and) => and,
96            _ => unreachable!(),
97        };
98
99        let common: Vec<_> = first_and
100            .operands
101            .iter()
102            .filter(|op| {
103                expr.operands[1..].iter().all(|other| {
104                    if let stmt::Expr::And(other_and) = other {
105                        other_and.operands.iter().any(|e| e.is_equivalent_to(op))
106                    } else {
107                        false
108                    }
109                })
110            })
111            .cloned()
112            .collect();
113
114        if common.is_empty() {
115            return None;
116        }
117
118        // Remove all common factors from each AND
119        for operand in &mut expr.operands {
120            if let stmt::Expr::And(and) = operand {
121                and.operands
122                    .retain(|op| !common.iter().any(|c| c.is_equivalent_to(op)));
123                // If only one operand left, unwrap the AND
124                if and.operands.len() == 1 {
125                    *operand = and.operands.pop().unwrap();
126                } else if and.operands.is_empty() {
127                    *operand = true.into();
128                }
129            }
130        }
131
132        // Common factors AND (the modified OR)
133        let mut result = common;
134        let or_expr = stmt::ExprOr {
135            operands: mem::take(&mut expr.operands),
136        };
137        result.push(stmt::Expr::Or(or_expr));
138        Some(stmt::Expr::and_from_vec(result))
139    }

141    /// Checks for complement law: `a or not(a)` → `true`
142    ///
143    /// Returns true if a complementary pair is found and both are non-nullable.
144    fn try_complement_or(&self, expr: &stmt::ExprOr) -> bool {
145        // Collect all NOT expressions and their inner expressions
146        let negated: Vec<_> = expr
147            .operands
148            .iter()
149            .filter_map(|op| {
150                if let stmt::Expr::Not(not_expr) = op {
151                    Some(not_expr.expr.as_ref())
152                } else {
153                    None
154                }
155            })
156            .collect();
157
158        // Check if any operand has its negation also present
159        for operand in &expr.operands {
160            // Skip NOT expressions themselves
161            if matches!(operand, stmt::Expr::Not(_)) {
162                continue;
163            }
164
165            // Check if not(operand) exists and operand is non-nullable
166            if negated.iter().any(|n| n.is_equivalent_to(operand))
167                && operand.is_always_non_nullable()
168            {
169                return true;
170            }
171        }
172
173        false
174    }

176    /// Converts disjunctive equality chains to IN lists.
177    ///
178    /// `a = 1 or a = 2 or b = 3` → `a in (1, 2) or b = 3`
179    /// `a = 1 or a = 2 or b = 3 or b = 4` → `a in (1, 2) or b in (3, 4)`
180    ///
181    /// Groups equality comparisons by their LHS and converts groups with 2+
182    /// values into IN lists. Non-equality operands are preserved.
183    fn try_or_to_in_list(&self, expr: &mut stmt::ExprOr) -> Option<stmt::Expr> {
184        use hashbrown::HashMap;
185
186        // Group equalities by their LHS expression
187        // Key: index into `lhs_exprs`, Value: list of RHS constant values
188        //
189        // TODO: this could be simplified to use `Expr` as the `HashMap` key
190        // directly if `Expr` ever implements `Hash`.
191        let mut lhs_exprs: Vec<stmt::Expr> = Vec::new();
192        let mut groups: HashMap<usize, Vec<stmt::Value>> = HashMap::new();
193        let mut other_operands: Vec<stmt::Expr> = Vec::new();
194
195        for operand in mem::take(&mut expr.operands) {
196            if let stmt::Expr::BinaryOp(bin_op) = &operand
197                && bin_op.op.is_eq()
198                && let stmt::Expr::Value(value) = bin_op.rhs.as_ref()
199            {
200                // Find or create index for this LHS. `is_equivalent_to` is
201                // crucial here: if the lhs is non-deterministic (e.g. `RAND()`),
202                // it is never equivalent to another occurrence of itself, so
203                // each instance falls into its own singleton group and no IN
204                // list is produced. Rewriting `RAND() = 1 OR RAND() = 2` into
205                // `RAND() IN (1, 2)` would collapse two independent draws into
206                // one, which is unsound.
207                let lhs_idx = lhs_exprs
208                    .iter()
209                    .position(|e| e.is_equivalent_to(bin_op.lhs.as_ref()))
210                    .unwrap_or_else(|| {
211                        lhs_exprs.push(bin_op.lhs.as_ref().clone());
212                        lhs_exprs.len() - 1
213                    });
214
215                groups.entry(lhs_idx).or_default().push(value.clone());
216                continue;
217            }
218
219            // Non-equality or non-constant RHS - keep as is
220            other_operands.push(operand);
221        }
222
223        // Check if any conversion will happen (any group with 2+ values)
224        let has_conversion = groups.values().any(|v| v.len() >= 2);
225        if !has_conversion {
226            // Restore original operands and return None
227            for (idx, values) in groups {
228                let lhs = &lhs_exprs[idx];
229                for value in values {
230                    other_operands.push(stmt::Expr::eq(lhs.clone(), stmt::Expr::Value(value)));
231                }
232            }
233            expr.operands = other_operands;
234            return None;
235        }
236
237        // Build result operands
238        let mut result_operands = other_operands;
239
240        for (idx, values) in groups {
241            let lhs = lhs_exprs[idx].clone();
242            if values.len() >= 2 {
243                // Convert to IN list
244                result_operands.push(stmt::Expr::in_list(lhs, stmt::Expr::list(values)));
245            } else {
246                // Keep as equality
247                for value in values {
248                    result_operands.push(stmt::Expr::eq(lhs.clone(), stmt::Expr::Value(value)));
249                }
250            }
251        }
252
253        if result_operands.len() == 1 {
254            Some(result_operands.remove(0))
255        } else {
256            expr.operands = result_operands;
257            None
258        }
259    }

1use super::Simplify;
2use toasty_core::stmt::{self, visit_mut};
3
4impl Simplify<'_> {
5    /// Inline a stable `Let` expression by substituting its bindings into the
6    /// body.
7    ///
8    /// A `Let` is inlinable when every binding is *stable* — that is, fully
9    /// resolved with no remaining `Stmt` nodes. After lowering converts
10    /// `Stmt → Arg`, the bindings become stable and the Let can be removed by
11    /// substituting each `Arg(position=i, nesting=0)` in the body with
12    /// `bindings[i]`.
13    pub(super) fn simplify_expr_let(&self, expr_let: &mut stmt::ExprLet) -> Option<stmt::Expr> {
14        if !expr_let.bindings.iter().all(|b| b.is_stable()) {
15            return None;
16        }
17
18        let mut body = *expr_let.body.clone();
19        substitute_let_bindings(&mut body, &expr_let.bindings);
20        Some(body)
21    }
22}
23
24/// Substitute Let bindings into a body expression.
25///
26/// - `Arg(position=i, nesting==depth)` where `i < bindings.len()` is replaced
27///   with `bindings[i].clone()`.
28/// - `Arg(nesting > depth)` gets `nesting -= 1` (the Let scope is removed).
29///
30/// Uses `walk_expr_scoped_mut` to automatically track scope depth through
31/// Let/Map scopes.
32fn substitute_let_bindings(expr: &mut stmt::Expr, bindings: &[stmt::Expr]) {
33    visit_mut::walk_expr_scoped_mut(expr, 0, |expr, scope_depth| match expr {
34        stmt::Expr::Arg(arg) if arg.nesting == scope_depth && arg.position < bindings.len() => {
35            *expr = bindings[arg.position].clone();
36            false
37        }
38        stmt::Expr::Arg(arg) if arg.nesting > scope_depth => {
39            arg.nesting -= 1;
40            false
41        }
42        _ => true,
43    });
44}

6/// A typed delete statement.
7///
8/// `Delete<T>` removes records that match the selection built by the
9/// originating [`Query`]. Obtain one by calling [`Query::delete`].
10///
11/// The type parameter `T` is the **returning type**, not the model being
12/// deleted. Currently `Query::delete` always produces `Delete<()>` because
13/// deletes do not return the removed records.
14///
15/// # Execution
16///
17/// Call [`exec`](Delete::exec) to run the delete, or convert it into a
18/// [`Statement`] with [`IntoStatement`] for batch execution.
19///
20/// ```
21/// # tokio::runtime::Runtime::new().unwrap().block_on(async {
22/// # #[derive(Debug, toasty::Model)]
23/// # struct User {
24/// #     #[key]
25/// #     id: i64,
26/// #     name: String,
27/// # }
28/// # let driver = toasty_driver_sqlite::Sqlite::in_memory();
29/// # let mut db = toasty::Db::builder().models(toasty::models!(User)).build(driver).await.unwrap();
30/// # db.push_schema().await.unwrap();
31/// use toasty::stmt::{List, Query};
32///
33/// Query::<List<User>>::filter(User::fields().id().eq(1))
34///     .delete()
35///     .exec(&mut db)
36///     .await
37///     .unwrap();
38/// # });
39/// ```
40pub struct Delete<T> {
41    pub(crate) untyped: stmt::Delete,
42    _p: PhantomData<T>,
43}
44
45impl<T> Delete<T> {
46    /// Wrap a raw untyped [`stmt::Delete`](toasty_core::stmt::Delete).
47    ///
48    /// # Examples
49    ///
50    /// ```
51    /// # #[derive(Debug, toasty::Model)]
52    /// # struct User {
53    /// #     #[key]
54    /// #     id: i64,
55    /// #     name: String,
56    /// # }
57    /// use toasty::stmt::{Delete, List, Query};
58    ///
59    /// // Build a delete from a query
60    /// let delete: Delete<()> = Query::<List<User>>::all().delete();
61    /// ```
62    pub const fn from_untyped(untyped: stmt::Delete) -> Self {
63        Self {
64            untyped,
65            _p: PhantomData,
66        }
67    }
68}

70impl<T> Delete<T> {
71    /// Attach a condition to this delete. The condition is evaluated after the
72    /// filter; if it fails, the operation returns an error (unlike a filter
73    /// failure, which silently produces count 0).
74    pub fn set_condition(mut self, condition: toasty_core::stmt::Condition) -> Self {
75        self.untyped.condition = condition;
76        self
77    }
78}

80impl<T: Load> Delete<T> {
81    /// Execute this delete statement against the given executor.
82    ///
83    /// Returns `Ok(())` on success. Any matching records are removed from the
84    /// database.
85    ///
86    /// # Examples
87    ///
88    /// ```
89    /// # tokio::runtime::Runtime::new().unwrap().block_on(async {
90    /// # #[derive(Debug, toasty::Model)]
91    /// # struct User {
92    /// #     #[key]
93    /// #     id: i64,
94    /// #     name: String,
95    /// # }
96    /// # let driver = toasty_driver_sqlite::Sqlite::in_memory();
97    /// # let mut db = toasty::Db::builder().models(toasty::models!(User)).build(driver).await.unwrap();
98    /// # db.push_schema().await.unwrap();
99    /// User::filter(User::fields().id().eq(1))
100    ///     .delete()
101    ///     .exec(&mut db)
102    ///     .await
103    ///     .unwrap();
104    /// # });
105    /// ```
106    pub async fn exec(self, executor: &mut dyn Executor) -> Result<T::Output> {
107        executor.exec(self.into()).await
108    }
109}

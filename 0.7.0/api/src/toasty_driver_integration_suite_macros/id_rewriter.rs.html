1use syn::{Type, TypePath, visit_mut::VisitMut};
2
3/// Visitor that rewrites type references to a configurable target type
4pub(crate) struct IdRewriter {
5    /// The identifier to replace (e.g., "ID")
6    ident: String,
7    /// The target type to replace with
8    target_type: Type,
9}
10
11impl IdRewriter {
12    pub(crate) fn new(ident: &str, target_type: Type) -> Self {
13        Self {
14            ident: ident.to_string(),
15            target_type,
16        }
17    }
18}
19
20impl VisitMut for IdRewriter {
21    fn visit_type_mut(&mut self, ty: &mut Type) {
22        if let Type::Path(TypePath { qself: None, path }) = ty {
23            // Check if this matches the identifier we're looking for
24            if path.segments.len() == 1 && path.segments[0].ident == self.ident {
25                // Replace with target type
26                *ty = self.target_type.clone();
27                return;
28            }
29        }
30
31        // Continue visiting nested types
32        syn::visit_mut::visit_type_mut(self, ty);
33    }
34}

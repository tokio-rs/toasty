6/// The parsed `update!` invocation.
7///
8/// `update!(target_expr { field_entry, ... })`
9///
10/// The target is any Rust expression you can call `.update()` on — a
11/// model instance (`&mut user`), a query builder
12/// (`User::filter_by_id(id)`), or a scoped query
13/// (`user.todos().filter_by_done(false)`).
14pub(crate) struct UpdateItem {
15    pub target: syn::Expr,
16    pub fields: FieldSet,
17}
18
19/// A set of field entries — the contents of the `{ ... }` block.
20pub(crate) struct FieldSet(pub Vec<FieldEntry>);
21
22/// A single field entry. The macro accepts four shapes:
23///
24/// - `name: value` — explicit value (set semantics by default)
25/// - `name` — shorthand for `name: name`
26/// - `name.method(args)` — method-call shorthand for
27///   `name: stmt::method(args)`
28pub(crate) struct FieldEntry {
29    pub name: syn::Ident,
30    pub value: FieldValue,
31}
32
33/// The right-hand side of a field entry.
34pub(crate) enum FieldValue {
35    /// Plain expression: literals, variables, function calls,
36    /// `stmt::*` builders the user spelled out.
37    Expr(Box<syn::Expr>),
38    /// Anonymous struct literal `{ sub: val, ... }` — for embedded-type
39    /// partial updates. Expands to a `stmt::apply([stmt::patch(...)])`
40    /// chain.
41    Single(FieldSet),
42    /// Array literal `[item, item, ...]`. Each item is itself a
43    /// `FieldValue`. For has-many fields, `Single` items become
44    /// `stmt::insert(...)` and the whole list is wrapped in
45    /// `stmt::apply([...])`. For lists of plain `Expr` items the array
46    /// passes through unchanged (set semantics, same as the chain form).
47    List(Vec<FieldValue>),
48    /// Method-call shorthand `name.method(args)` parsed from a field
49    /// entry without an explicit value. The `combinator` ident names a
50    /// function in the `toasty::stmt` module.
51    Method {
52        combinator: syn::Ident,
53        args: Punctuated<syn::Expr, syn::Token![,]>,
54    },
55}

57impl Parse for UpdateItem {
58    fn parse(input: ParseStream) -> syn::Result<Self> {
59        // Parse the target expression. Use `parse_without_eager_brace` so
60        // the parser doesn't eat `{ fields }` as a block expression — the
61        // brace block belongs to the macro, not the target expression.
62        let target = syn::Expr::parse_without_eager_brace(input)?;
63        let fields = parse_braced_fields(input)?;
64        Ok(UpdateItem { target, fields })
65    }
66}
67
68fn parse_braced_fields(input: ParseStream) -> syn::Result<FieldSet> {
69    let content;
70    braced!(content in input);
71    let entries = Punctuated::<FieldEntry, syn::Token![,]>::parse_terminated(&content)?;
72    Ok(FieldSet(entries.into_iter().collect()))
73}
74
75impl Parse for FieldSet {
76    fn parse(input: ParseStream) -> syn::Result<Self> {
77        parse_braced_fields(input)
78    }
79}

81impl Parse for FieldEntry {
82    fn parse(input: ParseStream) -> syn::Result<Self> {
83        let name = input.parse::<syn::Ident>()?;
84
85        // Decide what follows the field name. Each shape has a distinct
86        // lookahead token so there is no ambiguity:
87        //
88        //   `name: value`       → `:`
89        //   `name`              → `,` or `}` (end-of-entry)
90        //   `name.method(args)` → `.`
91        if input.peek(syn::Token![:]) && !input.peek(syn::Token![::]) {
92            // `name: value`
93            input.parse::<syn::Token![:]>()?;
94            let value = input.parse()?;
95            Ok(FieldEntry { name, value })
96        } else if input.peek(syn::Token![.]) {
97            // `name.method(args)` — method-call shorthand
98            input.parse::<syn::Token![.]>()?;
99            let combinator = input.parse::<syn::Ident>()?;
100
101            let args_content;
102            syn::parenthesized!(args_content in input);
103            let args = Punctuated::parse_terminated(&args_content)?;
104
105            // The shorthand is one method call deep. Anything after the
106            // closing paren that is not an entry terminator is an error.
107            if !input.is_empty() && !input.peek(syn::Token![,]) && !input.peek(token::Brace) {
108                return Err(input.error(
109                    "method shorthand is one call deep — chained calls are not supported; \
110                     use the explicit `field: expr` form for chained expressions",
111                ));
112            }
113
114            Ok(FieldEntry {
115                name,
116                value: FieldValue::Method { combinator, args },
117            })
118        } else {
119            // `name` shorthand — equivalent to `name: name`
120            let expr = syn::Expr::Path(syn::ExprPath {
121                attrs: vec![],
122                qself: None,
123                path: name.clone().into(),
124            });
125            Ok(FieldEntry {
126                name,
127                value: FieldValue::Expr(Box::new(expr)),
128            })
129        }
130    }
131}

133impl Parse for FieldValue {
134    fn parse(input: ParseStream) -> syn::Result<Self> {
135        if input.peek(token::Brace) {
136            // Nested struct literal `{ sub: value, ... }` — partial
137            // update of an embedded field.
138            let fields = parse_braced_fields(input)?;
139            Ok(FieldValue::Single(fields))
140        } else if input.peek(token::Bracket) {
141            // List literal `[item, item, ...]`. Items can themselves be
142            // brace blocks (nested builders for has-many insert) or
143            // plain expressions.
144            let content;
145            bracketed!(content in input);
146            let items = Punctuated::<FieldValue, syn::Token![,]>::parse_terminated(&content)?;
147            Ok(FieldValue::List(items.into_iter().collect()))
148        } else {
149            // Anything else is a plain Rust expression.
150            let expr = input.parse::<syn::Expr>()?;
151            Ok(FieldValue::Expr(Box::new(expr)))
152        }
153    }
154}

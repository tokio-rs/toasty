3/// ILIKE with an uppercase pattern matches rows regardless of their case.
4/// `.ilike()` maps to PostgreSQL's native `ILIKE`; backends without one reject
5/// the query (see `filter_ilike_unsupported`), so these tests are gated on
6/// `native_ilike`.
7#[driver_test(requires(native_ilike), scenario(crate::scenarios::fixed_item_name))]
8pub async fn ilike_uppercase_pattern(t: &mut Test) -> Result<()> {
9    let mut db = setup(t).await;
10
11    toasty::create!(Item::[
12        { id: 1_i64, name: "Alice"   },
13        { id: 2_i64, name: "ALICIA"  },
14        { id: 3_i64, name: "alfred"  },
15        { id: 4_i64, name: "Bob"     },
16        { id: 5_i64, name: "BARRY"   },
17    ])
18    .exec(&mut db)
19    .await?;
20
21    let mut items: Vec<Item> = Item::filter(Item::fields().name().ilike("AL%".to_string()))
22        .exec(&mut db)
23        .await?;
24
25    items.sort_by_key(|i| i.id);
26
27    assert_eq!(items.len(), 3);
28    assert_eq!(items[0].name, "Alice");
29    assert_eq!(items[1].name, "ALICIA");
30    assert_eq!(items[2].name, "alfred");
31
32    Ok(())
33}

89/// ILIKE on an `Option<String>` field — matches non-null values regardless of
90/// case; rows with NULL values are excluded since `NULL ILIKE pattern` is
91/// unknown.
92#[driver_test(requires(native_ilike))]
93pub async fn ilike_optional_field(test: &mut Test) -> Result<()> {
94    #[derive(Debug, toasty::Model)]
95    struct OptItem {
96        #[key]
97        id: i64,
98        nickname: Option<String>,
99    }
100
101    let mut db = test.setup_db(models!(OptItem)).await;
102
103    toasty::create!(OptItem::[
104        { id: 1_i64, nickname: Some("Alice".to_string())   },
105        { id: 2_i64, nickname: Some("ALICIA".to_string())  },
106        { id: 3_i64, nickname: Some("alfred".to_string())  },
107        { id: 4_i64, nickname: Some("Bob".to_string())     },
108        { id: 5_i64, nickname: None                        },
109    ])
110    .exec(&mut db)
111    .await?;
112
113    let mut items: Vec<OptItem> =
114        OptItem::filter(OptItem::fields().nickname().ilike("al%".to_string()))
115            .exec(&mut db)
116            .await?;
117
118    items.sort_by_key(|i| i.id);
119
120    assert_eq!(items.len(), 3);
121    assert_eq!(items[0].nickname.as_deref(), Some("Alice"));
122    assert_eq!(items[1].nickname.as_deref(), Some("ALICIA"));
123    assert_eq!(items[2].nickname.as_deref(), Some("alfred"));
124
125    Ok(())
126}

128/// On a SQL backend without a native `ILIKE` operator (SQLite, MySQL), `.ilike()`
129/// is rejected with an unsupported-feature error rather than silently degrading
130/// to plain `LIKE`.
131#[driver_test(
132    requires(and(sql, not(native_ilike))),
133    scenario(crate::scenarios::fixed_item_name)
134)]
135pub async fn ilike_unsupported_without_native_operator(t: &mut Test) -> Result<()> {
136    let mut db = setup(t).await;
137
138    toasty::create!(Item::[
139        { id: 1_i64, name: "Alice"   },
140        { id: 2_i64, name: "ALICIA"  },
141        { id: 3_i64, name: "alfred"  },
142        { id: 4_i64, name: "Bob"     },
143        { id: 5_i64, name: "BARRY"   },
144    ])
145    .exec(&mut db)
146    .await?;
147
148    let result: Result<Vec<Item>> = Item::filter(Item::fields().name().ilike("al%".to_string()))
149        .exec(&mut db)
150        .await;
151    let err = result.expect_err("ilike should be unsupported on a backend without a native ILIKE");
152
153    assert!(err.is_unsupported_feature(), "unexpected error: {err:?}");
154
155    Ok(())
156}

38/// Serialize a statement to a SQL string
39#[derive(Debug)]
40pub struct Serializer<'a> {
41    /// Schema against which the statement is to be serialized
42    schema: &'a db::Schema,
43
44    /// The database flavor handles the differences between SQL dialects and
45    /// supported features.
46    flavor: Flavor,
47
48    /// SQL emitted for [`TransactionMode::Default`] under the SQLite flavor.
49    /// Constructors that don't override this leave it at `"BEGIN"`, which is
50    /// SQLite's natural default (DEFERRED). A driver that wants `Default` to
51    /// mean something engine-specific — Turso under `concurrent_writes()`
52    /// uses `"BEGIN CONCURRENT"` — sets this through
53    /// [`Self::sqlite_with_default_begin`]. The non-`Default` variants
54    /// (`Deferred`, `Immediate`, `Exclusive`) always emit fixed SQL.
55    sqlite_default_begin: &'static str,
56}
57
58struct Formatter<'a> {
59    /// Handle to the serializer
60    serializer: &'a Serializer<'a>,
61
62    /// Expression-resolution context for the current scope. Re-scoped (via
63    /// [`Formatter::scope`]) each time serialization descends into a new
64    /// query level, so it travels with the formatter rather than as a
65    /// separate argument.
66    cx: ExprContext<'a>,
67
68    /// Where to write the serialized SQL
69    dst: &'a mut String,
70
71    /// Current query depth. This is used to determine the nesting level when
72    /// generating names
73    depth: usize,
74
75    /// True when table names should be aliased.
76    alias: bool,
77
78    /// True when inside an INSERT statement. Used by MySQL to decide whether
79    /// VALUES rows need the ROW() wrapper (required in subqueries but not in
80    /// INSERT).
81    in_insert: bool,
82
83    /// Target table whose stored columns must be qualified inside a PostgreSQL
84    /// upsert assignment to distinguish them from `excluded` columns.
85    assignment_table: Option<db::TableId>,
86
87    /// Collects `Expr::Arg(n)` positions in the order they appear in the SQL.
88    /// Used by MySQL (which uses positional `?` without indices) to reorder
89    /// the params vec to match placeholder occurrence order. Borrowed so a
90    /// scoped child formatter writes through to the root's vec.
91    arg_positions: &'a mut Vec<usize>,
92}
93
94impl<'a> Formatter<'a> {
95    /// Descend into a new expression scope, returning a child formatter that
96    /// shares this one's output sink and arg collector (so writes flow back
97    /// to the root) but resolves references against `target`.
98    ///
99    /// The child borrows `self`, so the parent scope stays live on the stack
100    /// for the child's lifetime — that is what keeps the `ExprContext` parent
101    /// chain valid for nested-reference resolution.
102    fn scope<'c>(&'c mut self, target: impl IntoExprTarget<'c, db::Schema>) -> Formatter<'c> {
103        Formatter {
104            serializer: self.serializer,
105            cx: self.cx.scope(target),
106            dst: &mut *self.dst,
107            depth: self.depth,
108            alias: self.alias,
109            in_insert: self.in_insert,
110            assignment_table: self.assignment_table,
111            arg_positions: &mut *self.arg_positions,
112        }
113    }
114}

116/// Expression context bound to a database-level schema.
117pub type ExprContext<'a> = toasty_core::stmt::ExprContext<'a, db::Schema>;

119impl<'a> Serializer<'a> {
120    /// Serializes a [`Statement`] to a SQL string with all values inlined as
121    /// literals (no bind parameters). Appends a trailing semicolon.
122    ///
123    /// Use this for DDL statements (`CREATE TABLE`, `CREATE TYPE`, etc.) where
124    /// bind parameters are not supported. DML statements should already have
125    /// their parameters extracted (as `Expr::Arg` placeholders) before reaching
126    /// the serializer.
127    pub fn serialize(&self, stmt: &Statement) -> String {
128        self.serialize_with_arg_order(stmt).0
129    }
130
131    /// Serializes a [`Statement`] and returns both the SQL string and the order
132    /// in which `Expr::Arg(n)` placeholders appear in the SQL.
133    ///
134    /// The arg order is needed by MySQL which uses positional `?` without
135    /// indices — the caller must reorder its params vec to match the occurrence
136    /// order. PostgreSQL and SQLite use indexed placeholders (`$1`, `?1`) so
137    /// they can ignore the arg order.
138    pub fn serialize_with_arg_order(&self, stmt: &Statement) -> (String, Vec<usize>) {
139        let mut ret = String::new();
140        let mut arg_positions = Vec::new();
141
142        {
143            let mut fmt = Formatter {
144                serializer: self,
145                cx: ExprContext::new(self.schema),
146                dst: &mut ret,
147                depth: 0,
148                alias: false,
149                in_insert: false,
150                assignment_table: None,
151                arg_positions: &mut arg_positions,
152            };
153
154            stmt.to_sql(&mut fmt);
155        }
156
157        ret.push(';');
158        (ret, arg_positions)
159    }
160
161    /// Serialize a transaction control operation to a SQL string.
162    ///
163    /// The generated SQL is flavor-specific (e.g., MySQL uses `START TRANSACTION`
164    /// while other databases use `BEGIN`). Savepoints are named `sp_{id}`.
165    pub fn serialize_transaction(&self, op: &Transaction) -> String {
166        let mut ret = String::new();
167        let mut arg_positions = Vec::new();
168
169        {
170            let mut f = Formatter {
171                serializer: self,
172                cx: ExprContext::new(self.schema),
173                dst: &mut ret,
174                depth: 0,
175                alias: false,
176                in_insert: false,
177                assignment_table: None,
178                arg_positions: &mut arg_positions,
179            };
180
181            match op {
182                Transaction::Start {
183                    isolation,
184                    read_only,
185                    mode,
186                } => fmt!(
187                    &mut f,
188                    self.serialize_transaction_start(*isolation, *read_only, *mode)
189                ),
190                Transaction::Commit => fmt!(&mut f, "COMMIT"),
191                Transaction::Rollback => fmt!(&mut f, "ROLLBACK"),
192                Transaction::Savepoint(name) => {
193                    fmt!(&mut f, "SAVEPOINT " Ident(name))
194                }
195                Transaction::ReleaseSavepoint(name) => {
196                    fmt!(&mut f, "RELEASE SAVEPOINT " Ident(name))
197                }
198                Transaction::RollbackToSavepoint(name) => {
199                    fmt!(&mut f, "ROLLBACK TO SAVEPOINT " Ident(name))
200                }
201            };
202        }
203
204        ret.push(';');
205        ret
206    }
207
208    fn serialize_transaction_start(
209        &self,
210        isolation: Option<IsolationLevel>,
211        read_only: bool,
212        mode: TransactionMode,
213    ) -> String {
214        fn isolation_level_str(level: IsolationLevel) -> &'static str {
215            match level {
216                IsolationLevel::ReadUncommitted => "READ UNCOMMITTED",
217                IsolationLevel::ReadCommitted => "READ COMMITTED",
218                IsolationLevel::RepeatableRead => "REPEATABLE READ",
219                IsolationLevel::Serializable => "SERIALIZABLE",
220            }
221        }
222
223        match self.flavor {
224            // MySQL has no SQLite-style lock-mode keyword; drivers
225            // reject non-Default `mode` before reaching the serializer.
226            Flavor::Mysql => {
227                let mut sql = String::new();
228                if let Some(level) = isolation {
229                    sql.push_str("SET TRANSACTION ISOLATION LEVEL ");
230                    sql.push_str(isolation_level_str(level));
231                    sql.push_str("; ");
232                }
233                sql.push_str("START TRANSACTION");
234                if read_only {
235                    sql.push_str(" READ ONLY");
236                }
237                sql
238            }
239            // PostgreSQL has no SQLite-style lock-mode keyword; drivers
240            // reject non-Default `mode` before reaching the serializer.
241            Flavor::Postgresql => {
242                let mut sql = String::from("BEGIN");
243                if let Some(level) = isolation {
244                    sql.push_str(" ISOLATION LEVEL ");
245                    sql.push_str(isolation_level_str(level));
246                }
247                if read_only {
248                    sql.push_str(" READ ONLY");
249                }
250                sql
251            }
252            // SQLite has no per-transaction isolation level or read-only
253            // keyword; the lock-acquisition mode is the only knob. `Default`
254            // emits whatever the serializer was configured with at
255            // construction (`BEGIN` by default, or e.g. `BEGIN CONCURRENT`
256            // for Turso under MVCC). `Deferred`/`Immediate`/`Exclusive` are
257            // explicit caller requests with fixed SQL.
258            Flavor::Sqlite => match mode {
259                TransactionMode::Default => self.sqlite_default_begin.to_string(),
260                TransactionMode::Deferred => "BEGIN".to_string(),
261                TransactionMode::Immediate => "BEGIN IMMEDIATE".to_string(),
262                TransactionMode::Exclusive => "BEGIN EXCLUSIVE".to_string(),
263            },
264        }
265    }

51/// A DynamoDB [`Driver`] backed by the AWS SDK.
52///
53/// Create one with [`DynamoDb::from_env`] to load AWS credentials and region
54/// from the environment, or [`DynamoDb::new`] / [`DynamoDb::with_sdk_config`]
55/// for manual setup.
56#[derive(Debug, Clone)]
57pub struct DynamoDb {
58    url: String,
59    client: Client,
60}
61
62impl DynamoDb {
63    /// Create driver with pre-built client (backward compatible, synchronous)
64    pub fn new(url: String, client: Client) -> Self {
65        Self { url, client }
66    }
67
68    /// Create driver loading AWS config from environment (async factory)
69    /// Reads: AWS_REGION, AWS_ENDPOINT_URL_DYNAMODB, AWS credentials, etc.
70    pub async fn from_env(url: String) -> Result<Self> {
71        use aws_config::BehaviorVersion;
72
73        let sdk_config = aws_config::defaults(BehaviorVersion::latest()).load().await;
74        let client = Client::new(&sdk_config);
75        Ok(Self::new(url, client))
76    }
77
78    /// Create driver with custom SdkConfig (synchronous)
79    pub fn with_sdk_config(url: String, sdk_config: &aws_config::SdkConfig) -> Self {
80        let client = Client::new(sdk_config);
81        Self::new(url, client)
82    }
83}

85#[async_trait]
86impl Driver for DynamoDb {
87    fn url(&self) -> Cow<'_, str> {
88        Cow::Borrowed(&self.url)
89    }
90
91    fn capability(&self) -> &'static Capability {
92        &Capability::DYNAMODB
93    }
94
95    async fn connect(
96        &self,
97        cx: &ConnectContext,
98    ) -> toasty_core::Result<Box<dyn toasty_core::driver::Connection>> {
99        // Clone the shared client - cheap operation (Client uses Arc internally)
100        let mut connection = Connection::new(self.client.clone());
101        connection.query_log = cx.query_log;
102        Ok(Box::new(connection))
103    }
104
105    fn generate_migration(&self, _schema_diff: &diff::Schema<'_>) -> Migration {
106        unimplemented!(
107            "DynamoDB migrations are not yet supported. DynamoDB schema changes require manual table updates through the AWS console or SDK."
108        )
109    }
110
111    async fn reset_db(&self) -> toasty_core::Result<()> {
112        // Use shared client directly
113        let mut exclusive_start_table_name = None;
114        loop {
115            let mut req = self.client.list_tables();
116            if let Some(start) = &exclusive_start_table_name {
117                req = req.exclusive_start_table_name(start);
118            }
119
120            let resp = req
121                .send()
122                .await
123                .map_err(toasty_core::Error::driver_operation_failed)?;
124
125            if let Some(table_names) = &resp.table_names {
126                for table_name in table_names {
127                    self.client
128                        .delete_table()
129                        .table_name(table_name)
130                        .send()
131                        .await
132                        .map_err(toasty_core::Error::driver_operation_failed)?;
133                }
134            }
135
136            exclusive_start_table_name = resp.last_evaluated_table_name;
137            if exclusive_start_table_name.is_none() {
138                break;
139            }
140        }
141
142        Ok(())
143    }
144}

164/// Resolves the table an operation targets, for the per-query event.
165fn op_table_name<'a>(schema: &'a Schema, op: &Operation) -> Option<&'a str> {
166    let table_id = match op {
167        Operation::GetByKey(op) => op.table,
168        Operation::QueryPk(op) => op.table,
169        Operation::DeleteByKey(op) => op.table,
170        Operation::UpdateByKey(op) => op.table,
171        Operation::Upsert(op) => op.stmt.target.as_table_unwrap().table,
172        Operation::FindPkByIndex(op) => op.table,
173        Operation::Scan(op) => op.table,
174        _ => return None,
175    };
176    Some(&schema.db.table(table_id).name)
177}

181    async fn exec(&mut self, schema: &Arc<Schema>, op: Operation) -> Result<ExecResponse> {
182        let log = QueryLog::operation(
183            &self.query_log,
184            "dynamodb",
185            op.name(),
186            op_table_name(schema, &op),

188        let result = self.exec2(schema, op).await;

195            tracing::debug!(table = %table.name, "creating table");

197        }
198        Ok(())
199    }

217impl Connection {
218    async fn exec2(&mut self, schema: &Arc<Schema>, op: Operation) -> Result<ExecResponse> {
219        match op {
220            Operation::GetByKey(op) => self.exec_get_by_key(schema, op).await,
221            Operation::QueryPk(op) => self.exec_query_pk(schema, op).await,
222            Operation::DeleteByKey(op) => self.exec_delete_by_key(&schema.db, op).await,
223            Operation::UpdateByKey(op) => self.exec_update_by_key(&schema.db, op).await,
224            Operation::Upsert(op) => self.exec_upsert(&schema.db, op).await,
225            Operation::FindPkByIndex(op) => self.exec_find_pk_by_index(schema, op).await,
226            Operation::QuerySql(op) => {
227                assert!(
228                    op.last_insert_id_hack.is_none(),
229                    "last_insert_id_hack is MySQL-specific and should not be set for DynamoDB"
230                );
231                match op.stmt {
232                    stmt::Statement::Insert(insert) => self.exec_insert(&schema.db, insert).await,
233                    _ => todo!("op={:#?}", op.stmt),
234                }
235            }
236            Operation::Scan(op) => self.exec_scan(schema, op).await,
237            Operation::RawSql(_) => Err(Error::unsupported_feature(
238                "raw SQL is only supported by SQL drivers",
239            )),
240            Operation::Transaction(_) => Err(Error::unsupported_feature(
241                "transactions are not supported by the DynamoDB driver",
242            )),
243            _ => todo!("op={op:#?}"),
244        }
245    }
246}

248fn ddb_key(table: &Table, key: &stmt::Value) -> HashMap<String, AttributeValue> {
249    let mut ret = HashMap::new();
250
251    for (index, column) in table.primary_key_columns().enumerate() {
252        let value = match key {
253            stmt::Value::Record(record) => &record[index],
254            value => value,
255        };
256
257        ret.insert(column.name.clone(), Value::from(value.clone()).to_ddb());
258    }
259
260    ret
261}

263/// Convert a DynamoDB AttributeValue to stmt::Value (type-inferred).
264fn attr_value_to_stmt_value(attr: &AttributeValue) -> stmt::Value {
265    use AttributeValue as AV;
266
267    match attr {
268        AV::S(s) => stmt::Value::String(s.clone()),
269        AV::N(n) => {
270            // Try to parse as i64 first (most common), fallback to string
271            n.parse::<i64>()
272                .map(stmt::Value::I64)
273                .unwrap_or_else(|_| stmt::Value::String(n.clone()))
274        }
275        AV::Bool(b) => stmt::Value::Bool(*b),
276        AV::B(bytes) => stmt::Value::Bytes(bytes.clone().into_inner()),
277        AV::Null(_) => stmt::Value::Null,
278        // For complex types, convert to string representation
279        _ => stmt::Value::String(format!("{:?}", attr)),
280    }
281}

283/// Serialize a DynamoDB LastEvaluatedKey (for pagination) into stmt::Value.
284/// Format: flat record [name1, value1, name2, value2, ...]
285/// Example: { "pk": S("abc"), "sk": N("42") } → Record([String("pk"), String("abc"), String("sk"), I64(42)])
286fn serialize_ddb_cursor(last_key: &HashMap<String, AttributeValue>) -> stmt::Value {
287    let mut fields = Vec::with_capacity(last_key.len() * 2);
288
289    for (name, attr_value) in last_key {
290        fields.push(stmt::Value::String(name.clone()));
291        fields.push(attr_value_to_stmt_value(attr_value));
292    }
293
294    stmt::Value::Record(stmt::ValueRecord::from_vec(fields))
295}

297/// Deserialize a stmt::Value cursor into a DynamoDB ExclusiveStartKey.
298/// Expects flat record format: [name1, value1, name2, value2, ...]
299fn deserialize_ddb_cursor(cursor: &stmt::Value) -> HashMap<String, AttributeValue> {
300    let mut ret = HashMap::new();
301
302    if let stmt::Value::Record(fields) = cursor {
303        // Process pairs: [name, value, name, value, ...]
304        for chunk in fields.chunks(2) {
305            if chunk.len() == 2
306                && let (stmt::Value::String(name), value) = (&chunk[0], &chunk[1])
307            {
308                ret.insert(name.clone(), Value::from(value.clone()).to_ddb());
309            }
310        }
311    }
312
313    ret
314}

362fn ddb_expression(
363    cx: &ExprContext<'_, db::Schema>,
364    attrs: &mut ExprAttrs,
365    primary: bool,
366    expr: &stmt::Expr,
367) -> String {
368    match expr {
369        stmt::Expr::Between(expr_between) => {
370            let field = ddb_expression(cx, attrs, primary, &expr_between.expr);
371            let low = ddb_expression(cx, attrs, primary, &expr_between.low);
372            let high = ddb_expression(cx, attrs, primary, &expr_between.high);
373            format!("{field} BETWEEN {low} AND {high}")
374        }
375        stmt::Expr::BinaryOp(expr_binary_op) => {
376            let lhs = ddb_expression(cx, attrs, primary, &expr_binary_op.lhs);
377            let rhs = ddb_expression(cx, attrs, primary, &expr_binary_op.rhs);
378
379            match expr_binary_op.op {
380                stmt::BinaryOp::Eq => format!("{lhs} = {rhs}"),
381                stmt::BinaryOp::Ne if primary => {
382                    todo!("!= conditions on primary key not supported")
383                }
384                stmt::BinaryOp::Ne => format!("{lhs} <> {rhs}"),
385                stmt::BinaryOp::Gt => format!("{lhs} > {rhs}"),
386                stmt::BinaryOp::Ge => format!("{lhs} >= {rhs}"),
387                stmt::BinaryOp::Lt => format!("{lhs} < {rhs}"),
388                stmt::BinaryOp::Le => format!("{lhs} <= {rhs}"),
389                // DynamoDB condition expressions don't support arithmetic
390                // between operands. Arithmetic ops belong in update
391                // expressions (handled by `update_by_key.rs`), not in
392                // condition/filter expressions.
393                stmt::BinaryOp::Add | stmt::BinaryOp::Sub => {
394                    todo!(
395                        "arithmetic operators in DynamoDB condition expressions are not supported"
396                    )
397                }
398            }
399        }
400        stmt::Expr::Reference(expr_reference) => {
401            let (column, col_alias) = column_alias(cx, attrs, expr_reference);
402            // A bare boolean column reference used as a predicate (result of
403            // `field = true` simplification) needs an explicit equality check.
404            if column.ty.is_bool() {
405                let true_val = attrs.ddb_value(aws_sdk_dynamodb::types::AttributeValue::Bool(true));
406                format!("{col_alias} = {true_val}")
407            } else {
408                col_alias
409            }
410        }
411        stmt::Expr::Value(val) => attrs.value(val),
412        // A projection into a `#[document]` column (`profile().name()`)
413        // arrives lowered as a `FuncJsonExtract` name path. DynamoDB filter
414        // expressions address nested Map attributes natively, so it renders
415        // as `#col_0.#doc_1.#doc_2`.
416        stmt::Expr::Func(stmt::ExprFunc::JsonExtract(func)) => {
417            let (path, leaf_ty) = document_path(cx, attrs, func);
418            // Like a bare bool column reference, a bool leaf in predicate
419            // position needs an explicit equality check.
420            if leaf_ty.is_bool() {
421                let true_val = attrs.ddb_value(aws_sdk_dynamodb::types::AttributeValue::Bool(true));
422                format!("{path} = {true_val}")
423            } else {
424                path
425            }
426        }
427        stmt::Expr::And(expr_and) => {
428            let operands = expr_and
429                .operands
430                .iter()
431                .map(|operand| ddb_expression(cx, attrs, primary, operand))
432                .collect::<Vec<_>>();
433            operands.join(" AND ")
434        }
435        stmt::Expr::Or(expr_or) => {
436            let operands = expr_or
437                .operands
438                .iter()
439                .map(|operand| ddb_expression(cx, attrs, primary, operand))
440                .collect::<Vec<_>>();
441            format!("({})", operands.join(" OR "))
442        }
443        stmt::Expr::InList(in_list) => {
444            let expr = ddb_expression(cx, attrs, primary, &in_list.expr);
445
446            // Extract the list items and create individual attribute values
447            let items = match &*in_list.list {
448                stmt::Expr::Value(stmt::Value::List(vals)) => vals
449                    .iter()
450                    .map(|val| attrs.value(val))
451                    .collect::<Vec<_>>()
452                    .join(", "),
453                _ => {
454                    // If it's not a literal list, treat it as a single expression
455                    ddb_expression(cx, attrs, primary, &in_list.list)
456                }
457            };
458
459            format!("{expr} IN ({items})")
460        }
461        stmt::Expr::IsNull(expr_is_null) => {
462            // `attribute_not_exists` takes a bare attribute path. Resolve a
463            // column alias or document path directly rather than through
464            // `ddb_expression`, which would expand a bool column/leaf to
465            // `#col = :true` — a comparison valid only in predicate position,
466            // not as a function argument. (Without this, `.is_none()` on any
467            // `Option<bool>` — including an `Option<Embed>` presence column —
468            // produces invalid syntax.)
469            let inner = match &*expr_is_null.expr {
470                stmt::Expr::Reference(expr_reference) => column_alias(cx, attrs, expr_reference).1,
471                stmt::Expr::Func(stmt::ExprFunc::JsonExtract(func)) => {
472                    document_path(cx, attrs, func).0
473                }
474                other => ddb_expression(cx, attrs, primary, other),
475            };
476            format!("attribute_not_exists({inner})")
477        }
478        stmt::Expr::Not(expr_not) => {
479            let inner = ddb_expression(cx, attrs, primary, &expr_not.expr);
480            format!("(NOT {inner})")
481        }
482        stmt::Expr::StartsWith(expr_starts_with) => {
483            let expr = ddb_expression(cx, attrs, primary, &expr_starts_with.expr);
484            let prefix = ddb_expression(cx, attrs, primary, &expr_starts_with.prefix);
485            format!("begins_with({expr}, {prefix})")
486        }
487        stmt::Expr::Like(_) => {
488            panic!(
489                "LIKE is not supported by the DynamoDB driver; use starts_with for prefix matching"
490            )
491        }
492        stmt::Expr::AnyOp(any) if matches!(any.op, stmt::BinaryOp::Eq) => {
493            // `Path::contains(value)` lowers to `value = ANY(col)`. On
494            // DynamoDB that's `contains(path, value)` — the standard List
495            // membership filter.
496            let value = ddb_expression(cx, attrs, primary, &any.lhs);
497            let path = ddb_expression(cx, attrs, primary, &any.rhs);
498            format!("contains({path}, {value})")
499        }
500        stmt::Expr::Length(expr) => {
501            let inner = ddb_expression(cx, attrs, primary, &expr.expr);
502            format!("size({inner})")
503        }
504        stmt::Expr::Cast(expr_cast) if expr_cast.ty == stmt::Type::Bool => {
505            // Bool key/index fields bridge through I8 (db::Type::Integer(1) via
506            // bridge_type). The lowering wraps the I8 column ref in
507            // Cast(col_ref, Bool) when the field appears as a bare predicate
508            // (result of `field = true` simplification). In predicate position
509            // this means "is true"; the `field = false` case arrives as
510            // Not(Cast(col_ref, Bool)) and is handled by the Not arm above.
511            let col_alias = ddb_expression(cx, attrs, primary, &expr_cast.expr);
512            let true_val =
513                attrs.ddb_value(aws_sdk_dynamodb::types::AttributeValue::N("1".to_string()));
514            format!("{col_alias} = {true_val}")
515        }
516        _ => todo!("FILTER = {:#?}", expr),
517    }
518}

520/// Resolves a column reference to its DynamoDB attribute alias (e.g. `#col_3`),
521/// registering the underlying attribute name in `attrs`. Returns the resolved
522/// column alongside the alias so callers can inspect its storage type.
523fn column_alias<'a>(
524    cx: &ExprContext<'a, db::Schema>,
525    attrs: &mut ExprAttrs,
526    expr_reference: &stmt::ExprReference,
527) -> (&'a Column, String) {
528    let column = cx.resolve_expr_reference(expr_reference).as_column_unwrap();
529    let alias = attrs.column(column).to_string();
530    (column, alias)
531}

533/// Renders a lowered document path ([`stmt::FuncJsonExtract`]) as a DynamoDB
534/// attribute path (`#col_0.#doc_1.#doc_2`), registering each path segment as
535/// an expression attribute name. The path arrives fully resolved from the
536/// engine's document lowering — segment names and leaf type live in the node,
537/// so no schema is consulted. Returns the rendered path and the leaf type.
538fn document_path<'a>(
539    cx: &ExprContext<'a, db::Schema>,
540    attrs: &mut ExprAttrs,
541    func: &'a stmt::FuncJsonExtract,
542) -> (String, &'a stmt::Type) {
543    let stmt::Expr::Reference(expr_reference) = func.base.as_ref() else {
544        todo!("document path base must be a column reference; func={func:#?}")
545    };
546    let (_, mut path) = column_alias(cx, attrs, expr_reference);
547
548    for name in &func.path {
549        path.push('.');
550        path.push_str(attrs.document_segment(name));
551    }
552
553    (path, &func.ty)
554}

556#[derive(Default)]
557struct ExprAttrs {
558    columns: HashMap<ColumnId, String>,
559    /// Placeholder per document path segment, keyed by the segment (field)
560    /// name so repeated mentions of the same key share one placeholder.
561    document_segments: HashMap<String, String>,
562    attr_names: HashMap<String, String>,
563    attr_values: HashMap<String, AttributeValue>,
564}
565
566impl ExprAttrs {
567    fn column(&mut self, column: &Column) -> &str {
568        use std::collections::hash_map::Entry;
569
570        match self.columns.entry(column.id) {
571            Entry::Vacant(e) => {
572                let name = format!("#col_{}", column.id.index);
573                self.attr_names.insert(name.clone(), column.name.clone());
574                e.insert(name)
575            }
576            Entry::Occupied(e) => e.into_mut(),
577        }
578    }
579
580    /// Registers one segment of a document path (a field name inside a Map
581    /// attribute) and returns its placeholder.
582    fn document_segment(&mut self, name: &str) -> &str {
583        use std::collections::hash_map::Entry;
584
585        match self.document_segments.entry(name.to_owned()) {
586            Entry::Vacant(e) => {
587                let placeholder = format!("#doc_{}", self.attr_names.len());
588                self.attr_names.insert(placeholder.clone(), name.to_owned());
589                e.insert(placeholder)
590            }
591            Entry::Occupied(e) => e.into_mut(),
592        }
593    }
594
595    fn value(&mut self, val: &stmt::Value) -> String {
596        self.ddb_value(Value::from(val.clone()).to_ddb())
597    }
598
599    fn ddb_value(&mut self, val: AttributeValue) -> String {
600        let i = self.attr_values.len();
601        let name = format!(":v_{i}");
602        self.attr_values.insert(name.clone(), val);
603        name
604    }

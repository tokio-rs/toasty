1use super::{
2    AttributeDefinition, BillingMode, Connection, GlobalSecondaryIndex, Projection, ProjectionType,
3    Result, Table, TypeExt, db, ddb_key_schema,
4};
5
6impl Connection {
7    pub(crate) async fn create_table(
8        &mut self,
9        schema: &db::Schema,
10        table: &Table,
11        reset: bool,
12    ) -> Result<()> {
13        if reset {
14            let _ = self
15                .client
16                .delete_table()
17                .table_name(&table.name)
18                .send()
19                .await;
20
21            for index in &table.indices {
22                if !index.primary_key && index.unique {
23                    let _ = self
24                        .client
25                        .delete_table()
26                        .table_name(&index.name)
27                        .send()
28                        .await;
29                }
30            }
31        }
32
33        // Calculate which attributes need to be defined
34        let mut defined_attributes = hashbrown::HashSet::new();
35
36        let pk_cols: Vec<&db::Column> = table.primary_key_columns().collect();
37
38        if pk_cols.is_empty() || pk_cols.len() > 2 {
39            return Err(toasty_core::Error::invalid_schema(format!(
40                "table '{}' primary key must have 1 or 2 columns, got {}",
41                table.name,
42                pk_cols.len()
43            )));
44        }
45
46        for col in &pk_cols {
47            defined_attributes.insert(col.id);
48        }
49
50        let pk_partition_cols = &pk_cols[..1];
51        let pk_range_cols = if pk_cols.len() > 1 {
52            &pk_cols[1..]
53        } else {
54            &[][..]
55        };
56
57        let mut gsis = vec![];
58
59        for index in &table.indices {
60            if index.primary_key {
61                continue;
62            }
63
64            if index.unique {
65                // Unique indices are materialized as separate tables below. Each
66                // index table is keyed on a single column and enforced with an
67                // `attribute_not_exists` condition, so composite (multi-column)
68                // unique indices are not supported. Reject them here, before any
69                // table is created, rather than partway through. SQL backends
70                // support them via `CREATE UNIQUE INDEX`.
71                if index.columns.len() != 1 {
72                    return Err(toasty_core::Error::unsupported_feature(format!(
73                        "DynamoDB does not support composite unique indices; \
74                         index '{}' spans {} columns",
75                        index.name,
76                        index.columns.len()
77                    )));
78                }
79
80                continue;
81            }
82
83            let partition_cols: Vec<&db::Column> = index
84                .columns
85                .iter()
86                .filter(|ic| ic.scope.is_partition())
87                .map(|ic| table.column(ic.column))
88                .collect();
89
90            let range_cols: Vec<&db::Column> = index
91                .columns
92                .iter()
93                .filter(|ic| ic.scope.is_local())
94                .map(|ic| table.column(ic.column))
95                .collect();
96
97            if partition_cols.is_empty() || partition_cols.len() > 4 {
98                return Err(toasty_core::Error::invalid_schema(format!(
99                    "GSI '{}' must have 1 to 4 partition (HASH) columns, got {}",
100                    index.name,
101                    partition_cols.len()
102                )));
103            }
104
105            if range_cols.len() > 4 {
106                return Err(toasty_core::Error::invalid_schema(format!(
107                    "GSI '{}' must have at most 4 range (RANGE) columns, got {}",
108                    index.name,
109                    range_cols.len()
110                )));
111            }
112
113            for col in &partition_cols {
114                defined_attributes.insert(col.id);
115            }
116
117            for col in &range_cols {
118                defined_attributes.insert(col.id);
119            }
120
121            gsis.push(
122                GlobalSecondaryIndex::builder()
123                    .index_name(&index.name)
124                    .set_key_schema(Some(ddb_key_schema(&partition_cols, &range_cols)))
125                    .projection(
126                        Projection::builder()
127                            .projection_type(ProjectionType::All)
128                            .build(),
129                    )
130                    .build()
131                    .unwrap(),
132            );
133        }
134
135        let attribute_definitions = defined_attributes
136            .iter()
137            .map(|column_id| {
138                let column = table.column(*column_id);
139                let ty = column.ty.to_ddb_type();
140
141                AttributeDefinition::builder()
142                    .attribute_name(&column.name)
143                    .attribute_type(ty)
144                    .build()
145                    .unwrap()
146            })
147            .collect();
148
149        self.client
150            .create_table()
151            .table_name(&table.name)
152            .set_attribute_definitions(Some(attribute_definitions))
153            .set_key_schema(Some(ddb_key_schema(pk_partition_cols, pk_range_cols)))
154            .set_global_secondary_indexes(if gsis.is_empty() { None } else { Some(gsis) })
155            .billing_mode(BillingMode::PayPerRequest)
156            .send()
157            .await
158            .map_err(toasty_core::Error::driver_operation_failed)?;
159
160        // Now, create separate tables for each unique index. Composite unique
161        // indices were already rejected above, so each index has one column.
162        for index in table.indices.iter().filter(|i| !i.primary_key && i.unique) {
163            let pk = schema.column(index.columns[0].column);
164
165            self.client
166                .create_table()
167                .table_name(&index.name)
168                .set_key_schema(Some(ddb_key_schema(&[pk], &[])))
169                .attribute_definitions(
170                    AttributeDefinition::builder()
171                        .attribute_name(&pk.name)
172                        .attribute_type(pk.ty.to_ddb_type())
173                        .build()
174                        .unwrap(),
175                )
176                .billing_mode(BillingMode::PayPerRequest)
177                .send()
178                .await
179                .map_err(toasty_core::Error::driver_operation_failed)?;
180        }
181
182        Ok(())
183    }

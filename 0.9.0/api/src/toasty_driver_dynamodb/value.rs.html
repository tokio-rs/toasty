20    /// Converts a DynamoDB AttributeValue to a Toasty value.
21    ///
22    /// Decoding is type-directed for scalar columns: `ty` (the column's
23    /// storage-level type) determines how each attribute is interpreted. A
24    /// `#[document]` column (`Type::Object`) decodes shape-directed instead —
25    /// a Map `M` attribute becomes the named `Value::Object` wire form
26    /// ([`from_ddb_any`](Self::from_ddb_any)), which the engine raises to the
27    /// embedded model's positional record. No schema is consulted here.
28    pub fn from_ddb(ty: &stmt::Type, val: &AttributeValue) -> Self {
29        use AttributeValue as AV;
30        use stmt::Type;
31
32        let core_value = match (ty, val) {
33            (Type::Bool, AV::Bool(val)) => stmt::Value::from(*val),
34            (Type::String, AV::S(val)) => stmt::Value::from(val.clone()),
35            (Type::I8, AV::N(val)) => stmt::Value::from(val.parse::<i8>().unwrap()),
36            (Type::I16, AV::N(val)) => stmt::Value::from(val.parse::<i16>().unwrap()),
37            (Type::I32, AV::N(val)) => stmt::Value::from(val.parse::<i32>().unwrap()),
38            (Type::I64, AV::N(val)) => stmt::Value::from(val.parse::<i64>().unwrap()),
39            (Type::U8, AV::N(val)) => stmt::Value::from(val.parse::<u8>().unwrap()),
40            (Type::U16, AV::N(val)) => stmt::Value::from(val.parse::<u16>().unwrap()),
41            (Type::U32, AV::N(val)) => stmt::Value::from(val.parse::<u32>().unwrap()),
42            (Type::U64, AV::N(val)) => stmt::Value::from(val.parse::<u64>().unwrap()),
43            (Type::F32, AV::N(val)) => stmt::Value::from(val.parse::<f32>().unwrap()),
44            (Type::F64, AV::N(val)) => stmt::Value::from(val.parse::<f64>().unwrap()),
45            (Type::Bytes, AV::B(val)) => stmt::Value::Bytes(val.clone().into_inner()),
46            (Type::Uuid, AV::S(val)) => stmt::Value::from(val.parse::<uuid::Uuid>().unwrap()),
47            (Type::List(elem), AV::L(items)) => {
48                let items = items
49                    .iter()
50                    .map(|item| Value::from_ddb(elem, item).into_inner())
51                    .collect();
52                stmt::Value::List(items)
53            }
54            // A `#[document]` column stored as a Map `M`: decode
55            // shape-directed to the named wire object. The engine raises it
56            // to the embed's positional record — the field layout is a
57            // model-level concept this driver does not know.
58            (Type::Object, val @ AV::M(_)) => Self::from_ddb_any(val),
59            (_, AV::Null(_)) => stmt::Value::Null,
60            // Scalars whose DynamoDB representation is their string form
61            // (temporals, decimals): recover the typed value through the same
62            // `Type::cast` conversions the engine uses for scalar columns.
63            (ty, AV::S(val)) => ty
64                .cast(&(), stmt::Value::String(val.clone()))
65                .expect("string attribute does not convert to the column type"),
66            _ => todo!("ty={:#?}; value={:#?}", ty, val),
67        };
68
69        Value(core_value)
70    }

72    /// Converts a DynamoDB AttributeValue to a Toasty value shape-directed —
73    /// the document-interior decode, with no type to consult. Every attribute
74    /// takes its wire-natural form: strings stay `String`, numbers decode by
75    /// integer fit (`I64`, then `U64`, then `F64` — mirroring the JSON codec),
76    /// maps become named `Value::Object`s in stored key order. The engine
77    /// casts the leaves to their field types when it raises the document.
78    fn from_ddb_any(val: &AttributeValue) -> CoreValue {
79        use AttributeValue as AV;
80
81        match val {
82            AV::Bool(val) => stmt::Value::Bool(*val),
83            AV::S(val) => stmt::Value::String(val.clone()),
84            AV::N(val) => {
85                if let Ok(v) = val.parse::<i64>() {
86                    stmt::Value::I64(v)
87                } else if let Ok(v) = val.parse::<u64>() {
88                    stmt::Value::U64(v)
89                } else {
90                    stmt::Value::F64(
91                        val.parse::<f64>()
92                            .expect("numeric attribute is not a number"),
93                    )
94                }
95            }
96            AV::B(val) => stmt::Value::Bytes(val.clone().into_inner()),
97            AV::Null(_) => stmt::Value::Null,
98            AV::L(items) => stmt::Value::List(items.iter().map(Self::from_ddb_any).collect()),
99            AV::M(map) => stmt::Value::Object(stmt::ValueObject::from_vec(
100                map.iter()
101                    .map(|(key, val)| (key.clone(), Self::from_ddb_any(val)))
102                    .collect(),
103            )),
104            _ => todo!("value={:#?}", val),
105        }
106    }

108    /// Converts a document-interior value to a DynamoDB AttributeValue,
109    /// mirroring the SQL drivers' JSON document encoding:
110    ///
111    /// - an object's `Null` entries (an `Option` leaf holding `None`) are
112    ///   omitted; they decode back from the missing key.
113    /// - temporal and decimal leaves store the shared document text form
114    ///   ([`stmt::Value::document_storage_text`], fixed six-digit sub-second
115    ///   precision) — the exact text the engine's document lowering binds as
116    ///   a comparison operand, so a stored leaf and an operand cannot drift
117    ///   apart.
118    /// - everything else shares the column wire forms of
119    ///   [`to_ddb`](Self::to_ddb).
120    fn to_ddb_document(value: &CoreValue) -> AttributeValue {
121        use AttributeValue as AV;
122
123        match value {
124            stmt::Value::Object(object) => {
125                let mut map = HashMap::new();
126                for (key, value) in object.iter() {
127                    if !value.is_null() {
128                        map.insert(key.clone(), Self::to_ddb_document(value));
129                    }
130                }
131                AV::M(map)
132            }
133            stmt::Value::List(items) => AV::L(items.iter().map(Self::to_ddb_document).collect()),
134            value => match value.document_storage_text() {
135                Some(text) => AV::S(text.to_string()),
136                None => Value(value.clone()).to_ddb(),
137            },
138        }
139    }

141    /// Converts a Toasty value to a DynamoDB AttributeValue, shape-directed.
142    pub fn to_ddb(&self) -> AttributeValue {
143        use AttributeValue as AV;
144
145        match &self.0 {
146            stmt::Value::Bool(val) => AV::Bool(*val),
147            stmt::Value::String(val) => AV::S(val.to_string()),
148            stmt::Value::I8(val) => AV::N(val.to_string()),
149            stmt::Value::I16(val) => AV::N(val.to_string()),
150            stmt::Value::I32(val) => AV::N(val.to_string()),
151            stmt::Value::I64(val) => AV::N(val.to_string()),
152            stmt::Value::U8(val) => AV::N(val.to_string()),
153            stmt::Value::U16(val) => AV::N(val.to_string()),
154            stmt::Value::U32(val) => AV::N(val.to_string()),
155            stmt::Value::U64(val) => AV::N(val.to_string()),
156            stmt::Value::F32(val) => AV::N(val.to_string()),
157            stmt::Value::F64(val) => AV::N(val.to_string()),
158            stmt::Value::Bytes(val) => AV::B(val.clone().into()),
159            stmt::Value::Uuid(val) => AV::S(val.to_string()),
160            stmt::Value::List(vals) => {
161                let items = vals
162                    .iter()
163                    .map(|val| Value(val.clone()).to_ddb())
164                    .collect::<Vec<_>>();
165                AV::L(items)
166            }
167            // A `#[document]` value, named by the engine's document lowering:
168            // encode through the document leaf encoding
169            // ([`to_ddb_document`](Self::to_ddb_document)).
170            value @ stmt::Value::Object(_) => Self::to_ddb_document(value),
171            stmt::Value::Null => AV::Null(true),
172            // Scalars whose DynamoDB representation is their string form
173            // (temporals, decimals): the same `Type::cast` conversions the
174            // engine applies to scalar columns of these types. The fixed
175            // sub-second precision keeps lexicographic string order
176            // chronological, so range filters on these attributes compare
177            // correctly.
178            value => match stmt::Type::String.cast(&(), value.clone()) {
179                Ok(stmt::Value::String(s)) => AV::S(s),
180                _ => todo!("{:#?}", self.0),
181            },
182        }
183    }

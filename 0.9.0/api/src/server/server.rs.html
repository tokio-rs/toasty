1//! service-ops (server binary): how you OPERATE a Toasty service — a pooled `Db`, structured
2//! tracing, and a sibling `migrate` binary for schema changes. Models live in `src/lib.rs`.
3//!
4//! Run it (see each statement Toasty issues):
5//!   RUST_LOG=toasty=debug cargo run -p example-service-ops --bin server
6
7use example_service_ops::{ApiKey, Tenant, build_db};
8use tracing_subscriber::EnvFilter;
9
10#[tokio::main]
11async fn main() -> anyhow::Result<()> {
12    // Toasty emits structured tracing events but prints NOTHING until a subscriber is
13    // installed. With one, `RUST_LOG=toasty=debug` shows each statement (db.system /
14    // db.statement). Parameter VALUES are never logged — log them yourself if you need them.
15    tracing_subscriber::fmt()
16        .with_env_filter(EnvFilter::from_default_env())
17        .init();
18
19    let mut db = build_db().await?;
20
21    // `push_schema` is the fast path for dev and tests. The `migrate` binary is the production
22    // path for evolving a schema that already holds data.
23    db.push_schema().await?;
24
25    // Get-or-create so the demo is re-runnable against the persistent database.
26    let tenant = match Tenant::get_by_slug(&mut db, "acme").await {
27        Ok(tenant) => tenant,
28        Err(_) => {
29            toasty::create!(Tenant {
30                slug: "acme",
31                name: "Acme Inc",
32                api_keys: [{ token: "key_live_1" }, { token: "key_live_2" }],
33            })
34            .exec(&mut db)
35            .await?
36        }
37    };
38
39    let keys = tenant.api_keys().exec(&mut db).await?;
40    println!("tenant {:?} has {} api key(s)", tenant.name, keys.len());
41
42    // Authenticate a request by its unique token.
43    let key = ApiKey::get_by_token(&mut db, "key_live_1").await?;
44    println!("token key_live_1 belongs to tenant {}", key.tenant_id);
45
46    Ok(())
47}

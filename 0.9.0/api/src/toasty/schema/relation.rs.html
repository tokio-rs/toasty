7/// A Rust field type that represents a `#[has_many]` relation.
8///
9/// Implemented by [`Vec<M>`](Vec) (eager) and
10/// [`Deferred<Vec<M>>`](super::Deferred) (lazy) where `M: Model`. The set of
11/// impls is the source of truth for which Rust shapes are valid as a
12/// has-many field: anything outside those two combinations does not satisfy
13/// the trait.
14///
15/// A direct `#[has_many]` pairs with a `BelongsTo` on its target. A
16/// `#[has_many(via = ...)]` follows a relation path instead. Many-to-many
17/// models use a direct field for join-model records and a `via` field for the
18/// opposite endpoints.
19pub trait RelationManyField: Load<Output = Self> {
20    /// The target model that this field references.
21    type Target: Model;
22
23    /// Whether the field stores its value in a deferred load slot.
24    const DEFERRED: bool;
25
26    /// Reloads this relation field from a returned value.
27    fn reload(target: &mut Self, value: stmt::Value) -> crate::Result<()>;
28
29    /// Build the [`FieldTy`] for a `HasMany` relation field, given the
30    /// singular name derived from the field identifier and an optional
31    /// paired `BelongsTo` field on the target model resolved from
32    /// `#[has_many(pair = <field>)]`. When `None`, the linker selects the
33    /// pair by searching the target for a unique `BelongsTo` back to the
34    /// source.
35    ///
36    /// `via` carries the fully resolved [`stmt::Path`] of a
37    /// `#[has_many(via = a.b)]` multi-step relation, rooted at the declaring
38    /// model. A `via` relation has no pair.
39    fn many_relation_field_ty(
40        singular: Name,
41        pair: Option<FieldId>,
42        via: Option<stmt::Path>,
43    ) -> FieldTy;
44}

1//! Helpers for resolving app-level relation paths.
2
3use toasty_core::schema::app;
4
5/// Walk a relation path, inlining each `via` field's resolved relation chain.
6///
7/// The result contains only direct relation fields. Scalar terminals are
8/// omitted because they project a value from the model reached by the
9/// relation chain; they do not add another relation step.
10pub(super) fn flatten_relation_path(
11    schema: &toasty_core::Schema,
12    source_model_id: app::ModelId,
13    initial_steps: &[usize],
14) -> Vec<app::FieldId> {
15    let mut result = Vec::with_capacity(initial_steps.len());
16    let mut current_model = source_model_id;
17    let mut queue: Vec<usize> = initial_steps.to_vec();
18    queue.reverse();
19
20    while let Some(index) = queue.pop() {
21        let field = &schema.app.model(current_model).as_root_unwrap().fields[index];
22        let field_id = app::FieldId {
23            model: current_model,
24            index,
25        };
26
27        if let app::FieldTy::Via(via) = &field.ty {
28            debug_assert_eq!(via.path.root.as_model(), Some(current_model));
29
30            for step in via_relation_steps(via).iter().rev() {
31                queue.push(*step);
32            }
33            continue;
34        }
35
36        current_model = field
37            .relation_target_id()
38            .expect("relation path step is not a relation");
39        result.push(field_id);
40    }
41
42    result
43}
44
45/// Return the direct relation fields that a `via` relation follows.
46pub(super) fn flatten_via_path(
47    schema: &toasty_core::Schema,
48    via: &app::Via,
49) -> Option<Vec<app::FieldId>> {
50    let root = via.path.root.as_model()?;
51    Some(flatten_relation_path(schema, root, via_relation_steps(via)))
52}
53
54fn via_relation_steps(via: &app::Via) -> &[usize] {
55    let projection = via.path.projection.as_slice();
56    match via.terminal {
57        Some(_) => &projection[..projection.len() - 1],
58        None => projection,
59    }
60}

1//! product-search: a storefront catalog/search endpoint — filter, sort, page, and project,
2//! the operations a product list or search screen runs on every request.
3//!
4//! Run it cold (`cargo run -p example-product-search`). Uses in-memory SQLite by default;
5//! set `TOASTY_CONNECTION_URL` for another backend. `.like()` is SQL-only; the portable
6//! prefix matcher `.starts_with()` works everywhere (see the note inline).
7
8#[derive(Debug, toasty::Model)]
9// A composite index over (category, price). It also generates the leftmost-prefix lookup
10// `filter_by_category`, so `category` needs no separate field-level `#[index]` — adding one
11// would emit duplicate methods and fail to compile.
12#[index(category, price)]
13struct Product {
14    #[key]
15    #[auto]
16    id: uuid::Uuid,
17    name: String,
18    // An open, extensible taxonomy — new categories appear without code changes — so this is a
19    // `String` (a real app might make it its own model). Contrast a small *closed* set of
20    // states, which is better modeled as an embedded enum (see store-operations' OrderStatus).
21    category: String,
22    price: i64, // cents
23    in_stock: bool,
24}
25
26#[tokio::main]
27async fn main() -> toasty::Result<()> {
28    let url =
29        std::env::var("TOASTY_CONNECTION_URL").unwrap_or_else(|_| "sqlite::memory:".to_string());
30    let mut db = toasty::Db::builder()
31        .models(toasty::models!(crate::*))
32        .connect(&url)
33        .await?;
34    db.push_schema().await?;
35
36    // --- seed: create_many bulk-inserts many rows of ONE model in one round-trip -----
37    // `.item(..)` takes a finished create builder; `.with_item(|c| ..)` configures one inline.
38    Product::create_many()
39        .item(toasty::create!(Product {
40            name: "Pro Keyboard",
41            category: "input",
42            price: 7999,
43            in_stock: true,
44        }))
45        .with_item(|c| c.name("Mouse").category("input").price(2999).in_stock(true))
46        .with_item(|c| {
47            c.name("4K Monitor")
48                .category("display")
49                .price(39999)
50                .in_stock(true)
51        })
52        .with_item(|c| {
53            c.name("Webcam")
54                .category("input")
55                .price(4999)
56                .in_stock(false)
57        })
58        .with_item(|c| {
59            c.name("USB Cable")
60                .category("input")
61                .price(999)
62                .in_stock(true)
63        })
64        .exec(&mut db)
65        .await?;
66
67    // --- filtering: predicates built from fields() accessors -------------------------
68    // Precedence is LEFT-TO-RIGHT method chaining, NOT SQL precedence: `a.or(b).and(c)`
69    // parses as `(a OR b) AND c`. Group differently by nesting the arguments.
70    let cheap_input = Product::filter(
71        Product::fields()
72            .category()
73            .eq("input")
74            .and(Product::fields().price().lt(3000)),
75    )
76    .exec(&mut db)
77    .await?;
78    println!("input items under $30: {}", cheap_input.len());
79
80    let two_cats = Product::filter(Product::fields().category().in_list(["input", "display"]))
81        .exec(&mut db)
82        .await?;
83    println!("items in input or display: {}", two_cats.len());
84
85    // `starts_with` is a portable, case-sensitive prefix match (works on every backend).
86    let pro = Product::filter(Product::fields().name().starts_with("Pro"))
87        .exec(&mut db)
88        .await?;
89    println!("names starting with \"Pro\": {}", pro.len());
90    // `.like()` is SQL-only and its case-sensitivity varies per backend (postgres-directory
91    // demonstrates PostgreSQL's case-insensitive `.ilike()`).
92    let cables = Product::filter(Product::fields().name().like("%Cable%"))
93        .exec(&mut db)
94        .await?;
95    println!("names matching %Cable%: {}", cables.len());
96
97    // --- generated lookups from the composite index ----------------------------------
98    // One method per leftmost prefix: `filter_by_category` and `filter_by_category_and_price`.
99    let input = Product::filter_by_category("input").exec(&mut db).await?;
100    let exact = Product::filter_by_category_and_price("input", 2999)
101        .exec(&mut db)
102        .await?;
103    println!(
104        "category=input: {}; category=input & price=2999: {}",
105        input.len(),
106        exact.len()
107    );
108
109    // --- sorting ---------------------------------------------------------------------
110    // A tuple is a multi-field sort: (primary, tie-breaker).
111    let sorted = Product::all()
112        .order_by((
113            Product::fields().price().asc(),
114            Product::fields().name().desc(),
115        ))
116        .exec(&mut db)
117        .await?;
118    let names: Vec<&str> = sorted.iter().map(|p| p.name.as_str()).collect();
119    println!("cheapest first: {names:?}");
120
121    // --- limit & offset --------------------------------------------------------------
122    // `offset` requires a `limit` first. Fine for small data; doesn't scale (use a cursor).
123    let page2 = Product::all()
124        .order_by(Product::fields().price().asc())
125        .limit(2)
126        .offset(2)
127        .exec(&mut db)
128        .await?;
129    let page2_names: Vec<&str> = page2.iter().map(|p| p.name.as_str()).collect();
130    println!("rows 3-4 by ascending price: {page2_names:?}");
131
132    // --- cursor pagination (requires order_by) ---------------------------------------
133    // `per_page` is an UPPER BOUND — walk `.next()` to the end rather than counting rows.
134    let mut page = Product::all()
135        .order_by(Product::fields().price().desc())
136        .paginate(2)
137        .exec(&mut db)
138        .await?;
139    let mut total = 0;
140    loop {
141        total += page.len();
142        match page.next(&mut db).await? {
143            Some(next) => page = next,
144            None => break,
145        }
146    }
147    println!("cursor-paginated through {total} products in pages of 2");
148
149    // --- projection: read only the columns you need ----------------------------------
150    // The return type follows the projection: one field -> `Vec<Field>`, a tuple -> `Vec<(..)>`.
151    let just_names: Vec<String> = Product::all()
152        .select(Product::fields().name())
153        .exec(&mut db)
154        .await?;
155    let id_price: Vec<(uuid::Uuid, i64)> = Product::all()
156        .select((Product::fields().id(), Product::fields().price()))
157        .exec(&mut db)
158        .await?;
159    println!(
160        "projected {} names and {} (id, price) pairs",
161        just_names.len(),
162        id_price.len()
163    );
164
165    Ok(())
166}

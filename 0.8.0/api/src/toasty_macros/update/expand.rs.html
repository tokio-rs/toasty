7/// Top-level entry point. Expand an `update!` invocation into the
8/// equivalent update-builder method chain.
9///
10/// The target expression is evaluated once at `.update()`, not at
11/// some earlier `let` binding. This is what lets instance updates use
12/// a bare identifier: `update!(user { ... })` expands to
13/// `user.update()`, which auto-borrows `&mut user` the same way the
14/// chain form does, leaving the original binding owned. Query
15/// targets pass straight through and are consumed.
16///
17/// The fields struct needed for embedded-patch paths is recovered
18/// from the bound update builder via the macro-only
19/// `__macro_fields_root()` method.
20///
21/// Field-name validation falls out of the chain expansion: each named
22/// field becomes a method call on the update builder, so a typo
23/// surfaces as a "no method" error at the macro call site without the
24/// macro doing anything specific.
25pub(crate) fn expand(item: &UpdateItem) -> TokenStream {
26    let target = &item.target;
27    let target_span = target.span();
28
29    let fields_path = quote!(__fields);
30    let field_calls = expand_field_set(&item.fields, &fields_path);
31
32    quote_spanned! { target_span=>
33        {
34            let __update = #target.update();
35
36            // Fields path used for embedded-patch path construction.
37            // `unused_variables` is silenced because the update may
38            // have only top-level scalar entries that don't need it.
39            #[allow(unused_variables)]
40            let #fields_path = __update.__macro_fields_root();
41
42            __update #(#field_calls)*
43        }
44    }
45}

47/// Expand a [`FieldSet`] into a list of method calls.
48fn expand_field_set(fields: &FieldSet, parent_path: &TokenStream) -> Vec<TokenStream> {
49    fields
50        .0
51        .iter()
52        .map(|entry| expand_field_entry(entry, parent_path))
53        .collect()
54}

56/// Expand a single field entry into a method-call token stream.
57///
58/// `parent_path` is the fields-path expression for the parent type
59/// (e.g. `__target_fields` at the top level, or
60/// `parent_path.embedded_field()` for a nested embedded entry).
61fn expand_field_entry(entry: &FieldEntry, parent_path: &TokenStream) -> TokenStream {
62    let name = &entry.name;
63    let span = name.span();
64
65    match &entry.value {
66        FieldValue::Expr(expr) => {
67            quote_spanned! { span=> .#name(#expr) }
68        }
69        FieldValue::Method { combinator, args } => {
70            quote_spanned! { span=> .#name(toasty::stmt::#combinator(#args)) }
71        }
72        FieldValue::Single(sub_fields) => {
73            // Embedded partial update — expand into
74            // `stmt::apply([stmt::patch(<sub_root>.sub(), val), ...])`.
75            let nested_path = quote_spanned! { span=> #parent_path.#name() };
76            let patches = expand_patch_entries(sub_fields, &nested_path);
77            quote_spanned! { span=>
78                .#name(toasty::stmt::apply([ #( #patches ),* ]))
79            }
80        }
81        FieldValue::List(items) => {
82            // Two emission shapes, depending on whether any list item is
83            // a `{ ... }` create-builder shorthand:
84            //
85            // - With a builder shorthand (only meaningful for has-many),
86            //   wrap each builder in `stmt::insert(..)` and the whole
87            //   list in `stmt::apply([..])` so the inserts compose with
88            //   any plain `stmt::*` siblings:
89            //
90            //     update!(user {
91            //         todos: [{ title: "x" }, stmt::remove(&old)],
92            //     })
93            //     // →
94            //     // .todos(stmt::apply([
95            //     //     stmt::insert(<child create>),
96            //     //     stmt::remove(&old),
97            //     // ]))
98            //
99            // - With every item a plain expression, pass the array
100            //   straight through. For `Vec<scalar>` this hits the
101            //   setter's `Assign<List<T>>` impl as set semantics:
102            //
103            //     update!(article { tags: ["a", "b"] })
104            //     // → .tags(["a", "b"])
105            let any_builder = items
106                .iter()
107                .any(|item| matches!(item, FieldValue::Single(_)));
108            let nested_path = quote_spanned! { span=> #parent_path.#name() };
109
110            if any_builder {
111                let entries = items
112                    .iter()
113                    .map(|item| expand_has_many_list_item(item, &nested_path, span));
114                quote_spanned! { span=>
115                    .#name(toasty::stmt::apply([ #( #entries ),* ]))
116                }
117            } else {
118                let entries = items.iter().map(|item| match item {
119                    FieldValue::Expr(e) => quote!(#e),
120                    // Nested `[ [...] ]` is rejected by the macro.
121                    FieldValue::List(_) => quote_spanned! { span=>
122                        compile_error!("nested lists are not supported in update!")
123                    },
124                    // Single/Method don't appear here because the
125                    // any_builder check above routed us to the other
126                    // branch when a Single was present, and Method only
127                    // appears as a top-level entry shape.
128                    FieldValue::Single(_) | FieldValue::Method { .. } => unreachable!(),
129                });
130                quote_spanned! { span=>
131                    .#name([ #( #entries ),* ])
132                }
133            }
134        }
135    }
136}

138/// Expand each entry of an embedded brace block into a
139/// `stmt::patch(<rooted>.sub_field(), value)` invocation.
140fn expand_patch_entries(fields: &FieldSet, parent_path: &TokenStream) -> Vec<TokenStream> {
141    fields
142        .0
143        .iter()
144        .map(|entry| expand_patch_entry(entry, parent_path))
145        .collect()
146}

148/// Expand a single sub-entry of an embedded brace block.
149fn expand_patch_entry(entry: &FieldEntry, parent_path: &TokenStream) -> TokenStream {
150    let name = &entry.name;
151    let span = name.span();
152
153    // The patch path is rooted at the embedded type itself, obtained
154    // via the `into_root()` inherent method on the parent fields path.
155    let rooted = quote_spanned! { span=> #parent_path.into_root() };
156
157    match &entry.value {
158        FieldValue::Expr(expr) => {
159            quote_spanned! { span=>
160                toasty::stmt::patch(#rooted.#name(), #expr)
161            }
162        }
163        FieldValue::Method { combinator, args } => {
164            quote_spanned! { span=>
165                toasty::stmt::patch(
166                    #rooted.#name(),
167                    toasty::stmt::#combinator(#args),
168                )
169            }
170        }
171        FieldValue::Single(sub_fields) => {
172            // Nested embedded patch — recurse.
173            let nested_path = quote_spanned! { span=> #parent_path.#name() };
174            let inner_patches = expand_patch_entries(sub_fields, &nested_path);
175            quote_spanned! { span=>
176                toasty::stmt::patch(
177                    #rooted.#name(),
178                    toasty::stmt::apply([ #( #inner_patches ),* ]),
179                )
180            }
181        }
182        FieldValue::List(_) => {
183            quote_spanned! { span=>
184                compile_error!("list values are not supported inside embedded patch blocks")
185            }
186        }
187    }
188}

190/// Expand a single item inside a has-many list literal.
191///
192/// A brace block builds a create builder and is wrapped in
193/// `stmt::insert`. A plain expression passes through as a complete
194/// `Assignment<List<T>>` value (e.g. `stmt::remove(&old)`).
195fn expand_has_many_list_item(
196    item: &FieldValue,
197    parent_path: &TokenStream,
198    span: proc_macro2::Span,
199) -> TokenStream {
200    match item {
201        FieldValue::Single(fields) => {
202            // Build a child create builder via the field-list struct's
203            // `.create()` method. A typo in any setter name surfaces as
204            // a "no method named …" error at the macro call site.
205            let setters = expand_create_setters(fields);
206            quote_spanned! { span=>
207                toasty::stmt::insert(#parent_path.create() #(#setters)*)
208            }
209        }
210        FieldValue::Expr(e) => quote!(#e),
211        FieldValue::List(_) => quote_spanned! { span=>
212            compile_error!("nested lists are not supported in update!")
213        },
214        FieldValue::Method { .. } => quote_spanned! { span=>
215            compile_error!("method shorthand is not supported as a list item")
216        },
217    }
218}

220/// Expand the setters for a create-builder list item. Same shape as
221/// `expand_field_entry`, but limited to entries that have a
222/// counterpart on a create builder. Deeper nesting inside a has-many
223/// list item is rejected for v1.
224fn expand_create_setters(fields: &FieldSet) -> Vec<TokenStream> {
225    fields
226        .0
227        .iter()
228        .map(|entry| {
229            let name = &entry.name;
230            let span = name.span();
231            match &entry.value {
232                FieldValue::Expr(expr) => quote_spanned! { span=> .#name(#expr) },
233                FieldValue::Method { combinator, args } => {
234                    quote_spanned! { span=> .#name(toasty::stmt::#combinator(#args)) }
235                }
236                FieldValue::Single(_) | FieldValue::List(_) => quote_spanned! { span=>
237                    .#name(compile_error!(
238                        "deeper nesting inside a has-many list item is not supported in update!"
239                    ))
240                },
241            }
242        })
243        .collect()
244}

1//! crm-embedded: a contact card composed from EMBEDDED value types — a typed id, a
2//! flattened address, and a tagged-union contact method. None of these gets its own table;
3//! their columns are flattened into the `contacts` table.
4//!
5//! Run it cold (`cargo run -p example-crm-embedded`). In-memory SQLite by default; set
6//! `TOASTY_CONNECTION_URL` for another backend.
7
8// A newtype embed maps to ONE un-prefixed column. As a `#[key]` with `#[auto]`, generation
9// proxies to the inner type — here a time-ordered UUID v7.
10#[derive(Debug, toasty::Embed)]
11struct ContactId(uuid::Uuid);
12
13// A multi-field embed flattens into prefixed columns: `address_street`, `address_city`,
14// `address_postal_code`. An `#[index]` on an embedded field indexes the flattened column.
15#[derive(Debug, toasty::Embed)]
16struct Address {
17    street: String,
18    city: String,
19    #[index]
20    postal_code: String,
21}
22
23// An embedded enum stores a discriminant column plus one nullable column per data field. With
24// no `#[column(variant)]`, the discriminant is the variant name in snake_case ("email",
25// "phone"); override it with `#[column(variant = "...")]` or an integer. Variants may carry
26// their own fields.
27#[derive(Debug, PartialEq, toasty::Embed)]
28enum ContactMethod {
29    Email { address: String },
30    Phone { number: String },
31}
32
33#[derive(Debug, toasty::Model)]
34struct Contact {
35    #[key]
36    #[auto]
37    id: ContactId,
38    name: String,
39    address: Address,
40    method: ContactMethod,
41}
42
43#[tokio::main]
44async fn main() -> toasty::Result<()> {
45    let url =
46        std::env::var("TOASTY_CONNECTION_URL").unwrap_or_else(|_| "sqlite::memory:".to_string());
47    let mut db = toasty::Db::builder()
48        .models(toasty::models!(crate::*))
49        .connect(&url)
50        .await?;
51    db.push_schema().await?;
52
53    // Build contacts entirely from embedded values — no extra tables involved.
54    let mut dana = toasty::create!(Contact {
55        name: "Dana",
56        address: Address {
57            street: "1 Main St".into(),
58            city: "Seattle".into(),
59            postal_code: "98101".into(),
60        },
61        method: ContactMethod::Email {
62            address: "dana@example.com".into(),
63        },
64    })
65    .exec(&mut db)
66    .await?;
67    toasty::create!(Contact {
68        name: "Ed",
69        address: Address {
70            street: "5 Oak Ave".into(),
71            city: "Portland".into(),
72            postal_code: "97201".into(),
73        },
74        method: ContactMethod::Phone {
75            number: "555-0100".into(),
76        },
77    })
78    .exec(&mut db)
79    .await?;
80
81    // Filter by a flattened sub-field of the embedded struct.
82    let seattle = Contact::filter(Contact::fields().address().city().eq("Seattle"))
83        .exec(&mut db)
84        .await?;
85    println!("contacts in Seattle: {}", seattle.len());
86
87    // Embedded enums generate `is_<variant>()` predicates, plus `.matches()` to reach into a
88    // data variant's fields from a query.
89    let emailable = Contact::filter(Contact::fields().method().is_email())
90        .exec(&mut db)
91        .await?;
92    println!("contacts reachable by email: {}", emailable.len());
93    let exact = Contact::filter(
94        Contact::fields()
95            .method()
96            .email()
97            .matches(|e| e.address().eq("dana@example.com")),
98    )
99    .exec(&mut db)
100    .await?;
101    println!("contacts with that exact email: {}", exact.len());
102
103    // PATCH one embedded sub-field; the others (street, postal_code) are preserved. Passing a
104    // whole `Address` value instead would REPLACE the entire embed.
105    toasty::update!(dana {
106        address: toasty::stmt::patch(Address::fields().city(), "Bellevue"),
107    })
108    .exec(&mut db)
109    .await?;
110    // Patch several fields at once with `stmt::apply`.
111    toasty::update!(dana {
112        address: toasty::stmt::apply([
113            toasty::stmt::patch(Address::fields().street(), "456 Oak Ave"),
114            toasty::stmt::patch(Address::fields().postal_code(), "98004"),
115        ]),
116    })
117    .exec(&mut db)
118    .await?;
119
120    let reloaded = Contact::get_by_id(&mut db, &dana.id).await?;
121    println!(
122        "Dana's patched address: {} {} {}",
123        reloaded.address.street, reloaded.address.city, reloaded.address.postal_code
124    );
125
126    Ok(())
127}

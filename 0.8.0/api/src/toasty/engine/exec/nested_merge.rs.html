8/// Combines parent and child data into nested structures.
9///
10/// The nested merge algorithm processes hierarchical data by:
11///
12/// 1. Loading all batch data upfront - fetches all input data for all levels
13///    before processing
14/// 2. Building all hash indexes from the pre-computed `indexes` list
15/// 3. Processing each root row:
16///    - For each nested child relationship at this level:
17///      - Filters batch-loaded child data using the qualification (hash lookup
18///        or scan predicate)
19///      - Recursively merges each matching child row with its own children
20///      - Collects results into a list, or a single value if `single` is `true`
21///    - Projects the final row by applying the projection function with the
22///      current row and all nested children
23///    - Adds the projected row to output
24/// 4. Returning all merged rows with their nested data
25///
26/// # Note
27///
28/// Rows loaded from batch queries are not the final projection. They may include
29/// extra fields needed for filtering or projecting nested children.
30#[derive(Debug, Clone)]
31pub(crate) struct NestedMerge {
32    /// Input sources. NestedLevel will reference their inputs by index in this vec.
33    pub(crate) inputs: Vec<VarId>,
34
35    /// Output variable, where to store the merged values
36    pub(crate) output: Output,
37
38    /// The root level
39    pub(crate) root: NestedLevel,
40
41    /// Flat list of hash indexes to build before the merge, computed at plan time.
42    ///
43    /// Each entry describes which input to index and which fields form the key.
44    /// `MergeQualification::HashLookup` references entries by position in this vec.
45    pub(crate) hash_indexes: Vec<MergeIndex>,
46
47    /// Flat list of sorted indexes to build before the merge, computed at plan time.
48    ///
49    /// `MergeQualification::SortLookup` references entries by position in this vec.
50    pub(crate) sort_indexes: Vec<MergeIndex>,
51}
52
53/// Describes one hash index to build over an input before the merge begins.
54///
55/// Computed entirely at plan time; execution builds the index mechanically.
56#[derive(Debug, Clone)]
57pub(crate) struct MergeIndex {
58    /// Index into `NestedMerge::inputs` — which input to build the index over.
59    pub(crate) source: usize,
60
61    /// Which field(s) of the child record form the hash key.
62    pub(crate) child_projections: Vec<stmt::Projection>,
63}
64
65/// A single level in the nesting hierarchy
66///
67/// Each level represents one child relationship and contains:
68/// - How to load the child data
69/// - How to filter it for each parent (can reference ANY ancestor)
70/// - How to project it (may include ExprArgs for its own children)
71/// - Its own children (recursive nesting)
72#[derive(Debug, Clone)]
73pub(crate) struct NestedLevel {
74    /// Input for this level as an index in `NestedMerge::inputs`
75    pub(crate) source: usize,
76
77    /// Projection for this level (before passing to parent) Argument 0 is the
78    /// row for the current level, all other arguments are the results of the
79    /// recursive nested merge.
80    pub(crate) projection: eval::Func,
81
82    /// This level's children (recursive nesting)
83    /// Empty for leaf nodes
84    pub(crate) nested: Vec<NestedChild>,
85}
86
87#[derive(Debug, Clone)]
88pub(crate) struct NestedChild {
89    /// The nested level for this child
90    pub(crate) level: NestedLevel,
91
92    /// How to filter rows to match the parent request
93    pub(crate) qualification: MergeQualification,
94
95    /// True if single value
96    pub(crate) single: bool,
97}

99/// How to filter nested records for a parent record
100#[derive(Debug, Clone)]
101pub(crate) enum MergeQualification {
102    /// Parent selects all records provided from sub statements.
103    ///
104    /// This is typically used when the parent query returns only one row, so all
105    /// nested records end up associated with that single parent record.
106    All,
107
108    /// Hash index lookup — O(1) per parent row after a one-time O(M) index build.
109    ///
110    /// For `single=true` (has_one / belongs_to) relationships only, since
111    /// `HashIndex` requires unique keys across the child set.
112    ///
113    /// `index` is a position into `NestedMerge::hash_indexes`.
114    /// `lookup_key` evaluates against the ancestor row stack to produce the key.
115    HashLookup {
116        /// Position in `NestedMerge::hash_indexes`.
117        index: usize,
118        /// Evaluates against the ancestor `RowStack` to produce the lookup key.
119        lookup_key: eval::Func,
120    },
121
122    /// Sorted index lookup — O(log M + k) per parent row after a one-time O(M log M) build.
123    ///
124    /// For `single=false` (has_many) relationships. Supports duplicate keys:
125    /// `find_range(Included(key), Included(key))` returns all matching children.
126    ///
127    /// `index` is a position into `NestedMerge::sort_indexes`.
128    SortLookup {
129        /// Position in `NestedMerge::sort_indexes`.
130        index: usize,
131        /// Evaluates against the ancestor `RowStack` to produce the lookup key.
132        lookup_key: eval::Func,
133    },
134
135    /// General predicate evaluation (nested loop — O(N×M)).
136    ///
137    /// Args: [ancestor_stack..., nested_record] -> bool
138    ///
139    /// The ancestor stack contains all ancestors from root to immediate parent:
140    /// [root, child, grandchild, ..., immediate_parent, nested_record]
141    ///
142    /// Execution: For each parent record, evaluate predicate against all
143    /// nested records. No indexing.
144    Scan(eval::Func),
145}

172impl Exec<'_> {
173    pub(super) async fn action_nested_merge(&mut self, action: &NestedMerge) -> Result<()> {
174        // Load all input data upfront
175        let mut inputs = Vec::with_capacity(action.inputs.len());
176
177        for var_id in &action.inputs {
178            let response = self.vars.load(*var_id).await?;
179            inputs.push(match response.values {
180                Rows::Count(count) => Input::Count(count),
181                Rows::Value(value) => Input::Value(match value {
182                    stmt::Value::List(items) => items,
183                    value => vec![value],
184                }),
185                Rows::Stream(value_stream) => Input::Value(value_stream.collect().await?),
186            });
187        }
188
189        // Build all hash and sorted indexes from the plan-time-computed flat lists.
190        let indices = Indices {
191            hash: action
192                .hash_indexes
193                .iter()
194                .map(|mi| {
195                    let Input::Value(values) = &inputs[mi.source] else {
196                        panic!("HashLookup source must be a Value input, not Count")
197                    };
198                    stmt::HashIndex::new(values, &mi.child_projections)
199                })
200                .collect(),
201            sort: action
202                .sort_indexes
203                .iter()
204                .map(|mi| {
205                    let Input::Value(values) = &inputs[mi.source] else {
206                        panic!("SortLookup source must be a Value input, not Count")
207                    };
208                    stmt::SortedIndex::new(values, &mi.child_projections)
209                })
210                .collect(),
211        };
212
213        // Load the root rows
214        let mut merged_rows = vec![];
215
216        match &inputs[action.root.source] {
217            Input::Count(count) => {
218                let row_stack = RowStack {
219                    parent: None,
220                    // Bit of a hack
221                    row: &stmt::Value::Null,
222                    position: 0,
223                };
224
225                for _ in 0..*count {
226                    merged_rows.push(self.merge_nested_row(
227                        &row_stack,
228                        &action.root,
229                        &inputs,
230                        &indices,
231                    )?);
232                }
233            }
234            Input::Value(root_rows) => {
235                // Iterate over each record to perform the nested merge
236                for row in root_rows {
237                    let stack = RowStack {
238                        parent: None,
239                        row,
240                        position: 0,
241                    };
242                    merged_rows.push(self.merge_nested_row(
243                        &stack,
244                        &action.root,
245                        &inputs,
246                        &indices,
247                    )?);
248                }
249            }
250        }
251
252        // Store the output
253        self.vars.store(
254            action.output.var,
255            action.output.num_uses,
256            ExecResponse::from_rows(Rows::value_stream(merged_rows)),
257        );
258
259        Ok(())
260    }

262    /// Recursively merges a single row with its nested child data.
263    ///
264    /// This is the core recursive function that processes one row at a time through
265    /// the nesting hierarchy. For each row, it:
266    ///
267    /// 1. **Processes each child relationship**: Iterates through all nested children
268    ///    defined at this level
269    /// 2. **Filters child rows**: For each child, uses either a pre-built hash index
270    ///    (O(1) lookup) or a linear scan with the qualification predicate
271    /// 3. **Recursively merges**: For each matching child row, recursively calls
272    ///    `merge_nested_row` to process its own children
273    /// 4. **Collects results**: Gathers all processed child rows into a list (or
274    ///    single value if `single: true`)
275    /// 5. **Projects final result**: Applies the projection function with the current
276    ///    row and all nested children as arguments to produce the final output
277    ///
278    /// # Arguments
279    ///
280    /// * `row_stack` - The ancestor stack containing this row and all parent rows,
281    ///   used for evaluating predicates that reference parent data
282    /// * `level` - The current nesting level descriptor with source, projection, and children
283    /// * `inputs` - All batch-loaded data for the entire merge, indexed by source
284    /// * `indexes` - Pre-built hash indexes, indexed by position in `NestedMerge::indexes`
285    ///
286    /// # Returns
287    ///
288    /// The projected value for this row, with all nested children merged in according
289    /// to the projection function.
290    fn merge_nested_row(
291        &self,
292        row_stack: &RowStack<'_>,
293        level: &NestedLevel,
294        inputs: &[Input],
295        indices: &Indices<'_>,
296    ) -> Result<stmt::Value> {
297        // Collected all nested rows for this row.
298        let mut nested = vec![];
299
300        for nested_child in &level.nested {
301            // Find the batch-loaded input
302            let Input::Value(nested_input) = &inputs[nested_child.level.source] else {
303                todo!("input={:#?}", inputs[nested_child.level.source])
304            };
305            let mut nested_rows_projected = vec![];
306
307            // Process a single matching child row: recurse and collect the result.
308            let mut process = |nested_row: &stmt::Value| -> Result<()> {
309                let nested_stack = RowStack {
310                    parent: Some(row_stack),
311                    row: nested_row,
312                    position: row_stack.position + 1,
313                };
314                nested_rows_projected.push(self.merge_nested_row(
315                    &nested_stack,
316                    &nested_child.level,
317                    inputs,
318                    indices,
319                )?);
320                Ok(())
321            };
322
323            match &nested_child.qualification {
324                MergeQualification::All => {
325                    for row in nested_input {
326                        process(row)?;
327                    }
328                }
329                MergeQualification::HashLookup { index, lookup_key } => {
330                    let key_val = lookup_key.eval(row_stack)?;
331                    if let Some(row) = indices.hash[*index].find(key_as_slice(&key_val)) {
332                        process(row)?;
333                    }
334                }
335                MergeQualification::SortLookup { index, lookup_key } => {
336                    let key_val = lookup_key.eval(row_stack)?;
337                    let key = key_as_slice(&key_val);
338                    for row in indices.sort[*index].find_range(
339                        std::ops::Bound::Included(key),
340                        std::ops::Bound::Included(key),
341                    ) {
342                        process(row)?;
343                    }
344                }
345                MergeQualification::Scan(func) => {
346                    for row in nested_input {
347                        let stack = RowStack {
348                            parent: Some(row_stack),
349                            row,
350                            position: row_stack.position + 1,
351                        };
352                        if func.eval_bool(&stack)? {
353                            process(row)?;
354                        }
355                    }
356                }
357            }
358
359            nested.push(if nested_child.single {
360                assert!(nested_rows_projected.len() <= 1, "TODO: error handling");
361
362                if let Some(row) = nested_rows_projected.into_iter().next() {
363                    row
364                } else {
365                    stmt::Value::Null
366                }
367            } else {
368                stmt::Value::List(nested_rows_projected)
369            });
370        }
371
372        // Project the row with the nested data as arguments.
373        let eval_input = RowAndNested {
374            row: row_stack.row,
375            nested: &nested[..],
376        };
377
378        level.projection.eval(&eval_input)
379    }
380}

382/// Extract a `&[Value]` key from a parent key value.
383///
384/// `Record` values are flattened to their field slice (composite key).
385/// All other values are wrapped in a single-element slice (scalar key).
386fn key_as_slice(val: &stmt::Value) -> &[stmt::Value] {
387    match val {
388        stmt::Value::Record(r) => r.as_slice(),
389        v => std::slice::from_ref(v),
390    }

393impl stmt::Input for &RowStack<'_> {
394    fn resolve_arg(
395        &mut self,
396        expr_arg: &stmt::ExprArg,
397        projection: &stmt::Projection,
398    ) -> Option<stmt::Expr> {
399        let mut current: &RowStack<'_> = self;
400
401        // Find the stack level that corresponds with the argument.
402        loop {
403            if current.position == expr_arg.position {
404                break;
405            }
406
407            let Some(parent) = current.parent else {
408                todo!()
409            };
410            current = parent;
411        }
412
413        // Get the value and apply projection
414        Some(current.row.entry(projection).to_expr())
415    }
416}

418impl stmt::Input for &RowAndNested<'_> {
419    fn resolve_arg(
420        &mut self,
421        expr_arg: &stmt::ExprArg,
422        projection: &stmt::Projection,
423    ) -> Option<stmt::Expr> {
424        let base = if expr_arg.position == 0 {
425            self.row
426        } else {
427            &self.nested[expr_arg.position - 1]
428        };
429
430        Some(base.entry(projection).to_expr())
431    }
